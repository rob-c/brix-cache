17/*
18 * Process-global effective seccomp mode: the STRICTEST value requested by ANY
19 * brix server, stream OR http (WebDAV/S3/cvmfs). Set at config-parse time by
20 * brix_conf_set_seccomp() (which every `brix_seccomp` directive, in either
21 * context, routes through) and read once per worker by brix_seccomp_install_once().
22 * A single process-wide value is correct because a seccomp filter is per-process
23 * and one nginx worker serves every configured server. It only ever ratchets UP
24 * within a master's lifetime (a SIGHUP reload cannot lower it — fail-secure;
25 * restart to drop the filter), which is the safe direction for a syscall filter.
26 */
27ngx_uint_t brix_seccomp_worker_mode = BRIX_SECCOMP_OFF;
28
29/*
30 * Process-global "allow execve/execveat under enforce" flag. DEFAULT ON: brix
31 * legitimately fork+execs helpers on common paths (the FRM "exec" MSS adapter,
32 * OIDC token fetch, native-TPC token-exchange, WebDAV HTTP-TPC oidc-agent, the
33 * kXR_prepare hook), and these are exercised by the E2E suite (test_seccomp_exec_frm.py),
34 * so killing exec by default would silently break them the moment an operator turns
35 * `brix_seccomp enforce` on. `brix_seccomp_allow_exec off` on ANY brix server (stream
36 * or http) opts back INTO the strict anti-shell posture — execve/execveat KILLED.
37 * Fail-secure ratchet: `off` wins and sticks (a later `on`, or a reload dropping the
38 * `off`, cannot re-enable exec — restart to reset), the same "ratchet toward secure"
39 * direction as the mode. The HARD kills (ptrace/process_vm_*) are unaffected by this
40 * flag and apply regardless.
41 */
42ngx_uint_t brix_seccomp_allow_exec = 1;
43
44/* Per-worker "already installed" latch so brix_seccomp_install_once() is a no-op
45 * after the first call — the install site is whichever brix init_process runs
46 * for this worker (stream for stream/mixed, webdav for http-only). */
47static ngx_uint_t brix_seccomp_installed = 0;

49char *
50brix_conf_set_seccomp(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
51{
52 ngx_uint_t *field = (ngx_uint_t *) ((char *) conf + cmd->offset);
53 ngx_conf_enum_t *e = cmd->post;
54 ngx_str_t *value = cf->args->elts;
55 ngx_uint_t i;
56
57 if (*field != NGX_CONF_UNSET_UINT && *field != BRIX_SECCOMP_OFF) {
58 return "is duplicate";
59 }
60 for (i = 0; e[i].name.len != 0; i++) {
61 if (e[i].name.len == value[1].len
62 && ngx_strcasecmp(e[i].name.data, value[1].data) == 0)
63 {
64 *field = e[i].value;
65 /* Bump the process-global to the strictest requested anywhere. */
66 if (e[i].value > brix_seccomp_worker_mode) {
67 brix_seccomp_worker_mode = e[i].value;
68 }
69 return NGX_CONF_OK;
70 }
71 }
72 return "invalid value";
73}

86char *
87brix_conf_set_seccomp_allow_exec(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
88{
89 ngx_str_t *value = cf->args->elts;
90
91 (void) cmd;
92 (void) conf;
93
94 if (value[1].len == 3 && ngx_strncasecmp(value[1].data,
95 (u_char *) "off", 3) == 0)
96 {
97 brix_seccomp_allow_exec = 0; /* ratchet toward strict: off wins */
98 return NGX_CONF_OK;
99 }
100 if (value[1].len == 2 && ngx_strncasecmp(value[1].data,
101 (u_char *) "on", 2) == 0)
102 {
103 /* on is the default; a no-op that must NOT override a prior `off`
104 * (fail-secure — once any server disabled exec, restart to re-enable). */
105 return NGX_CONF_OK;
106 }
107 return "must be \"on\" or \"off\"";
108}

112static void
113brix_seccomp_log_err(void *ud, const char *name, int rc)
114{
115 ngx_cycle_t *cycle = ud;
116
117 if (name != NULL) {
118 ngx_log_error(NGX_LOG_EMERG, cycle->log, 0,
119 "brix_seccomp: rule add for \"%s\" failed: %s",
120 name, strerror(-rc));
121 } else if (rc != 0) {
122 ngx_log_error(NGX_LOG_EMERG, cycle->log, 0,
123 "brix_seccomp: filter load failed: %s", strerror(-rc));
124 } else {
125 ngx_log_error(NGX_LOG_EMERG, cycle->log, 0,
126 "brix_seccomp: filter init failed");
127 }

130ngx_int_t
131brix_seccomp_install(ngx_cycle_t *cycle, ngx_uint_t mode, ngx_uint_t allow_exec)
132{
133 unsigned n_allow = 0;
134 unsigned n_deny = 0;
135 int rc;
136
137 if (mode == BRIX_SECCOMP_OFF) {
138 return NGX_OK;
139 }
140
141 rc = brix_seccomp_core_apply((unsigned) mode, (unsigned) allow_exec,
142 brix_seccomp_log_err, cycle, &n_allow, &n_deny);
143
144 if (rc == BRIX_SECCOMP_CORE_UNAVAIL) {
145 /* Fail closed: the operator asked for a syscall filter (audit or
146 * enforce) but this binary was built without libseccomp. Refusing the
147 * worker is the honest outcome — serving while the operator believes a
148 * filter is active would be a silent security downgrade. Rebuild with
149 * libseccomp-devel or set "brix_seccomp off". */
150 ngx_log_error(NGX_LOG_EMERG, cycle->log, 0,
151 "brix_seccomp: mode=%ui requested but this binary was "
152 "built without libseccomp (install libseccomp-devel and "
153 "rebuild, or set \"brix_seccomp off\")", mode);
154 return NGX_ERROR;
155 }
156
157 if (rc != BRIX_SECCOMP_CORE_OK) {
158 /* The sink already logged the specific failure. */
159 return NGX_ERROR;
160 }
161
162 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
163 "brix_seccomp: worker syscall filter active (mode=%s, exec=%s, "
164 "%ui allowed, %ui denied)",
165 (mode == BRIX_SECCOMP_ENFORCE) ? "enforce" : "audit",
166 allow_exec ? "allowed" : "killed",
167 (ngx_uint_t) n_allow, (ngx_uint_t) n_deny);
168 return NGX_OK;
169}
