1/*
2 * health.c — content handler for the `brix_health on;` location (phase-47 W2).
3 *
4 * WHAT: ngx_http_brix_health_handler() serves GET/HEAD /healthz as a small
5 * JSON document so an external load balancer or a Kubernetes liveness/
6 * readiness probe has a cheap endpoint to poll. Liveness is implicit: if
7 * this worker can accept the connection and run the handler it returns 200.
8 * `?verbose` adds cheap, non-secret readiness signals (metrics SHM mapped,
9 * worker pid, nginx version).
10 *
11 * WHY: The module already exposes /metrics (scrape) and the dashboard, but
12 * neither is a clean probe target — /metrics is a large text body and the
13 * dashboard needs auth. A dedicated 200/JSON endpoint keeps probes light and
14 * lets an operator wire `livenessProbe: GET /healthz` directly. It lives in
15 * the metrics module (not a new .so) and mirrors the read-only SRR/metrics
16 * handlers: method-gated, body discarded, no request input affects output.
17 *
18 * HOW: classify ?verbose → build the JSON in the request pool with ngx_snprintf
19 * (no jansson; the document is tiny and fixed-shape) → set status/headers →
20 * send_header → output the single memory buffer. HEAD stops after the header.
21 * Reads only process globals (ngx_pid) and the metrics SHM zone pointer; it
22 * never touches the request body and never emits a secret.
23 */
24

36health_verbose_requested(ngx_http_request_t *r)
37{
38 if (r->args.len == 0) {
39 return 0;

45/*
46 * Render the health document into a pool buffer. Non-verbose is a fixed
47 * liveness line; verbose appends the readiness object. Returns the buffer (and
48 * its length via *len), or NULL on allocation failure.
49 */
50static u_char *
51health_build_json(ngx_http_request_t *r, ngx_uint_t verbose, size_t *len)
52{
53 static const size_t cap = 2048;
54 u_char *buf;
55 u_char *p;
56 const char *shm_state;
57 ngx_brix_metrics_t *m;
58 ngx_uint_t generation = 0;
59 uint64_t config_hash = 0;
60
61 BRIX_PNALLOC_OR_RETURN(buf, r->pool, cap, NULL);
62
63 /* The metrics SHM zone is created at config time (metrics/config.c) and is
64 * the module's shared state — "mapped" once a stream server block exists.
65 * When mapped it also carries the config/reload fingerprint published by the
66 * master in init_module (brix_config_version_publish). */
67 m = (ngx_brix_shm_zone != NULL) ? ngx_brix_shm_zone->data : NULL;
68 if (m != NULL) {
69 generation = (ngx_uint_t) m->config_generation;
70 config_hash = m->config_hash;
71 }
72
73 if (!verbose) {
74 /* config_generation/config_version are cheap and reload-relevant, so
75 * they ride on the default (non-verbose) document: a probe can confirm a
76 * reload took effect without opting into the heavier readiness block. */
77 p = ngx_snprintf(buf, cap,
78 "{\"status\":\"ok\",\"service\":\"" BRIX_SERVER_NAME "\","
79 "\"version\":\"" BRIX_SERVER_VERSION "\","
80 "\"config_generation\":%ui,"
81 "\"config_version\":\"%016xL\"}\n",
82 generation, config_hash);
83 *len = (size_t) (p - buf);
84 return buf;
85 }
86
87 shm_state = (m != NULL) ? "mapped" : "unmapped";
88
89 p = ngx_snprintf(buf, cap,
90 "{\"status\":\"ok\",\"service\":\"" BRIX_SERVER_NAME "\","
91 "\"version\":\"" BRIX_SERVER_VERSION "\","
92 "\"config_generation\":%ui,"
93 "\"config_version\":\"%016xL\","
94 "\"checks\":{"
95 "\"metrics_shm\":\"%s\","
96 "\"worker_pid\":%P,"
97 "\"nginx_version\":\"%s\""
98 "}",
99 generation, config_hash,
100 shm_state, ngx_pid, NGINX_VERSION);
101
102 /* phase-68: per-origin health of every http-backed export (the CVMFS
103 * Stratum-1 sets). fail_score is the sd_http EWMA (0 = healthy). */
104 {
105 ngx_uint_t i, n_exports = brix_vfs_backend_export_count();
106 int first = 1;
107
108 p = ngx_snprintf(p, (size_t) (buf + cap - p), ",\"cvmfs_origins\":[");
109 for (i = 0; i < n_exports; i++) {
110 brix_vfs_backend_info_t info;
111 brix_sd_instance_t *inst;
112 char hosts[SD_HTTP_EP_MAX][256];
113 int ports[SD_HTTP_EP_MAX];
114 int scores[SD_HTTP_EP_MAX];
115 int j, n;
116
117 if (brix_vfs_backend_export_info(i, &info) != NGX_OK
118 || ngx_strcmp(info.backend, "http") != 0)
119 {
120 continue;
121 }
122 inst = brix_vfs_backend_resolve(info.root_canon,
123 r->connection->log);
124 while (inst != NULL
125 && ngx_strcmp(inst->driver->name, "http") != 0)
126 {
127 inst = brix_sd_cache_source_instance(inst);
128 }
129 n = sd_http_health_snapshot(inst, hosts, ports, scores,
130 SD_HTTP_EP_MAX);
131 for (j = 0; j < n; j++) {
132 p = ngx_snprintf(p, (size_t) (buf + cap - p),
133 "%s{\"host\":\"%s\",\"port\":%d,\"fail_score\":%d}",
134 first ? "" : ",", hosts[j], ports[j], scores[j]);
135 first = 0;
136 }
137 }
138 p = ngx_snprintf(p, (size_t) (buf + cap - p), "]}\n");
139 }
140 *len = (size_t) (p - buf);
141 return buf;
142}

145ngx_int_t
146ngx_http_brix_health_handler(ngx_http_request_t *r)
147{
148 ngx_http_brix_metrics_loc_conf_t *lcf;
149 ngx_int_t rc;
150 u_char *buf;
151 size_t len;
152 ngx_buf_t *b;
153 ngx_chain_t out;
154
155 lcf = ngx_http_get_module_loc_conf(r, ngx_http_brix_metrics_module);
156 if (!lcf->health) {
157 return NGX_DECLINED;
158 }
159
160 /* AGPL-3.0 sec.13: offer remote users the source (X-Source header). */
161 brix_http_source_offer(r);
162
163 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
164 return NGX_HTTP_NOT_ALLOWED;
165 }
166
167 rc = ngx_http_discard_request_body(r);
168 if (rc != NGX_OK) {
169 return rc;
170 }
171
172 buf = health_build_json(r, health_verbose_requested(r), &len);
173 if (buf == NULL) {
174 return NGX_HTTP_INTERNAL_SERVER_ERROR;
175 }
176
177 r->headers_out.status = NGX_HTTP_OK;
178 r->headers_out.content_length_n = (off_t) len;
179
180 {
181 ngx_str_t ct = ngx_string("application/json");
182 r->headers_out.content_type = ct;
183 r->headers_out.content_type_len = ct.len;
184 r->headers_out.content_type_lowcase = NULL;
185 }
186
187 rc = ngx_http_send_header(r);
188 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only) {
189 return rc;
190 }
191
192 BRIX_PCALLOC_OR_RETURN(b, r->pool, sizeof(*b), NGX_HTTP_INTERNAL_SERVER_ERROR);
193 b->pos = b->start = buf;
194 b->last = b->end = buf + len;
195 b->memory = 1;
196 b->last_buf = 1;
197
198 out.buf = b;
199 out.next = NULL;
200 return ngx_http_output_filter(r, &out);
201}
