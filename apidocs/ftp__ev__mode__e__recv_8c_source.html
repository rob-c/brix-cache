9/*
10 * ftp_ev_mode_e_recv.c — non-blocking MODE E (GFD.020 §3.4) STOR reassembly.
11 *
12 * WHAT: the STOR reassembly receiver — accept up to `Parallelism` passive data
13 * streams and fold their out-of-order extended blocks into the VFS writer, plus
14 * the 111 restart / 112 range markers the sink emits.
15 *
16 * HOW: STOR installs brix_ftp_ev_eb_accept() as the passive listener's read
17 * handler (in place of the single-stream ev_accept_handler): it accepts every
18 * pending stream, wraps each in a child connection, optionally starts its PROT P
19 * handshake, and arms a per-stream block reader. Child readers are kicked via
20 * ngx_post_event so a transfer that completes on the last EOD tears down the
21 * listener from a *posted* stack, never nested inside the accept loop. The
22 * committed-range table (shared on the parent dc) rejects overlapping/overflowing
23 * blocks; the transfer completes once the EOF-declared number of EODs has been
24 * seen, and the writer commit runs the whole-object read-back verify.
25 */

60/* Emit a GridFTP 111 restart/range marker: the contiguous byte ranges committed
61 * so far, coalesced, so a client can resume from what the sink already holds. The
62 * list is capped so a fragmented transfer cannot produce an unbounded reply. */
63static void
64ev_eb_marker_range(ftp_ev_dc_t *dc)
65{
66 ftp_ev_t *fc = dc->fc;
67 u_char line[64 * FTP_EV_EB_MARKER_MAX_RANGES + 32];
68 u_char *p = line;
69 u_char *end = line + sizeof(line);
70 size_t n = dc->eb_nranges, i, merged;
71
72 if (n == 0) {
73 return;
74 }
75 ngx_memcpy(dc->eb_scratch, dc->eb_ranges, n * sizeof(ftp_eb_range_t));
76 ngx_qsort(dc->eb_scratch, n, sizeof(ftp_eb_range_t), ev_eb_range_cmp);
77
78 /* Merge touching/adjacent ranges in place; rejected overlaps mean the sorted
79 * set only ever touches (hi == next.lo) or gaps, never crosses. */
80 merged = 0;
81 for (i = 1; i < n; i++) {
82 if (dc->eb_scratch[i].lo <= dc->eb_scratch[merged].hi) {
83 if (dc->eb_scratch[i].hi > dc->eb_scratch[merged].hi) {
84 dc->eb_scratch[merged].hi = dc->eb_scratch[i].hi;
85 }
86 } else {
87 dc->eb_scratch[++merged] = dc->eb_scratch[i];
88 }
89 }
90 merged++; /* count, not top index */
91
92 p = ngx_slprintf(p, end, "111 Range Marker ");
93 for (i = 0; i < merged && i < FTP_EV_EB_MARKER_MAX_RANGES; i++) {
94 p = ngx_slprintf(p, end, "%s%O-%O", i ? "," : "",
95 dc->eb_scratch[i].lo, dc->eb_scratch[i].hi);
96 }
97 *p = '\0';
98 (void) brix_ftp_ev_reply(fc, "%s\r\n", line);
99 (void) brix_ftp_ev_flush(fc);
100}

144/* A completed MODE E reassembly must tile the object gaplessly from offset 0.
145 * Every accepted block was overlap-checked, so the sorted ranges only ever touch
146 * (hi == next.lo) or leave a gap — they never cross. A gap means the sink holds a
147 * region the sender never delivered: a corrupted/forged EBLOCK offset (an in-path
148 * bit-flip of the 17-byte header past the TCP checksum), or a stream that emitted
149 * its EOD without covering its share. Committing then would publish a file with
150 * silently zero-filled holes as "complete" — the MODE E analogue of a truncated
151 * cache object accepted as whole. Require exactly one coalesced range
152 * [0, high-water) whose length equals the bytes actually received. */
153static int
154ev_eb_ranges_contiguous(ftp_ev_dc_t *dc)
155{
156 size_t n = dc->eb_nranges, i, merged;
157
158 if (n == 0) {
159 return dc->eb_received == 0; /* empty object: valid iff zero bytes */
160 }
161 ngx_memcpy(dc->eb_scratch, dc->eb_ranges, n * sizeof(ftp_eb_range_t));
162 ngx_qsort(dc->eb_scratch, n, sizeof(ftp_eb_range_t), ev_eb_range_cmp);
163
164 merged = 0;
165 for (i = 1; i < n; i++) {
166 if (dc->eb_scratch[i].lo <= dc->eb_scratch[merged].hi) {
167 if (dc->eb_scratch[i].hi > dc->eb_scratch[merged].hi) {
168 dc->eb_scratch[merged].hi = dc->eb_scratch[i].hi;
169 }
170 } else {
171 dc->eb_scratch[++merged] = dc->eb_scratch[i]; /* a hole opens here */
172 }
173 }
174 merged++; /* count, not top index */
175
176 return merged == 1
177 && dc->eb_scratch[0].lo == 0
178 && dc->eb_scratch[0].hi == dc->eb_received;
179}

182/* Retire a finished stream, then complete the transfer once every declared EOD
183 * has been seen (commit runs the whole-object read-back verify). */
184static void
185ev_eb_child_done(ftp_ev_eb_conn_t *ch)
186{
187 ftp_ev_dc_t *dc = ch->dc;
188
189 ev_eb_child_close(ch);
190
191 if (dc->eb_eof_total >= 0 && dc->eb_eod_seen >= dc->eb_eof_total) {
192 ngx_int_t rc;
193
194 if (!ev_eb_ranges_contiguous(dc)) {
195 /* Every declared EOD is in, yet the tiling has a hole: the client
196 * asserted the object is complete, so this partial is not an
197 * interrupted transfer awaiting a range-resume — it is a finished,
198 * holed poison (a forged/corrupted EBLOCK offset, or a short sender).
199 * The in-place random writer would otherwise leave the zero-filled
200 * object at the final path; abort it and unlink so nothing is
201 * published, then fail the transfer (550). */
202 brix_vfs_ctx_t vctx;
203
204 brix_vfs_writer_abort(dc->writer);
205 dc->writer = NULL;
206 brix_ftp_ev_vfs_ctx(dc->fc, dc->abs, &vctx);
207 (void) brix_vfs_unlink(&vctx);
208 brix_ftp_ev_data_finish(dc, NGX_ERROR);
209 return;
210 }
211 rc = brix_vfs_writer_commit(dc->writer);
212 dc->writer = NULL; /* finish must not abort it */
213 brix_ftp_ev_data_finish(dc, rc); /* closes listener + resumes */
214 }

239/* A fully-unpacked payload block: bounds-check its extent, reject overflow or a
240 * range overlapping an already-reserved one, then reserve it (before reading the
241 * payload, so a concurrent stream's overlap check sees it) and arm ch to drain. */
242static ev_eb_step_t
243ev_eb_reserve_range(ftp_ev_eb_conn_t *ch, ftp_ev_dc_t *dc,
244 uint64_t count, uint64_t offset)
245{
246 off_t lo, hi;
247
248 if (count > (uint64_t) INT64_MAX
249 || offset > (uint64_t) INT64_MAX - count)
250 {
251 brix_ftp_ev_data_finish(dc, NGX_ERROR); /* overflow */
252 return EV_EB_STEP_RET;
253 }
254 lo = (off_t) offset;
255 hi = (off_t) (offset + count);
256 if (dc->eb_nranges >= BRIX_FTP_EV_EB_MAX_RANGES
257 || ftp_eb_range_overlaps(dc->eb_ranges, dc->eb_nranges, lo, hi))
258 {
259 brix_ftp_ev_data_finish(dc, NGX_ERROR); /* overlap */
260 return EV_EB_STEP_RET;
261 }
262 dc->eb_ranges[dc->eb_nranges].lo = lo;
263 dc->eb_ranges[dc->eb_nranges].hi = hi;
264 dc->eb_nranges++;
265 ch->count = count;
266 ch->at = lo;
267 return EV_EB_STEP_OK;
268}

271/* Accumulate the 17-byte extended-block header; once complete, unpack it and set
272 * up the next block (EOF descriptors carry the EOD total in OFFSET and no payload;
273 * a non-empty block reserves its range). */
274static ev_eb_step_t
275ev_eb_recv_header(ngx_connection_t *c, ftp_ev_eb_conn_t *ch, ftp_ev_dc_t *dc)
276{
277 uint64_t count, offset;
278 ssize_t n = c->recv(c, ch->hdr + ch->hdr_got, FTP_EB_HDR - ch->hdr_got);
279
280 if (n == NGX_AGAIN) {
281 if (ev_eb_arm_read(c) != NGX_OK) {
282 brix_ftp_ev_data_finish(dc, NGX_ERROR);
283 }
284 return EV_EB_STEP_RET;
285 }
286 if (n == 0) {
287 if (ch->hdr_got == 0) {
288 ev_eb_child_done(ch); /* clean close at a boundary */
289 } else {
290 brix_ftp_ev_data_finish(dc, NGX_ERROR); /* EOF mid-header */
291 }
292 return EV_EB_STEP_RET;
293 }
294 if (n < 0) {
295 brix_ftp_ev_data_finish(dc, NGX_ERROR);
296 return EV_EB_STEP_RET;
297 }
298 ch->hdr_got += (size_t) n;
299 if (ch->hdr_got < FTP_EB_HDR) {
300 return EV_EB_STEP_MORE;
301 }
302
303 /* Full header: unpack and set up the block. */
304 ftp_eb_unpack(ch->hdr, &ch->desc, &count, &offset);
305 ch->hdr_got = 0;
306 ch->have_hdr = 1;
307 ch->count = 0;
308
309 if (ch->desc & FTP_EB_EOF) {
310 dc->eb_eof_total = (long) offset; /* no payload; EOD total is in OFFSET */
311 return EV_EB_STEP_OK;
312 }
313 if (count > 0) {
314 return ev_eb_reserve_range(ch, dc, count, offset);
315 }
316 return EV_EB_STEP_OK;
317}

320/* Drain this block's reserved payload into the writer at its absolute offset. */
321static ev_eb_step_t
322ev_eb_drain_payload(ngx_connection_t *c, ftp_ev_eb_conn_t *ch, ftp_ev_dc_t *dc)
323{
324 while (ch->count > 0) {
325 size_t want = (ch->count > BRIX_FTP_EV_XFER_BUF)
326 ? BRIX_FTP_EV_XFER_BUF : (size_t) ch->count;
327 ssize_t n = c->recv(c, dc->buf, want);
328 if (n == NGX_AGAIN) {
329 if (ev_eb_arm_read(c) != NGX_OK) {
330 brix_ftp_ev_data_finish(dc, NGX_ERROR);
331 }
332 return EV_EB_STEP_RET;
333 }
334 if (n <= 0) {
335 brix_ftp_ev_data_finish(dc, NGX_ERROR); /* EOF mid-payload / err */
336 return EV_EB_STEP_RET;
337 }
338 if (brix_vfs_writer_write(dc->writer, dc->buf, (size_t) n, ch->at)
339 != NGX_OK)
340 {
341 brix_ftp_ev_data_finish(dc, NGX_ERROR);
342 return EV_EB_STEP_RET;
343 }
344 ch->at += n;
345 ch->count -= (uint64_t) n;
346 dc->eb_received += n;
347 }
348 return EV_EB_STEP_OK;
349}

352/* Per-stream block reader: accumulate the 17-byte header, then drain `count`
353 * payload bytes into the writer at their absolute offset, block after block,
354 * until an EOD ends the stream or a clean close lands on a block boundary. */
355static void
356ev_eb_child_read(ngx_event_t *rev)
357{
358 ngx_connection_t *c = rev->data;
359 ftp_ev_eb_conn_t *ch = c->data;
360 ftp_ev_dc_t *dc = ch->dc;
361 ftp_ev_t *fc = dc->fc;
362
363 if (rev->timedout) {
364 brix_ftp_ev_data_finish(dc, NGX_ERROR);
365 return;
366 }
367 if (rev->timer_set) {
368 ngx_del_timer(rev);
369 }
370
371 for ( ;; ) {
372 if (!ch->have_hdr) {
373 ev_eb_step_t st = ev_eb_recv_header(c, ch, dc);
374 if (st == EV_EB_STEP_RET) {
375 return;
376 }
377 if (st == EV_EB_STEP_MORE) {
378 continue;
379 }
380 }
381
382 if (ev_eb_drain_payload(c, ch, dc) == EV_EB_STEP_RET) {
383 return;
384 }
385
386 /* Block complete: emit progress markers on the marker threshold. */
387 if (dc->eb_received - dc->eb_marked >= FTP_EV_EB_MARKER_BYTES) {
388 ev_eb_marker_range(dc);
389 ev_eb_marker_perf(fc, dc->eb_received);
390 dc->eb_marked = dc->eb_received;
391 }
392
393 if (ch->desc & FTP_EB_EOD) {
394 dc->eb_eod_seen++;
395 ev_eb_child_done(ch); /* last block on this stream */
396 return;
397 }
398 ch->have_hdr = 0; /* next block on this stream */
399 }
400}

465/* Passive listener read handler for a MODE E STOR: accept every pending stream,
466 * bring each up (PROT P handshake or straight), and arm its block reader. The
467 * listener stays open across the whole transfer — globus opens all parallel
468 * streams at once, so we must not retire it after the first accept. */
469void
470brix_ftp_ev_eb_accept(ngx_event_t *rev)
471{

473 ftp_ev_dc_t *dc = lc->data;
474 ftp_ev_t *fc = dc->fc;
475
476 if (rev->timedout) {
477 ngx_log_error(NGX_LOG_INFO, fc->c->log, NGX_ETIMEDOUT,
478 "brix: GridFTP(ev) MODE E passive accept timeout");
479 brix_ftp_ev_data_finish(dc, NGX_ERROR);
480 return;
481 }
482 if (rev->timer_set) {
483 ngx_del_timer(rev);
484 }
485
486 if (dc->writer == NULL && ev_eb_receiver_open(dc) != NGX_OK) {
487 brix_ftp_ev_data_finish(dc, NGX_ERROR);
488 return;
489 }
490

492 ftp_ev_eb_conn_t *ch;
493 ngx_connection_t *c;
494 int dfd = accept(lc->fd, NULL, NULL);
495
496 if (dfd < 0) {

498 break; /* drained pending accepts */
499 }

501 continue;
502 }
503 ngx_log_error(NGX_LOG_ERR, fc->c->log, ngx_socket_errno,
504 "brix: GridFTP(ev) MODE E accept failed");
505 brix_ftp_ev_data_finish(dc, NGX_ERROR);
506 return;
507 }
508
509 if (dc->eb_nconns >= BRIX_FTP_EV_EB_MAX_CONNS
510 || (ch = ev_eb_slot(dc)) == NULL)
511 {
512 (void) close(dfd); /* stream cap — refuse extra */
513 continue;
514 }
515 if (ngx_nonblocking(dfd) == -1) {
516 (void) close(dfd);
517 brix_ftp_ev_data_finish(dc, NGX_ERROR);
518 return;
519 }
520 c = brix_ftp_ev_wrap_conn(fc, dfd);
521 if (c == NULL) {
522 (void) close(dfd);
523 brix_ftp_ev_data_finish(dc, NGX_ERROR);
524 return;
525 }
526
527 ngx_memzero(ch, sizeof(*ch));
528 ch->dc = dc;
529 ch->in_use = 1;
530 ch->c = c;
531 c->data = ch;
532 dc->eb_nconns++;
533
534 if (fc->prot == 'P') {
535 /* Passive accept → TLS server role; completion arms the reader. */
536 if (brix_ftp_ev_tls_begin(fc, c, &ch->pool, 0 /* server */,
537 ev_eb_child_tls_done) == NGX_ERROR)
538 {
539 brix_ftp_ev_data_finish(dc, NGX_ERROR);
540 return;
541 }
542 } else {
543 c->read->handler = ev_eb_child_read;
544 if (ev_eb_arm_read(c) != NGX_OK) {
545 brix_ftp_ev_data_finish(dc, NGX_ERROR);
546 return;
547 }
548 /* Kick the reader off a posted event: a completion reached here must
549 * not tear the listener down from inside the accept loop. */
550 ngx_post_event(c->read, &ngx_posted_events);
551 }
552 }
553
554 /* Keep listening for the remaining parallel streams. */
555 if (ngx_handle_read_event(rev, 0) != NGX_OK) {
556 brix_ftp_ev_data_finish(dc, NGX_ERROR);
557 return;
