57/* ---- brix_pblock_drop_privilege ------------------------------------------
58 *
59 * WHAT: Guarantee the calling (worker) process is unprivileged before pblock
60 * creates any on-disk state, by permanently dropping to `want_user`
61 * (default "nobody"). Returns 0 on success or when already unprivileged,
62 * -1 (caller must fail closed) if a root worker cannot be dropped.
63 *
64 * WHY: pblock writes blob files (0600), block dirs (0700) and the SQLite
65 * catalog.db as the worker's OWN uid — it has no impersonation broker and
66 * never chowns (per-principal ownership is synthetic, in the catalog only).
67 * So a worker that runs as root (an explicit `user root;`, or any config
68 * that fails to drop) would create every blob/dir/DB owned by root — a
69 * privilege-escape foothold letting a client's data land as root. This
70 * backstop makes pblock on-disk data never owned by root (or the account
71 * that launched nginx), independent of the `user` directive.
72 *
73 * HOW: 1. euid != 0 → already unprivileged; nothing to do (the normal case,
74 * including impersonation `map` where the worker is a service account).
75 * This also makes the function idempotent: after the first drop every
76 * later call in the worker short-circuits here.
77 * 2. Resolve `want_user` (else "nobody"); refuse a uid/gid-0 target.
78 * 3. setgroups({gid}) → setgid → setuid — permanent (real+eff+saved).
79 * 4. Re-read getuid/geteuid; FAIL CLOSED if either is still 0.
80 * 5. Warn (fires only on the one transition). Diagnostics go to stderr,
81 * which nginx redirects into the worker error_log; this file is ngx-free
82 * (shared with the standalone unit test) so it cannot call ngx_log_*.
83 */
84static int
85brix_pblock_drop_privilege(const char *want_user)
86{
87 const char *acct;
88 struct passwd *pw;
89 gid_t gid;
90
91 if (geteuid() != 0) {
92 return 0;
93 }
94
95 acct = (want_user != NULL && want_user[0] != '\0') ? want_user : "nobody";
96 errno = 0;
97 pw = getpwnam(acct);
98 if (pw == NULL) {
99 fprintf(stderr, "pblock: refusing to run as root — unprivileged account "
100 "\"%s\" not found (set brix_pblock_unprivileged_user)\n",
101 acct);
102 return -1;
103 }
104 if (pw->pw_uid == 0 || pw->pw_gid == 0) {
105 fprintf(stderr, "pblock: refusing to drop to \"%s\" — it is a uid/gid 0 "
106 "account\n", acct);
107 return -1;
108 }
109
110 gid = pw->pw_gid;
111 if (setgroups(1, &gid) != 0 || setgid(gid) != 0 || setuid(pw->pw_uid) != 0) {
112 fprintf(stderr, "pblock: FAILED to drop root to \"%s\": %s — refusing to "
113 "serve (blobs/catalog.db must never be root-owned)\n",
114 acct, strerror(errno));
115 return -1;
116 }
117 if (getuid() == 0 || geteuid() == 0) {
118 fprintf(stderr, "pblock: privilege drop to \"%s\" did not stick — "
119 "refusing to serve\n", acct);
120 return -1;
121 }
122
123 fprintf(stderr, "pblock: worker was running as root; dropped to \"%s\" "
124 "(uid=%ld) so blobs and catalog.db are never root-owned — set "
125 "the nginx 'user <acct>;' directive to run the worker (and own "
126 "pblock data) as a chosen account instead\n",
127 acct, (long) pw->pw_uid);
128 return 0;
129}

131/* ---- lab / feature arming (phase-83) -------------------------------------- */

133/* Arm the lab control plane + its perf knob from the sidecar opts. With lab OFF
134 * (the default — no sidecar) this is a no-op and the driver stays byte-for-byte
135 * the production backend: st->lab stays NULL, the hot path never consults it. */
136static void
137pblock_arm_lab(brix_sd_instance_t *inst, pblock_state_t *st,
138 const pblock_opts_t *opts)
139{
140 if (opts->mem) { /* F16 */
141 (void) pblock_ctl_mem_pragmas(st->cat);
142 }
143 if (opts->lab) {
144 inst->caps = pblock_caps_apply(inst->caps, opts); /* F2 */
145 st->lab = pblock_lab_state_create(st->cat); /* F1/F8 */
146 if (st->lab != NULL) {
147 /* F9 rides the lab master gate (an anomaly simulator is a lab toy
148 * by definition). Best-effort: with no `recent` table the event
149 * writers no-op and consultation finds nothing. */
150 (void) pblock_anomaly_init(st);
151 }
152 }
153}

155/* Arm the per-instance data features that each advertise/enable only when their
156 * catalog table actually installed — an init failure for any one leaves that
157 * feature off and the byte-for-byte production path intact. */
158static void
159pblock_arm_data_features(brix_sd_instance_t *inst, pblock_state_t *st,
160 const pblock_opts_t *opts)
161{
162 if (opts->audit) { /* F17 */
163 /* Independent of the lab gate (its own opt): only turn audit on if the
164 * oplog table is actually present. */
165 st->audit = (pblock_audit_init(st->cat) == 0);
166 }
167 if (opts->csi) { /* F3 */
168 /* Advertise CAP_FSCS only when the csi table installed — an honest
169 * per-instance capability the driver will really honour. */
170 if (pblock_csi_init(st->cat) == 0) {
171 st->csi = 1;
172 inst->caps |= BRIX_SD_CAP_FSCS;
173 }
174 }
175 if (opts->nearline) { /* F4 */
176 /* Advertise CAP_NEARLINE only when the residency table installed (the
177 * residency seam and the cache tier's recall-at-fill key off it). */
178 if (pblock_nearline_init(st) == 0) {
179 st->nearline = 1;
180 inst->caps |= BRIX_SD_CAP_NEARLINE;
181 }
182 }
183 if (opts->locks) { /* F15 */
184 /* Mandatory lease enforcement only arms when the locks table installed;
185 * a failure leaves the production path (no lease reads anywhere). */
186 st->locks = (pblock_locks_init(st) == 0);
187 }
188 if (opts->dedup) { /* F10 */
189 /* Refcounted blobs + dedup only arm when the blobs table installed. */
190 st->refs = (pblock_refs_init(st) == 0);
191 }
192}

194/* Arm the F12/F13 per-block transform — the one HARD config gate. A bad spec
195 * (unknown transform, unreadable crypt keyfile, `zstd` without libzstd) fails
196 * instance init fail-closed rather than silently serving unreadable transformed
197 * bytes as raw. Returns 0 (incl. no xform requested), or -1 with errno set from
198 * pblock_xform_config; the caller owns st->cat/st cleanup on -1. A transformed
199 * export cannot hand out a block-0 fd, so the zero-copy caps are dropped. */
200static int
201pblock_arm_xform(brix_sd_instance_t *inst, pblock_state_t *st,
202 const pblock_opts_t *opts)
203{
204 if (opts->xform_len == 0) {
205 return 0;
206 }
207 if (pblock_xform_config(&st->xform, opts->xform, opts->xform_len) != 0) {
208 return -1; /* errno set by config */
209 }
210 if (pblock_xform_active(&st->xform)) {
211 inst->caps &= ~(uint32_t) (BRIX_SD_CAP_SENDFILE | BRIX_SD_CAP_IOURING);
212 }
213 return 0;
214}

216/* Arm the retention/accounting features. Snapshots (F6) and versioning/trash
217 * (F11) both HOLD prior blobs, so they build ON refcounted blobs (F10): arm
218 * refs first (idempotent), then the history tables — armed only when both
219 * installed, else the byte-for-byte production path is untouched. */
220static void
221pblock_arm_storage_features(pblock_state_t *st, const pblock_opts_t *opts)
222{
223 if (opts->snapshots) { /* F6 */
224 if (!st->refs) {
225 st->refs = (pblock_refs_init(st) == 0);
226 }
227 if (st->refs && pblock_snap_init(st) == 0) {
228 st->snap = 1;
229 }
230 }
231 if (opts->versions > 0 || opts->trash) { /* F11 */
232 if (!st->refs) {
233 st->refs = (pblock_refs_init(st) == 0);
234 }
235 if (st->refs && pblock_hist_init(st) == 0) {
236 st->versions = opts->versions;
237 st->trash = opts->trash;
238 }
239 }
240 if (opts->quota_bytes > 0 || opts->quota_inodes > 0) { /* F5 */
241 /* Its own opt: quota only arms when the rollup table + triggers actually
242 * installed, so an init failure leaves the production catalog path. */
243 if (pblock_quota_init(st) == 0) {
244 st->quota = 1;
245 st->quota_bytes = opts->quota_bytes;
246 st->quota_inodes = opts->quota_inodes;
247 }
248 }

253ngx_int_t
254sd_pblock_init(brix_sd_instance_t *inst, void *driver_conf)
255{
256 const brix_sd_pblock_conf_t *conf = driver_conf;
257 pblock_state_t *st;
258 char db[PATH_MAX];
259
260 if (conf == NULL || conf->root == NULL) {
261 errno = EINVAL;
262 return NGX_ERROR;
263 }
264
265 /* Never create pblock on-disk state (dirs/blobs/catalog.db) as root: a
266 * worker-time production build sets enforce_unprivileged, which drops a root
267 * worker to an unprivileged account here — BEFORE the pblock_mkdir_p below.
268 * Fail closed if the drop is impossible. */
269 if (conf->enforce_unprivileged
270 && brix_pblock_drop_privilege(conf->unpriv_user) != 0)
271 {
272 errno = EPERM;
273 return NGX_ERROR;
274 }
275
276 st = calloc(1, sizeof(*st));
277 if (st == NULL) {
278 errno = ENOMEM;
279 return NGX_ERROR;
280 }
281 snprintf(st->root, sizeof(st->root), "%s", conf->root);
282 snprintf(st->data_dir, sizeof(st->data_dir), "%s/data", conf->root);
283 st->block_size = conf->block_size > 0 ? conf->block_size
284 : PBLOCK_DEFAULT_BLOCK_SIZE;
285
286 if (pblock_mkdir_p(st->root) != 0 || pblock_mkdir_p(st->data_dir) != 0) {
287 int err = errno;
288
289 free(st);
290 errno = err;
291 return NGX_ERROR;
292 }
293
294 snprintf(db, sizeof(db), "%s/catalog.db", conf->root);
295 st->cat = pblock_catalog_open(db, conf->busy_timeout_ms);
296 if (st->cat == NULL) {
297 int err = errno;
298
299 free(st);
300 errno = err;
301 return NGX_ERROR;
302 }
303
304 /* Phase-83 lab control plane. The static opts sidecar (<root>/pblock.opts,
305 * written by the config finaliser when it strips a `?tail` off a pblock://
306 * root) selects the fail-closed master gate and the caps/mem knobs. With
307 * lab OFF (the default — no sidecar) the driver stays byte-for-byte the
308 * production backend: st->lab is NULL and the hot path never consults it. */
309 {
310 pblock_opts_t opts;
311
312 (void) pblock_opts_load_sidecar(st->root, &opts); /* absent ⇒ all-zero */
313
314 pblock_arm_lab(inst, st, &opts);
315 pblock_arm_data_features(inst, st, &opts);
316
317 /* F12/F13 transform is the one hard config gate — fail instance init
318 * fail-closed on a bad spec (the store is never built for the export,
319 * logged "backend init failed") rather than serving garbage as raw. */
320 if (pblock_arm_xform(inst, st, &opts) != 0) {
321 int err = errno;
322
323 pblock_catalog_close(st->cat);
324 free(st);
325 errno = err;
326 return NGX_ERROR;
327 }
328
329 pblock_arm_storage_features(st, &opts);
330 }
331
332 /* The export root "/" always exists (a directory), like a POSIX mount point —
333 * so stat("/")/opendir("/")/PROPFIND on the root succeed before anything is
334 * written. Created once; harmless if a concurrent worker also creates it.
335 * Sticky world-writable (/tmp semantics) so identity-enforced multi-user
336 * exports work out of the box: anyone may create top-level entries, only
337 * owners may remove them (pblock_ident_sticky_gate). Operators wanting a
338 * stricter top level chmod "/" or pre-create VO directories. */
339 if (pblock_catalog_lookup(st->cat, "/", NULL) == 1) {
340 pblock_meta root_meta;
341
342 memset(&root_meta, 0, sizeof(root_meta));
343 root_meta.is_dir = 1;
344 root_meta.mtime = root_meta.ctime = pblock_now();
345 root_meta.mode = S_IFDIR | S_ISVTX | 0777;
346 (void) pblock_catalog_put(st->cat, "/", &root_meta);
347 }
348
349 inst->state = st;
350 return NGX_OK;
351}

354sd_pblock_cleanup(brix_sd_instance_t *inst)
