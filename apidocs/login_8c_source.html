19/*
20 * WHAT: Validate the 8-byte fixed username field copied off the wire.
21 * WHY: A NUL in the middle silently truncates the field when treated as a C
22 * string, letting "a\x00evil" impersonate user "a"; the XRootD spec says
23 * the field is "null-padded ASCII", so reject NUL-embedded / non-printable
24 * bytes. Isolating the scan keeps the caller's control flow flat.
25 * HOW: Scan the up-to-8 leading non-NUL bytes; on the first out-of-range byte
26 * emit the exact kXR_ArgInvalid refusal and return NGX_DONE so the caller
27 * stops (brix_send_error itself returns NGX_OK, which must NOT continue).
28 * Returns NGX_OK when every byte is printable ASCII.
29 */
30static ngx_int_t
31brix_login_check_username(brix_ctx_t *ctx, ngx_connection_t *c,
32 const char user[9])
33{
34 int i;
35
36 for (i = 0; i < 8 && user[i] != '\0'; i++) {
37 if ((u_char) user[i] < 0x20 || (u_char) user[i] > 0x7e) {
38 brix_send_error(ctx, c, kXR_ArgInvalid,
39 "username contains invalid characters");
40 return NGX_DONE;
41 }
42 }
43
44 return NGX_OK;
45}

47/*
48 * WHAT: Run the pre-response login preconditions and parse the request into
49 * ctx->login (username, pid) plus a sanitized copy for logging.
50 * WHY: These checks (suspended, duplicate login, username validity) all short-
51 * circuit with an exact wire refusal before any response is built; keeping
52 * them together preserves the reference ordering and messages verbatim.
53 * HOW: On any refusal, emit the exact error and return NGX_DONE (brix_send_error
54 * returns NGX_OK, which must not fall through). On success populate
55 * ctx->login, mark the session logged_in, emit the debug trace, and return
56 * NGX_OK. `user`/`user_log` are caller-owned so the response stage can log.
57 */
58static ngx_int_t
59brix_login_precheck_and_parse(brix_ctx_t *ctx, ngx_connection_t *c,
60 ngx_stream_brix_srv_conf_t *conf, char user[9], char user_log[64])
61{
62 xrdw_login_req_t req;
63 ngx_int_t rc;
64
65 if (conf->cms.suspended) {
66 brix_send_error(ctx, c, kXR_Overloaded, "server suspended by manager");
67 return NGX_DONE;
68 }
69
70 /* The reference do_Login rejects a SECOND kXR_login on an already-logged-in
71 * connection ("if (Status) return Response.Send(kXR_InvalidRequest,
72 * \"duplicate login; already logged in\")", XrdXrootdXeq.cc:1095). Match the
73 * reference code and message verbatim. (Note: installed stock v5.9.5 happens
74 * to surface kXR_ArgMissing here, but the current reference source — and the
75 * correct semantics for a malformed-in-context request — is kXR_InvalidRequest;
76 * either way a live session cannot re-login.) */
77 if (ctx->login.logged_in) {
78 brix_send_error(ctx, c, kXR_InvalidRequest,
79 "duplicate login; already logged in");
80 return NGX_DONE;
81 }
82
83 xrdw_login_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
84
85 /* Username is an 8-byte fixed field on the wire, so copy and terminate it locally. */
86 ngx_memcpy(user, req.username, 8);
87 user[8] = '\0';
88
89 rc = brix_login_check_username(ctx, c, user);
90 if (rc != NGX_OK) {
91 return rc;
92 }
93
94 ngx_memcpy(ctx->login.user, user, sizeof(ctx->login.user));
95 ctx->login.pid = (uint32_t) req.pid;
96 brix_sanitize_log_string(user, user_log, 64);
97
98 ngx_log_debug3(NGX_LOG_DEBUG_STREAM, c->log, 0,
99 "brix: login user=\"%s\" pid=%d auth=%s",
100 user_log, (int) req.pid,
101 (conf->auth == BRIX_AUTH_GSI) ? "gsi" :
102 (conf->auth == BRIX_AUTH_TOKEN) ? "token" :
103 (conf->auth == BRIX_AUTH_BOTH) ? "both" :
104 (conf->auth == BRIX_AUTH_SSS) ? "sss" :
105 (conf->auth == BRIX_AUTH_UNIX) ? "unix" :
106 (conf->auth == BRIX_AUTH_KRB5) ? "krb5" : "none");
107
108 /* Login marks the session as known; auth_done is deferred for GSI mode. */
109 ctx->login.logged_in = 1;
110
111 return NGX_OK;
112}

114/*
115 * WHAT: Emit the anonymous-mode login response (sessid only, single round-trip).
116 * WHY: BRIX_AUTH_NONE completes login immediately with no auth negotiation, so
117 * its response, session registration, and access log differ from the
118 * authenticated path and are kept separate for clarity.
119 * HOW: Mark auth_done, build the kXR_ok header + sessid body, record session
120 * start/identity, register the session, log access, count the metric, and
121 * queue the response (returning whatever brix_queue_response returns).
122 */
123static ngx_int_t
124brix_login_respond_anon(brix_ctx_t *ctx, ngx_connection_t *c, const char *user)
125{
126 u_char *buf;
127 size_t total;
128
129 /* Anonymous mode completes login in one round-trip with only the sessid. */
130 ctx->login.auth_done = 1;
131
132 total = XRD_RESPONSE_HDR_LEN + BRIX_SESSION_ID_LEN;
133 BRIX_PALLOC_OR_RETURN(buf, c->pool, total, NGX_ERROR);
134
135 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok,
136 BRIX_SESSION_ID_LEN,
137 (ServerResponseHdr *) buf);
138 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, ctx->login.sessid,
139 BRIX_SESSION_ID_LEN);
140
141 /* Session timing starts at successful login so later disconnect stats have an origin. */
142 ctx->totals.start = ngx_current_msec;
143 if (ctx->identity != NULL) {
144 ctx->identity->auth_method = BRIX_AUTHN_NONE;
145 }
146 brix_sess_auth_once(ctx->sess, BRIX_SESS_AM_ANON, "-", "-");
147 brix_session_register(ctx->login.sessid, ctx->login.dn, ctx->login.vo_list, 0);
148 brix_log_access(ctx, c, "LOGIN", "-", user, 1, 0, NULL, 0);
149 brix_count_login_ok(ctx);
150
151 return brix_queue_response(ctx, c, buf, total);
152}

154/*
155 * WHAT: Format the "&P=..." security parameter block for an authenticated mode.
156 * WHY: The client parses this block to decide which security plugin to load;
157 * each auth mode advertises a distinct protocol string. Isolating the
158 * per-mode ladder here keeps the branch count out of the caller.
159 * HOW: Write into caller-owned `parms` (cap `sizeof`), returning the byte count
160 * INCLUDING the trailing NUL (clients treat the block as C-string data).
161 * The GSI-capable modes advertise a version that gates signed-DH.
162 */
163static size_t
164brix_login_build_parms(ngx_stream_brix_srv_conf_t *conf,
165 char *parms, size_t cap)
166{
167 /* Advertised GSI version drives the client's signed-DH decision:
168 * >=BRIX_GSI_VERS_DHSIGNED (10400) lets capable clients use the
169 * RSA-signed-DH variant. Default 10000 (unsigned, universally
170 * compatible) unless the brix_gsi_signed_dh policy opts in. */
171 unsigned gsi_ver = (conf->gsi_signed_dh != BRIX_GSI_SDH_OFF) ? 10600u : 10000u;
172
173 if (conf->auth == BRIX_AUTH_TOKEN) {
174 /* Token-only: advertise ztn, no CA hash needed. */
175 return (size_t) snprintf(parms, cap, "&P=ztn,v:10000") + 1;
176 }
177 if (conf->auth == BRIX_AUTH_SSS) {
178 /* XRootD SSS: bf32 ('0'), v2 server ('+'), client chooses keytab. */
179 return (size_t) snprintf(parms, cap, "&P=sss,0.+%d:",
180 (int) conf->sss_lifetime) + 1;
181 }
182 if (conf->auth == BRIX_AUTH_UNIX) {
183 return (size_t) snprintf(parms, cap, "&P=unix") + 1;
184 }
185 if (conf->auth == BRIX_AUTH_KRB5) {
186 return (size_t) snprintf(parms, cap, "&P=krb5,%s",
187 (const char *) conf->krb5.principal.data) + 1;
188 }
189 if (conf->auth == BRIX_AUTH_HOST) {
190 /* Phase 52 WS-C: host auth asserts no credential — bare protocol id. */
191 return (size_t) snprintf(parms, cap, "&P=host") + 1;
192 }
193 if (conf->auth == BRIX_AUTH_PWD) {
194 /* Phase 52 WS-B: XrdSecpwd password protocol (v:10100, ssl crypto). */
195 return (size_t) snprintf(parms, cap, "&P=pwd,v:10100,c:ssl") + 1;
196 }
197 if (conf->auth == BRIX_AUTH_BOTH) {
198 /* Both: token first (preferred), then GSI. */
199 return (size_t) snprintf(parms, cap,
200 "&P=ztn,v:10000&P=gsi,v:%u,c:ssl,ca:%s",
201 gsi_ver, conf->gsi_ca_hashes) + 1;
202 }
203
204 /* GSI-only. The advertised version drives the client's signed-DH
205 * decision: >=10400 makes capable clients use the RSA-signed-DH
206 * variant. Defaults to 10000 (unsigned, universal) unless the
207 * brix_gsi_signed_dh policy opts in. */
208 return (size_t) snprintf(parms, cap, "&P=gsi,v:%u,c:ssl,ca:%s",
209 gsi_ver, conf->gsi_ca_hashes) + 1;
210}

212/*
213 * WHAT: Emit the authenticated-mode login response (sessid + "&P=..." block).
214 * WHY: Authenticated modes send a text parameter block after the 16-byte
215 * session id so the client can select its security plugin; login succeeds
216 * here but auth negotiation continues in later round-trips.
217 * HOW: Re-fetch the live merged srv_conf, format the parameter block, guard its
218 * length (kXR_ServerError → NGX_DONE, since brix_send_error returns NGX_OK
219 * and must not fall through), build the kXR_ok header + sessid + parms,
220 * record session start, log access, count the metric, and queue.
221 */
222static ngx_int_t
223brix_login_respond_authenticated(brix_ctx_t *ctx, ngx_connection_t *c,
224 const char *user)
225{
226 ngx_stream_brix_srv_conf_t *conf;
227 char parms[256];
228 size_t parms_len;
229 u_char *buf;
230 size_t total;
231
232 /* Re-fetch the live merged srv_conf in case login inherited settings. */
233 conf = ngx_stream_get_module_srv_conf(ctx->session, ngx_stream_brix_module);
234
235 parms_len = brix_login_build_parms(conf, parms, sizeof(parms));
236
237 /* Include the trailing NUL because clients treat the parameter block as C-string data. */
238 if (parms_len > sizeof(parms)) {
239 brix_send_error(ctx, c, kXR_ServerError,
240 "auth parameter block too long");
241 return NGX_DONE;
242 }
243
244 total = XRD_RESPONSE_HDR_LEN + BRIX_SESSION_ID_LEN + parms_len;
245 BRIX_PALLOC_OR_RETURN(buf, c->pool, total, NGX_ERROR);
246
247 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok,
248 (uint32_t)(BRIX_SESSION_ID_LEN + parms_len),
249 (ServerResponseHdr *) buf);
250
251 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN, ctx->login.sessid,
252 BRIX_SESSION_ID_LEN);
253 ngx_memcpy(buf + XRD_RESPONSE_HDR_LEN + BRIX_SESSION_ID_LEN,
254 parms, parms_len);
255
256 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
257 "brix: login->kXGS_init parms=\"%s\" ca_hashes=%s",
258 parms, conf->gsi_ca_hashes);
259
260 /* Successful login still marks the start of the session even though auth continues. */
261 ctx->totals.start = ngx_current_msec;
262 brix_log_access(ctx, c, "LOGIN", "-", user, 1, 0, NULL, 0);
263 brix_count_login_ok(ctx);
264
265 return brix_queue_response(ctx, c, buf, total);
266}

268/* Handle kXR_login — accept a client username, generate a session id (sessid),
269 * and begin auth negotiation (advertising the configured security requirement). */
270ngx_int_t
271brix_handle_login(brix_ctx_t *ctx, ngx_connection_t *c,
272 ngx_stream_brix_srv_conf_t *conf)
273{
274 char user[9];
275 char user_log[64];
276 ngx_int_t rc;
277
278 rc = brix_login_precheck_and_parse(ctx, c, conf, user, user_log);
279 if (rc != NGX_OK) {
280 return (rc == NGX_DONE) ? NGX_OK : rc;
281 }
282
283 if (conf->auth == BRIX_AUTH_NONE) {
284 return brix_login_respond_anon(ctx, c, user);
285 }
286
287 rc = brix_login_respond_authenticated(ctx, c, user);
288 return (rc == NGX_DONE) ? NGX_OK : rc;
289}
