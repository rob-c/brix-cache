1/*
2 * resolve_path_variants.c — ngx_str_t-rooted convenience wrappers over the
3 * unified cstr resolver (config-time use only).
4 *
5 * WHAT: Thin adapters that take an nginx ngx_str_t export root, canonicalise it,
6 * and forward to brix_path_resolve_cstr() (unified.c) with a preset
7 * brix_path_opts_t. The internal brix_resolve_with_opts() does the
8 * canonicalise-then-resolve dance; the only surviving public variant is
9 * brix_resolve_path_noexist() (write semantics, parents may be missing).
10 *
11 * WHY: Most runtime client-path resolution moved off realpath() to the kernel-
12 * confined beneath API in Phase 8 (see the note further down). What remains
13 * here is intentionally limited to TRUSTED, CONFIG-TIME callers — namely
14 * brix_finalize_path_rules() canonicalising VO/group policy rule paths
15 * once at startup — where a realpath()-based canonicalisation is correct
16 * and is deliberately NOT migrated to the beneath API.
17 *
18 * HOW: brix_resolve_with_opts() calls brix_get_canonical_root() (canonical.c)
19 * then brix_path_resolve_cstr(); a TOO_LONG status is logged as a warning
20 * and any non-OK status maps to a 0 (false) return. _noexist() sets
21 * allow_missing_parents + is_write_operation before delegating.
22 */

30static int
31brix_resolve_with_opts(ngx_log_t *log, const ngx_str_t *root,
32 const char *reqpath, brix_path_opts_t opts,
33 char *resolved, size_t resolvsz)
34{
35 char root_canon[PATH_MAX];
36 brix_path_status_t rc;
37
38 if (!brix_get_canonical_root(log, root, root_canon, sizeof(root_canon))) {
39 return 0;
40 }
41
42 rc = brix_path_resolve_cstr(log, root_canon, reqpath, opts,
43 resolved, resolvsz, NULL);
44 if (rc == BRIX_PATH_STATUS_TOO_LONG) {
45 ngx_log_error(NGX_LOG_WARN, log, 0, "brix: path too long");
46 }
47
48 return rc == BRIX_PATH_STATUS_OK;
49}

51int
52brix_resolve_path_noexist(ngx_log_t *log, const ngx_str_t *root,
53 const char *reqpath, char *resolved,
54 size_t resolvsz)
55{
56 brix_path_opts_t opts;
57
58 opts = (brix_path_opts_t) { 0 };
59 opts.allow_missing_parents = 1;
60 opts.is_write_operation = 1;
61
62 return brix_resolve_with_opts(log, root, reqpath, opts,
63 resolved, resolvsz);
