1/*
2 * resolve.c — reverse-DNS peer resolution for XrdAcc host rules.
3 *
4 * WHAT: brix_acc_resolve_peer() reverse-resolves a peer socket address to a
5 * hostname via getnameinfo(NI_NAMEREQD) so authdb `h <host>` (exact) and
6 * `h .domain` (suffix) records can match. Writes the FQDN into `buf` and
7 * returns it on success; returns NULL on failure so the caller can fall back
8 * to the numeric peer IP.
9 *
10 * WHY: XrdAcc (XrdAccAccess::Resolve) matches host/domain records against the
11 * client's resolved hostname, not its IP. Without a reverse lookup those
12 * records never fire. Resolution is opt-in (brix_acc_resolve_hosts) and
13 * cached once per connection by the caller, which bounds the blocking-DNS
14 * cost and the DoS surface — the same cost-control XrdAcc gets by resolving
15 * only when a host looks like a raw IP literal and the authdb has host rules.
16 *
17 * HOW: a single getnameinfo() with NI_NAMEREQD — no numeric fallback, because
18 * we want NULL (not the IP) when there is no PTR record, so the caller can
19 * distinguish "unresolved" from "resolved to a name". This file owns the
20 * only <netdb.h> dependency; callers pass a plain stack buffer.
21 */

30/*
31 * Phase 51 (E3): circuit-breaker around the blocking reverse-DNS lookup so a slow
32 * or down resolver cannot block the event loop on every new connection's first
33 * resolve. After N consecutive slow lookups the breaker opens for a cooldown,
34 * during which we fail fast to NULL (caller falls back to the numeric IP — host
35 * rules simply don't fire, the conservative degraded behaviour). Per-worker,
36 * event-loop only → lock-free.
37 */
38#define ACC_DNS_SLOW_MS 2000
39#define ACC_DNS_TRIP_COUNT 5
40#define ACC_DNS_COOLDOWN_SECS 10

42static struct {
43 int consecutive_slow;
44 time_t open_until;
45} acc_dns_breaker;

47static int64_t
48acc_dns_monotonic_ms(void)
49{
50 struct timespec ts;
51 if (clock_gettime(CLOCK_MONOTONIC, &ts) != 0) {
52 return 0;
53 }
54 return (int64_t) ts.tv_sec * 1000 + ts.tv_nsec / 1000000;
55}

57const char *
58brix_acc_resolve_peer(struct sockaddr *sa, socklen_t salen,
59 char *buf, size_t buflen)
60{
61 time_t now;
62 int64_t t0, elapsed;
63 int rc;
64
65 if (sa == NULL || buf == NULL || buflen == 0) {
66 return NULL;
67 }
68
69 now = time(NULL);
70 if (acc_dns_breaker.open_until > now) {
71 return NULL; /* breaker open — skip the (slow) resolver */
72 }
73
74 t0 = acc_dns_monotonic_ms();
75 rc = getnameinfo(sa, salen, buf, (socklen_t) buflen, NULL, 0, NI_NAMEREQD);
76 elapsed = acc_dns_monotonic_ms() - t0;
77
78 if (elapsed >= ACC_DNS_SLOW_MS) {
79 if (++acc_dns_breaker.consecutive_slow >= ACC_DNS_TRIP_COUNT) {
80 acc_dns_breaker.open_until = now + ACC_DNS_COOLDOWN_SECS;
81 acc_dns_breaker.consecutive_slow = 0;
82 BRIX_RESIL_METRIC_INC(acc_dns_breaker_open_total);
83 ngx_log_error(NGX_LOG_WARN, ngx_cycle->log, 0,
84 "brix_acc: reverse DNS slow (%L ms) — opening "
85 "circuit breaker for %ds (host rules fall back to IP)",
86 (long long) elapsed, ACC_DNS_COOLDOWN_SECS);
87 }
88 } else {
89 acc_dns_breaker.consecutive_slow = 0;
90 }
91
92 if (rc != 0) {
93 return NULL;
94 }
95
96 return buf;
97}
