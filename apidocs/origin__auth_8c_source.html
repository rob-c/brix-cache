36/* Frame one kXR_auth request (ClientAuthRequest header, credtype = the 4-byte
37 * protocol id, + credential payload) on the connector stream. Shared by the ztn
38 * and gsi auth helpers. Returns 0, or -1 (errno set). */
39static int
40cache_origin_send_kxr_auth(brix_cache_origin_conn_t *oc, const char credtype[4],
41 const u_char *payload, uint32_t plen)
42{
43 ClientAuthRequest req;
44
45 ngx_memzero(&req, sizeof(req));
46 req.streamid[1] = 1; /* the connector stream */
47 req.requestid = htons(kXR_auth);
48 ngx_memcpy(req.credtype, credtype, 4);
49 req.dlen = htonl((kXR_int32) plen);
50
51 if (brix_cache_io_send(oc, &req, sizeof(req)) != 0
52 || (plen > 0 && brix_cache_io_send(oc, payload, plen) != 0))
53 {
54 return -1;
55 }
56 return 0;
57}

59/* XrdSecProtocolztn credential wire format (stock XRootD, byte-frozen).
60 *
61 * The ztn credential is an XrdSecProtocolztn::TokenResp: an 8-byte TokenHdr, a
62 * 2-byte big-endian length, then the token followed by one NUL terminator:
63 *
64 * off 0..3 id[4] = "ztn\0" (NUL-terminated protocol id)
65 * off 4 ver = 0 (XrdSecProtocolztn::ztnVersion)
66 * off 5 opr = 'T' (TokenHdr::IsTkn — "here is a token")
67 * off 6..7 rsvd[2] = 0, 0
68 * off 8..9 len = htons(tsz + 1) (token length INCLUDING its trailing NUL)
69 * off 10.. tkn[tsz] + one NUL byte
70 *
71 * The stock server (Authenticate) reads opr at off 5 to route the request,
72 * ntohs(len) at off 8, and the token at off 10 (pfxLen = sizeof(TokenHdr) +
73 * sizeof(uint16_t) = 10). It requires len >= 1, len's byte at (10 + len - 1) to
74 * be NUL, and (10 + len) <= credential size. The whole credential is
75 * (10 + tsz + 1) bytes. */
76#define BRIX_ZTN_VERSION 0
77#define BRIX_ZTN_OPR_ISTKN 'T'
78#define BRIX_ZTN_PREFIX_LEN 10 /* 8-byte TokenHdr + 2-byte length */
79
80/* cache_origin_build_ztn_credential — frame a bearer token as a stock XrdSecztn
81 * TokenResp credential.
82 *
83 * WHAT: allocate and fill the (10 + token->len + 1)-byte TokenResp blob for the
84 * given token and return it via *out / *outlen.
85 * WHY : the earlier code sent a raw "ztn\0" + token blob (the format OUR own
86 * parser tolerates); a stock XrdSecProtocolztn reads byte 5 as the opr
87 * code, sees a token character instead of 'T', and rejects the exchange
88 * with "Invalid ztn response code". This produces the exact bytes stock
89 * expects.
90 * HOW : write the 8-byte header (id/ver/opr/rsvd), the big-endian token length
91 * (token->len + 1, the trailing NUL counts), the token, and one NUL.
92 * Returns 0 with *out malloc'd (caller frees), -1 on allocation failure. */
93static int
94cache_origin_build_ztn_credential(const ngx_str_t *token, u_char **out,
95 size_t *outlen)
96{
97 u_char *blob;
98 size_t blen;
99 uint16_t tlen_be;
100
101 blen = BRIX_ZTN_PREFIX_LEN + token->len + 1; /* hdr + len + token + NUL */
102 blob = malloc(blen);
103 if (blob == NULL) {
104 return -1;
105 }
106
107 ngx_memcpy(blob, "ztn", 4); /* id[4], incl. trailing NUL */
108 blob[4] = BRIX_ZTN_VERSION; /* ver */
109 blob[5] = BRIX_ZTN_OPR_ISTKN; /* opr = IsTkn */
110 blob[6] = 0; /* rsvd[0] */
111 blob[7] = 0; /* rsvd[1] */
112
113 tlen_be = htons((uint16_t) (token->len + 1)); /* length includes the NUL */
114 ngx_memcpy(blob + 8, &tlen_be, sizeof(tlen_be));
115
116 ngx_memcpy(blob + BRIX_ZTN_PREFIX_LEN, token->data, token->len);
117 blob[BRIX_ZTN_PREFIX_LEN + token->len] = 0; /* required trailing NUL */
118
119 *out = blob;
120 *outlen = blen;
121 return 0;
122}

124/* brix_cache_origin_auth_ztn — present a WLCG/SciToken bearer to the origin via
125 * the XrdSecztn protocol after a kXR_login advertised "&P=ztn". The exchange is a
126 * single-round kXR_auth: credtype "ztn\0", payload = the stock XrdSecztn
127 * TokenResp (see cache_origin_build_ztn_credential for the exact framing). The
128 * server (which advertised its version/maxtsz in the login parms) validates the
129 * token and replies kXR_ok. Returns 0 on a kXR_ok auth, -1 otherwise (t error
130 * set). §14/C-3. */
131int
132brix_cache_origin_auth_ztn(brix_cache_fill_t *t,
133 brix_cache_origin_conn_t *oc, const ngx_str_t *token)
134{
135 u_char *blob;
136 size_t blen;
137 uint16_t status;
138 uint32_t dlen;
139 u_char *body;
140
141 if (cache_origin_build_ztn_credential(token, &blob, &blen) != 0) {
142 brix_cache_set_error(t, kXR_NoMemory, 0,
143 "cache origin ztn payload allocation failed");
144 return -1;
145 }
146
147 if (cache_origin_send_kxr_auth(oc, "ztn", blob, (uint32_t) blen) != 0) {
148 free(blob);
149 brix_cache_set_error(t, kXR_ServerError, errno,
150 "cache origin ztn auth write failed");
151 return -1;
152 }
153 free(blob);
154
155 body = NULL;
156 if (brix_cache_read_response(t, oc, &status, &body, &dlen, 4096) != 0) {
157 return -1;
158 }
159 if (status == kXR_error) {
160 brix_cache_set_origin_error(t, body, dlen,
161 "cache origin token auth rejected");
162 free(body);
163 return -1;
164 }
165 free(body);
166
167 if (status != kXR_ok) {
168 /* ztn is single-round; a second authmore (or anything else) is a failure. */
169 brix_cache_set_error(t, kXR_AuthFailed, 0,
170 "cache origin token auth incomplete");
171 return -1;
172 }
173 return 0;
174}

176/* Load the proxy cert chain as one contiguous PEM blob (certs only). The proxy is an
177 * operator-configured path (trusted, like the server's own cert/key); opened
178 * O_NOFOLLOW so a planted symlink cannot redirect it. malloc'd, *outlen set; NULL on
179 * failure. Mirrors client/lib/sec/sec_gsi.c load_proxy_pem. */
180static uint8_t *
181cache_origin_load_proxy_pem(const char *path, size_t *outlen)
182{
183 int fd;
184 BIO *in, *out;
185 X509 *cert;
186 BUF_MEM *bm;
187 uint8_t *buf = NULL;
188 int n = 0;
189
190 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC); /* vfs-seam-allow: config-domain X.509 proxy PEM (not export storage) */
191 if (fd < 0) {
192 return NULL;
193 }
194 in = BIO_new_fd(fd, BIO_CLOSE);
195 if (in == NULL) {
196 close(fd);
197 return NULL;
198 }
199 out = BIO_new(BIO_s_mem());
200 if (out == NULL) {
201 BIO_free(in);
202 return NULL;
203 }
204 while ((cert = PEM_read_bio_X509(in, NULL, NULL, NULL)) != NULL) {
205 PEM_write_bio_X509(out, cert);
206 X509_free(cert);
207 n++;
208 }
209 ERR_clear_error(); /* benign PEM EOF on the loop end */
210 BIO_free(in);
211
212 if (n > 0) {
213 BIO_get_mem_ptr(out, &bm);
214 buf = malloc(bm->length);
215 if (buf != NULL) {
216 ngx_memcpy(buf, bm->data, bm->length);
217 *outlen = bm->length;
218 }
219 }
220 BIO_free(out);
221 return buf;
222}

246/* originauth_gsi_certreq — round 1 of the origin GSI handshake: build a
247 * kXGC_certreq from the origin's advertised gsi parms and send it as a kXR_auth.
248 *
249 * WHAT: parse `gsi_parms` for crypto/version/CA, mint a signed-DH certreq bucket,
250 * and write it to the origin connector stream.
251 * WHY : isolates the pure request construction (parse + build) with its single
252 * side effect (the socket send) at the tail — the orchestrator stays flat.
253 * HOW : mirrors sec_gsi.c: default crypto "ssl", forced version 10600 (signed-DH),
254 * an 8-byte client rtag, PoP flag 0x80. Byte-frozen vs the pre-split code.
255 * Returns 0 on a sent certreq, -1 (t error set) on RNG/build/write failure. */
256static int
257originauth_gsi_certreq(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
258 const char *gsi_parms)
259{
260 uint32_t version = 0;
261 char crypto[16] = { 0 };
262 char ca[256] = { 0 };
263 uint8_t client_rtag[8] = { 0 };
264 uint8_t *certreq = NULL;
265 size_t certreq_len = 0;
266 int rc;
267
268 brix_gsi_parse_parms(gsi_parms, &version, crypto, sizeof(crypto),
269 ca, sizeof(ca));
270 if (crypto[0] == '\0') {
271 ngx_memcpy(crypto, "ssl", 4);
272 }
273 version = 10600; /* signed-DH default, as sec_gsi.c */
274 if (!brix_gsi_rand(client_rtag, sizeof(client_rtag))) {
275 brix_cache_set_error(t, kXR_ServerError, 0, "cache origin gsi RNG failed");
276 return -1;
277 }
278 certreq = brix_gsi_build_certreq(crypto, version, ca, 0x80u, client_rtag,
279 sizeof(client_rtag), &certreq_len);
280 if (certreq == NULL) {
281 brix_cache_set_error(t, kXR_ServerError, 0,
282 "cache origin gsi certreq build failed");
283 return -1;
284 }
285 rc = cache_origin_send_kxr_auth(oc, "gsi", certreq, (uint32_t) certreq_len);
286 free(certreq);
287 if (rc != 0) {
288 brix_cache_set_error(t, kXR_ServerError, errno,
289 "cache origin gsi certreq write failed");
290 return -1;
291 }
292 return 0;
293}

295/* originauth_gsi_read_cert — read the round-1 reply and require a kXGS_cert.
296 *
297 * WHAT: read the origin's response to the certreq and validate it is a
298 * kXR_authmore carrying a >= 16-byte kXGS_cert body.
299 * WHY : separates the wire read + status classification from the orchestrator;
300 * the caller owns *body and frees it exactly once.
301 * HOW : kXR_error → surface the origin error; anything but kXR_authmore with a
302 * usable body -> AuthFailed. On success body/dlen are set for round 2.
303 * Returns 0 with *body owned by the caller, -1 (t error set, *body freed). */
304static int
305originauth_gsi_read_cert(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
306 u_char **body, uint32_t *dlen)
307{
308 uint16_t status;
309
310 *body = NULL;
311 if (brix_cache_read_response(t, oc, &status, body, dlen, 1 << 16) != 0) {
312 return -1;
313 }
314 if (status == kXR_error) {
315 brix_cache_set_origin_error(t, *body, *dlen, "cache origin gsi rejected");
316 free(*body);
317 *body = NULL;
318 return -1;
319 }
320 if (status != kXR_authmore || *body == NULL || *dlen < 16) {
321 free(*body);
322 *body = NULL;
323 brix_cache_set_error(t, kXR_AuthFailed, 0,
324 "cache origin gsi expected kXGS_cert");
325 return -1;
326 }
327 return 0;
328}

330/* originauth_gsi_verify_srv_cert — MITM protection before agreeing a secret.
331 *
332 * WHAT: when the credential names a CA (conf->gsi_store), require the origin's
333 * server cert in the kXGS_cert body AND verify it against that store.
334 * WHY : the shared secret must not be agreed with an unverified peer; this is the
335 * security checkpoint. No store = operator opted out (unauthenticated origin).
336 * HOW : find the kXRS_x509 bucket, parse the PEM, run X509_verify_cert with
337 * ALLOW_PROXY_CERTS. Verdict byte-frozen vs the pre-split code.
338 * Returns 0 when verification passes (or is skipped), -1 (t error set) on failure.
339 * Never frees `body` — the caller owns it. */
340static int
341originauth_gsi_verify_srv_cert(brix_cache_fill_t *t, const u_char *body,
342 uint32_t dlen)
343{
344 const uint8_t *srv_pem = NULL;
345 size_t srv_pem_len = 0;
346 BIO *mbio;
347 X509 *srv;
348 X509_STORE_CTX *sctx;
349 int ok = 0;
350
351 if (t->conf->gsi_store == NULL) {
352 return 0;
353 }
354 if (brix_gsi_find_bucket(body, dlen, (uint32_t) kXRS_x509,
355 &srv_pem, &srv_pem_len) != 0
356 || srv_pem_len == 0)
357 {
358 brix_cache_set_error(t, kXR_NotAuthorized, 0,
359 "cache origin gsi: server presented no certificate to verify");
360 return -1;
361 }
362 mbio = BIO_new_mem_buf(srv_pem, (int) srv_pem_len);
363 srv = (mbio != NULL) ? PEM_read_bio_X509(mbio, NULL, NULL, NULL) : NULL;
364 if (mbio != NULL) { BIO_free(mbio); }
365 if (srv != NULL) {
366 sctx = X509_STORE_CTX_new();
367 if (sctx != NULL
368 && X509_STORE_CTX_init(sctx, t->conf->gsi_store, srv, NULL) == 1)
369 {
370 X509_STORE_CTX_set_flags(sctx, X509_V_FLAG_ALLOW_PROXY_CERTS);
371 ok = (X509_verify_cert(sctx) == 1);
372 }
373 if (sctx != NULL) { X509_STORE_CTX_free(sctx); }
374 X509_free(srv);
375 }
376 ERR_clear_error();
377 if (!ok) {
378 brix_cache_set_error(t, kXR_NotAuthorized, 0,
379 "cache origin gsi: server certificate verification failed");
380 return -1;
381 }
382 return 0;
383}

385/* originauth_gsi_cred_t — the round-2 proxy credential loaded from disk: the PEM
386 * cert chain (pem/pem_len) and its private key. Grouped so the load/warn/send
387 * steps thread one value and free it as a coherent pair. */
388typedef struct {

394/* originauth_gsi_load_credential — load the proxy PEM chain + private key.
395 *
396 * WHAT: load the cert chain from proxy_path and the key from either a separate
397 * configured key file (cache_origin_x509_key) or the same proxy PEM.
398 * WHY : a plain host cert/key pair then works without hand-concatenation into a
399 * proxy; both are loaded together so the caller frees them as a pair.
400 * HOW : cache_origin_load_proxy_pem + cache_origin_load_proxy_key; on any failure
401 * both are freed here and the struct is zeroed.
402 * Returns 0 with cred fields owned by the caller, -1 (t error set). */
403static int
404originauth_gsi_load_credential(brix_cache_fill_t *t, const char *proxy_path,
405 originauth_gsi_cred_t *cred)
406{
407 ngx_memzero(cred, sizeof(*cred));
408
409 cred->pem = cache_origin_load_proxy_pem(proxy_path, &cred->pem_len);
410 cred->key = cache_origin_load_proxy_key(
411 (t->conf != NULL && t->conf->cache_origin_x509_key.len > 0)
412 ? (const char *) t->conf->cache_origin_x509_key.data
413 : proxy_path);
414 if (cred->pem == NULL || cred->key == NULL) {
415 free(cred->pem);
416 EVP_PKEY_free(cred->key);
417 ngx_memzero(cred, sizeof(*cred));
418 brix_cache_set_error(t, kXR_AuthFailed, 0,
419 "cache origin gsi cannot load proxy credential");
420 return -1;
421 }
422 return 0;
423}

425/* originauth_gsi_warn_bare_cert — warn if the credential is not a proxy chain.
426 *
427 * WHAT: count the PEM certificates in the loaded credential and, if fewer than 2,
428 * log a WARN explaining a stock XRootD origin will reject it.
429 * WHY : a bare host cert (1 cert) fails at the origin with an opaque "received: 1,
430 * expected: >= 2" — this proactive message names the fix. Pure diagnostic;
431 * does NOT change the handshake outcome.
432 * HOW : scan for "-----BEGIN CERTIFICATE-----" markers; warn when count < 2. */
433static void
434originauth_gsi_warn_bare_cert(const uint8_t *proxy_pem, size_t proxy_pem_len,
435 const char *proxy_path)
436{
437 size_t ncerts = 0, off = 0;
438 static const char BEGIN[] = "-----BEGIN CERTIFICATE-----";
439 const size_t blen = sizeof(BEGIN) - 1;
440
441 while (off < proxy_pem_len) {
442 void *p = memmem(proxy_pem + off, proxy_pem_len - off, BEGIN, blen);
443 if (p == NULL) { break; }
444 ncerts++;
445 off = (size_t) ((u_char *) p - proxy_pem) + blen;
446 }
447 if (ncerts < 2) {
448 ngx_log_error(NGX_LOG_WARN, ngx_cycle->log, 0,
449 "brix: origin GSI credential \"%s\" has %uz certificate(s); a "
450 "stock XRootD origin requires a delegated PROXY chain (>= 2: "
451 "proxy + end-entity cert). A bare host cert/key will be rejected "
452 "with \"received: 1, expected: >= 2\". Generate a proxy, e.g. "
453 "voms-proxy-init -rfc -cert <hostcert> -key <hostkey> -out "
454 "<proxy.pem>, and point x509_proxy at it.",
455 proxy_path, ncerts);
456 }
457}

459/* originauth_gsi_send_response — round 2: build kXGC_cert and send it.
460 *
461 * WHAT: build the cert-response (proof-of-possession over the kXGS_cert body with
462 * the proxy chain + key) and write it as a kXR_auth to the origin.
463 * WHY : keeps the crypto build + its single socket-send side effect in one step;
464 * consumes `body` (frees it) so the orchestrator's ownership stays linear.
465 * HOW : brix_gsi_build_cert_response fills resp/resp_len (byte-frozen); on failure
466 * the kernel's `err` text is surfaced. `body` and the `cred` fields are all
467 * freed here regardless of outcome.
468 * Returns 0 on a sent response, -1 (t error set). */
469static int
470originauth_gsi_send_response(brix_cache_fill_t *t, brix_cache_origin_conn_t *oc,
471 u_char *body, uint32_t dlen, originauth_gsi_cred_t *cred)
472{

474 uint32_t resp_len = 0;
475 char err[160];
476 int rc;
477
478 err[0] = '\0';
479 rc = brix_gsi_build_cert_response(body, dlen, cred->pem, cred->pem_len,
480 cred->key, &resp, &resp_len,
481 err, sizeof(err));
482 free(body);
483 free(cred->pem);
484 EVP_PKEY_free(cred->key);
485 if (rc != 0) {
486 brix_cache_set_error(t, kXR_AuthFailed, 0,
487 err[0] ? err : "cache origin gsi round-2 failed");
488 return -1;
489 }
490
491 rc = cache_origin_send_kxr_auth(oc, "gsi", resp, resp_len);
492 free(resp);
493 if (rc != 0) {
494 brix_cache_set_error(t, kXR_ServerError, errno,
495 "cache origin gsi round-2 write failed");
496 return -1;
497 }
498 return 0;
499}

533/* brix_cache_origin_auth_gsi — present an X.509 proxy to the origin via the
534 * in-process XrdSecgsi two-round handshake after a login advertised "&P=gsi". The
535 * DH/cipher/proof-of-possession math is the SHARED gsi_core kernel (the exact
536 * implementation client/lib/sec/sec_gsi.c and src/tpc/gsi/gsi_outbound_exchange.c use):
537 * round 1 client → kXGC_certreq (build_certreq); server → kXGS_cert (kXR_authmore)
538 * round 2 client → kXGC_cert (build_cert_response w/ the proxy chain + key); → kXR_ok
539 * `gsi_parms` is the server's gsi v:/c:/ca: list; `proxy_path` the proxy PEM file.
540 * Returns 0 on a kXR_ok auth, -1 otherwise (t error set). C-3.
541 *
542 * Orchestrator only: each round/concern is a `static originauth_gsi_*` helper and
543 * this reads as a flat sequence of `if (step(...) != 0) return -1;`. */
544int
545brix_cache_origin_auth_gsi(brix_cache_fill_t *t,
546 brix_cache_origin_conn_t *oc, const char *gsi_parms, const char *proxy_path)
547{
548 u_char *body = NULL;
549 uint32_t dlen = 0;
550 originauth_gsi_cred_t cred = { 0 };
551
552 /* ---- round 1: kXGC_certreq -> kXGS_cert ---- */
553 if (originauth_gsi_certreq(t, oc, gsi_parms) != 0) {
554 return -1;
555 }
556 if (originauth_gsi_read_cert(t, oc, &body, &dlen) != 0) {
557 return -1;
558 }
559 if (originauth_gsi_verify_srv_cert(t, body, dlen) != 0) {
560 free(body);
561 return -1;
562 }
563
564 /* ---- round 2: kXGC_cert with the proxy chain ----
565 * The cert chain always comes from proxy_path. The key normally lives in the
566 * same file (a combined proxy PEM), but when the credential supplied a separate
567 * cert + key (cache_origin_x509_key set), load the key from there — a plain host
568 * cert/key pair then works without being hand-concatenated into a proxy. */
569 if (originauth_gsi_load_credential(t, proxy_path, &cred) != 0) {
570 free(body);
571 return -1;
572 }
573 originauth_gsi_warn_bare_cert(cred.pem, cred.pem_len, proxy_path);
574
575 /* Consumes body and the cred fields regardless of outcome. */
576 if (originauth_gsi_send_response(t, oc, body, dlen, &cred) != 0) {
577 return -1;
578 }
579
580 return originauth_gsi_read_final(t, oc);
581}

583/* cache_origin_load_sss_key — load the first usable key from an SSS keytab file into
584 * *out. The keytab is an operator-configured, trusted path (opened O_NOFOLLOW so a
585 * planted symlink cannot redirect it) parsed with the SHARED keytab line grammar
586 * (sss_keytab_parse_line) — the exact tokenisation the server's loader uses, so a key
587 * that works one side works the other. Returns 0 with *out filled, or -1 (unreadable /
588 * malformed / no usable key). */
589static int
590cache_origin_load_sss_key(const char *path, brix_sss_key_t *out)
591{
592 int fd;
593 FILE *fp;
594 char line[1024];
595 int found = 0;
596
597 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC); /* vfs-seam-allow: config-domain SSS keytab (not export storage) */
598 if (fd < 0) {
599 return -1;
600 }
601 fp = fdopen(fd, "r");
602 if (fp == NULL) {
603 close(fd);
604 return -1;
605 }
606 ngx_memzero(out, sizeof(*out));
607 while (!found && fgets(line, sizeof(line), fp) != NULL) {
608 sss_keytab_entry_t entry;
609 int rc = sss_keytab_parse_line(line, &entry,
610 (int64_t) ngx_time());
611
612 if (rc < 0) { /* malformed ⇒ fail closed */
613 (void) fclose(fp); /* read-only stream — nothing buffered to lose */
614 return -1;
615 }
616 if (rc == 0) { /* blank / comment / expired */
617 continue;
618 }
619 out->id = entry.id;
620 out->exp = (time_t) entry.exp;
621 out->key_len = entry.key_len;
622 ngx_memcpy(out->key, entry.key, entry.key_len);
623 ngx_cpystrn((u_char *) out->user, (u_char *) entry.user,
624 sizeof(out->user));
625 ngx_cpystrn((u_char *) out->group, (u_char *) entry.group,
626 sizeof(out->group));
627 ngx_cpystrn((u_char *) out->name, (u_char *) entry.name,
628 sizeof(out->name));
629 found = 1;
630 }
631 (void) fclose(fp); /* read-only stream — nothing buffered to lose */
632 return found ? 0 : -1;
633}

635/* brix_cache_origin_auth_sss — present an SSS (Simple Shared Secret) credential to
636 * the origin via the XrdSecsss protocol after a login advertised "&P=sss". Mints the
637 * SAME kXR_auth blob the proxy path sends (brix_sss_build_proxy_credential): a
638 * Blowfish-CFB block over a nonce + gen-time + the keytab user, keyed by the shared
639 * secret. Single-round: expect kXR_ok. Returns 0, or -1 (t error set). §14. */
640int
641brix_cache_origin_auth_sss(brix_cache_fill_t *t,
642 brix_cache_origin_conn_t *oc, const char *keytab_path)
643{
644 brix_sss_key_t key;
645 u_char cred[2048];
646 size_t cred_len = 0;
647 uint16_t status;
648 uint32_t dlen;
649 u_char *body = NULL;
650
651 if (cache_origin_load_sss_key(keytab_path, &key) != 0) {
652 brix_cache_set_error(t, kXR_AuthFailed, 0,
653 "cache origin SSS keytab unreadable or has no usable key");
654 return -1;
655 }
656 if (brix_sss_build_proxy_credential(&key, key.user, cred, sizeof(cred),
657 &cred_len) != NGX_OK)
658 {
659 brix_cache_set_error(t, kXR_AuthFailed, 0,
660 "cache origin SSS credential build failed");
661 return -1;
662 }
663 if (cache_origin_send_kxr_auth(oc, "sss", cred, (uint32_t) cred_len) != 0) {
664 brix_cache_set_error(t, kXR_ServerError, errno,
665 "cache origin SSS auth write failed");
666 return -1;
667 }
668 if (brix_cache_read_response(t, oc, &status, &body, &dlen, 4096) != 0) {
669 return -1;
670 }
671 if (status == kXR_error) {
672 brix_cache_set_origin_error(t, body, dlen,
673 "cache origin SSS auth rejected");
674 free(body);
675 return -1;
676 }
677 free(body);
678 if (status != kXR_ok) { /* SSS is single-round */
679 brix_cache_set_error(t, kXR_AuthFailed, 0,
680 "cache origin SSS auth incomplete");
681 return -1;
682 }
683 return 0;
684}
