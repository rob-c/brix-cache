36/* Frame one kXR_auth request (ClientAuthRequest header, credtype = the 4-byte
37 * protocol id, + credential payload) on the connector stream. Shared by the ztn
38 * and sss auth helpers here and the gsi helpers in origin_auth_gsi.c (prototype in
39 * cache_internal.h). Returns 0, or -1 (errno set). */
40int
41cache_origin_send_kxr_auth(brix_cache_origin_conn_t *oc, const char credtype[4],
42 const u_char *payload, uint32_t plen)
43{
44 ClientAuthRequest req;
45
46 ngx_memzero(&req, sizeof(req));
47 req.streamid[1] = 1; /* the connector stream */
48 req.requestid = htons(kXR_auth);
49 ngx_memcpy(req.credtype, credtype, 4);
50 req.dlen = htonl((kXR_int32) plen);
51
52 if (brix_cache_io_send(oc, &req, sizeof(req)) != 0
53 || (plen > 0 && brix_cache_io_send(oc, payload, plen) != 0))
54 {
55 return -1;
56 }
57 return 0;
58}

60/* XrdSecProtocolztn credential wire format (stock XRootD, byte-frozen).
61 *
62 * The ztn credential is an XrdSecProtocolztn::TokenResp: an 8-byte TokenHdr, a
63 * 2-byte big-endian length, then the token followed by one NUL terminator:
64 *
65 * off 0..3 id[4] = "ztn\0" (NUL-terminated protocol id)
66 * off 4 ver = 0 (XrdSecProtocolztn::ztnVersion)
67 * off 5 opr = 'T' (TokenHdr::IsTkn — "here is a token")
68 * off 6..7 rsvd[2] = 0, 0
69 * off 8..9 len = htons(tsz + 1) (token length INCLUDING its trailing NUL)
70 * off 10.. tkn[tsz] + one NUL byte
71 *
72 * The stock server (Authenticate) reads opr at off 5 to route the request,
73 * ntohs(len) at off 8, and the token at off 10 (pfxLen = sizeof(TokenHdr) +
74 * sizeof(uint16_t) = 10). It requires len >= 1, len's byte at (10 + len - 1) to
75 * be NUL, and (10 + len) <= credential size. The whole credential is
76 * (10 + tsz + 1) bytes. */
77#define BRIX_ZTN_VERSION 0
78#define BRIX_ZTN_OPR_ISTKN 'T'
79#define BRIX_ZTN_PREFIX_LEN 10 /* 8-byte TokenHdr + 2-byte length */
80
81/* cache_origin_build_ztn_credential — frame a bearer token as a stock XrdSecztn
82 * TokenResp credential.
83 *
84 * WHAT: allocate and fill the (10 + token->len + 1)-byte TokenResp blob for the
85 * given token and return it via *out / *outlen.
86 * WHY : the earlier code sent a raw "ztn\0" + token blob (the format OUR own
87 * parser tolerates); a stock XrdSecProtocolztn reads byte 5 as the opr
88 * code, sees a token character instead of 'T', and rejects the exchange
89 * with "Invalid ztn response code". This produces the exact bytes stock
90 * expects.
91 * HOW : write the 8-byte header (id/ver/opr/rsvd), the big-endian token length
92 * (token->len + 1, the trailing NUL counts), the token, and one NUL.
93 * Returns 0 with *out malloc'd (caller frees), -1 on allocation failure. */
94static int
95cache_origin_build_ztn_credential(const ngx_str_t *token, u_char **out,
96 size_t *outlen)
97{
98 u_char *blob;
99 size_t blen;
100 uint16_t tlen_be;
101
102 blen = BRIX_ZTN_PREFIX_LEN + token->len + 1; /* hdr + len + token + NUL */
103 blob = malloc(blen);
104 if (blob == NULL) {
105 return -1;
106 }
107
108 ngx_memcpy(blob, "ztn", 4); /* id[4], incl. trailing NUL */
109 blob[4] = BRIX_ZTN_VERSION; /* ver */
110 blob[5] = BRIX_ZTN_OPR_ISTKN; /* opr = IsTkn */
111 blob[6] = 0; /* rsvd[0] */
112 blob[7] = 0; /* rsvd[1] */
113
114 tlen_be = htons((uint16_t) (token->len + 1)); /* length includes the NUL */
115 ngx_memcpy(blob + 8, &tlen_be, sizeof(tlen_be));
116
117 ngx_memcpy(blob + BRIX_ZTN_PREFIX_LEN, token->data, token->len);
118 blob[BRIX_ZTN_PREFIX_LEN + token->len] = 0; /* required trailing NUL */
119
120 *out = blob;
121 *outlen = blen;
122 return 0;
123}

125/* brix_cache_origin_auth_ztn — present a WLCG/SciToken bearer to the origin via
126 * the XrdSecztn protocol after a kXR_login advertised "&P=ztn". The exchange is a
127 * single-round kXR_auth: credtype "ztn\0", payload = the stock XrdSecztn
128 * TokenResp (see cache_origin_build_ztn_credential for the exact framing). The
129 * server (which advertised its version/maxtsz in the login parms) validates the
130 * token and replies kXR_ok. Returns 0 on a kXR_ok auth, -1 otherwise (t error
131 * set). §14/C-3. */
132int
133brix_cache_origin_auth_ztn(brix_cache_fill_t *t,
134 brix_cache_origin_conn_t *oc, const ngx_str_t *token)
135{
136 u_char *blob;
137 size_t blen;
138 uint16_t status;
139 uint32_t dlen;
140 u_char *body;
141
142 if (cache_origin_build_ztn_credential(token, &blob, &blen) != 0) {
143 brix_cache_set_error(t, kXR_NoMemory, 0,
144 "cache origin ztn payload allocation failed");
145 return -1;
146 }
147
148 if (cache_origin_send_kxr_auth(oc, "ztn", blob, (uint32_t) blen) != 0) {
149 free(blob);
150 brix_cache_set_error(t, kXR_ServerError, errno,
151 "cache origin ztn auth write failed");
152 return -1;
153 }
154 free(blob);
155
156 body = NULL;
157 if (brix_cache_read_response(t, oc, &status, &body, &dlen, 4096) != 0) {
158 return -1;
159 }
160 if (status == kXR_error) {
161 brix_cache_set_origin_error(t, body, dlen,
162 "cache origin token auth rejected");
163 free(body);
164 return -1;
165 }
166 free(body);
167
168 if (status != kXR_ok) {
169 /* ztn is single-round; a second authmore (or anything else) is a failure. */
170 brix_cache_set_error(t, kXR_AuthFailed, 0,
171 "cache origin token auth incomplete");
172 return -1;
173 }
174 return 0;
175}

177/* cache_origin_load_sss_key — load the first usable key from an SSS keytab file into
178 * *out. The keytab is an operator-configured, trusted path (opened O_NOFOLLOW so a
179 * planted symlink cannot redirect it) parsed with the SHARED keytab line grammar
180 * (sss_keytab_parse_line) — the exact tokenisation the server's loader uses, so a key
181 * that works one side works the other. Returns 0 with *out filled, or -1 (unreadable /
182 * malformed / no usable key). */
183static int
184cache_origin_load_sss_key(const char *path, brix_sss_key_t *out)
185{
186 int fd;
187 FILE *fp;
188 char line[1024];
189 int found = 0;
190
191 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC); /* vfs-seam-allow: config-domain SSS keytab (not export storage) */
192 if (fd < 0) {
193 return -1;
194 }
195 fp = fdopen(fd, "r");
196 if (fp == NULL) {
197 close(fd);
198 return -1;
199 }
200 ngx_memzero(out, sizeof(*out));
201 while (!found && fgets(line, sizeof(line), fp) != NULL) {
202 sss_keytab_entry_t entry;
203 int rc = sss_keytab_parse_line(line, &entry,
204 (int64_t) ngx_time());
205
206 if (rc < 0) { /* malformed ⇒ fail closed */
207 (void) fclose(fp); /* read-only stream — nothing buffered to lose */
208 return -1;
209 }
210 if (rc == 0) { /* blank / comment / expired */
211 continue;
212 }
213 out->id = entry.id;
214 out->exp = (time_t) entry.exp;
215 out->key_len = entry.key_len;
216 ngx_memcpy(out->key, entry.key, entry.key_len);
217 ngx_cpystrn((u_char *) out->user, (u_char *) entry.user,
218 sizeof(out->user));
219 ngx_cpystrn((u_char *) out->group, (u_char *) entry.group,
220 sizeof(out->group));
221 ngx_cpystrn((u_char *) out->name, (u_char *) entry.name,
222 sizeof(out->name));
223 found = 1;
224 }
225 (void) fclose(fp); /* read-only stream — nothing buffered to lose */
226 return found ? 0 : -1;
227}

229/* brix_cache_origin_auth_sss — present an SSS (Simple Shared Secret) credential to
230 * the origin via the XrdSecsss protocol after a login advertised "&P=sss". Mints the
231 * SAME kXR_auth blob the proxy path sends (brix_sss_build_proxy_credential): a
232 * Blowfish-CFB block over a nonce + gen-time + the keytab user, keyed by the shared
233 * secret. Single-round: expect kXR_ok. Returns 0, or -1 (t error set). §14. */
234int
235brix_cache_origin_auth_sss(brix_cache_fill_t *t,
236 brix_cache_origin_conn_t *oc, const char *keytab_path)
237{
238 brix_sss_key_t key;
239 u_char cred[2048];
240 size_t cred_len = 0;
241 uint16_t status;
242 uint32_t dlen;
243 u_char *body = NULL;
244
245 if (cache_origin_load_sss_key(keytab_path, &key) != 0) {
246 brix_cache_set_error(t, kXR_AuthFailed, 0,
247 "cache origin SSS keytab unreadable or has no usable key");
248 return -1;
249 }
250 if (brix_sss_build_proxy_credential(&key, key.user, cred, sizeof(cred),
251 &cred_len) != NGX_OK)
252 {
253 brix_cache_set_error(t, kXR_AuthFailed, 0,
254 "cache origin SSS credential build failed");
255 return -1;
256 }
257 if (cache_origin_send_kxr_auth(oc, "sss", cred, (uint32_t) cred_len) != 0) {
258 brix_cache_set_error(t, kXR_ServerError, errno,
259 "cache origin SSS auth write failed");
260 return -1;
261 }
262 if (brix_cache_read_response(t, oc, &status, &body, &dlen, 4096) != 0) {
263 return -1;
264 }
265 if (status == kXR_error) {
266 brix_cache_set_origin_error(t, body, dlen,
267 "cache origin SSS auth rejected");
268 free(body);
269 return -1;
270 }
271 free(body);
272 if (status != kXR_ok) { /* SSS is single-round */
273 brix_cache_set_error(t, kXR_AuthFailed, 0,
274 "cache origin SSS auth incomplete");
275 return -1;
276 }
277 return 0;
