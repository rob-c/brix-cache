1/*
2 * cache/origin_auth.c — origin-side authentication for cache fills.
3 *
4 * Split out of origin_protocol.c: the ztn (WLCG bearer), GSI (X.509 proxy), and
5 * SSS auth handshakes a cache node performs against its upstream origin, plus
6 * their credential-loading helpers. Keeping the auth handshake (~430 lines) in

8 * and lets the security-sensitive origin-auth path be reviewed on its own.

10 * The three brix_cache_origin_auth_{ztn,gsi,sss}() entry points are declared in
11 * cache_internal.h and called from brix_cache_origin_bootstrap().

36/* Frame one kXR_auth request (ClientAuthRequest header, credtype = the 4-byte
37 * protocol id, + credential payload) on the connector stream. Shared by the ztn
38 * and gsi auth helpers. Returns 0, or -1 (errno set). */
39static int
40cache_origin_send_kxr_auth(brix_cache_origin_conn_t *oc, const char credtype[4],
41 const u_char *payload, uint32_t plen)
42{
43 ClientAuthRequest req;
44
45 ngx_memzero(&req, sizeof(req));
46 req.streamid[1] = 1; /* the connector stream */
47 req.requestid = htons(kXR_auth);
48 ngx_memcpy(req.credtype, credtype, 4);
49 req.dlen = htonl((kXR_int32) plen);
50
51 if (brix_cache_io_send(oc, &req, sizeof(req)) != 0
52 || (plen > 0 && brix_cache_io_send(oc, payload, plen) != 0))
53 {
54 return -1;
55 }
56 return 0;
57}

59/* brix_cache_origin_auth_ztn — present a WLCG/SciToken bearer to the origin via
60 * the XrdSecztn protocol after a kXR_login returned kXR_authmore. The credential is
61 * a single-round kXR_auth: credtype "ztn\0", payload "ztn\0" + <token> (the exact
62 * wire format the native client's sec_token.c sends and this server's gsi/token.c
63 * parses). Returns 0 on a kXR_ok auth, -1 otherwise (t error set). §14/C-3. */
64int
65brix_cache_origin_auth_ztn(brix_cache_fill_t *t,
66 brix_cache_origin_conn_t *oc, const ngx_str_t *token)
67{
68 u_char *blob;
69 size_t blen;
70 uint16_t status;
71 uint32_t dlen;
72 u_char *body;
73
74 blen = 4 + token->len; /* "ztn\0" + token */
75 blob = malloc(blen);
76 if (blob == NULL) {
77 brix_cache_set_error(t, kXR_NoMemory, 0,
78 "cache origin ztn payload allocation failed");
79 return -1;
80 }
81 ngx_memcpy(blob, "ztn", 4); /* copies the trailing NUL too */
82 ngx_memcpy(blob + 4, token->data, token->len);
83
84 if (cache_origin_send_kxr_auth(oc, "ztn", blob, (uint32_t) blen) != 0) {
85 free(blob);
86 brix_cache_set_error(t, kXR_ServerError, errno,
87 "cache origin ztn auth write failed");
88 return -1;
89 }
90 free(blob);
91
92 body = NULL;
93 if (brix_cache_read_response(t, oc, &status, &body, &dlen, 4096) != 0) {
94 return -1;
95 }
96 if (status == kXR_error) {
97 brix_cache_set_origin_error(t, body, dlen,
98 "cache origin token auth rejected");
99 free(body);
100 return -1;
101 }
102 free(body);
103
104 if (status != kXR_ok) {
105 /* ztn is single-round; a second authmore (or anything else) is a failure. */
106 brix_cache_set_error(t, kXR_AuthFailed, 0,
107 "cache origin token auth incomplete");
108 return -1;
109 }
110 return 0;
111}

113/* Load the proxy cert chain as one contiguous PEM blob (certs only). The proxy is an
114 * operator-configured path (trusted, like the server's own cert/key); opened
115 * O_NOFOLLOW so a planted symlink cannot redirect it. malloc'd, *outlen set; NULL on
116 * failure. Mirrors client/lib/sec/sec_gsi.c load_proxy_pem. */
117static uint8_t *
118cache_origin_load_proxy_pem(const char *path, size_t *outlen)
119{
120 int fd;
121 BIO *in, *out;
122 X509 *cert;
123 BUF_MEM *bm;
124 uint8_t *buf = NULL;
125 int n = 0;
126
127 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC); /* vfs-seam-allow: config-domain X.509 proxy PEM (not export storage) */
128 if (fd < 0) {
129 return NULL;
130 }
131 in = BIO_new_fd(fd, BIO_CLOSE);
132 if (in == NULL) {
133 close(fd);
134 return NULL;
135 }
136 out = BIO_new(BIO_s_mem());
137 if (out == NULL) {
138 BIO_free(in);
139 return NULL;
140 }
141 while ((cert = PEM_read_bio_X509(in, NULL, NULL, NULL)) != NULL) {
142 PEM_write_bio_X509(out, cert);
143 X509_free(cert);
144 n++;
145 }
146 ERR_clear_error(); /* benign PEM EOF on the loop end */
147 BIO_free(in);
148
149 if (n > 0) {
150 BIO_get_mem_ptr(out, &bm);
151 buf = malloc(bm->length);
152 if (buf != NULL) {
153 ngx_memcpy(buf, bm->data, bm->length);
154 *outlen = bm->length;
155 }
156 }
157 BIO_free(out);
158 return buf;
159}

183/* brix_cache_origin_auth_gsi — present an X.509 proxy to the origin via the
184 * in-process XrdSecgsi two-round handshake after a login advertised "&P=gsi". The
185 * DH/cipher/proof-of-possession math is the SHARED gsi_core kernel (the exact
186 * implementation client/lib/sec/sec_gsi.c and src/tpc/gsi/gsi_outbound_exchange.c use):
187 * round 1 client → kXGC_certreq (build_certreq); server → kXGS_cert (kXR_authmore)
188 * round 2 client → kXGC_cert (build_cert_response w/ the proxy chain + key); → kXR_ok
189 * `gsi_parms` is the server's gsi v:/c:/ca: list; `proxy_path` the proxy PEM file.
190 * Returns 0 on a kXR_ok auth, -1 otherwise (t error set). C-3. */
191int
192brix_cache_origin_auth_gsi(brix_cache_fill_t *t,
193 brix_cache_origin_conn_t *oc, const char *gsi_parms, const char *proxy_path)
194{
195 uint32_t version = 0;
196 char crypto[16] = { 0 };
197 char ca[256] = { 0 };
198 uint8_t client_rtag[8];
199 uint8_t *certreq = NULL;
200 size_t certreq_len = 0;
201 uint16_t status;
202 uint32_t dlen;
203 u_char *body = NULL;
204 uint8_t *proxy_pem = NULL;
205 size_t proxy_pem_len = 0;
206 EVP_PKEY *proxy_key = NULL;
207 uint8_t *resp = NULL;
208 uint32_t resp_len = 0;
209 char err[160];
210 int rc;
211
212 /* ---- round 1: kXGC_certreq ---- */
213 brix_gsi_parse_parms(gsi_parms, &version, crypto, sizeof(crypto),
214 ca, sizeof(ca));
215 if (crypto[0] == '\0') {
216 ngx_memcpy(crypto, "ssl", 4);
217 }
218 version = 10600; /* signed-DH default, as sec_gsi.c */
219 if (!brix_gsi_rand(client_rtag, sizeof(client_rtag))) {
220 brix_cache_set_error(t, kXR_ServerError, 0, "cache origin gsi RNG failed");
221 return -1;
222 }
223 certreq = brix_gsi_build_certreq(crypto, version, ca, 0x80u, client_rtag,
224 sizeof(client_rtag), &certreq_len);
225 if (certreq == NULL) {
226 brix_cache_set_error(t, kXR_ServerError, 0,
227 "cache origin gsi certreq build failed");
228 return -1;
229 }
230 rc = cache_origin_send_kxr_auth(oc, "gsi", certreq, (uint32_t) certreq_len);
231 free(certreq);
232 if (rc != 0) {
233 brix_cache_set_error(t, kXR_ServerError, errno,
234 "cache origin gsi certreq write failed");
235 return -1;
236 }
237
238 if (brix_cache_read_response(t, oc, &status, &body, &dlen, 1 << 16) != 0) {
239 return -1;
240 }
241 if (status == kXR_error) {
242 brix_cache_set_origin_error(t, body, dlen, "cache origin gsi rejected");
243 free(body);
244 return -1;
245 }
246 if (status != kXR_authmore || body == NULL || dlen < 16) {
247 free(body);
248 brix_cache_set_error(t, kXR_AuthFailed, 0,
249 "cache origin gsi expected kXGS_cert");
250 return -1;
251 }

253 /* MITM protection: when the credential names a CA (cache_origin_ca_dir →
254 * conf->gsi_store), the origin's server cert MUST be present in the kXGS_cert AND
255 * verify against that store BEFORE we agree a shared secret with it. No store =
256 * the operator opted out of origin-cert verification (unauthenticated origin). */
257 if (t->conf->gsi_store != NULL) {
258 const uint8_t *srv_pem = NULL;
259 size_t srv_pem_len = 0;
260 BIO *mbio;
261 X509 *srv;
262 X509_STORE_CTX *sctx;
263 int ok = 0;
264
265 if (brix_gsi_find_bucket(body, dlen, (uint32_t) kXRS_x509,
266 &srv_pem, &srv_pem_len) != 0
267 || srv_pem_len == 0)
268 {
269 free(body);
270 brix_cache_set_error(t, kXR_NotAuthorized, 0,
271 "cache origin gsi: server presented no certificate to verify");
272 return -1;
273 }
274 mbio = BIO_new_mem_buf(srv_pem, (int) srv_pem_len);
275 srv = (mbio != NULL) ? PEM_read_bio_X509(mbio, NULL, NULL, NULL) : NULL;
276 if (mbio != NULL) { BIO_free(mbio); }
277 if (srv != NULL) {
278 sctx = X509_STORE_CTX_new();
279 if (sctx != NULL
280 && X509_STORE_CTX_init(sctx, t->conf->gsi_store, srv, NULL) == 1)
281 {
282 X509_STORE_CTX_set_flags(sctx, X509_V_FLAG_ALLOW_PROXY_CERTS);
283 ok = (X509_verify_cert(sctx) == 1);
284 }
285 if (sctx != NULL) { X509_STORE_CTX_free(sctx); }
286 X509_free(srv);
287 }
288 ERR_clear_error();
289 if (!ok) {
290 free(body);
291 brix_cache_set_error(t, kXR_NotAuthorized, 0,
292 "cache origin gsi: server certificate verification failed");
293 return -1;
294 }
295 }
296
297 /* ---- round 2: kXGC_cert with the proxy chain ---- */
298 proxy_pem = cache_origin_load_proxy_pem(proxy_path, &proxy_pem_len);
299 proxy_key = cache_origin_load_proxy_key(proxy_path);
300 if (proxy_pem == NULL || proxy_key == NULL) {
301 free(proxy_pem);
302 EVP_PKEY_free(proxy_key);
303 free(body);
304 brix_cache_set_error(t, kXR_AuthFailed, 0,
305 "cache origin gsi cannot load proxy credential");
306 return -1;
307 }
308
309 err[0] = '\0';
310 rc = brix_gsi_build_cert_response(body, dlen, proxy_pem, proxy_pem_len,
311 proxy_key, &resp, &resp_len,
312 err, sizeof(err));
313 free(body);
314 body = NULL;
315 free(proxy_pem);
316 EVP_PKEY_free(proxy_key);
317 if (rc != 0) {
318 brix_cache_set_error(t, kXR_AuthFailed, 0,
319 err[0] ? err : "cache origin gsi round-2 failed");
320 return -1;
321 }
322
323 rc = cache_origin_send_kxr_auth(oc, "gsi", resp, resp_len);
324 free(resp);
325 if (rc != 0) {
326 brix_cache_set_error(t, kXR_ServerError, errno,
327 "cache origin gsi round-2 write failed");
328 return -1;
329 }
330
331 body = NULL;
332 if (brix_cache_read_response(t, oc, &status, &body, &dlen, 4096) != 0) {
333 return -1;
334 }
335 if (status == kXR_error) {
336 brix_cache_set_origin_error(t, body, dlen,
337 "cache origin gsi authentication failed");
338 free(body);
339 return -1;
340 }
341 free(body);
342 if (status != kXR_ok) {
343 brix_cache_set_error(t, kXR_AuthFailed, 0,
344 "cache origin gsi authentication incomplete");
345 return -1;
346 }
347 return 0;
348}

350/* cache_origin_load_sss_key — load the first usable key from an SSS keytab file into
351 * *out. The keytab is an operator-configured, trusted path (opened O_NOFOLLOW so a
352 * planted symlink cannot redirect it) parsed with the SHARED keytab line grammar
353 * (sss_keytab_parse_line) — the exact tokenisation the server's loader uses, so a key
354 * that works one side works the other. Returns 0 with *out filled, or -1 (unreadable /
355 * malformed / no usable key). */
356static int
357cache_origin_load_sss_key(const char *path, brix_sss_key_t *out)
358{
359 int fd;
360 FILE *fp;
361 char line[1024];
362 int found = 0;
363
364 fd = open(path, O_RDONLY | O_NOFOLLOW | O_CLOEXEC); /* vfs-seam-allow: config-domain SSS keytab (not export storage) */
365 if (fd < 0) {
366 return -1;
367 }
368 fp = fdopen(fd, "r");
369 if (fp == NULL) {
370 close(fd);
371 return -1;
372 }
373 ngx_memzero(out, sizeof(*out));
374 while (!found && fgets(line, sizeof(line), fp) != NULL) {
375 sss_keytab_entry_t entry;
376 int rc = sss_keytab_parse_line(line, &entry,
377 (int64_t) ngx_time());
378
379 if (rc < 0) { /* malformed ⇒ fail closed */
380 fclose(fp);
381 return -1;
382 }
383 if (rc == 0) { /* blank / comment / expired */
384 continue;
385 }
386 out->id = entry.id;
387 out->exp = (time_t) entry.exp;
388 out->key_len = entry.key_len;
389 ngx_memcpy(out->key, entry.key, entry.key_len);
390 ngx_cpystrn((u_char *) out->user, (u_char *) entry.user,
391 sizeof(out->user));
392 ngx_cpystrn((u_char *) out->group, (u_char *) entry.group,
393 sizeof(out->group));
394 ngx_cpystrn((u_char *) out->name, (u_char *) entry.name,
395 sizeof(out->name));
396 found = 1;
397 }
398 fclose(fp);
399 return found ? 0 : -1;
400}
