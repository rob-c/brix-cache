1/*
2 * multipart.c — S3 Multipart Upload (CreateMultipartUpload / UploadPart /
3 * CompleteMultipartUpload / AbortMultipartUpload /
4 * ListParts / ListMultipartUploads).
5 *
6 * Design overview
7 *
8 * Parts are staged as files named "part.<N>" inside a hidden working directory
9 * alongside the target object:
10 *
11 * <root>/<bucket>/<key> ← final object (created at Complete time)
12 * <root>/<bucket>/.<key>.mpu-<id>/ ← staging dir
13 * <root>/<bucket>/.<key>.mpu-<id>/part.1
14 * <root>/<bucket>/.<key>.mpu-<id>/part.2
15 * ...
16 *
17 * CompleteMultipartUpload streams part.1, part.2, … in order into a temp file
18 * then atomically renames it to the final path. The staging dir is then
19 * deleted with a safe opendir/unlinkat recursive helper — no shell subprocess.
20 *
21 * AbortMultipartUpload removes the staging dir in the same safe way.
22 *
23 * Security notes
24 *
25 * - partNumber is range-checked (1–10000) and validated to be a decimal
26 * integer before it appears in any path — no shell metachar risk.
27 * - All filesystem access goes through the VFS seam (s3_mpu_reap_stale scans via
28 * brix_vfs_opendir_quiet/readdir_kind/probe; mpu_rmdir_recursive delegates to
29 * brix_fs_remove_tree_confined) to prevent path traversal.
30 */

77/*
78 * s3_get_mpu_dir — build the staging directory path for an in-progress MPU.
79 *
80 * Given fs_path = "/export/bucket/subdir/key.root" and upload_id "abc123",
81 * produces "/export/bucket/subdir/.key.root.mpu-abc123".
82 *
83 * The leading '.' keeps the directory hidden from S3 LIST responses and
84 * ordinary directory listings.
85 */
86void
87s3_get_mpu_dir(const char *fs_path, const char *upload_id,
88 char *out_dir, size_t outsz)
89{
90 const char *last_slash = strrchr(fs_path, '/');
91 u_char *p;
92
93 if (last_slash != NULL) {
94 size_t dir_prefix_len = (size_t)(last_slash - fs_path) + 1; /* includes '/' */
95 p = ngx_snprintf((u_char *)out_dir, outsz - 1,
96 "%*s.%s.mpu-%s",
97 (int)dir_prefix_len, fs_path,
98 last_slash + 1, upload_id);
99 } else {
100 p = ngx_snprintf((u_char *)out_dir, outsz - 1,
101 ".%s.mpu-%s", fs_path, upload_id);
102 }
103 *p = '\0';
104}

106/*
107 * mpu_validate_upload_id — return 1 if upload_id contains only safe hex
108 * characters (as generated by s3_handle_multipart_initiate). Rejects any
109 * value that could affect path construction.
110 */
111int
112mpu_validate_upload_id(const char *upload_id)
113{
114 const char *p = upload_id;
115
116 if (*p == '\0') {
117 return 0;
118 }
119 for (; *p != '\0'; p++) {
120 if (!((*p >= '0' && *p <= '9') || (*p >= 'a' && *p <= 'f'))) {
121 return 0;
122 }
123 }
124 return 1;
125}

146/*
147 * Phase 39 (WS8/HTTP-2): reap abandoned multipart staging directories.
148 *
149 * WHAT: scan the single directory holding final_path for staging dirs named
150 * ".<obj>.mpu-<uploadid>" (the layout s3_get_mpu_dir produces) whose mtime
151 * is older than max_age_secs, and remove each via mpu_rmdir_recursive.
152 * Returns the number reaped; no-op (0) when max_age_secs <= 0.
153 *
154 * WHY: a client that uploads parts and then never sends Complete or Abort leaks
155 * its staging dir forever — unbounded disk growth → ENOSPC DoS. Reaping on
156 * MPU initiate keeps active directories self-cleaning, mirroring AWS S3's
157 * incomplete-multipart lifecycle expiry.
158 *
159 * HOW: one bounded readdir of final_path's parent directory (NOT recursive, not
160 * a whole-tree walk). An active upload keeps its staging-dir mtime fresh
161 * (each new part.<N> bumps it), so only genuinely idle uploads are reaped;
162 * mpu_rmdir_recursive enforces root_canon confinement on every removal.
163 */
164int
165s3_mpu_reap_stale(ngx_http_request_t *r, ngx_http_s3_loc_conf_t *cf,
166 const char *final_path, time_t max_age_secs)
167{
168 ngx_log_t *log = r->connection->log;
169 const char *root_canon = cf->common.root_canon;
170 char dir[PATH_MAX];
171 const char *slash;
172 brix_vfs_ctx_t vctx;
173 brix_vfs_dir_t *d;
174 time_t now;
175 size_t dlen;
176 int reaped = 0;
177
178 if (max_age_secs <= 0 || final_path == NULL) {
179 return 0;
180 }
181
182 slash = strrchr(final_path, '/');
183 if (slash == NULL || slash == final_path) {
184 return 0; /* no usable parent directory */
185 }
186 dlen = (size_t) (slash - final_path);
187 if (dlen >= sizeof(dir)) {
188 return 0;
189 }
190 ngx_memcpy(dir, final_path, dlen);
191 dir[dlen] = '\0';
192
193 /* Scan the staging dir AS THE MAPPED USER under impersonation (0700, owned by
194 * that user) — a bare worker opendir would EACCES. Non-metered (this reaper
195 * is opportunistic maintenance, not a user LIST). */
196 s3_build_vfs_ctx(r, dir, cf, &vctx);
197 d = brix_vfs_opendir_quiet(&vctx, NULL);
198 if (d == NULL) {
199 return 0;
200 }
201
202 now = time(NULL);
203
204 for ( ;; ) {
205 ngx_str_t name;
206 const char *dname;
207 char full[PATH_MAX];
208 brix_vfs_ctx_t cctx;
209 brix_vfs_stat_t cst;
210
211 if (brix_vfs_readdir_kind(d, &name, NULL) != NGX_OK) {
212 break; /* end of stream or error */
213 }
214 dname = (const char *) name.data;
215
216 /* Match the staging-dir naming ".<objname>.mpu-<uploadid>" only. */
217 if (dname[0] != '.' || strstr(dname, ".mpu-") == NULL) {
218 continue;
219 }
220 if (snprintf(full, sizeof(full), "%s/%s", dir, dname)
221 >= (int) sizeof(full)) {
222 continue;
223 }
224 /* Confined no-follow probe for the dir check + mtime (non-metered). */
225 s3_build_vfs_ctx(r, full, cf, &cctx);
226 if (brix_vfs_probe(&cctx, 1 /* no-follow */, &cst) != NGX_OK
227 || !cst.is_directory)
228 {
229 continue;
230 }
231 if ((now - cst.mtime) <= max_age_secs) {
232 continue; /* recently active — keep */
233 }
234 if (mpu_rmdir_recursive(log, root_canon, full) == 0) {
235 reaped++;
236 }
237 }
238
239 brix_vfs_closedir(d, log);
240
241 if (reaped > 0 && log != NULL) {
242 ngx_log_error(NGX_LOG_WARN, log, 0,
243 "s3: reaped %d abandoned multipart staging dir(s) "
244 "(idle >%T s) under \"%s\"",
245 reaped, max_age_secs, dir);
246 }
247 return reaped;
248}
