32/* Plan the ops whose argument rides the Mode field (mkdir/mkpath/chmod/trunc).
33 * Returns 0 with plan->action/mode/size set, -1 on a missing mandatory arg. */
34static int
35node_plan_mode_op(unsigned char code, const char *mode,
36 brix_cms_node_plan_t *plan)
37{
38 switch (code) {
39
40 case K_MKDIR:
41 case K_MKPATH:
42 plan->action = (code == K_MKDIR) ? XRDCMS_NACT_MKDIR
43 : XRDCMS_NACT_MKPATH;
44 plan->mode = mode ? (mode_t) strtol(mode, NULL, 8)
45 : (mode_t) XRDCMS_NODE_DEFAULT_DIR_MODE;
46 return 0;
47
48 case K_CHMOD:
49 if (mode == NULL) {
50 return -1;
51 }
52 plan->action = XRDCMS_NACT_CHMOD;
53 plan->mode = (mode_t) strtol(mode, NULL, 8);
54 return 0;
55
56 case K_TRUNC:
57 /* fwdArgA reuses the Mode field to carry the truncate size (decimal). */
58 if (mode == NULL) {
59 return -1;
60 }
61 plan->action = XRDCMS_NACT_TRUNC;
62 plan->size = strtoll(mode, NULL, 10);
63 return 0;
64
65 default:
66 return -1; /* not a Mode-field op — caller routes those four only */
67 }
68}

70int
71brix_cms_node_plan(unsigned char code, const brix_cms_rrdata_t *d,
72 brix_cms_node_plan_t *plan)
73{
74 const char *path = field_str(d->path, d->path_len);
75 const char *path2 = field_str(d->path2, d->path2_len);
76 const char *mode = field_str(d->mode, d->mode_len);
77
78 plan->action = XRDCMS_NACT_NONE;
79 plan->path = path;
80 plan->path2 = NULL;
81 plan->mode = 0;
82 plan->size = 0;
83 plan->reqid = field_str(d->reqid, d->reqid_len);
84 plan->notify = field_str(d->notify, d->notify_len);
85 plan->prty = field_str(d->prty, d->prty_len);
86
87 /* prepdel carries only a reqid — route it before the path guard. */
88 if (code == K_PREPDEL) {
89 if (plan->reqid == NULL) {
90 return -1;
91 }
92 plan->action = XRDCMS_NACT_PREPDEL;
93 return 0;
94 }
95
96 /* Every other executed op needs a primary path. */
97 if (path == NULL) {
98 return -1;
99 }
100
101 switch (code) {
102
103 case K_MKDIR:
104 case K_MKPATH:
105 case K_CHMOD:
106 case K_TRUNC:
107 return node_plan_mode_op(code, mode, plan);
108
109 case K_RM:
110 plan->action = XRDCMS_NACT_RM;
111 return 0;
112
113 case K_RMDIR:
114 plan->action = XRDCMS_NACT_RMDIR;
115 return 0;
116
117 case K_MV:
118 if (path2 == NULL) {
119 return -1;
120 }
121 plan->action = XRDCMS_NACT_MV;
122 plan->path2 = path2;
123 return 0;
124
125 case K_PREPADD:
126 /* padArgs: a stage-in needs the manager's reqid to key the later
127 * prepdel; notify/prty stay optional (ADR-2b sidecar map). */
128 if (plan->reqid == NULL) {
129 return -1;
130 }
131 plan->action = XRDCMS_NACT_PREPADD;
132 return 0;
133
134 default:
135 return -1; /* anything else: not executed on this node */
136 }
137}
