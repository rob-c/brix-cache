50test_percentile(void)

53
54 /* nearest-rank: p50 of 1..10 -> rank ceil(0.5*10)=5 -> value 5 */

71static void
72test_bench_compute(void)
73{
74 /* Unsorted input: must be sorted internally without mutating the caller. */
75 double lat[5] = {10.0, 2.0, 8.0, 4.0, 6.0};
76 double lat_copy[5];
77 storascan_bench_result r;
78
79 memcpy(lat_copy, lat, sizeof(lat));
80
81 /* 5 ops, 50 MiB total, in 2.0 s */
82 storascan_bench_compute(lat, 5, 50ull << 20, 2.0, &r);
83
84 CHECK(r.ops == 5, "ops counted");
85 CHECK(r.bytes == (50ull << 20), "bytes carried");
86 CHECK(CLOSE(r.elapsed_s, 2.0), "elapsed carried");
87 /* 50 MiB / 2 s = 25 MiB/s */
88 CHECK(CLOSE(r.throughput_mibps, 25.0), "throughput 25 MiB/s");
89 /* 5 ops / 2 s = 2.5 iops */
90 CHECK(CLOSE(r.iops, 2.5), "iops 2.5");
91 /* sorted 2,4,6,8,10 -> p50 rank ceil(2.5)=3 -> 6 */
92 CHECK(CLOSE(r.p50_ms, 6.0), "bench p50 == 6");
93 /* input array left untouched */
94 CHECK(memcmp(lat, lat_copy, sizeof(lat)) == 0, "input array not mutated");
95
96 /* zero elapsed -> throughput/iops 0, no div-by-zero */
97 storascan_bench_compute(lat, 5, 50ull << 20, 0.0, &r);
98 CHECK(CLOSE(r.throughput_mibps, 0.0), "zero elapsed -> 0 throughput");
99 CHECK(CLOSE(r.iops, 0.0), "zero elapsed -> 0 iops");
100}
