1/* File: tpc io — low-level socket helpers for native TPC data transfer
2 * WHAT: Three blocking I/O primitives used exclusively by the TPC thread-pool worker
3 * to read/write bytes between a remote XRootD origin and nginx's local file.
4 * All functions run inside NGX_THREADS detached workers — allocations use
5 * malloc/free (not ngx_palloc). No SSL wrapping; plain TCP sockets only.
6 *
7 * WHY: TPC transfers require the destination server to connect directly to the
8 * source XRootD endpoint and pull data over raw TCP. Unlike event-loop I/O
9 * which accumulates incrementally via callbacks, these helpers guarantee
10 * exact byte counts — partial sends/recv loop until complete. The recv_response
11 * function additionally parses XRootD ServerResponseHdr framing (status +
12 * payload length) and allocates the body for caller use.
13 *
14 * HOW: tpc_send_all iterates remaining bytes with send() loop, continues on EINTR,
15 * returns -1 on any other failure. tpc_recv_exact mirrors this pattern with
16 * recv(). tpc_recv_response reads fixed-size ServerResponseHdr first, then
17 * mallocs the payload body and calls tpc_recv_exact for the remainder.
18 * Caller must free the returned body pointer.
19 *
20 * phase-57 §F5: when the pull upgraded to TLS (t->tls != NULL, set by
21 * tpc_start_tls after a kXR_gotoTLS), send/recv route through SSL_write/
22 * SSL_read instead of the raw fd. The SSL sits on a blocking fd in the
23 * thread-pool worker, so a single SSL_write/SSL_read completes or errors;
24 * <=0 is treated as a hard failure exactly like the plain recv()/send() path. */

34/* WHAT: Send all bytes from buf over fd using send() loop — continues on EINTR, returns -1 on any other failure. Returns 0 on full write success. Caller: thread.c, bootstrap.c, source.c (wire I/O pipeline). */

36int
37tpc_send_all(brix_tpc_pull_t *t, int fd, const void *buf, size_t len)
38{
39 const u_char *cursor = buf;
40 SSL *ssl = (t != NULL) ? (SSL *) t->tls : NULL;
41
42 while (len > 0) {
43 ssize_t bytes_sent;
44
45 if (ssl != NULL) {
46 int n = SSL_write(ssl, cursor, (int) (len > INT_MAX ? INT_MAX : len));
47 if (n <= 0) {
48 return -1;
49 }
50 bytes_sent = n;
51 } else {
52 bytes_sent = send(fd, cursor, len, 0);
53 if (bytes_sent < 0 && errno == EINTR) {
54 continue;
55 }
56 if (bytes_sent <= 0) {
57 return -1;
58 }
59 }
60
61 cursor += (size_t) bytes_sent;
62 len -= (size_t) bytes_sent;
63 }
64
65 return 0;
66}

101/*
102 *
103 * WHAT: Reads a full XRootD ServerResponseHdr (status code + payload length), then
104 * allocates and reads the payload body. Returns parsed status, malloc'd body,
105 * and payload length via output pointers. NULL body when dlen==0.
106 *
107 * WHY: TPC transfers require parsing wire protocol response frames before processing
108 * data — the header encodes whether the operation succeeded (kXR_ok) or failed
109 * with a specific kXR error code, plus how many bytes of payload follow. The
110 * malloc'd body lets callers handle variable-length responses without fixed buffers.
111 *
112 * HOW: Three-phase read → 1) tpc_recv_exact fetches sizeof(ServerResponseHdr) bytes,
113 * then ntohs/ntohl decode status and dlen from network byte order. 2) If dlen==0
114 * returns immediately with NULL body. 3) Validates dlen <= TPC_RESP_MAX_BODY,
115 * mallocs (dlen + 1) bytes for safety, calls tpc_recv_exact to read payload,
116 * null-terminates the buffer, sets output pointers, and returns 0 on success.
117 * Caller must free(*body) after use. Returns -1 on any I/O or allocation error. */
118int
119tpc_recv_response(brix_tpc_pull_t *t, int fd, uint16_t *status,
120 u_char **body, uint32_t *dlen)
121{
122 ServerResponseHdr hdr;
123 u_char *response_body;
124
125 if (tpc_recv_exact(t, fd, &hdr, sizeof(hdr)) != 0) {
126 return -1;
127 }
128
129 *status = ntohs(hdr.status);
130 *dlen = (uint32_t) ntohl(hdr.dlen);
131 *body = NULL;
132
133 if (*dlen == 0) {
134 return 0;
135 }
136
137 if (*dlen > TPC_RESP_MAX_BODY) {
138 return -1;
139 }
140
141 response_body = malloc((size_t) (*dlen) + 1);
142 if (response_body == NULL) {
143 return -1;
144 }
145
146 if (tpc_recv_exact(t, fd, response_body, *dlen) != 0) {
147 free(response_body);
148 return -1;
149 }
150
151 response_body[*dlen] = '\0';
152 *body = response_body;
153 return 0;
154}
