23/* ---- directory listing (kXR_dirlist, fetch-all-then-iterate) --------------
24 *
25 * WHAT: opendir issues ONE kXR_dirlist against the origin, buffers every
26 * name into a brix_cache_dirlist_t held as the brix_sd_dir_t's driver-
27 * private state, then closes the origin session immediately (mirrors
28 * every other ns op in this file — no op here holds a live origin
29 * connection across separate VFS calls). readdir yields one buffered
30 * name per call; closedir frees the buffer.
31 * WHY: brix_sd_dir_t iteration (opendir/readdir/closedir) is 3 separate VFS
32 * calls with no guarantee they run back-to-back on the same thread, so
33 * a streamed/session-held approach would need to keep an origin TCP
34 * connection open across arbitrary VFS-call gaps — fragile and unlike
35 * every other sd_xroot op. Fetch-all-then-iterate trades a larger
36 * up-front read for a stateless, connection-free readdir/closedir.
37 * HOW: opendir_common does the session + dirlist fetch + dir handle alloc;
38 * opendir/opendir_cred differ only in which credential (NULL vs the
39 * caller's) sd_xroot_session presents at the origin. readdir walks the
40 * buffered brix_cache_dirlist_t by index; closedir frees it.
41 */
42
43/* Driver-private dir state: the fetched name buffer plus a read cursor. */
44typedef struct {
45 brix_cache_dirlist_t dl; /* heap-owned name array (see cache_internal.h) */
46 size_t next; /* index of the next name readdir will yield */
47} sd_xroot_dir_state;

49/* sd_xroot_opendir_common — shared body for opendir/opendir_cred: open a
50 * session under `cred` (NULL ⇒ service credential), fetch the whole listing,
51 * close the session, and wrap the result in a brix_sd_dir_t. Returns the dir
52 * handle, or NULL with *err_out set. */
53static brix_sd_dir_t *
54sd_xroot_opendir_common(brix_sd_instance_t *inst, const char *path,
55 const brix_sd_cred_t *cred, int *err_out)
56{
57 sd_xroot_inst_state *is = inst->state;
58 brix_cache_origin_conn_t oc;
59 brix_cache_fill_t *t;
60 sd_xroot_dir_state *ds;
61 brix_sd_dir_t *dir;
62 int e = 0;
63
64 if (sd_xroot_session(is->conf, cred, &oc, &t, &e) != 0) {
65 if (err_out) { *err_out = e; }
66 return NULL;
67 }
68
69 ds = calloc(1, sizeof(*ds));
70 if (ds == NULL) {
71 brix_cache_origin_close(&oc);
72 free(t);
73 if (err_out) { *err_out = ENOMEM; }
74 return NULL;
75 }
76 brix_cache_dirlist_init(&ds->dl);
77
78 if (brix_cache_origin_dirlist(t, &oc, path, &ds->dl) != 0) {
79 e = sd_xroot_errno(t);
80 brix_cache_dirlist_free(&ds->dl);
81 free(ds);
82 brix_cache_origin_close(&oc);
83 free(t);
84 if (err_out) { *err_out = e; }
85 return NULL;
86 }
87
88 /* The listing is fully buffered — the origin session is no longer
89 * needed for readdir/closedir (fetch-all-then-iterate, see above). */
90 brix_cache_origin_close(&oc);
91 free(t);
92
93 dir = calloc(1, sizeof(*dir));
94 if (dir == NULL) {
95 brix_cache_dirlist_free(&ds->dl);
96 free(ds);
97 if (err_out) { *err_out = ENOMEM; }
98 return NULL;
99 }
100 dir->inst = inst;
101 dir->state = ds;
102 return dir;
103}

121/* readdir — vtable readdir slot: yield the next buffered name, bound-copied
122 * into out->name (255 chars + NUL — brix_sd_dirent_t's fixed field). Returns
123 * NGX_OK (out filled), NGX_DONE (end of the buffered listing — matches the
124 * POSIX driver's readdir contract that brix_vfs_readdir relies on), or
125 * NGX_ERROR (malformed state; errno set). */
126ngx_int_t
127sd_xroot_readdir(brix_sd_dir_t *d, brix_sd_dirent_t *out)
128{
129 sd_xroot_dir_state *ds = d->state;
130
131 if (ds == NULL) {
132 errno = EINVAL;
133 return NGX_ERROR;
134 }
135 if (ds->next >= ds->dl.count) {
136 return NGX_DONE;
137 }
138 ngx_cpystrn((u_char *) out->name, (u_char *) ds->dl.names[ds->next],
139 sizeof(out->name));
140 ds->next++;
141 return NGX_OK;
142}

144/* closedir — vtable closedir slot: free the buffered name array, the dir
145 * state, and the brix_sd_dir_t handle itself. No origin session to close
146 * (opendir already released it). The VFS (brix_vfs_closedir, vfs_dir.c)
147 * calls driver->closedir(dh->sd_dir) but never frees dh->sd_dir itself, so
148 * — unlike sd_posix's pool-allocated dir handle, which the pool reclaims —
149 * this heap-allocated handle (calloc'd in sd_xroot_opendir_common) MUST be
150 * freed here or every dirlist leaks one brix_sd_dir_t. NULL-safe (both on
151 * `d` and on `d->state`) so a double-close is harmless. */
152ngx_int_t
153sd_xroot_closedir(brix_sd_dir_t *d)
154{
155 sd_xroot_dir_state *ds;
156
157 if (d == NULL) {
158 return NGX_OK;
159 }
160 ds = d->state;
161 if (ds != NULL) {
162 brix_cache_dirlist_free(&ds->dl);
163 free(ds);
164 d->state = NULL;
165 }
166 free(d);
167 return NGX_OK;
168}
