1/*
2 * privs.h — XrdAcc privilege + operation model (pure leaf, no nginx deps).
3 *
4 * WHAT: the faithful re-implementation of XRootD's XrdAccPrivs.hh privilege
5 * bitmask, the Access_Operation enum (AOP_*), the operation->privilege
6 * requirement table, the single-letter privilege parser, and the grant test.
7 *
8 * WHY: this is the heart of the `xrdacc` authorization engine — keeping the
9 * numeric privilege values byte-identical to XrdAccPrivs.hh means a stock
10 * XRootD `authdb` file yields exactly the same access decisions here. This
11 * header deliberately depends only on <stdint.h>/<stddef.h> (no nginx types)
12 * so the privilege algebra can be unit-tested as a standalone leaf.
13 *
14 * HOW: bits mirror XrdAccPrivs.hh exactly; brix_acc_parse_privs() ports
15 * XrdAccConfig::PrivsConvert() (positive privileges, then optional '-' then
16 * negatives); brix_acc_op_needs() + brix_acc_test() port
17 * XrdAccAccess::Test() (grant iff every required bit is present).
18 */
19

36/*
37 * Privilege bits — numerically identical to enum XrdAccPrivs (XrdAccPrivs.hh).
38 * Single primitives plus the composite operations XRootD pre-computes.
39 */
40#define BRIX_ACC_PRIV_NONE 0x000u
41#define BRIX_ACC_PRIV_DELETE 0x001u /* 'd' */
42#define BRIX_ACC_PRIV_INSERT 0x002u /* 'i' (also mkdir, create target) */
43#define BRIX_ACC_PRIV_LOCK 0x004u /* 'k' (historically unused) */
44#define BRIX_ACC_PRIV_LOOKUP 0x008u /* 'l' (stat / traversal) */
45#define BRIX_ACC_PRIV_RENAME 0x010u /* 'n' (mv source) */
46#define BRIX_ACC_PRIV_READ 0x020u /* 'r' (open r/o, readdir) */
47#define BRIX_ACC_PRIV_WRITE 0x040u /* 'w' (open w/append) */
48#define BRIX_ACC_PRIV_POLL 0x100u /* stage polling */
49#define BRIX_ACC_PRIV_ALL 0x1ffu /* 'a' */

51/* Composite operations (XrdAccPrivs.hh): */
52#define BRIX_ACC_PRIV_UPDATE (BRIX_ACC_PRIV_READ | BRIX_ACC_PRIV_WRITE) /* 0x060 */
53#define BRIX_ACC_PRIV_CREATE (BRIX_ACC_PRIV_INSERT | BRIX_ACC_PRIV_UPDATE) /* 0x062 */
54#define BRIX_ACC_PRIV_CHMOD (BRIX_ACC_PRIV_INSERT | BRIX_ACC_PRIV_UPDATE \
55 | BRIX_ACC_PRIV_DELETE) /* 0x063 */

94/*
95 * brix_acc_op_needs() — the privilege bits an operation requires.
96 *
97 * Ports XrdAccAccess::Test()'s need[] table. AOP_EXCL_* return 0xffff (a value
98 * no real privilege set can satisfy) because exclusivity is enforced outside the
99 * privilege model. AOP_ANY returns 0 (no requirement). Out-of-range ops return
100 * 0xffff (deny).
101 */
102brix_acc_privs_t brix_acc_op_needs(brix_acc_op_t op);

104/*
105 * brix_acc_test() — does `priv` satisfy `op`? Returns 1 (granted) iff every
106 * required bit is present: (need & priv) == need. Ports XrdAccAccess::Test().
107 */
108int brix_acc_test(brix_acc_privs_t priv, brix_acc_op_t op);

110/*
111 * brix_acc_parse_privs() — parse a privilege letter string [s, s+len) into
112 * positive/negative capabilities. Ports XrdAccConfig::PrivsConvert():
113 * letters a/d/i/k/l/n/r/w accumulate into the positive set, a single '-'
114 * switches the remainder into the negative set; a second '-' or any unknown
115 * character is an error. Returns 0 on success, -1 on a malformed string.
116 * Note (vs the native engine): 'r' does NOT imply 'l'.
117 */
118int brix_acc_parse_privs(const char *s, size_t len, brix_acc_priv_caps_t *caps);
