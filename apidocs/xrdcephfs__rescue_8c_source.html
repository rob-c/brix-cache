1/*
2 * xrdcephfs_rescue.c — offline CephFS recovery tool (operator utility).
3 *
4 * WHAT: Reads a CephFS directly from its RADOS pools — no kernel mount, no MDS,
5 * no libcephfs — to list, stat, cat, and recursively copy data OUT when the
6 * filesystem cannot be mounted but the pools are intact. It is the
7 * command-line face of the read-only `cephfsro` Storage Driver: it builds a
8 * driver instance and drives the same vtable the nginx export uses, so the
9 * decode/read path is shared and identically tested.
10 *
11 * SAFETY: like the driver, this assumes a QUIESCED filesystem (MDS down / fs
12 * failed, journal flushed). It only ever reads.
13 *
14 * USAGE:
15 * xrdcephfs_rescue <meta_pool> <data_pool> ls <path>
16 * xrdcephfs_rescue <meta_pool> <data_pool> stat <path>
17 * xrdcephfs_rescue <meta_pool> <data_pool> cat <path>
18 * xrdcephfs_rescue <meta_pool> <data_pool> get <path> <local_file>
19 * xrdcephfs_rescue <meta_pool> <data_pool> cp -r <path> <local_dir>
20 * (env CEPH_CONF overrides /etc/ceph/ceph.conf)
21 *
22 * BUILD: `make -C client ceph-tools` (dep-gated), or by hand where
23 * librados-devel exists:
24 * gcc -DXRDPROTO_NO_NGX -DBRIX_HAVE_CEPH -I src/fs/backend -I src/fs/backend/rados \
25 * -include client/apps/ceph/ngx_shim.h client/apps/ceph/xrdcephfs_rescue.c \
26 * src/fs/backend/rados/sd_cephfs_ro.c src/fs/backend/rados/sd_ceph.c \
27 * src/fs/backend/rados/sd_ceph_compat.c src/fs/backend/rados/cephfs_layout.c \
28 * src/fs/backend/rados/cephfs_denc.c -lrados -o xrdcephfs_rescue
29 */

46/* Stream an open object's bytes to a FILE* via the driver's pread. */
47static int
48copy_to_stream(brix_sd_obj_t *o, FILE *out)
49{
50 char buf[1u << 20];
51 off_t off = 0;
52 ssize_t n;
53
54 while ((n = DRV->pread(o, buf, sizeof(buf), off)) > 0) {
55 if (fwrite(buf, 1, (size_t) n, out) != (size_t) n) { return -1; }
56 off += n;
57 }
58 return (n < 0) ? -1 : 0;
59}

61static int
62do_cat_or_get(const char *path, const char *local)
63{
64 brix_sd_obj_t *o;
65 FILE *out;
66 int err = 0, rc;
67
68 o = DRV->open(&INST, path, BRIX_SD_O_READ, 0, &err);
69 if (o == NULL) { fprintf(stderr, "open %s: %s\n", path, strerror(err)); return 1; }
70
71 out = (local != NULL) ? fopen(local, "wb") : stdout;
72 if (out == NULL) { fprintf(stderr, "create %s: %s\n", local, strerror(errno));
73 DRV->close(o); return 1; }
74
75 rc = copy_to_stream(o, out);
76 if (local != NULL) { fclose(out); }
77 DRV->close(o);
78 if (rc != 0) { fprintf(stderr, "read %s failed\n", path); return 1; }
79 return 0;
80}

104 fprintf(stderr, "stat %s: %s\n", path, strerror(errno));

116/* Recursively copy a CephFS subtree to a local directory. Directories are
117 * recreated; regular files are streamed out; other types (symlinks, etc.) are
118 * reported and skipped. Returns the number of failures. */
119static int
120copy_tree(const char *src, const char *dst)
121{
122 brix_sd_stat_t sb;
123 brix_sd_dir_t *d;
124 brix_sd_dirent_t de;
125 int err = 0, fails = 0;
126
127 if (DRV->stat(&INST, src, &sb) != NGX_OK) {
128 fprintf(stderr, "stat %s: %s\n", src, strerror(errno));
129 return 1;
130 }
131 if (sb.is_reg) {
132 if (do_cat_or_get(src, dst) != 0) { return 1; }
133 printf(" %s -> %s (%lld bytes)\n", src, dst, (long long) sb.size);
134 return 0;
135 }
136 if (!sb.is_dir) {
137 fprintf(stderr, " skip (not a file or dir): %s\n", src);
138 return 0;
139 }
140
141 if (mkdir(dst, 0755) != 0 && errno != EEXIST) {
142 fprintf(stderr, "mkdir %s: %s\n", dst, strerror(errno));
143 return 1;
144 }
145 d = DRV->opendir(&INST, src, &err);
146 if (d == NULL) { fprintf(stderr, "opendir %s: %s\n", src, strerror(err)); return 1; }
147 while (DRV->readdir(d, &de) == NGX_OK) {
148 char csrc[4096], cdst[4096];
149 snprintf(csrc, sizeof(csrc), "%s/%s", src, de.name);
150 snprintf(cdst, sizeof(cdst), "%s/%s", dst, de.name);
151 fails += copy_tree(csrc, cdst);
152 }
153 DRV->closedir(d);
154 return fails;
155}

158main(int argc, char **argv)
159{
160 brix_sd_cephfs_ro_conf_t conf;
161 const char *cmd, *path;
162 int rc;
163
164 if (argc < 5) {
165 fprintf(stderr,
166 "usage: %s <meta_pool> <data_pool> <ls|stat|cat|get|cp> <path> [args]\n",
167 argv[0]);
168 return 2;
169 }
170
171 DRV = &brix_sd_cephfs_ro_driver;
172 memset(&INST, 0, sizeof(INST));
173 INST.driver = DRV;
174
175 memset(&conf, 0, sizeof(conf));
176 conf.meta_pool = argv[1];
177 conf.data_pool = argv[2];
178 conf.conf_file = getenv("CEPH_CONF") ? getenv("CEPH_CONF")
179 : "/etc/ceph/ceph.conf";
180 conf.assume_quiesced = 1; /* the operator runs this against a quiesced fs */
181
182 if (DRV->init(&INST, &conf) != NGX_OK) {
183 fprintf(stderr, "connect (meta=%s data=%s): %s\n",
184 argv[1], argv[2], strerror(errno));
185 return 1;
186 }
187
188 cmd = argv[3];
189 if (strcmp(cmd, "ls") == 0) {
190 rc = do_ls(argv[4]);
191 } else if (strcmp(cmd, "stat") == 0) {
192 rc = do_stat(argv[4]);
193 } else if (strcmp(cmd, "cat") == 0) {
194 rc = do_cat_or_get(argv[4], NULL);
195 } else if (strcmp(cmd, "get") == 0) {
196 if (argc < 6) { fprintf(stderr, "get needs <path> <local_file>\n"); rc = 2; }
197 else { rc = do_cat_or_get(argv[4], argv[5]); }
198 } else if (strcmp(cmd, "cp") == 0) {
199 /* "cp -r <path> <local_dir>" */
200 if (argc < 7 || strcmp(argv[4], "-r") != 0) {
201 fprintf(stderr, "cp needs: cp -r <path> <local_dir>\n"); rc = 2;
202 } else {
203 path = argv[5];
204 rc = copy_tree(path, argv[6]);
205 if (rc == 0) { printf("recovered %s -> %s\n", path, argv[6]); }
206 }
207 } else {
208 fprintf(stderr, "unknown command '%s'\n", cmd);
209 rc = 2;
210 }
211
212 DRV->cleanup(&INST);
213 return rc;
214}
