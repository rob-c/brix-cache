21/*
22 * read_clamped_total — bytes this memory-path read will actually deliver.
23 *
24 * WHAT: the request length clamped to what the file holds, for read-only
25 * handles whose size was cached at open time.
26 * WHY: Phase 31 W2.1 bounds resident heap for large memory-backed reads; the
27 * windowed-vs-single-shot decision must be made on real bytes, not the
28 * client's (possibly EOF-crossing) rlen.
29 * HOW: read-only handles clamp against cached_size; writable handles (size
30 * unknown) use rlen and let a short read at EOF terminate early.
31 */
32size_t
33read_clamped_total(brix_ctx_t *ctx, const brix_read_io_t *io)
34{
35 size_t total = io->rlen;
36
37 if (!ctx->files[io->idx].writable && ctx->files[io->idx].cached_size > 0) {
38 off_t avail = ctx->files[io->idx].cached_size - (off_t) io->offset;
39 total = (avail <= 0) ? 0
40 : ((off_t) total > avail ? (size_t) avail : total);
41 }
42
43 return total;
44}

46/*
47 * read_serve_windowed — stream a large memory-path read as kXR_oksofar chunks.
48 *
49 * WHAT: admits one window's worth of budget, arms the windowed-read state
50 * machine in ctx->rd and kicks the pump.
51 * WHY: a request bigger than one streaming window must not buffer whole in
52 * heap — serve it as a sequence of window-sized kXR_oksofar chunks ending in
53 * kXR_ok, holding only ~one window in read_scratch at a time.
54 * HOW: budget-rejected requests get kXR_wait; otherwise the pump (and any
55 * resumption after a partial flush) reads from rd_win_* rather than this
56 * request's locals, so it survives across event-loop returns.
57 */
58ngx_int_t
59read_serve_windowed(brix_ctx_t *ctx, ngx_connection_t *c,
60 ngx_stream_brix_srv_conf_t *rconf, const brix_read_io_t *io,
61 size_t total)
62{
63 /* Admit one window's worth — a windowed stream holds ~2 MiB, not
64 * the full request, so many more fit under the budget. */
65 if (!brix_budget_admit(ctx, rconf->memory_budget,
66 (size_t) BRIX_READ_WINDOW)) {
67 return brix_send_wait(ctx, c, 1);
68 }
69
70 /*
71 * Arm the windowed-read state machine: the pump below (and any
72 * resumption after a partial flush) reads from rd_win_* rather than
73 * this request's locals, so it survives across event-loop returns.
74 * cur_streamid is snapshotted into rd_win_streamid because each
75 * kXR_oksofar/kXR_ok chunk must echo the originating request's stream
76 * id, but cur_streamid will be overwritten by the next inbound header
77 * before this stream finishes draining.
78 */
79 ctx->rd.win_active = 1;
80 ctx->rd.win_fd = io->fd;
81 ctx->rd.win_idx = io->idx;
82 ctx->rd.win_offset = (off_t) io->offset;
83 ctx->rd.win_remaining = total;
84 ctx->rd.win_streamid[0] = ctx->recv.cur_streamid[0];
85 ctx->rd.win_streamid[1] = ctx->recv.cur_streamid[1];
86
87 brix_prefetch_read_file(c->log, &ctx->files[io->idx], (off_t) io->offset,
88 total,
89 ctx->files[io->idx].writable
90 ? 0 : ctx->files[io->idx].cached_size);
91
92 if (rconf->access_log_fd != NGX_INVALID_FILE) {
93 char read_detail[64];
94 snprintf(read_detail, sizeof(read_detail), "%lld+%zu",
95 (long long) io->offset, io->rlen);
96 brix_log_access(ctx, c, "READ", ctx->files[io->idx].path,
97 read_detail, 1, 0, NULL, total);
98 }
99
100 brix_read_window_pump(ctx, c, rconf);
101 return NGX_OK;
102}

104/*
105 * read_prefetch_buffered — readahead hint for the single-shot memory path.
106 *
107 * WHAT: issues the prefetch hint for a regular-file buffered read, clamped to
108 * the cached file size when known.
109 * WHY: the buffered path bypasses the sendfile branch's clamp, so the hint
110 * length must be recomputed here or readahead would be asked for bytes past
111 * EOF.
112 * HOW: read-only handles use the size cached at open; writable handles pass 0
113 * (unknown) and hint the full rlen.
114 */
115static void
116read_prefetch_buffered(brix_ctx_t *ctx, ngx_connection_t *c,
117 const brix_read_io_t *io)
118{
119 off_t file_size;
120 size_t hint_len;
121
122 if (!ctx->files[io->idx].is_regular) {
123 return;
124 }
125
126 file_size = ctx->files[io->idx].writable ? 0
127 : ctx->files[io->idx].cached_size;
128 hint_len = io->rlen;
129
130 if (file_size > 0) {
131 if ((off_t) io->offset >= file_size) {
132 hint_len = 0;
133 } else if ((off_t) hint_len > file_size - (off_t) io->offset) {
134 hint_len = (size_t) (file_size - (off_t) io->offset);
135 }
136 }
137
138 brix_prefetch_read_file(c->log, &ctx->files[io->idx], (off_t) io->offset,
139 hint_len, file_size);
140}

142/*
143 * read_try_warm — Phase 32 WS4 warm-cache fast path.
144 *
145 * WHAT: probes the page cache with a non-blocking preadv2(RWF_NOWAIT); on a
146 * full hit, verifies CSI page checksums and attributes backend byte metrics.
147 * WHY: if the whole request is resident it returns rlen bytes immediately and
148 * completes inline — skipping the thread-pool round-trip (hundreds of µs)
149 * that otherwise dominates a cache-hot read.
150 * HOW: returns 1 on a full hit with *nread_out set (or -1 with errno=EIO on a
151 * CSI mismatch); returns 0 on a (partial) miss so the caller falls through to
152 * the AIO thread / synchronous pread, which reads the full data blocking off
153 * the event loop. Only attempted for regular files (RWF_NOWAIT is meaningful
154 * against the page cache) with a thread pool configured, matching the paths
155 * the probe would otherwise short-circuit.
156 */
157static ngx_flag_t
158read_try_warm(brix_ctx_t *ctx, ngx_stream_brix_srv_conf_t *rconf,
159 const brix_read_io_t *io, ssize_t *nread_out)
160{
161 ssize_t warm = -1;
162 ssize_t nread;
163
164#if defined(RWF_NOWAIT)
165 if (rconf->common.thread_pool != NULL && ctx->files[io->idx].is_regular) {
166 struct iovec iov;
167 brix_sd_obj_t obj;
168 iov.iov_base = io->databuf;
169 iov.iov_len = io->rlen;
170 brix_sd_posix_wrap(&obj, io->fd); /* phase-55: SD seam */
171 warm = obj.driver->preadv2(&obj, &iov, 1,
172 (off_t) io->offset, RWF_NOWAIT);
173 }
174#endif
175
176 /*
177 * Only an exact rlen match counts as a hit: a short warm result means
178 * part of the range was not resident (or EOF), and re-issuing a blocking
179 * read for the missing tail from the event loop would stall it — so any
180 * non-exact result falls through to the thread pool / sync path, which
181 * re-reads the full range from offset (the partial warm bytes are simply
182 * overwritten).
183 */
184 if (warm != (ssize_t) io->rlen) {
185 return 0;
186 }
187
188 nread = warm; /* full page-cache hit — databuf is filled; complete inline */
189
190 /* phase-59 W2: the warm fast path bypasses the VFS job, so verify
191 * the page CRCs here too; a mismatch fails the read (EIO). */
192 if (ctx->files[io->idx].csi != NULL && nread > 0
193 && brix_csi_verify_read(
194 (brix_csi_t *) ctx->files[io->idx].csi, io->databuf,
195 (off_t) io->offset, (size_t) nread) == BRIX_CSI_MISMATCH)
196 {
197 nread = -1;
198 errno = EIO;
199 }
200
201 /* The warm fast path bypasses brix_vfs_io_execute (where the other
202 * read paths attribute), so charge the per-backend read total here. */
203 if (nread > 0) {
204 brix_metric_backend_bytes(
205 ctx->files[io->idx].sd_obj.driver != NULL
206 ? ctx->files[io->idx].sd_obj.driver->name : "posix",
207 BRIX_METRIC_OP_READ, (size_t) nread);
208 }
209
210 *nread_out = nread;
211 return 1;
212}

214/*
215 * read_post_aio — post the buffered read to the thread pool.
216 *
217 * WHAT: allocates (once per session) or resets the reusable read AIO task,
218 * fills the job fields and posts it to the configured thread pool.
219 * WHY: the blocking pread must run off the event loop; the done-callback owns
220 * databuf and finishes the response, so a successful post is the end of this
221 * request's event-loop work.
222 * HOW: one reusable task per session (ctx->rd.read_aio_task): allocate it the
223 * first time, otherwise reset the two fields ngx reuse requires — unlink from
224 * any prior queue (next) and clear the completion flag so the event loop will
225 * fire the done-callback again. Returns NGX_ERROR on allocation failure
226 * (databuf already released); otherwise NGX_OK with *posted saying whether the
227 * pool accepted the task (not posted => caller must read synchronously so the
228 * read never silently drops).
229 */
230static ngx_int_t
231read_post_aio(brix_ctx_t *ctx, ngx_connection_t *c,
232 ngx_stream_brix_srv_conf_t *rconf, const brix_read_io_t *io,
233 ngx_flag_t *posted)
234{
235 ngx_thread_task_t *task;
236 brix_read_aio_t *t;
237
238 task = ctx->rd.read_aio_task;
239 if (task == NULL) {
240 task = ngx_thread_task_alloc(c->pool, sizeof(brix_read_aio_t));
241 if (task == NULL) {
242 brix_release_read_buffer(ctx, c, io->databuf);
243 return NGX_ERROR;
244 }
245 ctx->rd.read_aio_task = task;
246 } else {
247 task->next = NULL;
248 task->event.complete = 0;
249 }
250
251 t = task->ctx;
252 t->c = c;
253 t->ctx = ctx;
254 t->fd = io->fd;
255 t->handle_idx = io->idx;
256 t->offset = (off_t) io->offset;
257 t->rlen = io->rlen;
258 t->databuf = io->databuf;
259 t->streamid[0] = ctx->recv.cur_streamid[0];
260 t->streamid[1] = ctx->recv.cur_streamid[1];
261 t->nread = 0;
262 t->io_errno = 0;
263 t->csi = ctx->files[io->idx].csi; /* phase-59 W2: verify on read */
264 t->obj = ctx->files[io->idx].sd_obj; /* Layer 3: driver obj (or zeroed) */
265
266 brix_task_bind(task, brix_read_aio_thread, brix_read_aio_done);
267
268 (void) brix_aio_post_task(ctx, c, rconf->common.thread_pool, task,
269 "brix: thread_task_post failed, sync read fallback",
270 posted);
271 return NGX_OK;
272}

274/*
275 * read_sync_fill — blocking buffered read via the VFS I/O seam.
276 *
277 * WHAT: fills io->databuf with up to rlen bytes from offset through
278 * brix_vfs_io_execute, with CSI verification attached.
279 * WHY: the fallback when no thread pool is configured or the pool rejected the
280 * post — the read runs inline on the event loop rather than dropping.
281 * HOW: returns the byte count (or -1 with errno set). A CSI page-checksum
282 * mismatch surfaces here as EIO (job.csi_mismatch set); the caller's nread<0
283 * path fails the read so corrupt data is never served.
284 */
285static ssize_t
286read_sync_fill(brix_ctx_t *ctx, const brix_read_io_t *io)
287{
288 brix_vfs_job_t job;
289
290 brix_vfs_job_read_init(&job, io->fd, (off_t) io->offset, io->rlen,
291 io->databuf, io->rlen, 0);
292 job.csi = ctx->files[io->idx].csi; /* phase-59 W2: verify on read */
293 brix_vfs_job_set_obj(&job, &ctx->files[io->idx].sd_obj);
294 brix_vfs_io_execute(&job);
295 if (job.io_errno != 0) {
296 errno = job.io_errno;
297 }
298 return job.nio;
299}

301/*
302 * read_finish_buffered — account, log and queue a completed memory-path read.
303 *
304 * WHAT: turns the (nread, databuf) result of the warm/sync buffered paths into
305 * the wire response: error on nread<0, otherwise byte accounting, dashboard
306 * slot update, access log, chunked chain build and queue.
307 * WHY: the warm-hit and sync-fallback paths converge here so the response
308 * assembly (and its ordering) exists exactly once — INVARIANT: this path
309 * serves memory-backed buffers (TLS-safe), never sendfile.
310 * HOW: on queue park (still SENDING) marks the response pipelinable —
311 * per-in-flight buffer + per-slot header make this memory-backed read safe to
312 * pipeline; on any other outcome the buffer is released back to the pool.
313 */
314static ngx_int_t
315read_finish_buffered(brix_ctx_t *ctx, ngx_connection_t *c,
316 ngx_stream_brix_srv_conf_t *rconf, const brix_read_io_t *io,
317 ssize_t nread)
318{
319 size_t data_total;
320 ngx_chain_t *rsp_chain;
321 u_char *databuf = io->databuf;
322 int idx = io->idx;
323
324 if (nread < 0) {
325 brix_read_io_failure_log(c->log, "buffered", io->fd,
326 (off_t) io->offset, io->rlen, errno);
327 brix_release_read_buffer(ctx, c, databuf);
328 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
329 ctx->files[idx].path, "-",
330 kXR_IOError, strerror(errno));
331 }
332
333 data_total = (size_t) nread;
334
335 ctx->files[idx].bytes_read += data_total;
336 ctx->totals.bytes += data_total;
337
338 if (ctx->files[idx].dashboard_slot >= 0 &&
339 ngx_brix_dashboard_shm_zone != NULL)
340 {
341 brix_transfer_slot_update(ngx_brix_dashboard_shm_zone->data,
342 ctx->files[idx].dashboard_slot,
343 (ngx_atomic_int_t) data_total,
344 (int64_t) ngx_current_msec);
345 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
346 ctx->files[idx].dashboard_slot, "read");
347 }
348
349 if (rconf->access_log_fd != NGX_INVALID_FILE) {
350 char read_detail[64];
351
352 snprintf(read_detail, sizeof(read_detail), "%lld+%zu",
353 (long long) io->offset, io->rlen);
354 brix_log_access(ctx, c, "READ", ctx->files[idx].path,
355 read_detail, 1, 0, NULL, data_total);
356 }
357 BRIX_OP_OK(ctx, BRIX_OP_READ);
358
359 rsp_chain = brix_build_chunked_chain(ctx, c, databuf, data_total);
360 if (rsp_chain == NULL) {
361 brix_release_read_buffer(ctx, c, databuf);
362 return NGX_ERROR;
363 }
364
365 {
366 ngx_int_t rc = brix_queue_response_chain(ctx, c, rsp_chain, databuf);
367
368 if (rc != NGX_OK || ctx->state != XRD_ST_SENDING) {
369 brix_release_read_buffer(ctx, c, databuf);
370 } else {
371 /*
372 * Parked and draining: per-in-flight buffer + per-slot header make
373 * this memory-backed (TLS) read safe to pipeline, so let the recv loop
374 * queue the next read behind it instead of idling while it drains a
375 * jittered socket. (A single-chunk response only: the non-windowed
376 * path is bounded by BRIX_READ_WINDOW < BRIX_READ_CHUNK_MAX.)
377 */
378 ctx->out.resp_pipelinable = 1;
379 }
380 return rc;
381 }
382}

384/*
385 * read_serve_buffered — single-shot memory-path read (<= one window).
386 *
387 * WHAT: admits the request against the memory budget, acquires a
388 * per-in-flight buffer and fills it via the warm-cache probe, the AIO thread
389 * pool, or a synchronous VFS read — then completes through
390 * read_finish_buffered().
391 * WHY: this is the memory path (TLS / non-regular file / CSI / driver-backed
392 * handle) — INVARIANT: it serves memory-backed buffers only, never sendfile.
393 * HOW: budget-rejected requests get kXR_wait. Warm hit completes inline; a
394 * miss posts to the thread pool when configured (a successful post returns
395 * early — the done-callback owns databuf); a rejected post or no pool falls
396 * back to read_sync_fill() so the read never silently drops.
397 */
398ngx_int_t
399read_serve_buffered(brix_ctx_t *ctx, ngx_connection_t *c,
400 ngx_stream_brix_srv_conf_t *rconf, brix_read_io_t *io)
401{
402 ssize_t nread;
403
404 if (!brix_budget_admit(ctx, rconf->memory_budget, io->rlen)) {
405 return brix_send_wait(ctx, c, 1);
406 }
407
408 /*
409 * Per-in-flight read buffer (read pipelining): each outstanding memory read
410 * gets its OWN buffer from rd_pool rather than the single shared read_scratch,
411 * so this response can keep draining the (possibly jittered) socket while the
412 * recv loop already issues the next read into a different buffer. Released
413 * back to the pool when this response's out_ring slot drains.
414 */
415 io->databuf = brix_acquire_read_buffer(ctx, c, io->rlen);
416 if (io->databuf == NULL) {
417 return NGX_ERROR;
418 }
419
420 /* Charge the (possibly grown) read-pool footprint to the budget now so a
421 * concurrent connection's admission check sees this allocation promptly. */
422 brix_budget_sync(ctx);
423
424 read_prefetch_buffered(ctx, c, io);
425
426 if (!read_try_warm(ctx, rconf, io, &nread)) {
427 if (rconf->common.thread_pool != NULL) {
428 ngx_flag_t posted;
429
430 if (read_post_aio(ctx, c, rconf, io, &posted) != NGX_OK) {
431 return NGX_ERROR;
432 }
433 /*
434 * Posted: the read now completes off-thread; the done-callback owns
435 * databuf and finishes the response, so return early — nothing more to
436 * do on the event loop. Not posted (queue full / post error): fall
437 * through to a blocking pread here so the read never silently drops.
438 */
439 if (posted) {
440 return NGX_OK;
441 }
442 }
443
444 /* No thread pool configured (or post rejected): read inline on the
445 * event loop. */
446 nread = read_sync_fill(ctx, io);
447 }
448
449 return read_finish_buffered(ctx, c, rconf, io, nread);
450}
