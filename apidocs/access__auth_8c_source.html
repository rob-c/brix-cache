2 * access_auth.c - WebDAV access-phase authentication gate.
3 *
4 * WHAT: The credential-source tier of the WebDAV access phase: GSI proxy
5 * cert, bearer token, and Basic password sources, the RFC 7617 Basic
6 * challenge for browser clients, and the policy gate (access_authenticate)
7 * that runs them in order and applies auth=required/optional/none.
8 *
9 * WHY: Split out of access.c so the authentication surface is grouped by
10 * concern and individually reviewable while access.c keeps the access-phase

12 *
13 * HOW: access_authenticate is the sole cross-split entry point (declared in
14 * access_internal.h); the per-source wrappers and the Basic challenge stay
15 * file-local statics. Each outcome increments exactly the same metric slot
16 * as before the decomposition.
17 */

28/*
29 * access_try_cert — GSI proxy-certificate credential source.
30 *
31 * WHAT: Attempts client-certificate authentication and records the cert-OK
32 * metrics on success.
33 *
34 * WHY: The cert tier is tried FIRST (before bearer token), matching the
35 * root:// auth ordering; metrics must attribute the session to GSI.
36 *
37 * HOW: Thin wrapper over webdav_verify_proxy_cert; returns its verdict.
38 */
39static ngx_int_t
40access_try_cert(ngx_http_request_t *r,
41 ngx_http_brix_webdav_loc_conf_t *conf)
42{
43 ngx_int_t rc = webdav_verify_proxy_cert(r, conf);
44
45 if (rc == NGX_OK) {
46 BRIX_WEBDAV_METRIC_INC(
47 auth_total[BRIX_WEBDAV_AUTH_RESULT_CERT_OK]);
48 brix_metric_auth(BRIX_PROTO_WEBDAV, BRIX_AUTHN_GSI, 1);
49 }
50 return rc;
51}

103/*
104 * access_basic_challenge — RFC 7617 Basic challenge for browser clients.
105 *
106 * WHAT: Attaches `WWW-Authenticate: Basic realm="brix"` to the response and
107 * returns NGX_HTTP_UNAUTHORIZED, or NGX_HTTP_INTERNAL_SERVER_ERROR on
108 * allocation failure.
109 *
110 * WHY: A browser only shows its login prompt (and re-prompts after a wrong
111 * password) on 401 + a challenge header; a bare 403 is a dead end. Emitted
112 * only when Basic is actually enabled on the export, so cert/token-only
113 * exports keep their historical 403 and never invite a password prompt that
114 * cannot succeed.
115 *
116 * HOW: The same headers_out wiring nginx's own auth_basic module uses — push
117 * the header and point headers_out.www_authenticate at it so the special-
118 * response path emits it.
119 */
120static ngx_int_t
121access_basic_challenge(ngx_http_request_t *r)
122{
123 ngx_table_elt_t *h = ngx_list_push(&r->headers_out.headers);
124
125 if (h == NULL) {
126 return NGX_HTTP_INTERNAL_SERVER_ERROR;
127 }
128 h->hash = 1;
129#if (nginx_version >= 1023000)
130 h->next = NULL;
131#endif
132 ngx_str_set(&h->key, "WWW-Authenticate");
133 ngx_str_set(&h->value, "Basic realm=\"brix\"");
134 r->headers_out.www_authenticate = h;
135 return NGX_HTTP_UNAUTHORIZED;
136}

138/*
139 * access_authenticate — the authentication gate.
140 *
141 * WHAT: Runs the credential sources in order (GSI proxy cert, bearer token,
142 * then Basic password) and applies the location's auth policy to the outcome.
143 *
144 * WHY: auth=required rejects unauthenticated requests — with a 401 Basic
145 * challenge when a pwd db is configured (so browsers prompt for credentials),
146 * else the historical 403; auth=optional lets them proceed as anonymous;
147 * auth=none skips verification entirely. Each outcome increments exactly
148 * the same metric slot as before the decomposition.
149 *
150 * HOW: Returns NGX_OK to continue (authenticated or anonymous) or the
151 * metrics-counted rejection.
152 */
153ngx_int_t
154access_authenticate(ngx_http_request_t *r,
155 ngx_http_brix_webdav_loc_conf_t *conf)
156{
157 ngx_int_t auth_rc;
158
159 if (conf->auth == WEBDAV_AUTH_NONE) {
160 BRIX_WEBDAV_METRIC_INC(
161 auth_total[BRIX_WEBDAV_AUTH_RESULT_NONE]);
162 brix_metric_auth(BRIX_PROTO_WEBDAV, BRIX_AUTHN_NONE, 1);
163 return NGX_OK;
164 }
165
166 auth_rc = access_try_cert(r, conf);
167 if (auth_rc != NGX_OK) {
168 auth_rc = access_try_token(r, conf);
169 }
170 if (auth_rc != NGX_OK) {
171 auth_rc = access_try_basic(r, conf);
172 }
173
174 if (auth_rc != NGX_OK && conf->auth == WEBDAV_AUTH_REQUIRED) {
175 BRIX_WEBDAV_METRIC_INC(
176 auth_total[BRIX_WEBDAV_AUTH_RESULT_REJECTED]);
177 brix_metric_auth(BRIX_PROTO_WEBDAV, BRIX_AUTHN_NONE, 0);
178 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
179 "brix_webdav: unauthenticated request rejected"
180 " (auth=required)");
181 if (conf->pwd_file.len > 0) {
182 return webdav_metrics_return(r, access_basic_challenge(r));
183 }
184 return webdav_metrics_return(r, NGX_HTTP_FORBIDDEN);
185 }
186
187 if (auth_rc != NGX_OK) {
188 BRIX_WEBDAV_METRIC_INC(
189 auth_total[BRIX_WEBDAV_AUTH_RESULT_ANONYMOUS]);
190 brix_metric_auth(BRIX_PROTO_WEBDAV, BRIX_AUTHN_NONE, 1);
191 }
192
193 return NGX_OK;
194}
