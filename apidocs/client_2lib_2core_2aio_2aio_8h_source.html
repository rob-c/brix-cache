1/*
2 * aio.h — asynchronous, multi-in-flight transport for libbrix (M1).
3 *
4 * WHAT: A single epoll event loop, on its own thread, that drives one or more
5 * already-brought-up XRootD connections (brix_conn) in NON-BLOCKING mode,
6 * pipelining many requests per connection and demultiplexing replies by
7 * streamid. Requests complete via a callback on the loop thread; a blocking
8 * convenience wrapper (brix_aio_call) lets the FUSE driver's worker threads
9 * issue ordinary synchronous calls that nonetheless pipeline over a shared
10 * socket.
11 * WHY: The synchronous core (sock.c/frame.c) is one-request-in-flight and a flat
12 * 30 s timeout — on a high-RTT / lossy link that serialises throughput and
13 * turns any transient drop into a stalled, then EIO'd, operation. The async
14 * core hides RTT (pipelining), bounds every request with its own deadline,
15 * and is the substrate the resilience + resumption layers (M2/M3) build on.
16 * HOW: Bring a connection up synchronously the usual way (brix_connect: connect →
17 * [TLS] → login → auth), then hand it to the loop with brix_aconn_attach;
18 * the loop sets the fd non-blocking and owns all subsequent socket I/O. The
19 * per-connection state (out queue, in-flight map, parse buffer) is touched
20 * ONLY by the loop thread; other threads interact strictly through a
21 * mutex-guarded command queue woken by an eventfd. So the data plane needs
22 * no per-field locking.
23 *
24 * Clean-room: epoll/eventfd/pthreads + the existing wire framing only. No XrdCl.
25 *
26 * Limitation (M1): a connection with GSI request-signing active (conn->signing_
27 * active) cannot go async yet — each request would need an inline kXR_sigver
28 * round-trip. brix_aconn_attach rejects such a conn so the caller can fall back to
29 * the synchronous pool. Anonymous / token (ztn) / unix sessions pipeline fine.
30 */

40/*
41 * Completion callback, invoked on the loop thread exactly once per request.
42 * rc == 0 → a terminal server frame arrived: `kxr` is the status (kXR_ok,
43 * and for the upper layers kXR_redirect / kXR_wait / kXR_error-as
44 * -rc<0). `body`/`blen` is the accumulated reply body (oksofar frames
45 * are concatenated); ownership passes to the callback, which must
46 * free(body).
47 * rc < 0 → the request failed at the transport/deadline level: `st` describes
48 * it (kxr is its kXR_* / XRDC_E* code), body is NULL.
49 * The callback must not block the loop (it runs inline); the sync wrapper below
50 * just signals a condvar.
51 */
52typedef void (*brix_aio_cb)(void *ctx, int rc, uint16_t kxr,
53 uint8_t *body, uint32_t blen, const brix_status *st);

55/* ---- loop lifecycle ---- */
56/* Create the loop and start its thread. NULL + st on failure. */
57brix_loop *brix_loop_create(brix_status *st);
58/* Stop the loop thread and free it. Any still-attached aconns are closed first

61void brix_loop_destroy(brix_loop *l);

63/* ---- connection lifecycle ---- */
64/* Hand a brought-up conn to the loop (fd set non-blocking, registered in epoll).
65 * The aconn borrows `conn` (host/port/opts/creds for reconnect, M2); the caller
66 * keeps ownership and must not touch conn->io while it is attached. Returns NULL +
67 * st on failure (including a signing-active conn — see the file header). */
68brix_aconn *brix_aconn_attach(brix_loop *l, brix_conn *conn, brix_status *st);
69/* Detach + tear down an aconn (synchronous: returns once the loop has failed all
70 * its in-flight requests and removed it). The underlying conn is left closed-by-
71 * caller — call brix_close on it afterwards if you own it. */
72void brix_aconn_close(brix_aconn *ac);

74/* ---- resilience tuning (M2) ---- */
75/* Override an aconn's resilience policy (else sensible defaults apply at attach):
76 * max_stall_ms — total patience across reconnect attempts before a parked
77 * request is failed (the "bad wifi" budget). 0 disables
78 * reconnect entirely (a drop fails in-flight immediately).
79 * keepalive_ms — idle time before a kXR_ping heartbeat probes a quiet link
80 * (proactive dead-peer detection). 0 disables the heartbeat.
81 * max_retries — default transport-level re-issues for a retry-safe request.
82 * Pass a negative value to leave that field at its current/default value. */
83void brix_aconn_set_resilience(brix_aconn *ac, int max_stall_ms,
84 int keepalive_ms, int max_retries);

86/* ---- request submission ---- */
87/* Per-request options for brix_aio_submit_ex. */
88typedef struct {
89 int deadline_ms; /* hard ceiling for this request; <= 0 ⇒ adaptive (RTT-based,
90 * bounded by the aconn's max_stall_ms) */
91 int max_retries; /* transport re-issues after a reconnect; < 0 ⇒ aconn default */
92 int retry_safe; /* 1 ⇒ idempotent + not handle-bound, so the engine may
93 * transparently re-issue it verbatim after a reconnect */
94} brix_aio_opts;

103/* Submit one request. `hdr24` is a 24-byte ClientRequestHdr with requestid + the
104 * 16-byte body already filled; the engine assigns the streamid and writes dlen.
105 * `payload` (plen bytes, may be NULL) is copied. `cb`/`ctx` receive the result.
106 * deadline_ms <= 0 ⇒ adaptive deadline. Not auto-retried (retry_safe=0); use
107 * brix_aio_submit_ex to opt an idempotent request into transparent reconnect
108 * retry. Thread-safe; returns 0 (queued) or -1 (st). */

113/* Blocking convenience: submit + wait for completion on the calling thread. Many
114 * callers may be in flight at once over a single aconn (that is the point). On
115 * success returns 0 with *kxr set and the *body / *blen out-params holding the
116 * malloc'd reply (caller frees; pass body=NULL to discard). -1 (st) on error. */

130/* ============================================================================
131 * Connection manager + resilient open files (M3)
132 *
133 * brix_mgr owns the loop and a small pool of attached connections; metadata calls
134 * round-robin across them (transparently surviving reconnects when retry_safe).
135 * brix_mfile is an open file that survives a connection drop: on a transport
136 * failure or a stale handle it reopens (fresh fhandle, NON-destructively — never
137 * re-truncating) and re-issues the read/write at the same absolute offset (which
138 * is idempotent), so a mid-transfer cat/dd survives a server restart with no EIO.
139 * ==========================================================================*/
140
141typedef struct brix_mgr brix_mgr;
142typedef struct brix_mfile brix_mfile;
143
144/* Create the loop + a pool of `nconns` connections to `u` (opts `o`, may be NULL),
145 * each with the given resilience policy. `eager` (clamped to [1, nconns]) connect
146 * up front — in parallel, so mount-time wall is ~1×RTT not eager×RTT; the rest
147 * connect lazily on first brix_mgr_pick. NULL + st on failure (an eager connect
148 * failed, so a bad endpoint / auth still fails fast). */
149brix_mgr *brix_mgr_create(const brix_url *u, const brix_opts *o, int nconns,
150 int eager, int max_stall_ms, int keepalive_ms,
151 int max_retries, brix_status *st);
152void brix_mgr_destroy(brix_mgr *m);
153/* Round-robin one of the pool's connections (for a new file or a metadata op). */
154brix_aconn *brix_mgr_pick(brix_mgr *m);
155/* Issue a metadata/path request over a pooled connection. retry_safe=1 for
156 * idempotent ops (stat/dirlist/query/...) so they survive a reconnect. */
157int brix_mgr_call(brix_mgr *m, const void *hdr24, const void *payload,
158 uint32_t plen, int retry_safe, uint16_t *kxr,
