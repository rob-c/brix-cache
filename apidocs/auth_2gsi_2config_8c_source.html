5/*
6 * Build (or rebuild) the X509_STORE used for GSI certificate verification.
7 *
8 * Loads the trusted CA from xcf->trusted_ca, then loads CRLs from xcf->crl.
9 * Sets X509_V_FLAG_ALLOW_PROXY_CERTS so GSI proxy certificate chains verify.
10 *
11 * On success the new store is atomically swapped into xcf->gsi_store and any
12 * previous store is freed. On failure the old store is left intact so
13 * existing connections are not disrupted.
14 */
15ngx_int_t
16brix_rebuild_gsi_store(ngx_stream_brix_srv_conf_t *xcf, ngx_log_t *log)
17{
18 X509_STORE *store;
19 X509_STORE *old_store;
20 int crl_count = 0;
21 struct stat ca_st;
22 int ca_is_dir;
23
24 /* brix_trusted_ca may name a single CA bundle file OR a hashed CA
25 * directory (e.g. /etc/grid-security/certificates). A directory is loaded
26 * as an OpenSSL CApath so on-demand hash lookup verifies real grid proxy
27 * chains (any CA under the dir), which a single-file bundle cannot cover. */
28 ca_is_dir = (stat((char *) xcf->trusted_ca.data, &ca_st) == 0
29 && S_ISDIR(ca_st.st_mode));
30
31 store = brix_build_ca_store(log,
32 ca_is_dir ? (char *) xcf->trusted_ca.data
33 : NULL,
34 ca_is_dir ? NULL
35 : (char *) xcf->trusted_ca.data,
36 xcf->crl.len > 0
37 ? (char *) xcf->crl.data : NULL,
38 X509_V_FLAG_ALLOW_PROXY_CERTS,
39 &crl_count,
40 (brix_sp_mode_t) xcf->signing_policy_mode,
41 (int) xcf->crl_mode);
42 if (store == NULL) {
43 return NGX_ERROR;
44 }
45
46 if (xcf->crl.len > 0) {
47 ngx_log_error(NGX_LOG_NOTICE, log, 0,
48 "brix: loaded %d CRL(s) from \"%s\"",
49 crl_count, xcf->crl.data);
50 }
51
52 /* Atomic swap */
53 old_store = xcf->gsi_store;
54 xcf->gsi_store = store;
55 if (old_store != NULL) {
56 X509_STORE_free(old_store);
57 }
58
59 return NGX_OK;
60}

62/*
63 * WHAT: Full GSI authentication configuration loader — validates auth method (GSI or BOTH), checks all three trust inputs (certificate, private key, trusted CA) are present and path-valid, loads server certificate via PEM_read_X509(), serializes it into cached PEM buffer for kXGS_cert responses on every GSI login, loads private key via PEM_read_PrivateKey(), builds X509_STORE with trusted CA + CRLs via brix_rebuild_gsi_store() (with proxy cert flag and CRL_CHECK_ALL), runs PKI/CRL consistency check, computes CA hash via X509_subject_name_hash() for kXRS_issuer_hash advertisement during GSI bootstrap.
64 * WHY: All three trust inputs must be present simultaneously — missing any one makes GSI auth meaningless. Cert serialization caching avoids per-request PEM encoding overhead on every client login. CRL checking enables revocation detection across the full certificate chain. CA hash allows clients to confirm which CA the server trusts before initiating DH exchange.
65 */
66ngx_int_t
67brix_configure_gsi(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
68{
69 if (xcf->auth != BRIX_AUTH_GSI && xcf->auth != BRIX_AUTH_BOTH) {
70 return NGX_OK;
71 }
72
73 /* GSI mode is only meaningful when all three trust inputs are present. */
74 if (xcf->certificate.len == 0 || xcf->certificate_key.len == 0
75 || xcf->trusted_ca.len == 0)
76 {
77 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
78 "brix_auth gsi requires brix_certificate, "
79 "brix_certificate_key and brix_trusted_ca");
80 return NGX_ERROR;
81 }
82
83 if (brix_validate_path(cf, "brix_certificate", &xcf->certificate,
84 BRIX_PATH_REGULAR_FILE, R_OK) != NGX_OK
85 || brix_validate_path(cf, "brix_certificate_key",
86 &xcf->certificate_key,
87 BRIX_PATH_REGULAR_FILE, R_OK) != NGX_OK
88 || brix_validate_path(cf, "brix_trusted_ca", &xcf->trusted_ca,
89 BRIX_PATH_FILE_OR_DIRECTORY, R_OK) != NGX_OK
90 || brix_validate_path(cf, "brix_crl", &xcf->crl,
91 BRIX_PATH_FILE_OR_DIRECTORY, R_OK) != NGX_OK)
92 {
93 return NGX_ERROR;
94 }
95
96 /* Load server certificate */
97 {
98 FILE *fp = fopen((char *) xcf->certificate.data, "r");
99 if (fp == NULL) {
100 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
101 "brix: cannot open certificate \"%s\"",
102 xcf->certificate.data);
103 return NGX_ERROR;
104 }
105 fcntl(fileno(fp), F_SETFD, FD_CLOEXEC);
106 xcf->gsi_cert = PEM_read_X509(fp, NULL, NULL, NULL);
107 fclose(fp);
108 if (xcf->gsi_cert == NULL) {
109 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
110 "brix: cannot parse certificate \"%s\"",
111 xcf->certificate.data);
112 return NGX_ERROR;
113 }
114 }
115
116 /* Cache the PEM serialization once; kXGS_cert sends it on every GSI login. */
117 {
118 BIO *bio;
119 BUF_MEM *bptr;
120
121 bio = BIO_new(BIO_s_mem());
122 if (bio == NULL) {
123 return NGX_ERROR;
124 }
125
126 if (!PEM_write_bio_X509(bio, xcf->gsi_cert)) {
127 BIO_free(bio);
128 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
129 "brix: cannot serialize certificate \"%s\"",
130 xcf->certificate.data);
131 return NGX_ERROR;
132 }
133
134 BIO_get_mem_ptr(bio, &bptr);
135 xcf->gsi_cert_pem = ngx_pnalloc(cf->pool, bptr->length);
136 if (xcf->gsi_cert_pem == NULL) {
137 BIO_free(bio);
138 return NGX_ERROR;
139 }
140
141 ngx_memcpy(xcf->gsi_cert_pem, bptr->data, bptr->length);
142 xcf->gsi_cert_pem_len = bptr->length;
143 BIO_free(bio);
144 }
145
146 /* Load server private key */
147 {
148 FILE *fp = fopen((char *) xcf->certificate_key.data, "r");
149 if (fp == NULL) {
150 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
151 "brix: cannot open private key \"%s\"",
152 xcf->certificate_key.data);
153 return NGX_ERROR;
154 }
155 fcntl(fileno(fp), F_SETFD, FD_CLOEXEC);
156 xcf->gsi_key = PEM_read_PrivateKey(fp, NULL, NULL, NULL);
157 fclose(fp);
158 if (xcf->gsi_key == NULL) {
159 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
160 "brix: cannot parse private key \"%s\"",
161 xcf->certificate_key.data);
162 return NGX_ERROR;
163 }
164 }
165
166 /* Build trusted CA X509_STORE. This parses every CA (and CRL) under the
167 * trusted-CA path, so its cost scales with the bundle size — on a full grid
168 * CA distribution (hundreds of CAs + CRLs) it dominates GSI config time.
169 * Time it independently so a slow startup is attributable at a glance. */
170 {
171 uint64_t t0 = brix_phase_now_ns();
172
173 if (brix_rebuild_gsi_store(xcf, cf->log) != NGX_OK) {
174 return NGX_ERROR;
175 }
176
177 /* Run a lightweight PKI/CRL consistency check and log any problems. */
178 (void) brix_check_pki_consistency_stream(cf->log, xcf);
179
180 ngx_log_error(NGX_LOG_NOTICE, cf->log, 0,
181 "brix: GSI trust store built from \"%V\" in %uLus",
182 &xcf->trusted_ca,
183 (brix_phase_now_ns() - t0) / 1000ull);
184 }
185
186 /* Compute CA hash (for kXRS_issuer_hash in kXGS_init) */
187 {
188 FILE *fp;
189 X509 *ca;
190
191 fp = fopen((char *) xcf->trusted_ca.data, "r");
192 if (fp) {
193 fcntl(fileno(fp), F_SETFD, FD_CLOEXEC);
194 ca = PEM_read_X509(fp, NULL, NULL, NULL);
195 fclose(fp);
196 if (ca) {
197 /*
198 * The protocol advertises the issuer hash during the GSI
199 * bootstrap so clients can confirm which CA the server wants.
200 */
201 xcf->gsi_ca_hash = (uint32_t) X509_subject_name_hash(ca);
202 X509_free(ca);
203 }
204 }
205 }
206
207 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
208 "brix: GSI auth configured - cert=%s ca_hash=%08xd",
209 xcf->certificate.data, xcf->gsi_ca_hash);
210
211 return NGX_OK;
212}
