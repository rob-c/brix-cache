5/*
6 * Build (or rebuild) the X509_STORE used for GSI certificate verification.
7 *
8 * Loads the trusted CA from xcf->trusted_ca, then loads CRLs from xcf->crl.
9 * Sets X509_V_FLAG_ALLOW_PROXY_CERTS so GSI proxy certificate chains verify.
10 *
11 * On success the new store is atomically swapped into xcf->gsi_store and any
12 * previous store is freed. On failure the old store is left intact so
13 * existing connections are not disrupted.
14 */
15ngx_int_t
16brix_rebuild_gsi_store(ngx_stream_brix_srv_conf_t *xcf, ngx_log_t *log,
17 void *cache_scope)
18{
19 X509_STORE *store;
20 X509_STORE *old_store;
21 int crl_count = 0;
22 struct stat ca_st;
23 int ca_is_dir;
24
25 /* brix_trusted_ca may name a single CA bundle file OR a hashed CA
26 * directory (e.g. /etc/grid-security/certificates). A directory is loaded
27 * as an OpenSSL CApath so on-demand hash lookup verifies real grid proxy
28 * chains (any CA under the dir), which a single-file bundle cannot cover. */
29 ca_is_dir = (stat((char *) xcf->trusted_ca.data, &ca_st) == 0
30 && S_ISDIR(ca_st.st_mode));
31
32 store = brix_build_ca_store_cached(cache_scope, log,
33 ca_is_dir ? (char *) xcf->trusted_ca.data
34 : NULL,
35 ca_is_dir ? NULL
36 : (char *) xcf->trusted_ca.data,
37 xcf->crl.len > 0
38 ? (char *) xcf->crl.data : NULL,
39 X509_V_FLAG_ALLOW_PROXY_CERTS,
40 &crl_count,
41 (brix_sp_mode_t) xcf->signing_policy_mode,
42 (int) xcf->crl_mode);
43 if (store == NULL) {
44 return NGX_ERROR;
45 }
46
47 if (xcf->crl.len > 0) {
48 ngx_log_error(NGX_LOG_NOTICE, log, 0,
49 "brix: loaded %d CRL(s) from \"%s\"",
50 crl_count, xcf->crl.data);
51 }
52
53 /* Atomic swap */
54 old_store = xcf->gsi_store;
55 xcf->gsi_store = store;
56 if (old_store != NULL) {
57 X509_STORE_free(old_store);
58 }
59
60 return NGX_OK;
61}

63/*
64 * Compute the CA name-hash list advertised in the kXGS_init sec token
65 * ("ca:hash1|hash2").
66 *
67 * WHY: stock XrdSecgsi clients look the advertised hashes up in their own CA
68 * directory and abort the handshake with "unknown CA: cannot verify server
69 * certificate" when none match, so the list must name the REAL issuer chain
70 * of our server certificate. (An earlier version PEM-parsed brix_trusted_ca
71 * directly — wrong for a multi-CA bundle and silently empty when the
72 * directive names a Grid CA DIRECTORY, which advertised ca:00000000 and broke
73 * every stock client whose host cert hangs off a real CA chain.)
74 *
75 * HOW: verify our own certificate against the freshly built trust store and
76 * hash every CA in the resulting chain (intermediates + root). Falls back to
77 * the leaf's issuer-name hash when chain building fails (e.g. a CRL gap), and
78 * to "00000000" only when there is no certificate at all.
79 */
80static void
81brix_gsi_compute_ca_hashes(ngx_stream_brix_srv_conf_t *xcf)
82{
83 X509_STORE_CTX *sctx;
84 STACK_OF(X509) *chain;
85 size_t off = 0;
86 int i, n, w;
87
88 xcf->gsi_ca_hashes[0] = '\0';
89
90 sctx = X509_STORE_CTX_new();
91 if (sctx != NULL && xcf->gsi_store != NULL && xcf->gsi_cert != NULL
92 && X509_STORE_CTX_init(sctx, xcf->gsi_store, xcf->gsi_cert, NULL) == 1
93 && X509_verify_cert(sctx) == 1)
94 {
95 chain = X509_STORE_CTX_get1_chain(sctx);
96 if (chain != NULL) {
97 n = sk_X509_num(chain);
98 for (i = 1; i < n; i++) { /* skip the leaf itself */
99 w = snprintf(xcf->gsi_ca_hashes + off,
100 sizeof(xcf->gsi_ca_hashes) - off,
101 off > 0 ? "|%08lx" : "%08lx",
102 X509_subject_name_hash(sk_X509_value(chain, i)));
103 if (w <= 0
104 || off + (size_t) w >= sizeof(xcf->gsi_ca_hashes))
105 {
106 break;
107 }
108 off += (size_t) w;
109 }
110 sk_X509_pop_free(chain, X509_free);
111 }
112 }
113 if (sctx != NULL) {
114 X509_STORE_CTX_free(sctx);
115 }
116
117 if (xcf->gsi_ca_hashes[0] == '\0' && xcf->gsi_cert != NULL) {
118 (void) snprintf(xcf->gsi_ca_hashes, sizeof(xcf->gsi_ca_hashes),
119 "%08lx", X509_issuer_name_hash(xcf->gsi_cert));
120 }
121 if (xcf->gsi_ca_hashes[0] == '\0') {
122 (void) snprintf(xcf->gsi_ca_hashes, sizeof(xcf->gsi_ca_hashes),
123 "%08x", 0u);
124 }
125}

127/* ---- Validate that all GSI trust inputs are present and path-usable ----
128 *
129 * WHAT: Returns NGX_OK when the certificate, private key, trusted CA and
130 * (optional) CRL directives all point at readable filesystem paths of the
131 * expected kind; returns NGX_ERROR after logging an NGX_LOG_EMERG line when a
132 * mandatory directive is absent or any path fails validation.
133 *
134 * WHY: GSI mode is only meaningful when the certificate, private key and
135 * trusted CA are all present simultaneously — missing any one makes GSI auth
136 * meaningless, so the presence check must reject the config before any file is
137 * touched. Path validation is grouped here so the loader body stays a flat
138 * sequence of load steps.
139 *
140 * HOW:
141 * 1. Reject the config when any of certificate/certificate_key/trusted_ca is
142 * empty, with the combined "requires ..." emergency message.
143 * 2. Validate certificate and certificate_key as readable regular files.
144 * 3. Validate trusted_ca and crl as readable file-or-directory paths (crl may
145 * be empty; brix_validate_path treats an unset directive as a pass).
146 */
147static ngx_int_t
148brix_gsi_require_trust_inputs(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
149{
150 if (xcf->certificate.len == 0 || xcf->certificate_key.len == 0
151 || xcf->trusted_ca.len == 0)
152 {
153 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
154 "brix_auth gsi requires brix_certificate, "
155 "brix_certificate_key and brix_trusted_ca");
156 return NGX_ERROR;
157 }
158
159 if (brix_validate_path(cf, "brix_certificate", &xcf->certificate,
160 BRIX_PATH_REGULAR_FILE, R_OK) != NGX_OK
161 || brix_validate_path(cf, "brix_certificate_key",
162 &xcf->certificate_key,
163 BRIX_PATH_REGULAR_FILE, R_OK) != NGX_OK
164 || brix_validate_path(cf, "brix_trusted_ca", &xcf->trusted_ca,
165 BRIX_PATH_FILE_OR_DIRECTORY, R_OK) != NGX_OK
166 || brix_validate_path(cf, "brix_crl", &xcf->crl,
167 BRIX_PATH_FILE_OR_DIRECTORY, R_OK) != NGX_OK)
168 {
169 return NGX_ERROR;
170 }
171
172 return NGX_OK;
173}

175/* ---- Load the server certificate PEM into xcf->gsi_cert ----
176 *
177 * WHAT: Opens xcf->certificate, parses the first X509 with PEM_read_X509() and
178 * stores it in xcf->gsi_cert; returns NGX_OK on success or NGX_ERROR after an
179 * NGX_LOG_EMERG line when the file cannot be opened or parsed.
180 *
181 * WHY: The parsed certificate is needed both to answer kXGS_cert and to build
182 * the CA-hash advertisement, so it is loaded once at config time and kept.
183 *
184 * HOW:
185 * 1. fopen the certificate path; on failure log with ngx_errno and return.
186 * 2. Mark the descriptor FD_CLOEXEC so it does not leak across exec.
187 * 3. PEM_read_X509 into xcf->gsi_cert and close the read-only stream.
188 * 4. Reject when parsing yields no certificate.
189 */
190static ngx_int_t
191brix_gsi_load_certificate(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
192{
193 FILE *fp = fopen((char *) xcf->certificate.data, "r");
194 if (fp == NULL) {
195 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
196 "brix: cannot open certificate \"%s\"",
197 xcf->certificate.data);
198 return NGX_ERROR;
199 }
200 fcntl(fileno(fp), F_SETFD, FD_CLOEXEC);
201 xcf->gsi_cert = PEM_read_X509(fp, NULL, NULL, NULL);
202 (void) fclose(fp); /* phase74-fp: read-only stream, close failure cannot lose data */
203 if (xcf->gsi_cert == NULL) {
204 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
205 "brix: cannot parse certificate \"%s\"",
206 xcf->certificate.data);
207 return NGX_ERROR;
208 }
209
210 return NGX_OK;
211}

213/* ---- Cache the certificate PEM serialization for kXGS_cert replies ----
214 *
215 * WHAT: Serializes xcf->gsi_cert back to PEM into a pool-allocated buffer at
216 * xcf->gsi_cert_pem (length xcf->gsi_cert_pem_len); returns NGX_OK on success,
217 * NGX_ERROR on any OpenSSL/allocation failure (with an NGX_LOG_EMERG line for a
218 * serialization failure).
219 *
220 * WHY: kXGS_cert sends the certificate on every GSI login, so encoding it once
221 * at config time avoids per-request PEM encoding overhead on the hot path.
222 *
223 * HOW:
224 * 1. Allocate a memory BIO; bail on allocation failure.
225 * 2. PEM_write_bio_X509 the loaded certificate into it.
226 * 3. Copy the BIO's bytes into a pool buffer and record the length.
227 * 4. Free the BIO on every exit so no OpenSSL memory leaks.
228 */
229static ngx_int_t
230brix_gsi_cache_cert_pem(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
231{
232 BIO *bio;
233 BUF_MEM *bptr;
234
235 bio = BIO_new(BIO_s_mem());
236 if (bio == NULL) {
237 return NGX_ERROR;
238 }
239
240 if (!PEM_write_bio_X509(bio, xcf->gsi_cert)) {
241 BIO_free(bio);
242 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
243 "brix: cannot serialize certificate \"%s\"",
244 xcf->certificate.data);
245 return NGX_ERROR;
246 }
247
248 BIO_get_mem_ptr(bio, &bptr);
249 xcf->gsi_cert_pem = ngx_pnalloc(cf->pool, bptr->length);
250 if (xcf->gsi_cert_pem == NULL) {
251 BIO_free(bio);
252 return NGX_ERROR;
253 }
254
255 ngx_memcpy(xcf->gsi_cert_pem, bptr->data, bptr->length);
256 xcf->gsi_cert_pem_len = bptr->length;
257 BIO_free(bio);
258
259 return NGX_OK;
260}

300/* ---- Build the trusted-CA X509_STORE and run the PKI consistency check ----
301 *
302 * WHAT: Builds xcf->gsi_store from trusted_ca + CRLs via brix_rebuild_gsi_store,
303 * runs the lightweight PKI/CRL consistency check, and logs the build duration;
304 * returns NGX_OK, or NGX_ERROR when the store cannot be built.
305 *
306 * WHY: The store parse walks every CA (and CRL) under the trusted-CA path, so
307 * its cost scales with the bundle size — on a full grid CA distribution
308 * (hundreds of CAs + CRLs) it dominates GSI config time. It is timed
309 * independently so a slow startup is attributable at a glance.
310 *
311 * HOW:
312 * 1. Snapshot a start timestamp.
313 * 2. Rebuild the store scoped to cf->cycle so sibling GSI server blocks
314 * sharing the same trusted_ca/CRL dir load the IGTF CRLs once, not per
315 * block; bail on failure.
316 * 3. Run brix_check_pki_consistency_stream to surface CRL gaps as warnings.
317 * 4. Log the elapsed build time in microseconds.
318 */
319static ngx_int_t
320brix_gsi_build_trust_store(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
321{
322 uint64_t t0 = brix_phase_now_ns();
323
324 if (brix_rebuild_gsi_store(xcf, cf->log, cf->cycle) != NGX_OK) {
325 return NGX_ERROR;
326 }
327
328 (void) brix_check_pki_consistency_stream(cf->log, xcf);
329
330 ngx_log_error(NGX_LOG_NOTICE, cf->log, 0,
331 "brix: GSI trust store built from \"%V\" in %uLus",
332 &xcf->trusted_ca,
333 (brix_phase_now_ns() - t0) / 1000ull);
334
335 return NGX_OK;
336}

338/* ---- Full GSI authentication configuration loader ----
339 *
340 * WHAT: When the auth mode selects GSI (or BOTH), validates the trust inputs,
341 * loads and caches the server certificate + private key, builds the trusted-CA
342 * store, and computes the advertised CA hashes; returns NGX_OK (also for a
343 * non-GSI auth mode, a no-op) or NGX_ERROR on any validation/load failure.
344 *
345 * WHY: All three trust inputs must be present simultaneously — missing any one
346 * makes GSI auth meaningless. Cert serialization caching avoids per-request
347 * PEM encoding on every client login, CRL checking enables revocation detection
348 * across the full chain, and the CA hash lets clients confirm which CA the
349 * server trusts before initiating the DH exchange.
350 *
351 * HOW:
352 * 1. Return early when auth is neither GSI nor BOTH.
353 * 2. Validate that all trust-input directives are present and path-usable.
354 * 3. Load the certificate, cache its PEM, and load the private key.
355 * 4. Build the trusted-CA store (with the PKI consistency check).
356 * 5. Compute the CA-hash advertisement and log the configured summary.
357 */
358ngx_int_t
359brix_configure_gsi(ngx_conf_t *cf, ngx_stream_brix_srv_conf_t *xcf)
360{
361 if (xcf->auth != BRIX_AUTH_GSI && xcf->auth != BRIX_AUTH_BOTH) {
362 return NGX_OK;
363 }
364
365 if (brix_gsi_require_trust_inputs(cf, xcf) != NGX_OK) {
366 return NGX_ERROR;
367 }
368
369 if (brix_gsi_load_certificate(cf, xcf) != NGX_OK) {
370 return NGX_ERROR;
371 }
372
373 if (brix_gsi_cache_cert_pem(cf, xcf) != NGX_OK) {
374 return NGX_ERROR;
375 }
376
377 if (brix_gsi_load_private_key(cf, xcf) != NGX_OK) {
378 return NGX_ERROR;
379 }
380
381 if (brix_gsi_build_trust_store(cf, xcf) != NGX_OK) {
382 return NGX_ERROR;
383 }
384
385 brix_gsi_compute_ca_hashes(xcf);
386
387 ngx_conf_log_error(NGX_LOG_NOTICE, cf, 0,
388 "brix: GSI auth configured - cert=%s ca_hashes=%s",
389 xcf->certificate.data, xcf->gsi_ca_hashes);
390
391 return NGX_OK;
392}
