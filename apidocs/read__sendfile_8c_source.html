20/*
21 * brix_ktls_send_active — true when kernel-TLS transmit is active on this
22 * connection (Phase 29 kTLS).
23 *
24 * Without kTLS, a TLS data stream must encrypt in userspace and therefore cannot
25 * use sendfile(2) — the historical reason the read path gates the zero-copy
26 * sendfile branch on !c->ssl. When the kernel TLS ULP is negotiated for the send
27 * side (OpenSSL SSL_OP_ENABLE_KTLS + a kTLS-offloadable cipher), the kernel does
28 * the record encryption inside sendfile, so a file-backed chain is legal over TLS
29 * and the read inherits the cleartext sendfile fast path (and its Phase-2
30 * pipelining). Returns 0 whenever kTLS is unavailable or not negotiated, so the
31 * caller transparently falls back to the memory/window path — the relaxation is
32 * always safe.
33 */
34static ngx_flag_t
35brix_ktls_send_active(ngx_connection_t *c)
36{
37#ifdef BIO_get_ktls_send
38 if (c->ssl != NULL && c->ssl->connection != NULL) {
39 return BIO_get_ktls_send(SSL_get_wbio(c->ssl->connection)) > 0 ? 1 : 0;
40 }
41#endif
42 return 0;
43}

45/* Zero-copy sendfile serve path for a regular-file cleartext (or kTLS) read:
46 * clamps the chunk to EOF, charges bytes/bandwidth/dashboard + access log,
47 * builds the sendfile chain and queues it. Always completes the request --
48 * the caller tail-calls this under the is_regular && (!ssl || kTLS) gate. */
49ngx_int_t
50brix_read_serve_sendfile(brix_ctx_t *ctx, ngx_connection_t *c,
51 ngx_stream_brix_srv_conf_t *rconf, const brix_read_io_t *io)
52{
53 size_t data_total;
54 u_char *send_base = NULL;
55 ngx_chain_t *rsp_chain;
56 int idx = io->idx;
57
58 off_t file_size;
59 off_t avail;
60
61 /*
62 * Read-only handles: file size is stable, use the value cached at open
63 * time to skip the fstat(2) syscall on every chunk request.
64 * Writable handles (kXR_open_updt): re-stat so a write on the same
65 * session is visible to subsequent reads.
66 */
67 if (!ctx->files[idx].writable) {
68 file_size = ctx->files[idx].cached_size;
69 } else {
70 struct stat st;
71 if (fstat(io->fd, &st) != 0) {
72 BRIX_RETURN_ERR(ctx, c, BRIX_OP_READ, "READ",
73 ctx->files[idx].path, "-",
74 kXR_IOError, strerror(errno));
75 }
76 file_size = st.st_size;
77 }
78
79 /*
80 * Clamp the chunk to bytes actually present: sendfile would otherwise be
81 * asked for data past EOF. offset at/after EOF yields a zero-length OK
82 * (legal short read); otherwise serve min(rlen, remaining-to-EOF). This
83 * is also why data_total — not the client's requested rlen — drives every
84 * accounting counter below.
85 */
86 if ((off_t) io->offset >= file_size) {
87 data_total = 0;
88 } else {
89 avail = file_size - (off_t) io->offset;
90 data_total = (avail < (off_t) io->rlen) ? (size_t) avail : io->rlen;
91 }
92
93 brix_prefetch_read_file(c->log, &ctx->files[idx], (off_t) io->offset,
94 data_total, file_size);
95
96 ctx->files[idx].bytes_read += data_total;
97 ctx->totals.bytes += data_total;
98 brix_rl_charge_ctx(ctx, data_total); /* Phase 25 bandwidth */
99
100 /* Per-backend storage byte totals: this zero-copy branch never reaches
101 * brix_vfs_io_execute (the kernel moves the bytes), so attribute here.
102 * The branch gate (sd_obj.driver == NULL) means the backend is always the
103 * default POSIX driver. Buffered/driver-backed reads attribute at the
104 * io_execute seam instead — no double count. */
105 brix_metric_backend_bytes("posix", BRIX_METRIC_OP_READ, data_total);
106
107 if (ctx->files[idx].dashboard_slot >= 0 &&
108 ngx_brix_dashboard_shm_zone != NULL)
109 {
110 brix_transfer_slot_update(ngx_brix_dashboard_shm_zone->data,
111 ctx->files[idx].dashboard_slot,
112 (ngx_atomic_int_t) data_total,
113 (int64_t) ngx_current_msec);
114 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
115 ctx->files[idx].dashboard_slot,
116 "read");
117 }
118
119 if (rconf->access_log_fd != NGX_INVALID_FILE) {
120 char read_detail[64];
121
122 snprintf(read_detail, sizeof(read_detail), "%lld+%zu",
123 (long long) io->offset, io->rlen);
124 brix_log_access(ctx, c, "READ", ctx->files[idx].path,
125 read_detail, 1, 0, NULL, data_total);
126 }
127 BRIX_OP_OK(ctx, BRIX_OP_READ);
128
129 rsp_chain = brix_build_sendfile_chain(ctx, c, io->fd,
130 ctx->files[idx].path,
131 (off_t) io->offset, data_total,
132 &send_base);
133 if (rsp_chain == NULL) {
134 brix_release_read_buffer(ctx, c, send_base);
135 return NGX_ERROR;
136 }
137
138 {
139 ngx_int_t rc = brix_queue_response_chain(ctx, c, rsp_chain,
140 send_base);
141
142 if (rc != NGX_OK || ctx->state != XRD_ST_SENDING) {
143 brix_release_read_buffer(ctx, c, send_base);
144 }
145 return rc;
146 }
147}

149/*
150 * read_sendfile_eligible — gate for the zero-copy sendfile fast path.
151 *
152 * WHAT: true when this handle/connection pair may serve the read via a
153 * file-backed sendfile chain instead of the memory/window path.
154 * WHY: INVARIANT — TLS serves memory-backed buffers and cleartext serves
155 * file-backed + sendfile, never mixed; this predicate is the single place
156 * that split is decided for kXR_read.
157 * HOW: two conditions must both hold:
158 * - is_regular: sendfile(2) only works against a real file, not a pipe/dir.
159 * - !c->ssl OR kTLS active: a userspace-TLS stream cannot sendfile because
160 * nginx must encrypt each record in user memory (INVARIANT: TLS =>
161 * memory-backed buffers). kTLS lifts that — the kernel encrypts inside
162 * sendfile — so a TLS connection with kTLS negotiated rejoins this branch.
163 * Anything that fails the gate (TLS without kTLS, irregular file) drops to the
164 * memory/window path.
165 */
166ngx_flag_t
167read_sendfile_eligible(brix_ctx_t *ctx, ngx_connection_t *c, int idx)
168{
169 return ctx->files[idx].is_regular
170 && (!c->ssl || brix_ktls_send_active(c))
171 && ctx->files[idx].csi == NULL /* phase-59 W2/ADR-6: CSI needs the
172 * bytes in memory to verify, so an
173 * integrity-checked handle takes the
174 * buffered path, not zero-copy sendfile */
175 && ctx->files[idx].sd_obj.driver == NULL; /* Layer 3: a driver-backed
176 * handle's bare fd is only block 0 — a
177 * sendfile over it cannot span striped
178 * blocks, so serve via the buffered
179 * io_core path (driver preadv) instead */
180}
