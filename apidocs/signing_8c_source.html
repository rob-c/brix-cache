7/*
8 * brix_handle_sigver - XRootD request signing (kXR_sigver).
9 *
10 * Protocol flow:
11 * 1. Client sends kXR_sigver with HMAC-SHA256(signing_key, seqno || next_hdr
12 * [|| next_payload]) as the body, and expectrid = opcode of the next request.
13 * 2. We save the HMAC and seqno in pending state on ctx.
14 * 3. brix_dispatch() verifies the HMAC before routing the following request.
15 *
16 * For GSI sessions signing_active is 1 and signing_key = SHA-256(DH secret).
17 * For token/anonymous sessions we accept sigver without verification; legitimate
18 * clients should not send it unsolicited, but some do (e.g. when connecting to an
19 * unknown server type).
20 *
21 * Only HMAC-SHA256 without RSA (kXR_SHA256, kXR_rsaKey unset) is verified.
22 * RSA-signed requests are accepted without checking the asymmetric signature.
23 */
24/* Handle kXR_sigver — store the expected HMAC-SHA256 signature and monotonic
25 * sequence number for verifying the NEXT signed request (the seqno blocks
26 * replay). */
27ngx_int_t
28brix_handle_sigver(brix_ctx_t *ctx, ngx_connection_t *c)
29{
30 xrdw_sigver_req_t req;
31 uint16_t expectrid;
32 uint64_t seqno;
33
34 xrdw_sigver_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
35 expectrid = req.expectrid;
36 seqno = req.seqno;
37
38 if (ctx->sigver.signing_active) {
39 /* Reject replays: seqno must strictly increase across the session. */
40 if (seqno <= ctx->sigver.last_seqno) {
41 ngx_log_error(NGX_LOG_WARN, c->log, 0,
42 "brix: sigver replay (seqno=%llu <= last=%llu)",
43 (unsigned long long) seqno,
44 (unsigned long long) ctx->sigver.last_seqno);
45 return brix_send_error(ctx, c, kXR_NotAuthorized,
46 "sigver replay detected");
47 }
48 ctx->sigver.last_seqno = seqno;
49
50 if ((req.crypto & kXR_HashMask_sig) == kXR_SHA256_sig
51 && !(req.crypto & kXR_rsaKey_sig))
52 {
53 /* Need exactly 32 bytes of HMAC in the body. */
54 if (ctx->recv.cur_dlen < 32 || ctx->recv.payload == NULL) {
55 return brix_send_error(ctx, c, kXR_ArgInvalid,
56 "sigver body too short");
57 }
58
59 ctx->sigver.pending = 1;
60 ctx->sigver.expectrid = expectrid;
61 ctx->sigver.seqno = seqno;
62 ctx->sigver.nodata = (req.flags & kXR_nodata_sig) ? 1 : 0;
63 ngx_memcpy(ctx->sigver.hmac, ctx->recv.payload, 32);
64
65 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
66 "brix: sigver pending expectrid=%d seqno=%llu",
67 (int) expectrid, (unsigned long long) seqno);
68 } else {
69 ngx_log_debug1(NGX_LOG_DEBUG_STREAM, c->log, 0,
70 "brix: sigver crypto=0x%02xd not verified (RSA path)",
71 (unsigned) req.crypto);
72 }
73 } else {
74 ngx_log_debug0(NGX_LOG_DEBUG_STREAM, c->log, 0,
75 "brix: sigver accepted without verification (no GSI key)");
76 }
77
78 brix_log_access(ctx, c, "SIGVER", "-", "-", 1, 0, NULL, 0);
79 /*
80 * kXR_sigver is a request PREFIX, not a standalone request: per the XRootD
81 * protocol (XrdXrootdProtocol::ProcSig) the server sends NO response on
82 * success — the single response belongs to the signed request that follows,
83 * which brix_verify_pending_sigver() validates at the top of its dispatch.
84 * Returning NGX_OK without queuing a frame lets the recv loop read that next
85 * request. Emitting a kXR_ok here desynchronised every stock-protocol client
86 * (go-hep, official XrdCl): they read the ack as the signed request's reply
87 * (e.g. go-hep stat → statinfo "" parse error). The error paths above still
88 * reply, since a sigver failure aborts the exchange.
89 */
90 return NGX_OK;
91}
