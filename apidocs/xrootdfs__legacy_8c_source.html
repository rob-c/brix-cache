1/*
2 * xrootdfs_legacy.c — simple synchronous XRootD root:// FUSE mount.
3 *
4 * WHAT: `xrootdfs_legacy root://host[:port]/ /mnt [-f] [auth opts]` — exposes a
5 * remote XRootD namespace as a directory tree that ordinary tools (ls, cat,
6 * cp, md5sum, …) can read and write through libfuse3.
7 * WHY: The simple, synchronous fallback driver. The DEFAULT driver is xrootdfs(1)
8 * (the async, network-resilient core); this one is kept as a minimal
9 * safety-net implementation, reachable via `xrd mount --legacy`. It has NO
10 * libXrdCl / XrdFfs dependency.
11 * HOW: Concurrency model — XRootD connections are one-request-in-flight and not
12 * thread-safe, so:
13 * * metadata / path ops (getattr, readdir, mkdir, …) borrow a connection
14 * from a thread-safe POOL (brix_pool) for the duration of the call;
15 * * each OPEN FILE pins its own dedicated brix_conn for its lifetime
16 * (the server's fhandle is only valid on the connection that opened
17 * it), serialised by a per-handle mutex so concurrent reads/writes on
18 * the same fd are safe while different files run in parallel.
19 * The mount runs multi-threaded (libfuse's thread pool). Each callback is a
20 * thin map onto a libbrix call, errors translated to -errno via
21 * brix_kxr_to_errno. Open handles live in a malloc'd xfs_handle in fi->fh.
22 *
23 * Scope: full metadata + read + write incl. random/in-place updates (O_RDWR
24 * without truncate), create/truncate/mkdir/unlink/rmdir/rename/chmod,
25 * statfs (df), access, readdir-plus, kernel-side attr/data caching, and
26 * per-handle read-ahead + write-back buffering (coalesce sequential reads /
27 * small writes into larger server round-trips; --readahead/--writeback).
28 * Hard wire gaps (no XRootD opcode): set-mtime (utimens) and chown are
29 * accepted as no-ops so `cp -p` succeeds; symlink/hardlink are unsupported.
30 * Opt-in --xattr adds extended attributes (kXR_fattr) plus a read-only
31 * user.XrdCks.<algo> checksum view.
32 *
33 * Clean-room: composes the public libbrix API + libfuse3 only.
34 */

166xfs_readdir(const char *path, void *buf, fuse_fill_dir_t filler, off_t offset,

168{

201/* Allocate a handle with its own dedicated connection (the server fhandle is only
202 * valid on the connection that opened it). Returns the handle or NULL (+ -errno in
203 * *err) on connect failure. */
204static xfs_handle *
205xfs_handle_new(int *err)
206{
207 xfs_handle *h = calloc(1, sizeof(*h));
208 brix_status st;
209
210 if (h == NULL) {
211 *err = -ENOMEM;
212 return NULL;
213 }
214 pthread_mutex_init(&h->lock, NULL);
215 brix_status_clear(&st);
216 if (brix_connect(&h->conn, &g_url, &g_opts, &st) != 0) {
217 pthread_mutex_destroy(&h->lock);
218 free(h);
219 *err = xfs_err(&st);
220 return NULL;
221 }
222 return h;
223}

234static int
235xfs_open(const char *path, struct fuse_file_info *fi)
236{
237 xfs_handle *h;
238 brix_status st;
239 int acc = fi->flags & O_ACCMODE;
240 int rc, err = 0;
241
242 h = xfs_handle_new(&err);
243 if (h == NULL) {
244 return err;
245 }
246 brix_status_clear(&st);
247 if (acc == O_RDONLY) {
248 rc = brix_rfile_open_read(&h->conn, path, NULL, 0, -1, &h->f, &st);
249 } else {
250 /* Writable open: O_TRUNC → overwrite from empty (force 1); otherwise
251 * in-place update (force 2), so partial edits (sed -i, random writes)
252 * preserve unwritten content. */
253 rc = brix_rfile_open_write(&h->conn, path,
254 brix_open_force_for_open(fi->flags),
255 0, 0, -1, &h->f, &st);
256 h->writable = 1;
257 }
258 if (rc != 0) {
259 xfs_handle_free(h);
260 return xfs_err(&st);
261 }
262 brix_iobuf_init(&h->io, h, xfs_io_pread, xfs_io_pwrite, h->writable,
263 g_readahead, g_writeback);
264 fi->fh = (uint64_t) (uintptr_t) h;
265 return 0;
266}

469/* statfs — report capacity from kXR_Qspace ("oss.space=…&oss.free=…"); reported in
470 * 1 MiB blocks so total/free fit f_blocks/f_bfree comfortably for huge backends. */
471static int
472xfs_statfs(const char *path, struct statvfs *stbuf)
473{
474 char text[1024];
475 brix_status st;
476 brix_conn *c;
477 int rc;
478 unsigned long long total = 0, freeb = 0;
479
480 memset(stbuf, 0, sizeof(*stbuf));
481 brix_status_clear(&st);
482 c = brix_pool_checkout(g_pool, &st);
483 if (c == NULL) {
484 return xfs_err(&st);
485 }
486 /* kXR_Qspace returns "oss.cgroup=…&oss.space=<bytes>&oss.free=<bytes>&…";
487 * kXR_stat|kXR_vfs (brix_statvfs) uses a different compact format, so query
488 * Qspace directly for the parseable byte totals. */
489 rc = brix_query(c, kXR_Qspace, (path != NULL && path[0]) ? path : "/",
490 text, sizeof(text), &st);
491 brix_pool_checkin(g_pool, c, rc == 0 ? 1 : xfs_conn_healthy(&st));
492 if (rc != 0) {
493 return xfs_err(&st);
494 }
495
496 brix_parse_qspace(text, &total, &freeb);
497
498 stbuf->f_bsize = 1024 * 1024;
499 stbuf->f_frsize = stbuf->f_bsize;
500 stbuf->f_blocks = (fsblkcnt_t) (total / stbuf->f_bsize);
501 stbuf->f_bfree = (fsblkcnt_t) (freeb / stbuf->f_bsize);
502 stbuf->f_bavail = stbuf->f_bfree;
503 stbuf->f_namemax = 255;
504 return 0;
505}

543static int
544xfs_getxattr(const char *path, const char *name, char *value, size_t size)
545{
546 brix_status st;
547 brix_conn *c;
548 int rc;
549 size_t vlen = 0;
550 const char *fname;
551
552 if (!g_xattr) {
553 return -ENOTSUP;
554 }
555 brix_status_clear(&st);
556
557 /* Virtual read-only checksum xattr: user.XrdCks.<algo> → kXR_Qcksum hex. */
558 if (strncmp(name, XFS_CKS_XATTR_PFX, sizeof(XFS_CKS_XATTR_PFX) - 1) == 0) {
559 const char *algo = name + sizeof(XFS_CKS_XATTR_PFX) - 1;
560 char hex[160];
561
562 if (algo[0] == '\0') {
563 return -ENODATA;
564 }
565 c = brix_pool_checkout(g_pool, &st);
566 if (c == NULL) {
567 return xfs_err(&st);
568 }
569 rc = brix_query_cksum(c, path, algo, hex, sizeof(hex), &st);
570 brix_pool_checkin(g_pool, c, rc == 0 ? 1 : xfs_conn_healthy(&st));
571 if (rc != 0) {
572 return xfs_err(&st);
573 }
574 vlen = strlen(hex);
575 if (size == 0) {
576 return (int) vlen;
577 }
578 if (vlen > size) {
579 return -ERANGE;
580 }
581 memcpy(value, hex, vlen);
582 return (int) vlen;
583 }
584
585 fname = xfs_xattr_to_fattr(name);
586 if (fname == NULL) {
587 return -ENODATA;
588 }
589 c = brix_pool_checkout(g_pool, &st);
590 if (c == NULL) {
591 return xfs_err(&st);
592 }
593 rc = brix_fattr_get(c, path, fname, value, size, &vlen, &st);
594 brix_pool_checkin(g_pool, c, rc == 0 ? 1 : xfs_conn_healthy(&st));
595 if (rc != 0) {
596 return xfs_err(&st);
597 }
598 if (size == 0) {
599 return (int) vlen;
600 }
601 if (vlen > size) {
602 return -ERANGE;
603 }
604 return (int) vlen;
605}

607static int
608xfs_setxattr(const char *path, const char *name, const char *value,
609 size_t size, int flags)
610{
611 brix_status st;
612 brix_conn *c;
613 int rc;
614 const char *fname;
615
616 if (!g_xattr) {
617 return -ENOTSUP;
618 }
619 if (strncmp(name, XFS_CKS_XATTR_PFX, sizeof(XFS_CKS_XATTR_PFX) - 1) == 0) {
620 return -EACCES; /* checksum xattr is read-only */
621 }
622 fname = xfs_xattr_to_fattr(name);

764/* WHAT: handle a no-value dash-option (a bare flag), updating the driver's
765 * file-scope option globals in place.
766 * WHY: splitting the flag group from the value group (mirrors the async driver's
767 * aio_opt_novalue) keeps each option handler under the complexity gate; the
768 * option state already lives in file-scope globals, so this maps onto them
769 * directly (no new state introduced).
770 * HOW: returns 1 if `a` was a recognised flag, 0 otherwise. */
771static int
772xfs_opt_novalue(const char *a)
773{
774 if (strcmp(a, "--tls") == 0) { g_opts.want_tls = 1; return 1; }
775 if (strcmp(a, "--notlsok") == 0) { g_opts.notlsok = 1; return 1; }
776 if (strcmp(a, "--noverifyhost") == 0) { g_opts.verify_host = 0; return 1; }
777 if (strcmp(a, "--kernel-cache") == 0) { g_kernel_cache = 1; return 1; }
778 if (strcmp(a, "--xattr") == 0) { g_xattr = 1; return 1; }
779 return 0;
780}
781
782/* WHAT: handle a value-taking dash-option `a` with its already-extracted value
783 * word `v`, updating the driver's file-scope option globals in place.
784 * WHY: the value group split from the flag group (mirrors aio_opt_conn_value /
785 * aio_opt_cache_value) keeps this handler under the complexity gate.
786 * HOW: the caller only calls this when a next word exists, so `v` is always
787 * valid (a trailing value option therefore falls through to libfuse exactly
788 * as before). Returns 1 if `a` was a recognised value option, 0 otherwise. */
789static int
790xfs_opt_value(const char *a, char *v)
791{
792 if (strcmp(a, "--auth") == 0) { g_opts.auth_force = v; return 1; }
793 if (strcmp(a, "--max-conns") == 0) {
794 g_max_conns = atoi(v);
795 if (g_max_conns < 1) { g_max_conns = 1; }
796 return 1;
797 }
798 if (strcmp(a, "--attr-timeout") == 0) { g_attr_timeout = atof(v); return 1; }
799 if (strcmp(a, "--entry-timeout") == 0) { g_entry_timeout = atof(v); return 1; }
800 if (strcmp(a, "--readahead") == 0) {
801 long n = atol(v);
802 g_readahead = (n > 0) ? (size_t) n : 0;
803 return 1;
804 }
805 if (strcmp(a, "--writeback") == 0) {
806 long n = atol(v);
807 g_writeback = (n > 0) ? (size_t) n : 0;
808 return 1;
809 }
810 if (strcmp(a, "--connect-timeout") == 0) { brix_tmo_set_connect_ms(atoi(v)); return 1; }
811 if (strcmp(a, "--io-timeout") == 0) { brix_tmo_set_io_ms(atoi(v)); return 1; }
812 return 0;
813}

815/* WHAT: split the command line into our options, the endpoint, and the libfuse
816 * passthrough vector.
817 * WHY: our known options are honored ANYWHERE on the line (before OR after the
818 * endpoint); unknown dash-args fall through to libfuse (so -f/-d/-s/-o still
819 * work); the first bare word is the endpoint, the next the mountpoint.
820 * HOW: each dash-arg is offered to the flag group then (when a next word exists)
821 * the value group, matching the original inline argc guards; -h/--help print
822 * usage and stop. Returns -1 to proceed with the mount, or a process exit
823 * code (0 for -h/--help encountered inline). */
824static int
825xfs_parse_args(int argc, char **argv, char **fuse_argv, int *fuse_argc,
826 const char **endpoint)
827{
828 int i;
829
830 for (i = 1; i < argc; i++) {
831 char *a = argv[i];
832
833 if (a[0] == '-') {
834 if (xfs_opt_novalue(a)) { continue; }
835 if (i + 1 < argc && xfs_opt_value(a, argv[i + 1])) { i++; continue; }
836 if (strcmp(a, "-h") == 0 || strcmp(a, "--help") == 0) {
837 usage();
838 return 0;
839 }
840 if (*fuse_argc < 61) { fuse_argv[(*fuse_argc)++] = argv[i]; } /* fuse opt */
841 } else if (*endpoint == NULL) {
842 *endpoint = a; /* first non-option = the root:// URL */
843 } else if (*fuse_argc < 61) {
844 fuse_argv[(*fuse_argc)++] = argv[i]; /* mountpoint + fuse flags */
845 }
846 }
847 return -1;
848}

850/* WHAT: parse the endpoint, bring up the metadata connection pool, run the FUSE
851 * mount, and tear the pool down afterwards.
852 * WHY: isolates the mount bring-up + teardown ordering from the entry point so
853 * main() stays a short arg-parse → mount sequence.
854 * HOW: the pool connects one conn eagerly, so a bad endpoint/auth fails here
855 * (before fuse_main); the pool is destroyed after fuse_main returns.
856 * Returns the process exit code. */
857static int
858xfs_root_mount(int fuse_argc, char **fuse_argv, const char *endpoint)
859{
860 brix_status st;
861 int rc;
862
863 brix_status_clear(&st);
864 if (brix_endpoint_parse(endpoint, &g_url, &st) != 0) {
865 fprintf(stderr, "xrootdfs: %s\n", st.msg);
866 return 2;
867 }
868 /* The pool connects one conn eagerly, so a bad endpoint/auth fails here. */
869 g_pool = brix_pool_create(&g_url, &g_opts, g_max_conns, &st);
870 if (g_pool == NULL) {
871 fprintf(stderr, "xrootdfs: connect %s:%d: %s\n",
872 g_url.host, g_url.port, st.msg);
873 return brix_shellcode(&st);
874 }
875 fprintf(stderr, "xrootdfs: mounted %s:%d (pool=%d, multi-threaded)\n",
876 g_url.host, g_url.port, g_max_conns);
877
878 rc = fuse_main(fuse_argc, fuse_argv, &xfs_ops, NULL);
879
880 brix_pool_destroy(g_pool);
881 return rc;
882}

884/* Entry point for the synchronous fallback driver. Invoked by the unified
885 * xrootdfs front-end (apps/xrootdfs_main.c) when --legacy is given. */
886int
887xrootdfs_legacy_main(int argc, char **argv)
888{
889 const char *endpoint = NULL;
890 char *fuse_argv[64];
891 int fuse_argc = 0;
892 int rc;
893
894 if (argc < 3) {
895 usage();
896 return 2;
897 }
898 memset(&g_opts, 0, sizeof(g_opts));
899 g_opts.verify_host = 1;
900 brix_crypto_init();
901
902 fuse_argv[fuse_argc++] = argv[0];
903 /* This entry point only runs in --legacy mode (the binary is `xrootdfs`), so
904 * tag the kernel mount subtype explicitly: mounts then show as
905 * fuse.xrootdfs_legacy and `xrd mount` can still tell the two apart. */
906 fuse_argv[fuse_argc++] = (char *) "-osubtype=xrootdfs_legacy";
907
908 rc = xfs_parse_args(argc, argv, fuse_argv, &fuse_argc, &endpoint);
909 if (rc >= 0) {
910 return rc;
911 }
912
913 if (endpoint == NULL || fuse_argc < 2) {
914 usage();
915 return 2;
916 }
917 fuse_argv[fuse_argc] = NULL;
918
919 return xfs_root_mount(fuse_argc, fuse_argv, endpoint);
920}
