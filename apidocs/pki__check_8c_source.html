30/*
31 *
32 * WHAT: Converts an X509_NAME structure to a human-readable string via X509_NAME_oneline — produces the standard OpenSSL DN format used for CA/CRL issuer identification in error logs. Returns caller-owned malloc'd string that must be freed with OPENSSL_free.
33 *
34 * WHY: PKI verification functions need to log issuer DNs when CRL/CA mismatches occur; without converting the binary X509_NAME structure to readable text, operators can't determine which certificate in their trust store is causing the issue.
35 *
36 * HOW: Calls X509_NAME_oneline(name, NULL, 0) which returns a malloc'd string in standard DN format. Caller retains ownership and must free via OPENSSL_free.
37 */
38
39static char *
40brix_pki_name_to_str(const X509_NAME *name)
41{
42 return X509_NAME_oneline(name, NULL, 0);
43}

70/*
71 *
72 * WHAT: Compares the issuer DN of a CRL against the subject DN of a CA certificate — returns 1 if they match, 0 otherwise. Uses HAVE_X509_CRL_GET0_ISSUER compile-time macro to select between modern X509_CRL_get0_issuer (const pointer) and legacy X509_CRL_get_issuer (malloc'd pointer).
73 *
74 * WHY: CRL issuer matching is the first step of cross-signature verification; DN comparison selects candidate CA, then signature check proves the CRL was actually signed by that CA's private key. Without this helper, each caller would duplicate the issuer/subject extraction + comparison logic.
75 *
76 * HOW: Extracts crl_issuer via CRL_GET_ISSUER macro (conditional on OpenSSL version), extracts ca_subject via X509_get_subject_name, compares with X509_NAME_cmp — returns 1 if zero difference.
77 */
78
79static ngx_flag_t
80brix_pki_crl_matches_ca(X509_CRL *crl, X509 *ca_cert)
81{
82 X509_NAME *crl_issuer;
83 X509_NAME *ca_subject;
84
85 crl_issuer = CRL_GET_ISSUER(crl);
86 ca_subject = X509_get_subject_name(ca_cert);
87
88 return X509_NAME_cmp(crl_issuer, ca_subject) == 0;
89}

91/*
92 *
93 * WHAT: Verifies that a CRL's digital signature was produced by the specified issuer CA's public key — returns 1 if valid, 0 on failure with structured error log. Extracts public key from CA cert, calls X509_CRL_verify, frees key.
94 *
95 * WHY: DN matching alone is insufficient to prove a CRL belongs to a CA; a different CA could have the same subject DN in misconfigured setups. The signature check proves cryptographic ownership — the CRL was actually signed by that CA's private key, not just nominally issued by it.
96 *
97 * HOW: Extracts issuer_subject and issuer_public_key from issuer_ca cert; if pubkey extraction fails, logs "CA has no public key" error; calls X509_CRL_verify(crl, issuer_public_key); frees pubkey via EVP_PKEY_free; on failure logs CRL signature verification error with issuer DN.
98 */
99
100static ngx_flag_t
101brix_pki_crl_signature_valid(ngx_log_t *log, X509_CRL *crl,
102 X509 *issuer_ca, const char *log_prefix)
103{
104 EVP_PKEY *issuer_public_key;
105 X509_NAME *issuer_subject;
106 int verify_ok;
107
108 issuer_subject = X509_get_subject_name(issuer_ca);
109 issuer_public_key = X509_get_pubkey(issuer_ca);
110 if (issuer_public_key == NULL) {
111 brix_pki_log_name_error(log, log_prefix,
112 "CA has no public key",
113 issuer_subject);
114 return 0;
115 }
116
117 /*
118 * Matching issuer DN selects the candidate CA; the signature check proves
119 * the CRL was actually issued by that CA key.
120 */
121 verify_ok = X509_CRL_verify(crl, issuer_public_key);
122 EVP_PKEY_free(issuer_public_key);
123
124 if (verify_ok != 1) {
125 brix_pki_log_name_error(log, log_prefix,
126 "CRL signature verification failed for CRL issuer",
127 issuer_subject);
128 return 0;
129 }
130
131 return 1;
132}

134/*
135 * brix_pki_verify_crls — at startup, check that every loaded CRL has a
136 * matching CA certificate and a valid signature.
137 *
138 * This is a configuration-time consistency check, not an online revocation
139 * check. Its purpose is to catch mismatched ca_dir / crl_dir settings before
140 * the server accepts any connections.
141 *
142 * Issuer matching: the CRL's issuer DN is compared with each CA's subject DN.
143 * DN matching selects the candidate CA; the signature check then proves the
144 * CRL was actually signed by that CA's private key.
145 *
146 * Errors are logged but do not cause the function to return NGX_ERROR — the
147 * server starts even with a broken CRL so that a CA update doesn't take the
148 * whole service down. Operators should monitor the error log.
149 *
150 * Returns: always NGX_OK (errors are non-fatal, logged only).
151 */
152ngx_int_t
153brix_pki_verify_crls(ngx_log_t *log, STACK_OF(X509) *ca_certs,
154 STACK_OF(X509_CRL) *crls, const char *log_prefix)
155{
156 int crl_count;
157 int ca_count;
158 int crl_index;
159
160 crl_count = sk_X509_CRL_num(crls);
161 ca_count = sk_X509_num(ca_certs);
162
163 for (crl_index = 0; crl_index < crl_count; crl_index++) {
164 X509_CRL *crl;
165 X509_NAME *crl_issuer;
166 ngx_flag_t issuer_found;
167 int ca_index;
168
169 crl = sk_X509_CRL_value(crls, crl_index);
170 crl_issuer = CRL_GET_ISSUER(crl);
171 issuer_found = 0;
172
173 for (ca_index = 0; ca_index < ca_count; ca_index++) {
174 X509 *candidate_ca;
175
176 candidate_ca = sk_X509_value(ca_certs, ca_index);
177 if (brix_pki_crl_matches_ca(crl, candidate_ca)) {
178 issuer_found = 1;
179 (void) brix_pki_crl_signature_valid(log, crl,
180 candidate_ca,
181 log_prefix);
182 break;
183 }
184 }
185
186 if (!issuer_found) {
187 brix_pki_log_name_error(log, log_prefix,
188 "no matching CA found for CRL issuer",
189 crl_issuer);
190 }
191 }
192
193 return NGX_OK;
194}
