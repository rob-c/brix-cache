1/*
2 * sd_pblock_ident.c — identity resolution + POSIX access checks for the pblock
3 * storage driver's catalog-internal ownership model.
4 *
5 * WHAT: Resolves a request credential (brix_sd_cred_t: authenticated principal
6 * + comma-separated VO list) into pblock_ids_t — the catalog-internal
7 * synthetic uid and VO gids from the catalog's `ids` registry — and
8 * provides the POSIX mode-bit checks the *_cred vtable slots
9 * (sd_pblock_cred.c) enforce: object access by class (owner/group/other),
10 * parent-directory checks for create/remove, and the sticky-bit delete
11 * gate.
12 *
13 * WHY: pblock is self-contained multi-user storage: no unix accounts, no
14 * impersonation — the catalog is the identity authority (the settled
15 * "catalog-internal registry" owner model). Group membership IS VO
16 * membership, so a VO-shared directory is just a group-writable
17 * directory owned by the VO's gid. Splitting resolution/checks from the
18 * slots keeps both files under the one-concept size cap.
19 *
20 * HOW: Resolution maps principal→uid and each VO name→gid through
21 * pblock_catalog_id_map (auto-assigning, immutable, process-safe); a
22 * VO-less principal gets a user-private group (gid = uid — the id space
23 * is a single sequence, so uids and gids never collide). No cache: the
24 * maps are indexed point SELECTs on a WAL/FULLMUTEX connection, safe from
25 * worker threads where a shared mutable cache would not be. A NULL/empty
26 * principal resolves to `service` (all checks pass), preserving
27 * single-user semantics for identity-less deployments. ngx-free; gated by
28 * BRIX_HAVE_SQLITE like the rest of the backend.
29 */

49/* ---- identity resolution --------------------------------------------------
50 *
51 * WHAT: Fill *out with the credential's synthetic ids: principal→uid, each
52 * VO name in the comma-separated cred->vos→one gid (first VO = primary
53 * gid). NULL cred / NULL or empty principal → out->service = 1 (bypass).
54 * NGX_OK, or NGX_ERROR with errno (registry failure).
55 *
56 * WHY: Every *_cred slot needs the same cred→ids translation; ids must be
57 * stable across workers and restarts, which the catalog registry
58 * guarantees and an in-process table could not.
59 *
60 * HOW: 1. Service bypass on missing identity. 2. id_map the principal (USER
61 * kind). 3. Walk the VO CSV, id_mapping each non-empty token (GROUP
62 * kind) up to PBLOCK_MAX_GIDS (excess VOs are ignored — membership
63 * checks just won't see them; 16 covers real WLCG credentials). 4. A
64 * VO-less principal gets gid = uid (user-private group).
65 */
66ngx_int_t
67pblock_ident_resolve(pblock_state_t *st, const brix_sd_cred_t *cred,
68 pblock_ids_t *out)
69{
70 const char *p;
71 int64_t id;
72
73 memset(out, 0, sizeof(*out));
74 if (cred == NULL || cred->principal == NULL || cred->principal[0] == '\0') {
75 out->service = 1;
76 return NGX_OK;
77 }
78
79 if (pblock_catalog_id_map(st->cat, PBLOCK_ID_USER, cred->principal,
80 &id) != 0)
81 {
82 return NGX_ERROR;
83 }
84 out->uid = (uint32_t) id;
85
86 for (p = cred->vos; p != NULL && *p != '\0'
87 && out->ngids < PBLOCK_MAX_GIDS; /* void */) {
88 const char *end = strchr(p, ',');
89 size_t len = end != NULL ? (size_t) (end - p) : strlen(p);
90 char name[256];
91
92 while (len > 0 && p[0] == ' ') { p++; len--; }
93 while (len > 0 && p[len - 1] == ' ') { len--; }
94 if (len > 0 && len < sizeof(name)) {
95 memcpy(name, p, len);
96 name[len] = '\0';
97 if (pblock_catalog_id_map(st->cat, PBLOCK_ID_GROUP, name,
98 &id) != 0)
99 {
100 return NGX_ERROR;
101 }
102 out->gids[out->ngids++] = (uint32_t) id;
103 }
104 p = end != NULL ? end + 1 : "";
105 }
106
107 /* Primary gid: the first VO, else a user-private group. The id space is
108 * one sequence across kinds, so uid-as-gid collides with no VO's gid. */
109 out->gid = out->ngids > 0 ? out->gids[0] : out->uid;
110 return NGX_OK;
111}

113/* ---- POSIX mode-bit access check ------------------------------------------
114 *
115 * WHAT: Grant/deny `want` (OR of R_OK/W_OK/X_OK) on a catalog row per POSIX
116 * class selection: owner bits when ids->uid owns the row, group bits
117 * when the row's gid is any of the requester's VO gids (or the private
118 * group), other bits otherwise. NGX_OK, or NGX_ERROR with errno EACCES.
119 *
120 * WHY: This one function IS the enforcement model — everything else just
121 * decides which row and which `want` bits to hand it.
122 *
123 * HOW: Class selection is exclusive (an owner is judged by owner bits alone,
124 * like the kernel). `service` identities pass unconditionally.
125 */
126ngx_int_t
127pblock_ident_access(const pblock_meta *meta, const pblock_ids_t *ids,
128 int want)
129{
130 unsigned shift = 0;
131 int i;
132
133 if (ids->service) {
134 return NGX_OK;
135 }
136
137 if (meta->uid == ids->uid) {
138 shift = 6;
139 } else if (meta->gid == ids->gid) {
140 shift = 3;
141 } else {
142 for (i = 0; i < ids->ngids; i++) {
143 if (meta->gid == ids->gids[i]) {
144 shift = 3;
145 break;
146 }
147 }
148 }
149
150 if ((int) ((meta->mode >> shift) & 7 & (unsigned) want) != want) {
151 errno = EACCES;
152 return NGX_ERROR;
153 }
154 return NGX_OK;
155}

157/* ---- object gate: traverse + lookup + access in one call -------------------
158 *
159 * WHAT: Demand X (search) on `path`'s immediate parent, then look `path` up
160 * and demand `want` on it; fills *meta_out (optional). NGX_OK, or
161 * NGX_ERROR with errno (EACCES traverse/access denied, ENOENT absent).
162 *
163 * WHY: Nearly every slot starts with exactly this sequence; folding it keeps
164 * the slots one-screen and the error mapping in one place. The traverse
165 * check is what makes a 0770 group directory actually private: without
166 * it a world-readable file INSIDE the directory would still be reachable
167 * by non-members (mode bits are only checked on the entry itself).
168 *
169 * HOW: pblock_ident_check_parent(X_OK) — immediate parent only, matching the
170 * documented no-ancestor-walk simplification — then catalog lookup
171 * (rc 1 → ENOENT), then pblock_ident_access. Traverse denial reports
172 * EACCES before the entry's existence is consulted, like the kernel.
173 */
174ngx_int_t
175pblock_ident_check(pblock_state_t *st, const char *path,
176 const pblock_ids_t *ids, int want, pblock_meta *meta_out)
177{
178 pblock_meta meta;
179 int rc;
180
181 if (!ids->service
182 && pblock_ident_check_parent(st, path, ids, X_OK, NULL) != NGX_OK)
183 {
184 return NGX_ERROR;
185 }
186
187 rc = pblock_catalog_lookup(st->cat, path, &meta);
188 if (rc < 0) {
189 return NGX_ERROR;
190 }
191 if (rc == 1) {
192 errno = ENOENT;
193 return NGX_ERROR;
194 }
195 if (meta_out != NULL) {
196 *meta_out = meta;
197 }
198 return pblock_ident_access(&meta, ids, want);
199}

201/* ---- parent-directory gate -------------------------------------------------
202 *
203 * WHAT: Demand `want` (normally W_OK) on `path`'s immediate parent directory;
204 * fills *parent_out (optional, for the sticky gate). NGX_OK, or
205 * NGX_ERROR with errno (ENOENT/ENOTDIR from the lookup, EACCES denied).
206 *
207 * WHY: Create/remove/rename authority lives on the containing directory, not
208 * the entry (POSIX). Only the IMMEDIATE parent is consulted — no
209 * ancestor X-bit walk — a documented simplification: exports present a
210 * flat authority model where ancestor traversal is the VFS/export's
211 * concern, not per-op.
212 *
213 * HOW: A top-level entry's parent is the root, which usually has no catalog
214 * row: synthesize a service-owned, world-writable, STICKY directory
215 * (mode 01777, like /tmp) so multi-user exports work out of the box —
216 * anyone may create top-level entries, only owners may remove them.
217 * Operators wanting stricter top-level control mkdir real VO
218 * directories with real modes. Deeper paths use the real parent row
219 * (pblock_catalog_parent_lookup).
220 */
221ngx_int_t
222pblock_ident_check_parent(pblock_state_t *st, const char *path,
223 const pblock_ids_t *ids, int want, pblock_meta *parent_out)
224{
225 pblock_meta parent;
226 int rc;
227
228 if (path == NULL || path[0] != '/' || strchr(path + 1, '/') == NULL) {
229 /* parent is the root */
230 rc = pblock_catalog_lookup(st->cat, "/", &parent);
231 if (rc < 0) {
232 return NGX_ERROR;
233 }
234 if (rc == 1) {
235 memset(&parent, 0, sizeof(parent));
236 parent.is_dir = 1;
237 parent.mode = S_IFDIR | S_ISVTX | 0777;
238 }
239 } else if (pblock_catalog_parent_lookup(st->cat, path, &parent) != 0) {
240 return NGX_ERROR;
241 }
242
243 if (parent_out != NULL) {
244 *parent_out = parent;
245 }
246 return pblock_ident_access(&parent, ids, want);
247}

249/* ---- sticky-bit delete gate -------------------------------------------------
250 *
251 * WHAT: In a sticky (S_ISVTX) directory, only the entry's owner or the
252 * directory's owner may remove/rename the entry even with write
253 * permission on the directory. NGX_OK, or NGX_ERROR with errno EPERM.
254 *
255 * WHY: The synthetic root is sticky-world-writable (/tmp semantics): without
256 * this gate any authenticated user could delete any other user's
257 * top-level files. Real sticky directories created by operators get the
258 * same protection.
259 *
260 * HOW: Exactly the kernel's rule, minus the no-CAP_FOWNER cases pblock does
261 * not model. Service identities pass.
262 */
263ngx_int_t
264pblock_ident_sticky_gate(const pblock_meta *parent, const pblock_meta *entry,
265 const pblock_ids_t *ids)
266{
267 if (ids->service || (parent->mode & S_ISVTX) == 0
268 || entry->uid == ids->uid || parent->uid == ids->uid)
269 {
270 return NGX_OK;
271 }
272 errno = EPERM;
273 return NGX_ERROR;
274}
