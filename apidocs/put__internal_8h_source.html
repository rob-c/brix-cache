1/*
2 * put_internal.h - shared declarations for the WebDAV PUT split.
3 *
4 * The PUT path is split across three translation units by concern:
5 * put.c - orchestrator + finalize (body_inner / handle_put_body / commit)
6 * put_setup.c - precondition/setup (precheck / open_target / start_dashboard /
7 * resumable Content-Range branch)
8 * put_body.c - body-write concern (thread offload / codec / sync write / async
9 * done + checksum-on-ingest)
10 *
11 * Symbols DEFINED in one of those units and REFERENCED from another are declared
12 * here. Single-file symbols stay `static` in their own .c. Types come from the
13 * include block each .c copies (webdav.h + fs/vfs/vfs.h) ahead of this header.
14 */

18/*
19 * Outcome of a per-phase PUT helper, telling the orchestrator how to proceed.
20 * The helpers own their own error responses (metrics/status already sent), so
21 * the orchestrator only branches on whether to continue, stop, or (for the
22 * resumable branch) treat the request as fully handled.
23 */
24typedef enum {
25 WEBDAV_PUT_CONTINUE = 0, /* phase succeeded — proceed to the next phase */
26 WEBDAV_PUT_DONE /* request fully handled/responded — stop, return */
