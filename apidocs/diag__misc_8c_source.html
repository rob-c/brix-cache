8int
9do_probe_robustness(const diag_args *a)
10{
11 brix_url u;
12 brix_status st;
13 char ip[128];
14 char urlbuf[300];
15 int is_loop = 0, tmo;
16
17 brix_status_clear(&st);
18 if (brix_endpoint_parse(a->url, &u, &st) != 0) {
19 fprintf(stderr, "xrddiag: %s\n", st.msg);
20 return 50;
21 }
22 if (resolve_once(u.host, u.port, ip, sizeof(ip), &is_loop, &st) != 0) {
23 fprintf(stderr, "xrddiag: %s\n", st.msg);
24 return 51;
25 }
26 /* Safety gate: this is a fuzzing-class tool. Refuse a non-loopback resolved
27 * address unless the operator explicitly asserts authorisation. */
28 if (!is_loop && !a->authorized) {
29 fprintf(stderr,
30 "xrddiag: refusing to fuzz non-loopback target %s (%s) — this is "
31 "an adversarial auditor; re-run with --i-am-authorized only "
32 "against hosts you are authorised to test.\n", u.host, ip);
33 return 3;
34 }
35
36 tmo = a->probe_timeout_ms > 0 ? a->probe_timeout_ms : 5000;
37 snprintf(urlbuf, sizeof(urlbuf), "%s://%s:%d/",
38 a->conn.want_tls ? "roots" : "root", ip, u.port);
39 printf("probe-robustness %s (%s:%d)%s\n", u.host, ip, u.port,
40 is_loop ? "" : " [AUTHORISED non-loopback]");
41 printf("Probes (each must be REJECTED cleanly, server must survive):\n");
42
43 /* P1 — path-escape (well-formed; must be refused AND the conn survive). */
44 {
45 brix_conn c;
46 brix_status cs;
47 brix_statinfo si;
48 brix_status_clear(&cs);
49 if (probe_open(&c, urlbuf, a, tmo, &cs) == 0) {
50 brix_status est;
51 brix_statinfo s2;
52 brix_status s2t;
53 int served, alive;
54 brix_status_clear(&est);
55 brix_status_clear(&s2t);
56 served = (brix_stat(&c, "/../../../../../../etc/passwd", &si, &est) == 0);
57 /* survive: a normal stat still completes (transport intact). */
58 alive = (brix_stat(&c, "/", &s2, &s2t) == 0) || (s2t.kxr > 0);
59 probe("path-escape", !served && alive,
60 served ? "ESCAPE SERVED — confinement broken!"
61 : "refused (%s), connection alive",
62 served ? "" : brix_kxr_name(est.kxr));
63 brix_close(&c);
64 } else {
65 probe("path-escape", 0, "connect: %s", cs.msg);
66 }
67 }
68
69 /* P2 — unknown opcode (must be refused). */
70 {
71 brix_conn c;
72 brix_status cs;
73 brix_status_clear(&cs);
74 if (probe_open(&c, urlbuf, a, tmo, &cs) == 0) {
75 uint8_t hdr[24];
76 int v;
77 memset(hdr, 0, sizeof(hdr));
78 hdr[2] = 0x27; hdr[3] = 0x0f; /* requestid 9999 = not a kXR_* op */
79 v = raw_send_expect_reject(&c, hdr, NULL, 0, 0, 0);
80 probe("bad-opcode", v == 1,
81 v == 0 ? "SERVED an unknown opcode!" : "rejected");
82 brix_close(&c);
83 } else {
84 probe("bad-opcode", 0, "connect: %s", cs.msg);
85 }
86 }
87
88 /* P3 — oversized dlen (header claims ~1 GiB, no body): must reject/close,
89 * never buffer it (the server caps payload). */
90 {
91 brix_conn c;
92 brix_status cs;
93 brix_status_clear(&cs);
94 if (probe_open(&c, urlbuf, a, tmo, &cs) == 0) {
95 uint8_t hdr[24];
96 int v;
97 memset(hdr, 0, sizeof(hdr));
98 hdr[2] = (uint8_t) (kXR_stat >> 8);
99 hdr[3] = (uint8_t) (kXR_stat & 0xff);
100 v = raw_send_expect_reject(&c, hdr, NULL, 0, 1, 0x40000000u);
101 probe("oversized-dlen", v == 1,
102 v == 0 ? "accepted a 1 GiB dlen!" : "rejected/closed");
103 brix_close(&c);
104 } else {
105 probe("oversized-dlen", 0, "connect: %s", cs.msg);
106 }
107 }
108
109 /* P4 — read on a bogus file handle with a huge length (OOB): must reject. */
110 {
111 brix_conn c;
112 brix_status cs;
113 brix_status_clear(&cs);
114 if (probe_open(&c, urlbuf, a, tmo, &cs) == 0) {
115 uint8_t hdr[24];
116 int v;
117 memset(hdr, 0, sizeof(hdr));
118 hdr[2] = (uint8_t) (kXR_read >> 8);
119 hdr[3] = (uint8_t) (kXR_read & 0xff);
120 memset(hdr + 4, 0xff, 4); /* fhandle = never-opened */
121 hdr[16] = 0x7f; hdr[17] = 0xff; /* rlen = 0x7fffffff (huge) */
122 hdr[18] = 0xff; hdr[19] = 0xff;
123 v = raw_send_expect_reject(&c, hdr, NULL, 0, 0, 0);
124 probe("oob-read", v == 1,
125 v == 0 ? "served a read on a bogus handle!" : "rejected");
126 brix_close(&c);
127 } else {
128 probe("oob-read", 0, "connect: %s", cs.msg);
129 }
130 }
131
132 /* P5 — truncated/slowloris partial header (must not crash the server). */
133 {
134 brix_conn c;
135 brix_status cs;
136 brix_status_clear(&cs);
137 if (probe_open(&c, urlbuf, a, tmo, &cs) == 0) {
138 uint8_t half[12] = { 0 };
139 half[2] = (uint8_t) (kXR_stat >> 8);
140 half[3] = (uint8_t) (kXR_stat & 0xff);
141 (void) brix_write_full(&c.io, half, sizeof(half), &cs);
142 probe("partial-frame", 1, "sent 12/24 header bytes then closed");
143 brix_close(&c); /* abandon mid-frame */
144 } else {
145 probe("partial-frame", 0, "connect: %s", cs.msg);
146 }
147 }
148
149 /* Server-survives gate: a fresh session + stat must still work after the
150 * battery — proves nothing above crashed or wedged the server. */
151 {
152 brix_conn c;
153 brix_status cs;
154 brix_statinfo si;
155 brix_status_clear(&cs);
156 if (probe_open(&c, urlbuf, a, tmo, &cs) == 0) {
157 brix_status s2;
158 int alive;
159 brix_status_clear(&s2);
160 alive = (brix_stat(&c, "/", &si, &s2) == 0) || (s2.kxr > 0);
161 probe("server-survives", alive,
162 alive ? "fresh session OK after battery"
163 : "server unreachable after battery!");
164 brix_close(&c);
165 } else {
166 probe("server-survives", 0, "reconnect: %s", cs.msg);
167 }
168 }
169
170 printf("Result: %d failure(s)\n", g_fails);
171 return g_fails ? 1 : 0;
172}

197/* xrddiag srr <http[s]-url> — fetch the WLCG Storage Resource Reporting document
198 * and summarize the site's shares + capacity. Closes the SRR client gap. */
199int
200do_srr(const diag_args *a)
201{
202 dx_proto proto;
203 int tls, port;
204 char host[256], path[XRDC_PATH_MAX], name[128];
205 brix_http_resp r;
206 brix_status st;
207 int tmo = a->probe_timeout_ms > 0 ? a->probe_timeout_ms : 8000;
208
209 if (dx_url_parse(a->url, &proto, &tls, host, sizeof(host), &port, path,
210 sizeof(path)) != 0 || proto == DXP_ROOT) {
211 fprintf(stderr, "xrddiag: srr needs an http(s):// URL\n");
212 return 50;
213 }
214 if (path[0] == '\0' || strcmp(path, "/") == 0) {
215 snprintf(path, sizeof(path), "/.well-known/wlcg-storage-resource-reporting");
216 }
217 brix_status_clear(&st);
218 if (brix_http_req(host, port, tls, "GET", path, NULL, NULL, 0, tmo,
219 a->verify_tls, NULL, &r, &st) != 0) {
220 fprintf(stderr, "xrddiag: srr GET %s:%d%s: %s\n", host, port, path, st.msg);
221 return 51;
222 }
223 if (r.status != 200) {
224 fprintf(stderr, "xrddiag: srr returned HTTP %d\n", r.status);
225 brix_http_resp_free(&r);
226 return 51;
227 }
228 if (a->json) {
229 fwrite(r.body, 1, r.body_len, stdout);
230 } else {
231 long long total = js_sum(r.body, "totalsize");
232 long long used = js_sum(r.body, "usedsize");
233 int shares = js_count(r.body, "totalsize");
234 if (js_str(r.body, "implementation", name, sizeof(name))) {
235 printf("SRR: implementation=%s", name);
236 if (js_str(r.body, "implementationversion", name, sizeof(name))) {
237 printf(" %s", name);
238 }
239 printf("\n");
240 }
241 printf(" shares: %d\n", shares);
242 printf(" capacity: %lld bytes total, %lld used (%.1f%% full)\n",
243 total, used, total > 0 ? 100.0 * (double) used / (double) total : 0.0);
244 }
245 brix_http_resp_free(&r);
246 return 0;
247}

250/* xrddiag tape <http[s]-url//path> — drive the WLCG/FRM Tape REST API: POST a stage
251 * request for the path, then poll its status once and report locality. Closes the
252 * FRM Tape-REST client gap. */
253int
254do_tape(const diag_args *a)
255{
256 dx_proto proto;
257 int tls, port;
258 char host[256], path[XRDC_PATH_MAX], body[1024], reqid[128], state[32];
259 brix_http_resp r;
260 brix_status st;
261 int tmo = a->probe_timeout_ms > 0 ? a->probe_timeout_ms : 8000;
262
263 if (dx_url_parse(a->url, &proto, &tls, host, sizeof(host), &port, path,
264 sizeof(path)) != 0 || proto == DXP_ROOT || path[0] != '/'
265 || path[1] == '\0') {
266 fprintf(stderr, "xrddiag: tape needs an http(s):// URL with a file path\n");
267 return 50;
268 }
269 /* POST /api/v1/stage {"files":[{"path":"<path>"}]} — reject (don't silently
270 * truncate) an over-long path, which would stage the wrong file. */
271 if (strlen(path) > 900) {
272 fprintf(stderr, "xrddiag: tape path too long (max 900 bytes)\n");
273 return 50;
274 }
275 snprintf(body, sizeof(body), "{\"files\":[{\"path\":\"%s\"}]}", path);
276 brix_status_clear(&st);
277 if (brix_http_req(host, port, tls, "POST", "/api/v1/stage",
278 "Content-Type: application/json\r\n", body, strlen(body),
279 tmo, a->verify_tls, NULL, &r, &st) != 0) {
280 fprintf(stderr, "xrddiag: tape stage POST: %s\n", st.msg);
281 return 51;
282 }
283 if (r.status != 200 && r.status != 201) {
284 fprintf(stderr, "xrddiag: tape stage returned HTTP %d\n", r.status);
285 brix_http_resp_free(&r);
286 return 51;
287 }
288 if (!js_str(r.body, "id", reqid, sizeof(reqid))
289 && !js_str(r.body, "requestId", reqid, sizeof(reqid))) {
290 fprintf(stderr, "xrddiag: tape stage: no request id in response\n");
291 brix_http_resp_free(&r);
292 return 51;
293 }
294 state[0] = '\0';
295 (void) js_str(r.body, "state", state, sizeof(state));
296 printf("stage accepted: request-id=%s state=%s\n", reqid, state[0] ? state : "?");
297 brix_http_resp_free(&r);
298
299 /* poll GET /api/v1/stage/{id} once. */
300 {
301 char poll[256];
302 snprintf(poll, sizeof(poll), "/api/v1/stage/%s", reqid);
303 brix_status_clear(&st);
304 if (brix_http_req(host, port, tls, "GET", poll, NULL, NULL, 0, tmo,
305 a->verify_tls, NULL, &r, &st) == 0) {
306 if (r.status == 200) {
307 char ondisk[16];
308 state[0] = '\0';
309 (void) js_str(r.body, "state", state, sizeof(state));
310 ondisk[0] = '\0';
311 (void) js_str(r.body, "onDisk", ondisk, sizeof(ondisk));
312 printf("poll: state=%s onDisk=%s\n", state[0] ? state : "?",
313 ondisk[0] ? ondisk : "?");
314 }
315 brix_http_resp_free(&r); /* free on every successful request */
316 }
317 }
318 return 0;
319}
