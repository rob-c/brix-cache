24/* ---- credential-scoped namespace wrappers (Phase 2 Task 1) -----------------
25 *
26 * WHAT: One wrapper per supported ns op that opens a per-user origin session
27 * (via sd_xroot_session with a non-NULL cred) instead of the anonymous /
28 * service-credential session the plain slots use.
29 *
30 * WHY: A deny-mode request whose data-plane credential gate fires must also
31 * present the user proxy for every pre-flight probe and metadata mutation
32 * (unlink, rename, copy, xattr). Without these wrappers the probe stat
33 * still runs under the static service credential even when the protocol
34 * handler correctly denied the operation.
35 *
36 * HOW: Each wrapper delegates the operation body verbatim to the matching plain
37 * op helper but passes the cred to sd_xroot_session so the bootstrap uses
38 * the per-user proxy. The plain ops now call sd_xroot_session(…, NULL, …);
39 * these wrappers call sd_xroot_session(…, cred, …) — the only difference.
40 * stat_cred is implemented in sd_xroot.c (alongside sd_xroot_stat, which
41 * reuses sd_xroot_origin_open — a file-private type). */

46 const brix_sd_cred_t *cred)
47{

49 brix_cache_origin_conn_t oc;
50 brix_cache_fill_t *t;
51 int rc, e = 0;
52
53 if (sd_xroot_session(is->conf, cred, &oc, &t, &e) != 0) {
54 errno = e; return NGX_ERROR;
55 }
56 rc = is_dir ? brix_cache_origin_rmdir(t, &oc, path)

58 e = errno;
59 brix_cache_origin_close(&oc);
60 free(t);
61 errno = e;
62 return rc == 0 ? NGX_OK : NGX_ERROR;
63}

143/* listxattr_cred: enumerate extended attributes under the user's credential. */
144ssize_t
145sd_xroot_listxattr_cred(brix_sd_instance_t *inst, const char *path,
146 void *buf, size_t cap, const brix_sd_cred_t *cred)
147{
148 sd_xroot_inst_state *is = inst->state;
149 brix_cache_origin_conn_t oc;
150 brix_cache_fill_t *t;
151 char *raw;
152 const size_t rawcap = 65536;
153 ssize_t n;
154 size_t out = 0, i;
155 int e = 0;
156
157 raw = malloc(rawcap);
158 if (raw == NULL) { errno = ENOMEM; return -1; }
159 if (sd_xroot_session(is->conf, cred, &oc, &t, &e) != 0) {
160 free(raw); errno = e; return -1;
161 }
162 n = brix_cache_origin_listfattr(t, &oc, path, raw, rawcap);
163 e = errno;
164 brix_cache_origin_close(&oc);
165 free(t);
166 if (n < 0) { free(raw); errno = e; return -1; }
167
168 for (i = 0; i < (size_t) n; ) {
169 size_t nl = strnlen(raw + i, (size_t) n - i);
170
171 if (nl == 0) { i += 1; continue; }
172 if (buf != NULL && cap > 0) {
173 if (out + SD_XROOT_FATTR_PFX_LEN + nl + 1 > cap) {
174 free(raw); errno = ERANGE; return -1;
175 }
176 ngx_memcpy((char *) buf + out, SD_XROOT_FATTR_PFX,
177 SD_XROOT_FATTR_PFX_LEN);
178 ngx_memcpy((char *) buf + out + SD_XROOT_FATTR_PFX_LEN, raw + i, nl);
179 ((char *) buf)[out + SD_XROOT_FATTR_PFX_LEN + nl] = '\0';
180 }
181 out += SD_XROOT_FATTR_PFX_LEN + nl + 1;
182 i += nl + 1;
183 }
184 free(raw);
185 return (ssize_t) out;
186}
