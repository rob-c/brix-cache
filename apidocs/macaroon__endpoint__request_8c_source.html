26/*
27 * mac_iso_component — parse one "<digits><unit>" duration component of a
28 * restricted ISO-8601 duration at s[*ip] and accumulate its seconds into
29 * *total. Units D/H/M/S only; both the digit run and the running total are
30 * overflow-checked. Advances *ip past the unit letter. Returns NGX_OK /
31 * NGX_ERROR. Pure helper for mac_iso8601_secs — code motion only.
32 */
33static ngx_int_t
34mac_iso_component(const char *s, size_t len, size_t *ip, uint64_t *total)
35{
36 size_t i = *ip;
37 uint64_t v = 0;
38 size_t digits = 0;
39
40 while (i < len && s[i] >= '0' && s[i] <= '9') {
41 if (v > (UINT64_MAX - 9) / 10) { return NGX_ERROR; }
42 v = v * 10 + (uint64_t) (s[i] - '0');
43 i++; digits++;
44 }
45 if (digits == 0 || i >= len) { return NGX_ERROR; }
46 switch (s[i]) {
47 case 'D': *total += v * 86400; break;
48 case 'H': *total += v * 3600; break;
49 case 'M': *total += v * 60; break;
50 case 'S': *total += v; break;
51 default: return NGX_ERROR;
52 }
53 if (*total > (uint64_t) 0x7fffffffffffffffLL) { return NGX_ERROR; }
54 *ip = i + 1;
55 return NGX_OK;
56}

58/*
59 * mac_iso8601_secs — parse a restricted ISO-8601 duration ("PT1H","PT30M","P1D",
60 * "PT3600S") to seconds, clamped to [1,max]. Returns NGX_OK / NGX_ERROR. Units
61 * D/H/M/S only (W/Y/Mo unsupported — WLCG macaroons use these). EITHER.
62 */
63static ngx_int_t
64mac_iso8601_secs(const char *s, size_t len, time_t max, time_t *out)
65{
66 size_t i = 0;
67 uint64_t total = 0;
68 int saw_unit = 0;
69
70 if (s == NULL || out == NULL || len == 0) {
71 return NGX_ERROR;
72 }
73 if (s[i] == 'P') { i++; }
74 if (i < len && s[i] == 'T') { i++; }
75 while (i < len) {
76 if (mac_iso_component(s, len, &i, &total) != NGX_OK) {
77 return NGX_ERROR;
78 }
79 saw_unit = 1;
80 }
81 if (!saw_unit) { return NGX_ERROR; }
82 if (total == 0) { total = 1; }
83 if ((time_t) total > max) { total = (uint64_t) max; }
84 *out = (time_t) total;
85 return NGX_OK;
86}

100/*
101 * mac_caveat_apply — classify one "..." caveat array element:
102 * "activity:<csv>" is accumulated (comma-joined) into out->act,
103 * "path:<abs>" overwrites out->path (last one wins). Unknown caveat
104 * kinds are ignored. Bounded copies only — code motion from the scan.
105 */
106static void
107mac_caveat_apply(const u_char *p, size_t elen, mac_caveats_t *out)
108{
109 if (elen > 9
110 && ngx_strncasecmp((u_char *) p, (u_char *) "activity:", 9) == 0)
111 {
112 size_t cur = ngx_strlen(out->act);
113 size_t cp, room;
114 if (cur > 0 && cur + 1 < out->actsz) { out->act[cur++] = ','; }
115 cp = elen - 9;
116 room = (cur < out->actsz) ? out->actsz - cur - 1 : 0;
117 if (cp > room) { cp = room; }
118 ngx_memcpy(out->act + cur, p + 9, cp);
119 out->act[cur + cp] = '\0';
120 } else if (elen > 5
121 && ngx_strncasecmp((u_char *) p, (u_char *) "path:", 5) == 0)
122 {
123 size_t cp = elen - 5;
124 if (cp > out->pathsz - 1) { cp = out->pathsz - 1; }
125 ngx_memcpy(out->path, p + 5, cp);
126 out->path[cp] = '\0';
127 }
128}

130/*
131 * mac_caveats_scan — extract activity/path caveats from a dCache request body's
132 * "caveats":["activity:DOWNLOAD,LIST","path:/foo"] array. json_min has no array
133 * support, so this is a bounded string scan: for each "..." element it recognises
134 * "activity:<csv>" (accumulated, comma-joined into out->act) and "path:<abs>"
135 * (last one wins into out->path). Tolerant of whitespace; unknown caveat kinds
136 * ignored. EITHER.
137 */
138static void
139mac_caveats_scan(const char *body, size_t len, mac_caveats_t *out)
140{
141 const u_char *p = (const u_char *) body;
142 const u_char *end = p + len;
143 const u_char *arr;
144
145 out->act[0] = '\0';
146 out->path[0] = '\0';
147 arr = ngx_strlcasestrn((u_char *) p, (u_char *) end,
148 (u_char *) "\"caveats\"", sizeof("\"caveats\"") - 2);
149 if (arr == NULL) {
150 return;
151 }
152 p = ngx_strlchr((u_char *) arr, (u_char *) end, '[');
153 if (p == NULL) {
154 return;
155 }
156 p++;
157 while (p < end && *p != ']') {
158 const u_char *q;
159 if (*p != '"') { p++; continue; }
160 q = ++p;
161 while (q < end && *q != '"') { q++; }
162 if (q >= end) { break; }
163 mac_caveat_apply(p, (size_t) (q - p), out);
164 p = q + 1;
165 }
166}

168/*
169 * mac_request_respond — build and send the dCache issuance response:
170 * {macaroon, uri{target,targetWithMacaroon,base,baseWithMacaroon}}.
171 * Response-assembly order is frozen (behavioral freeze) — code motion only.
172 */
173static void
174mac_request_respond(ngx_http_request_t *r, const char *token_out)
175{
176 char json[BRIX_MACAROON_ISSUE_OUT_MAX + 768];
177 const char *scheme;
178 ngx_str_t host;
179 int n;
180
181 scheme = (r->connection->ssl != NULL) ? "https" : "http";
182 if (r->headers_in.host != NULL) {
183 host = r->headers_in.host->value;
184 } else {
185 ngx_str_set(&host, "localhost");
186 }
187 n = (int) (ngx_snprintf((u_char *) json, sizeof(json) - 1,
188 "{\"macaroon\":\"%s\",\"uri\":{"
189 "\"target\":\"%s://%V%V\","
190 "\"targetWithMacaroon\":\"%s://%V%V?authz=%s\","
191 "\"base\":\"%s://%V/\","
192 "\"baseWithMacaroon\":\"%s://%V/?authz=%s\"}}",
193 token_out,
194 scheme, &host, &r->uri,
195 scheme, &host, &r->uri, token_out,
196 scheme, &host,
197 scheme, &host, token_out) - (u_char *) json);
198 if (n < 0) { n = 0; }
199 mac_respond(r, NGX_HTTP_OK, json, (size_t) n);
200}

202void
203webdav_handle_macaroon_request(ngx_http_request_t *r)
204{
205 ngx_http_brix_webdav_loc_conf_t *conf;
206 u_char *body = NULL;
207 size_t body_len = 0;
208 u_char root_key[64];
209 ssize_t key_len;
210 char activities[128];
211 char cav_path[1024];
212 char location[256];
213 char identifier[96];
214 char token_out[BRIX_MACAROON_ISSUE_OUT_MAX];
215 time_t validity = 0, exp;
216 mac_caveats_t cavs;
217
218 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
219
220 /* Secret configured + authenticated + body read (order load-bearing) */
221 if (mac_gate_and_read_body(r, conf, 16384, &body, &body_len) != NGX_OK) {
222 return;
223 }
224
225 /* caveats[] → activities + path (default base = request path). */
226 cavs.act = activities;
227 cavs.actsz = sizeof(activities);
228 cavs.path = cav_path;
229 cavs.pathsz = sizeof(cav_path);
230 mac_caveats_scan((const char *) body, body_len, &cavs);
231 if (cav_path[0] == '\0') {
232 size_t cp = ngx_min(r->uri.len, sizeof(cav_path) - 1);
233 ngx_memcpy(cav_path, r->uri.data, cp);
234 cav_path[cp] = '\0';
235 }
236
237 /* Bearer-token callers cannot exceed their own scope. When activities
238 * is empty the issued macaroon carries no activity restriction, so we
239 * still require at least read scope on the base path to prevent
240 * zero-scope tokens from issuing unconstrained macaroons. */
241 if (mac_authorize(r, activities, cav_path) != NGX_OK) {
242 return;
243 }
244
245 /* validity (ISO-8601) → seconds, clamped to the configured max. */
246 {
247 char vbuf[32];
248 if (brix_json_get_str((const char *) body, body_len, "validity",
249 vbuf, sizeof(vbuf)) > 0)
250 {
251 (void) mac_iso8601_secs(vbuf, ngx_strlen(vbuf),
252 (time_t) conf->macaroon_max_validity,
253 &validity);
254 }
255 }
256 if (validity <= 0) { validity = (time_t) conf->macaroon_max_validity; }
257 exp = (time_t) ngx_time() + validity;
258
259 /* Unique identifier "v=1;t=<unix>;n=<16-hex-random>" (same scheme as the
260 * OAuth2 path — brix_macaroon_issue requires a non-NULL identifier). */
261 mac_make_identifier(r, identifier, sizeof(identifier));
262
263 key_len = brix_macaroon_secret_parse(
264 (const char *) conf->token_macaroon_secret.data,
265 conf->token_macaroon_secret.len, root_key, sizeof(root_key));
266 if (key_len <= 0) {
267 mac_respond(r, NGX_HTTP_INTERNAL_SERVER_ERROR,
268 J_SERVER_ERROR, sizeof(J_SERVER_ERROR) - 1);
269 return;
270 }
271
272 /* location: pin to the configured issuer (else macaroon_location, else
273 * Host) — same rule as the OAuth2 path so our own macaroon re-validates
274 * here. */
275 mac_build_location(r, conf, 1, location, sizeof(location));
276
277 if (brix_macaroon_issue(r->connection->log, root_key, (size_t) key_len,
278 location, identifier,
279 activities[0] ? activities : NULL,
280 cav_path, exp, token_out, sizeof(token_out)) != NGX_OK)
281 {
282 mac_respond(r, NGX_HTTP_INTERNAL_SERVER_ERROR,
283 J_SERVER_ERROR, sizeof(J_SERVER_ERROR) - 1);
284 return;
285 }
286
287 /* dCache response: {macaroon, uri{target,targetWithMacaroon,base,baseWithMacaroon}} */
288 mac_request_respond(r, token_out);
289}
