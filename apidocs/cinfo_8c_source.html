1/*
2 * cinfo.c — block-present bitmap sidecar ("<cachefile>.cinfo"). See cinfo.h.
3 *
4 * The header is written/read verbatim (native byte order on the x86-64 target,
5 * matching meta.c); the bitmap follows it on disk. Block recording serialises
6 * its read-modify-write with flock(2) so that concurrent slice-fill workers —
7 * each completing a different window of the same file — never lose one another's
8 * bits. A torn or inconsistent sidecar is treated by the loader as "nothing
9 * recorded" (NGX_DECLINED), so a crash mid-write is always safe: the affected
10 * blocks merely look absent and are refetched.
11 */
12

49int
50brix_cache_cinfo_block_present(const uint8_t *bitmap, uint64_t blk)
51{
52 return (bitmap[blk >> 3] >> (blk & 7u)) & 1u;

55uint64_t
56brix_cache_cinfo_present_count(const uint8_t *bitmap, uint64_t nblocks)
57{
58 uint64_t count = 0;
59 uint64_t blk;
60
61 for (blk = 0; blk < nblocks; blk++) {
62 if (brix_cache_cinfo_block_present(bitmap, blk)) {
63 count++;
64 }
65 }
66 return count;
67}

69void
70brix_cache_cinfo_refresh_flags(brix_cache_cinfo_t *hdr, const uint8_t *bitmap)
71{
72 uint64_t present;
73
74 hdr->flags &= (uint16_t) ~(BRIX_CINFO_F_COMPLETE | BRIX_CINFO_F_PARTIAL);
75
76 if (hdr->nblocks == 0) {
77 hdr->flags |= BRIX_CINFO_F_COMPLETE; /* an empty file is trivially whole */
78 return;
79 }
80 present = brix_cache_cinfo_present_count(bitmap, hdr->nblocks);
81 if (present == hdr->nblocks) {
82 hdr->flags |= BRIX_CINFO_F_COMPLETE;
83 } else if (present > 0) {
84 hdr->flags |= BRIX_CINFO_F_PARTIAL;
85 }
86}

98ngx_int_t
99brix_cache_cinfo_to_xmeta(const brix_cache_cinfo_t *hdr,
100 const uint8_t *bitmap, size_t bitmap_len, brix_xmeta_t *m)
101{
102 uint32_t bs = hdr->block_size ? hdr->block_size : BRIX_CACHE_DIRTY_BLOCK;
103 size_t need;
104
105 if (brix_xmeta_init(m, (int64_t) hdr->size, (int64_t) bs)
106 != BRIX_XMETA_OK)
107 {
108 return NGX_ERROR;
109 }
110 m->have_blockcrc = 0; /* block CRCs are CSI state (P3) */
111 m->access_cnt = hdr->access_count;
112 if (hdr->last_access != 0 || hdr->bytes_served != 0) {
113 m->astat_count = 1;
114 m->astat.attach_time = (int64_t) hdr->last_access;
115 m->astat.detach_time = (int64_t) hdr->last_access;
116 m->astat.bytes_hit = (int64_t) hdr->bytes_served;
117 }
118 m->origin_mtime = hdr->mtime;
119 m->dirty_lo = hdr->dirty_lo;
120 m->dirty_hi = hdr->dirty_hi;
121 m->dirty_since = hdr->dirty_since;
122 m->flush_gen = hdr->flush_gen;
123 m->last_flush = hdr->last_flush;
124 m->bytes_flushed = hdr->bytes_flushed;
125 m->expires_at = hdr->expires_at;
126 m->filled_at = hdr->filled_at;
127 m->mode = hdr->mode;
128 m->state_flags =
129 ((hdr->flags & BRIX_CINFO_F_VERIFIED) ? BRIX_XMETA_F_VERIFIED : 0u)
130 | ((hdr->flags & BRIX_CINFO_F_EXPIRES) ? BRIX_XMETA_F_EXPIRES : 0u);
131 /* Clamp the validity-string lengths at this boundary: a hand-built or
132 * corrupted header must not be able to drive an over-long encode. */
133 m->etag_len = (hdr->etag_len <= sizeof(hdr->etag)) ? hdr->etag_len : 0;
134 ngx_memcpy(m->etag, hdr->etag, m->etag_len);
135 m->cks_alg_len = (hdr->cks_alg_len <= sizeof(hdr->cks_alg))
136 ? hdr->cks_alg_len : 0;
137 ngx_memcpy(m->cks_alg, hdr->cks_alg, m->cks_alg_len);
138 m->cks_len = (hdr->cks_len <= sizeof(hdr->cks_hex)) ? hdr->cks_len : 0;
139 ngx_memcpy(m->cks_hex, hdr->cks_hex, m->cks_len);
140
141 /* Present bitmap: adopt the caller's when it covers this geometry;
142 * otherwise synthesize whole-file state from the COMPLETE flag. */
143 need = brix_cache_cinfo_bitmap_len(m->nblocks);
144 if (bitmap != NULL && bitmap_len >= need && need > 0) {
145 ngx_memcpy(m->bitmap, bitmap, need);
146 } else if ((hdr->flags & BRIX_CINFO_F_COMPLETE) && need > 0) {
147 ngx_memset(m->bitmap, 0xFF, need);
148 }
149 return NGX_OK;
150}

198/* ---- raw path carrier (shared: fs/meta/xmeta_path.c) -----------------------
199 * The record rides in the data file's user.xrd.cinfo xattr when it fits, else
200 * as the stock-readable "<path>.cinfo" sidecar; CSI and the cache persist into
201 * the SAME record under the SAME per-file lock. Thin NGX-return wrappers. */

203static ngx_int_t
204cinfo_xmeta_read(const char *cache_path, brix_xmeta_t *xm)
205{
206 int rc = brix_xmeta_path_load(cache_path, xm);
207
208 if (rc == BRIX_XMETA_OK) {
209 return NGX_OK;
210 }
211 return (rc == BRIX_XMETA_FOREIGN) ? NGX_DECLINED : NGX_ERROR;
212}

214static ngx_int_t
215cinfo_xmeta_write(const char *cache_path, const brix_xmeta_t *xm)
216{
217 return (brix_xmeta_path_save(cache_path, xm) == BRIX_XMETA_OK)
218 ? NGX_OK : NGX_ERROR;
219}

233ngx_int_t
234brix_cache_cinfo_load(const char *cache_path, brix_cache_cinfo_t *hdr,
235 uint8_t **bitmap, size_t *bitmap_len)
236{
237 brix_xmeta_t xm;
238 size_t blen;
239 ngx_int_t rc;
240
241 if (cache_path == NULL || hdr == NULL || bitmap == NULL
242 || bitmap_len == NULL)
243 {
244 errno = EINVAL;
245 return NGX_ERROR;
246 }
247 *bitmap = NULL;
248 *bitmap_len = 0;
249
250 rc = cinfo_xmeta_read(cache_path, &xm);
251 if (rc != NGX_OK) {
252 return rc;
253 }
254 brix_cache_cinfo_from_xmeta(&xm, hdr);
255
256 blen = brix_cache_cinfo_bitmap_len(hdr->nblocks);
257 if (blen > 0) {
258 uint8_t *bits = malloc(blen);
259
260 if (bits == NULL) {
261 brix_xmeta_free(&xm);
262 errno = ENOMEM;
263 return NGX_ERROR;
264 }
265 ngx_memcpy(bits, xm.bitmap, blen);
266 *bitmap = bits;
267 *bitmap_len = blen;
268 }
269 brix_xmeta_free(&xm);
270 return NGX_OK;
271}

340ngx_int_t
341brix_cache_cinfo_record_block(const char *cache_path, uint64_t size,
342 uint32_t block_size, uint64_t mtime, uint32_t mode, uint64_t blk,
343 ngx_log_t *log)
344{
345 brix_cache_cinfo_t hdr;
346 brix_xmeta_t xm;
347 uint64_t nblocks;
348 int lfd;
349 ngx_int_t rc;
350
351 (void) log;
352
353 if (cache_path == NULL || block_size == 0) {
354 errno = EINVAL;
355 return NGX_ERROR;
356 }
357 nblocks = brix_cache_cinfo_nblocks(size, block_size);
358 if (blk >= nblocks) {
359 errno = ERANGE;
360 return NGX_ERROR; /* block out of range for this size */
361 }
362
363 lfd = cinfo_rmw_lock(cache_path);
364 if (lfd < 0) {
365 return NGX_ERROR;
366 }
367
368 /* Adopt the recorded state only if it describes the SAME origin file; an
369 * absent/garbage/stale (changed origin) record starts the bitmap fresh. */
370 if (cinfo_xmeta_read(cache_path, &xm) == NGX_OK) {
371 if ((uint64_t) xm.file_size == size
372 && (uint64_t) xm.buffer_size == block_size
373 && xm.origin_mtime == mtime)
374 {
375 brix_cache_cinfo_from_xmeta(&xm, &hdr);
376 } else {
377 brix_xmeta_free(&xm);
378 cinfo_init(&hdr, size, block_size, mtime);
379 if (brix_cache_cinfo_to_xmeta(&hdr, NULL, 0, &xm) != NGX_OK) {
380 cinfo_rmw_unlock(lfd);
381 return NGX_ERROR;
382 }
383 }
384 } else {
385 cinfo_init(&hdr, size, block_size, mtime);
386 if (brix_cache_cinfo_to_xmeta(&hdr, NULL, 0, &xm) != NGX_OK) {
387 cinfo_rmw_unlock(lfd);
388 return NGX_ERROR;
389 }
390 }
391
392 if (mode != 0) {
393 xm.mode = mode; /* origin perms (0 = caller has none) */
394 }
395 brix_xmeta_block_set(&xm, blk);
396
397 rc = cinfo_xmeta_write(cache_path, &xm);
398 brix_xmeta_free(&xm);
399 cinfo_rmw_unlock(lfd);
400 return rc;
401}

403/* ---- write-back (dirty) record-keeping --------------------------------- */
404
405/* Under the per-file RMW lock, load the record describing the SAME origin
406 * file (size/mtime/block_size) into *xm; an absent/garbage/stale record
407 * starts fresh (clean dirty state, empty bitmap). NGX_OK / NGX_ERROR. */
408static ngx_int_t
409cinfo_rmw_load(const char *cache_path, uint64_t size, uint32_t block_size,
410 uint64_t mtime, brix_xmeta_t *xm)
411{
412 brix_cache_cinfo_t fresh;
413
414 if (cinfo_xmeta_read(cache_path, xm) == NGX_OK) {
415 if ((uint64_t) xm->file_size == size
416 && (uint64_t) xm->buffer_size == block_size
417 && xm->origin_mtime == mtime)
418 {
419 return NGX_OK;
420 }
421 brix_xmeta_free(xm);
422 }
423 cinfo_init(&fresh, size, block_size, mtime);
424 return brix_cache_cinfo_to_xmeta(&fresh, NULL, 0, xm);
425}

427ngx_int_t
428brix_cache_cinfo_mark_dirty(const char *cache_path, uint64_t size,
429 uint32_t block_size, uint64_t mtime, uint64_t off, uint64_t len,
430 ngx_log_t *log)
431{
432 brix_xmeta_t xm;
433 int lfd;
434 ngx_int_t rc;
435
436 (void) log;
437
438 if (cache_path == NULL || block_size == 0 || len == 0) {
439 errno = EINVAL;
440 return NGX_ERROR;
441 }
442 lfd = cinfo_rmw_lock(cache_path);
443 if (lfd < 0) {
444 return NGX_ERROR;
445 }
446 if (cinfo_rmw_load(cache_path, size, block_size, mtime, &xm) != NGX_OK) {
447 cinfo_rmw_unlock(lfd);
448 return NGX_ERROR;
449 }
450
451 if (xm.dirty_lo >= xm.dirty_hi) { /* clean -> dirty */
452 xm.dirty_lo = off;
453 xm.dirty_hi = off + len;
454 xm.dirty_since = (uint64_t) time(NULL);
455 } else { /* widen; keep dirty_since */
456 if (off < xm.dirty_lo) {
457 xm.dirty_lo = off;
458 }
459 if (off + len > xm.dirty_hi) {
460 xm.dirty_hi = off + len;
461 }
462 }
463
464 rc = cinfo_xmeta_write(cache_path, &xm);
465 brix_xmeta_free(&xm);
466 cinfo_rmw_unlock(lfd);
467 return rc;
468}

470ngx_int_t
471brix_cache_cinfo_mark_clean(const char *cache_path, uint64_t bytes,
472 ngx_log_t *log)
473{
474 brix_xmeta_t xm;
475 int lfd;
476 ngx_int_t rc;
477
478 (void) log;
479
480 if (cache_path == NULL) {
481 errno = EINVAL;
482 return NGX_ERROR;
483 }
484 lfd = cinfo_rmw_lock(cache_path);
485 if (lfd < 0) {
486 return NGX_ERROR;
487 }
488 rc = cinfo_xmeta_read(cache_path, &xm);
489 if (rc != NGX_OK) {
490 cinfo_rmw_unlock(lfd);

492 }
493
494 xm.dirty_lo = 0;
495 xm.dirty_hi = 0;
496 xm.dirty_since = 0;
497 xm.flush_gen += 1;
498 xm.last_flush = (uint64_t) time(NULL);
499 xm.bytes_flushed += bytes;
500
501 rc = cinfo_xmeta_write(cache_path, &xm);
502 brix_xmeta_free(&xm);
503 cinfo_rmw_unlock(lfd);
504 return rc;
505}

507ngx_int_t
508brix_cache_cinfo_dirty_extent(const char *cache_path, uint64_t *lo,
509 uint64_t *hi, uint64_t *dirty_since)
510{
511 brix_cache_cinfo_t h;
512 uint8_t *bm = NULL;
513 size_t bl = 0;
514 ngx_int_t rc;
515
516 rc = brix_cache_cinfo_load(cache_path, &h, &bm, &bl);
517 if (rc != NGX_OK) {

520 free(bm);
521

523 return NGX_DECLINED; /* clean */
524 }
525 if (lo != NULL) {
526 *lo = h.dirty_lo;
527 }
528 if (hi != NULL) {
529 *hi = h.dirty_hi;
530 }
531 if (dirty_since != NULL) {
532 *dirty_since = h.dirty_since;
533 }
534 return NGX_OK;
535}
