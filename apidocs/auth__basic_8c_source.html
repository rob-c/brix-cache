1/*
2 * auth_basic.c — HTTP Basic password authentication for WebDAV/HTTP exports.
3 *
4 * WHAT: Verifies an `Authorization: Basic <base64 user:password>` header
5 * against the same PBKDF2 password db (`user:salthex:hashhex[:vo,..]`)
6 * the stream-side `brix_auth pwd` uses, and stamps the request identity
7 * (dn = username, BRIX_AUTHN_PWD, VO list from the entry's 4th field).
8 *
9 * WHY: Gives http(s):// and dav(s):// clients the same low-friction
10 * username/password entry point root:// has (curl -u, browsers, simple
11 * tooling) without inventing a second credential store. Password auth
12 * is poor practice for production — the config-time summary warns
13 * loudly — but it is invaluable for labs and first-run setups.
14 *
15 * HOW: Opt-in via `brix_webdav_pwd_file`; returns NGX_DECLINED when
16 * unconfigured, when no Basic header is present, or when the credential
17 * fails verification (so the auth policy in access.c decides between
18 * anonymous fallback and 403). Verification is brix_pwd_file_lookup +
19 * brix_pwd_verify (PBKDF2-HMAC-SHA1, constant-time compare). On success
20 * the req ctx is created/marked verified exactly like the cert/token
21 * tiers, so authdb/VO ACLs and identity-aware backends (e.g. pblock's
22 * catalog-internal ownership) see one canonical identity object.
23 */

32/*
33 * WHAT: Extract and base64-decode the Basic credential from the request's
34 * Authorization header into a NUL-terminated `user` cstring and a
35 * password byte range. NGX_OK, or NGX_DECLINED (absent header,
36 * different scheme, malformed base64/credential) — never an error the
37 * caller must distinguish, so other credential tiers can still run.
38 *
39 * WHY: RFC 7617 credential parsing kept out of the verify path so the crypto
40 * step reads linearly; the decoded buffer is pool-owned, letting the
41 * username be NUL-terminated in place for the C-string pwd-file API.
42 *
43 * HOW: Case-insensitive "Basic" match, skip the separating spaces, decode
44 * with ngx_decode_base64, split at the FIRST ':' (passwords may contain
45 * colons). An empty username is rejected; an empty password is left to
46 * the verifier (it will not match any stored hash).
47 */
48static ngx_int_t
49wb_parse_basic_userpass(ngx_http_request_t *r, ngx_str_t *user,
50 ngx_str_t *pass)
51{
52 ngx_table_elt_t *h = r->headers_in.authorization;
53 ngx_str_t b64, plain;
54 u_char *colon;
55
56 if (h == NULL || h->value.len < sizeof("Basic ") - 1
57 || ngx_strncasecmp(h->value.data, (u_char *) "Basic ",
58 sizeof("Basic ") - 1) != 0)
59 {
60 return NGX_DECLINED;
61 }
62
63 b64.data = h->value.data + sizeof("Basic ") - 1;
64 b64.len = h->value.len - (sizeof("Basic ") - 1);
65 while (b64.len > 0 && b64.data[0] == ' ') {
66 b64.data++;
67 b64.len--;
68 }
69 if (b64.len == 0) {
70 return NGX_DECLINED;
71 }
72
73 plain.data = ngx_pnalloc(r->pool, ngx_base64_decoded_length(b64.len) + 1);
74 if (plain.data == NULL) {
75 return NGX_DECLINED;
76 }
77 if (ngx_decode_base64(&plain, &b64) != NGX_OK) {
78 return NGX_DECLINED;
79 }
80 plain.data[plain.len] = '\0';
81
82 colon = memchr(plain.data, ':', plain.len);
83 if (colon == NULL || colon == plain.data) {
84 return NGX_DECLINED;
85 }
86
87 user->data = plain.data;
88 user->len = (size_t) (colon - plain.data);
89 *colon = '\0';
90 pass->data = colon + 1;
91 pass->len = plain.len - user->len - 1;
92 return NGX_OK;
93}

95/*
96 * WHAT: Get-or-create the WebDAV request context with an attached identity.
97 * Returns the ctx or NULL on allocation failure.
98 *
99 * WHY: Basic auth may be the FIRST credential tier that runs on a plain-http
100 * listener (no TLS handshake created a ctx), so it must be able to
101 * create the ctx the later authz/backends stages read.
102 *
103 * HOW: Mirrors the cert/token tiers' ensure step: ngx_http_get_module_ctx,
104 * allocate+attach on NULL, back-fill a missing identity.
105 */
106static ngx_http_brix_webdav_req_ctx_t *
107wb_ensure_ctx(ngx_http_request_t *r)
108{
109 ngx_http_brix_webdav_req_ctx_t *ctx;
110
111 ctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
112 if (ctx == NULL) {
113 ctx = ngx_pcalloc(r->pool, sizeof(*ctx));
114 if (ctx == NULL) {
115 return NULL;
116 }
117 ngx_http_set_ctx(r, ctx, ngx_http_brix_webdav_module);
118 }
119 if (ctx->identity == NULL) {
120 ctx->identity = brix_identity_alloc(r->pool);
121 if (ctx->identity == NULL) {
122 return NULL;
123 }
124 }
125 return ctx;
126}

128/*
129 * WHAT: The Basic-password auth gate: verify the request's Basic credential
130 * against conf->pwd_file and mark the request authenticated (dn =
131 * username, BRIX_AUTHN_PWD, VOs from the entry). NGX_OK on success;
132 * NGX_DECLINED when unconfigured/absent/invalid (policy decides);
133 * NGX_HTTP_INTERNAL_SERVER_ERROR on allocation failure.
134 *
135 * WHY: Completes credential parity with root:// so operators can test every
136 * protocol scheme with one pwd db; runs LAST in the tier order (cert,
137 * token, basic) so stronger credentials always win.
138 *
139 * HOW: Parse header → pwd-file lookup → constant-time PBKDF2 verify → stamp
140 * req ctx + identity. A failed verify logs at NOTICE and counts a
141 * failed BRIX_AUTHN_PWD auth, matching the stream-side behaviour.
142 */
143ngx_int_t
144webdav_verify_basic_pwd(ngx_http_request_t *r,
145 ngx_http_brix_webdav_loc_conf_t *conf)
146{
147 ngx_http_brix_webdav_req_ctx_t *ctx;
148 ngx_str_t user, pass;
149 char path[PATH_MAX];
150 uint8_t salt[BRIX_PWD_MAX_SALT];
151 uint8_t hash[BRIX_PWD_HASH_LEN];
152 size_t saltlen = 0, hashlen = 0;
153 char vos[512];
154 int verified = 0;
155
156 if (conf->pwd_file.len == 0 || conf->pwd_file.len >= sizeof(path)) {
157 return NGX_DECLINED;
158 }
159 if (wb_parse_basic_userpass(r, &user, &pass) != NGX_OK) {
160 return NGX_DECLINED;
161 }
162
163 ngx_memcpy(path, conf->pwd_file.data, conf->pwd_file.len);
164 path[conf->pwd_file.len] = '\0';
165
166 vos[0] = '\0';
167 if (brix_pwd_file_lookup(path, (const char *) user.data, salt, &saltlen,
168 hash, &hashlen, vos, sizeof(vos)) == NGX_OK)
169 {
170 verified = brix_pwd_verify(pass.data, pass.len, salt, saltlen,
171 hash, hashlen);
172 }
173 ngx_memzero(salt, sizeof(salt));
174 ngx_memzero(hash, sizeof(hash));
175
176 if (!verified) {
177 ngx_log_error(NGX_LOG_NOTICE, r->connection->log, 0,
178 "brix_webdav: Basic pwd auth denied for user \"%s\"",
179 user.data);
180 brix_metric_auth(BRIX_PROTO_WEBDAV, BRIX_AUTHN_PWD, 0);
181 return NGX_DECLINED;
182 }
183
184 ctx = wb_ensure_ctx(r);
185 if (ctx == NULL) {
186 return NGX_HTTP_INTERNAL_SERVER_ERROR;
187 }
188
189 ngx_cpystrn((u_char *) ctx->dn, user.data, sizeof(ctx->dn));
190 ctx->verified = 1;
191 ctx->auth_source = "basic";
192
193 if (brix_identity_set_dn(ctx->identity, r->pool, ctx->dn,
194 BRIX_AUTHN_PWD) != NGX_OK
195 || (vos[0] != '\0'
196 && brix_identity_set_vos_csv(ctx->identity, r->pool,
197 vos) != NGX_OK))
198 {
199 return NGX_HTTP_INTERNAL_SERVER_ERROR;
200 }
201
202 ngx_log_error(NGX_LOG_INFO, r->connection->log, 0,
203 "brix_webdav: Basic pwd auth OK user=\"%s\"", ctx->dn);
204 return NGX_OK;
205}
