73/*
74 * s3_delete_xml_append_error — append one per-key <Error> element (Key/Code/
75 * Message) to the buffered DeleteResult XML.
76 *
77 * WHY: S3 DeleteObjects reports per-key failures inline in the batch result
78 * rather than aborting the whole request; each failed key contributes an
79 * <Error> block with its escaped key plus the S3 code/message pair.
80 * HOW: Emit the open tag, the escaped Key/Code/Message text elements, then the
81 * close tag — short-circuiting to NGX_ERROR if any append overflows the
82 * pre-sized buffer.
83 */
84ngx_int_t
85s3_delete_xml_append_error(ngx_buf_t *xml_buf, size_t *xml_len,
86 const u_char *key, size_t key_len, const s3_del_err_t *err)
87{
88 return s3_delete_xml_append_raw(xml_buf, xml_len, "<Error>") == NGX_OK

90 == NGX_OK
91 && s3_delete_xml_append_elem(xml_buf, xml_len, "Code",
92 (const u_char *) err->code,
