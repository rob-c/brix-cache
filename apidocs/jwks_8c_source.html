1/* JWKS (JSON Web Key Set) loading from disk — parse RSA and EC P-256 public keys for OIDC token verification.
2 *
3 * WHAT: Loads a JWKS JSON file from disk, parses the "keys" array to extract RSA (n/e) and EC P-256 (crv/x/y)
4 * key parameters, converts them into EVP_PKEY objects via OpenSSL, stores kid + pkey pairs in caller-supplied
5 * keys array. Also provides cleanup function to free all loaded EVP_PKEY handles.
6 *
7 * WHY: OIDC token verification requires the public key matching the token's "kid" header claim. JWKS is the
8 * standard format for publishing a provider's rotating set of public keys. Loading from disk at startup lets
9 * the server verify bearer tokens without HTTP round-trips to the provider's JWKS endpoint.
10 *
11 * HOW: brix_jwks_load() — fopen(path,"r"), fcntl FD_CLOEXEC, fseek/ftell for size validation (0 < fsize ≤ 65536),
12 * malloc(fsize+1), fread full file content, null-terminate buf, call brix_jwks_load_jansson(log,path,buf,fsize,keys,max_keys),
13 * free(buf), return count. brix_jwks_free() — iterate i=0..count, EVP_PKEY_free non-null pkey, nullify keys[i].pkey. */

25static int
26brix_jwks_load_jansson(ngx_log_t *log, const char *path,
27 const char *buf, size_t len, brix_jwks_key_t *keys, int max_keys)
28{
29 json_error_t err;
30 json_t *root, *arr, *item;
31 size_t index;
32 int count;
33
34 root = json_loadb(buf, len, 0, &err);
35 if (root == NULL || !json_is_object(root)) {
36 if (root != NULL) {
37 json_decref(root);
38 }
39 BRIX_DIAG_ERR(log, 0,
40 "brix_token: JWKS \"%s\" is not valid JSON: %s",
41 "the JWKS file is truncated, HTML (an error page), or was fetched "
42 "to the wrong path",
43 "verify the file holds the IdP's real JWKS JSON; until it parses, "
44 "ALL bearer-token authentication is rejected",
45 path, err.text);
46 return -1;
47 }
48
49 arr = json_object_get(root, "keys");
50 if (!json_is_array(arr)) {
51 json_decref(root);
52 BRIX_DIAG_ERR(log, 0,
53 "brix_token: JWKS has no \"keys\" array",
54 "the JSON parsed but is not a JWKS document (no top-level "
55 "\"keys\": [...] member)",
56 "point the JWKS setting at the IdP's jwks_uri output; a bare "
57 "single JWK or an OIDC discovery document will not work");
58 return -1;
59 }
60
61 count = 0;
62 json_array_foreach(arr, index, item) {
63 const char *kty, *kid, *n_b64, *e_b64, *crv, *x_b64, *y_b64;
64 json_t *v;
65
66 if (!json_is_object(item) || count >= max_keys) {
67 continue;
68 }
69
70#define JWKS_STRING_FIELD(obj, field) \
71 (v = json_object_get((obj), (field)), json_is_string(v) ? json_string_value(v) : "")
72
73 kty = JWKS_STRING_FIELD(item, "kty");
74 kid = JWKS_STRING_FIELD(item, "kid");
75 n_b64 = JWKS_STRING_FIELD(item, "n");
76 e_b64 = JWKS_STRING_FIELD(item, "e");
77 crv = JWKS_STRING_FIELD(item, "crv");
78 x_b64 = JWKS_STRING_FIELD(item, "x");
79 y_b64 = JWKS_STRING_FIELD(item, "y");
80
81#undef JWKS_STRING_FIELD
82
83 if (strcmp(kty, "RSA") == 0 && n_b64[0] && e_b64[0]) {
84 EVP_PKEY *pkey;
85
86 pkey = brix_token_rsa_pubkey_from_ne(n_b64, strlen(n_b64),
87 e_b64, strlen(e_b64),
88 log);
89 if (pkey != NULL) {
90 ngx_cpystrn((u_char *) keys[count].kid,
91 (u_char *) kid, sizeof(keys[count].kid));
92 keys[count].pkey = pkey;
93 count++;
94 ngx_log_error(NGX_LOG_INFO, log, 0,
95 "brix_token: loaded RSA JWKS key kid=\"%s\"",
96 kid);
97 }
98 } else if (strcmp(kty, "EC") == 0 && strcmp(crv, "P-256") == 0
99 && x_b64[0] && y_b64[0])
100 {
101 EVP_PKEY *pkey;
102
103 pkey = brix_token_ec_pubkey_from_xy(x_b64, strlen(x_b64),
104 y_b64, strlen(y_b64),
105 log);
106 if (pkey != NULL) {
107 ngx_cpystrn((u_char *) keys[count].kid,
108 (u_char *) kid, sizeof(keys[count].kid));
109 keys[count].pkey = pkey;
110 count++;
111 ngx_log_error(NGX_LOG_INFO, log, 0,
112 "brix_token: loaded EC P-256 JWKS key kid=\"%s\"",
113 kid);
114 }
115 } else {
116 ngx_log_error(NGX_LOG_WARN, log, 0,
117 "brix_token: skipping JWKS key kty=\"%s\" crv=\"%s\" "
118 "(only RSA and EC P-256 supported)", kty, crv);
119 }
120 }
121
122 json_decref(root);
123 ngx_log_error(NGX_LOG_NOTICE, log, 0,
124 "brix_token: loaded %d JWKS key(s) from \"%s\" using jansson",
125 count, path);
126 return count;
127}

129/* WHAT: Static helper — parse JWKS JSON from memory buffer into EVP_PKEY key array.
130 * WHY: Separates the Jansson-specific parsing logic from disk I/O so brix_jwks_load() can swap backends
131 * (e.g., file-based vs HTTP-fetched JWKS) without changing parser internals.
132 * HOW: json_loadb() parses buffer into json_t root; validate root is object then extract "keys" array (return -1 if missing);
133 * json_array_foreach() iterates each item, extracting kty/kid/n_b64/e_b64/crv/x_b64/y_b64 via JWKS_STRING_FIELD macro;
134 * for RSA keys: call brix_token_rsa_pubkey_from_ne(n_b64,e_b64) → EVP_PKEY; for EC P-256: call brix_token_ec_pubkey_from_xy(x_b64,y_b64);
135 * store kid+pkey into keys[count], increment count, skip unsupported kty with WARN log; json_decref(root), return count. */

137int
138brix_jwks_load(ngx_log_t *log, const char *path,
139 brix_jwks_key_t *keys, int max_keys)
140/* WHAT: Load JWKS public keys from disk file into caller-supplied key array for OIDC token verification.
141 * WHY: Provides the startup-time key loading entry point — callers open a JWKS JSON file and receive parsed
142 * EVP_PKEY handles indexed by kid, ready for brix_token_verify_bearer() lookup.
143 * HOW: fopen(path,"r"), fcntl(fileno(fp),F_SETFD,FD_CLOEXEC) for safe fork behavior; fseek/ftell validate size (0 < fsize ≤ 65536);
144 * malloc(fsize+1), fread full content, null-terminate buf; call brix_jwks_load_jansson(log,path,buf,fsize,keys,max_keys); free(buf); return count. */
145{
146 FILE *fp;
147 char *buf;
148 long fsize;
149 int count;
150
151 fp = fopen(path, "r");
152 if (fp == NULL) {
153 BRIX_DIAG_ERR(log, ngx_errno,
154 "brix_token: cannot open JWKS file \"%s\"",
155 "the path is wrong, or the file is unreadable by the nginx user",
156 "check brix_token_jwks points at a readable file (the JWKS "
157 "refresh job must write it where nginx can read); the OS reason "
158 "is appended below",
159 path);
160 return -1;
161 }
162 fcntl(fileno(fp), F_SETFD, FD_CLOEXEC);
163
164 fseek(fp, 0, SEEK_END);
165 fsize = ftell(fp);
166 fseek(fp, 0, SEEK_SET);
167
168 if (fsize <= 0 || fsize > 65536) {
169 BRIX_DIAG_ERR(log, 0,
170 "brix_token: JWKS file is empty or too large (%ld bytes)",
171 "an empty file usually means the refresh job failed mid-write; "
172 ">64 KiB means it is not a JWKS document",
173 "ensure the JWKS is written atomically and contains just the key "
174 "set; until then token authentication is rejected",
175 fsize);
176 fclose(fp);
177 return -1;
178 }
179
180 /* Defense-in-depth: reject a negative or wraparound fsize before malloc. */
181 size_t buf_sz;
182 if (fsize < 0 || brix_size_add((size_t) fsize, 1, &buf_sz) != NGX_OK) {
183 fclose(fp);
184 return -1;
185 }
186 buf = malloc(buf_sz);
187 if (buf == NULL) {
188 fclose(fp);
189 return -1;
190 }
191
192 if (fread(buf, 1, (size_t) fsize, fp) != (size_t) fsize) {
193 ngx_log_error(NGX_LOG_ERR, log, ngx_errno,
194 "brix_token: failed to read JWKS file");
195 free(buf);
196 fclose(fp);
197 return -1;
198 }
199 buf[fsize] = '\0';
200 fclose(fp);
201
202 count = brix_jwks_load_jansson(log, path, buf, (size_t) fsize,
203 keys, max_keys);
204 free(buf);
205 return count;
206}

208/* Pool-cleanup shim: holds the array + a pointer to the live count so the
209 * handler frees whatever set of keys is current when the pool is destroyed
210 * (the refresh path rewrites the array in place and updates the count). */

244void
245brix_jwks_free(brix_jwks_key_t *keys, int count)
246/* WHAT: Free all EVP_PKEY handles loaded by brix_jwks_load() to prevent memory leaks.
247 * WHY: JWKS keys are allocated at startup; this cleanup function must be called before shutdown or when reloading
248 * a new JWKS file to avoid leaking OpenSSL EVP_PKEY resources.
249 * HOW: Iterate i=0..count-1, for each non-null keys[i].pkey call EVP_PKEY_free(), then nullify keys[i].pkey to prevent double-free. */
250{
251 int i;
252
253 for (i = 0; i < count; i++) {
254 if (keys[i].pkey != NULL) {
255 EVP_PKEY_free(keys[i].pkey);
256 keys[i].pkey = NULL;
257 }
258 }
