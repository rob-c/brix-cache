38/*
39 * brix_ssi_service <name> — enable a non-default SSI provider. The built-in
40 * test/reference services always resolve; the flagship CTA tape service is opt-in
41 * (it exposes a storage-control surface). Extend the recognised-name list here as
42 * more native services gain config gating.
43 */
44static char *
45brix_ssi_service_directive(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
46{
47 ngx_stream_brix_srv_conf_t *xcf = conf;
48 ngx_str_t *value = cf->args->elts;
49
50 (void) cmd;
51 if (value[1].len == 3 && ngx_strncmp(value[1].data, "cta", 3) == 0) {
52 xcf->ssi_cta_enable = 1;
53 return NGX_CONF_OK;
54 }
55 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
56 "unknown SSI service \"%V\" (known: cta)", &value[1]);
57 return NGX_CONF_ERROR;
58}
