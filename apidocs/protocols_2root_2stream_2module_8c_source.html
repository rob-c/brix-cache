68/*
69 * brix_conf_set_stream_mint_ca — setter for "brix_storage_credential_mint_ca
70 * <cert> <key>" on the stream (root://) plane (phase-3 T1). Mirrors
71 * brix_conf_set_mint_ca (src/core/config/http_common.c) exactly — a
72 * stream-local copy is required because the HTTP setter is file-static and
73 * the two conf struct types differ. Validates both PEM files load-parse at
74 * config time (nginx -t fails loudly on a bad mint CA instead of every mint
75 * request failing at runtime) and stores their paths into the shared
76 * preamble's storage_credential_mint_ca_cert / _key fields. TRUST NOTE:
77 * configuring this directive means the frontend will sign per-user x509
78 * proxies with this CA key — the ORIGIN must trust this CA for minted
79 * credentials to be usable; see src/fs/backend/cred_mint.h for the full
80 * trust-model note.
81 */
82static char *
83brix_conf_set_stream_mint_ca(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
84{
85 ngx_stream_brix_srv_conf_t *xcf = conf;
86 ngx_str_t *value = cf->args->elts;
87 FILE *f;
88 X509 *cert;
89 EVP_PKEY *key;
90
91 (void) cmd;
92
93 f = fopen((const char *) value[1].data, "r");
94 cert = (f != NULL) ? PEM_read_X509(f, NULL, NULL, NULL) : NULL;
95 if (f != NULL) {
96 (void) fclose(f); /* read-only stream; the PEM parse result is the gate */
97 }
98 if (cert == NULL) {
99 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
100 "brix_storage_credential_mint_ca: cannot parse CA cert \"%V\"",
101 &value[1]);
102 return NGX_CONF_ERROR;
103 }
104 X509_free(cert);
105
106 f = fopen((const char *) value[2].data, "r");
107 key = (f != NULL) ? PEM_read_PrivateKey(f, NULL, NULL, NULL) : NULL;
108 if (f != NULL) {
109 (void) fclose(f); /* read-only stream; the PEM parse result is the gate */
110 }
111 if (key == NULL) {
112 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
113 "brix_storage_credential_mint_ca: cannot parse CA key \"%V\"",
114 &value[2]);
115 return NGX_CONF_ERROR;
116 }
117 EVP_PKEY_free(key);
118
119 xcf->common.storage_credential_mint_ca_cert = value[1];
120 xcf->common.storage_credential_mint_ca_key = value[2];
121 return NGX_CONF_OK;
122}

124/*
125 * brix_ssi_service <name> — enable a non-default SSI provider. The built-in
126 * test/reference services always resolve; the flagship CTA tape service is opt-in
127 * (it exposes a storage-control surface). Extend the recognised-name list here as
128 * more native services gain config gating.
129 */
130static char *
131brix_ssi_service_directive(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
132{
133 ngx_stream_brix_srv_conf_t *xcf = conf;
134 ngx_str_t *value = cf->args->elts;
135
136 (void) cmd;
137 if (value[1].len == 3 && ngx_strncmp(value[1].data, "cta", 3) == 0) {
138 xcf->ssi_cta_enable = 1;
139 return NGX_CONF_OK;
140 }
141 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
142 "unknown SSI service \"%V\" (known: cta)", &value[1]);
143 return NGX_CONF_ERROR;
144}
