1/*
2 * stream_mirror_launch.c — Phase 24 XRootD stream mirror: eligibility gates and
3 * per-target launch (see stream_mirror.h / stream_mirror_internal.h).
4 *
5 * WHAT: Owns the "should we mirror this request, and if so to whom" policy: the
6 * opcode->bit classifier, the replayability predicate, the write-op gate,
7 * the collapsed eligibility ladder, the per-target context allocation +
8 * request snapshot, and the public brix_stream_mirror_maybe() fan-out hook
9 * called after the primary handler has answered the client.
10 * WHY: split out (phase-79 file-size cap) from the async shadow-connection state
11 * machine in stream_mirror.c so each file owns one concept under the
12 * 500-line cap. This file decides WHAT to replay and hands a ready mir
13 * context to brix_mir_start(); stream_mirror.c owns HOW the shadow exchange
14 * runs. The two share only the symbols in stream_mirror_internal.h.
15 * HOW: brix_stream_mirror_maybe -> brix_stream_mirror_eligible (guard ladder) ->
16 * per-target brix_stream_mirror_launch_target -> brix_mir_start. Pure
17 * policy predicates (opcode_bit, request_replayable, write_op_allowed) feed
18 * the eligibility gate; the only side effects are the sample-drop metric
19 * and the launch itself.
20 */

31static ngx_uint_t
32brix_mirror_opcode_bit(uint16_t reqid)
33{
34 switch (reqid) {
35 case kXR_stat: return BRIX_MIRROR_OP_STAT;
36 case kXR_locate: return BRIX_MIRROR_OP_LOCATE;
37 case kXR_open: return BRIX_MIRROR_OP_OPEN;
38 case kXR_read: return BRIX_MIRROR_OP_READ;
39 case kXR_readv: return BRIX_MIRROR_OP_READV;
40 case kXR_dirlist: return BRIX_MIRROR_OP_DIRLIST;
41 case kXR_statx: return BRIX_MIRROR_OP_STATX;
42 case kXR_query: return BRIX_MIRROR_OP_QUERY;
43 /* Write/mutation opcodes (Phase 24 write mirroring, gated by
44 * brix_mirror_writes). mkdir/rm/rmdir/mv/truncate/chmod are self-contained
45 * path-based metadata ops replayable by this stateless mirror; data writes
46 * (open-write/write/pgwrite/close) map to OP_WRITE but are handled by the
47 * stateful write-mirror (W3), not here. */
48 case kXR_mkdir: return BRIX_MIRROR_OP_MKDIR;
49 case kXR_rm: return BRIX_MIRROR_OP_RM;
50 case kXR_rmdir: return BRIX_MIRROR_OP_RMDIR;
51 case kXR_mv: return BRIX_MIRROR_OP_MV;
52 case kXR_truncate: return BRIX_MIRROR_OP_TRUNCATE;
53 case kXR_chmod: return BRIX_MIRROR_OP_CHMOD;
54 default: return 0;
55 }
56}

58/* Write-implying open flags. A read-path mirror must never replay these to a
59 * shadow: they would create/truncate/append/mkdir on the official server. */
60#define BRIX_MIRROR_OPEN_WRITE_FLAGS \
61 (kXR_delete | kXR_new | kXR_open_updt | kXR_open_apnd | kXR_mkpath)

63/*
64 * Can this request be faithfully replayed to a fresh shadow session?
65 *
66 * The mirror is stateless — it opens a new connection, bootstraps, and sends ONE
67 * saved request frame. That only works for SELF-CONTAINED, side-effect-free
68 * requests whose target lives entirely in the frame:
69 *
70 * - locate / dirlist / query (incl. Qcksum) : path + args are in the payload.
71 * - stat / statx with dlen>0 : path-based (dlen==0 is by open
72 * handle and cannot be replayed).
73 * - open WITHOUT write flags : a read-only open of a path.
74 *
75 * Handle-based ops (read, readv, stat/statx by handle) carry the CLIENT's file
76 * handle, which is meaningless on the shadow's separate session, and write/
77 * create opens would mutate the official server. Neither is replayable by a
78 * stateless mirror, so they are skipped — this is what lets the mirror "just
79 * work" in front of an official xrootd instead of spuriously diverging.
80 */
81static int
82brix_mirror_request_replayable(brix_ctx_t *ctx)
83{
84 switch (ctx->recv.cur_reqid) {
85 case kXR_locate:
86 case kXR_dirlist:
87 case kXR_query:
88 return 1;
89 case kXR_stat:
90 case kXR_statx:
91 return ctx->recv.cur_dlen > 0;
92 case kXR_open: {
93 uint16_t options; /* ClientOpenRequest.options — header byte 6 */
94 ngx_memcpy(&options, ctx->recv.hdr_buf + 6, sizeof(options));
95 options = ntohs(options);
96 return (options & BRIX_MIRROR_OPEN_WRITE_FLAGS) == 0;
97 }
98 /* Metadata mutations (W1): self-contained and path-based — the path[s] live
99 * in the payload, so a fresh shadow session replays them faithfully.
100 * truncate may instead be by open-handle (dlen==0); that form is not
101 * replayable, same as handle-based stat. These reach here only when
102 * brix_mirror_writes is on (gated in brix_stream_mirror_maybe). */
103 case kXR_mkdir:
104 case kXR_rm:
105 case kXR_rmdir:
106 case kXR_mv:
107 case kXR_chmod:
108 case kXR_truncate:
109 return ctx->recv.cur_dlen > 0;
110 default:
111 return 0;
112 }
113}

118/*
119 * Is this write/mutation opcode cleared to proceed on the one-shot mirror?
120 *
121 * WHAT: for a WRITE-class @opbit, return 1 only when brix_mirror_writes is on
122 * AND the op is a self-contained metadata mutation (not OP_WRITE); for
123 * any non-write opcode return 1 unconditionally.
124 * WHY: write mirroring needs a second, independent gate beyond the opcode
125 * allowlist, and data writes (OP_WRITE) belong to the stateful write-
126 * mirror, never this stateless path — factoring it keeps the driver flat.
127 * HOW: short-circuits on the non-write case; otherwise checks the flag and
128 * excludes OP_WRITE.
129 */
130static int
131brix_mirror_write_op_allowed(ngx_uint_t opbit,
132 ngx_stream_brix_srv_conf_t *conf)
133{
134 if ((opbit & BRIX_MIRROR_OP_WRITE_ALL) == 0) {
135 return 1;
136 }
137 if (!conf->mirror.mirror_writes || opbit == BRIX_MIRROR_OP_WRITE) {
138 return 0;
139 }
140 return 1;
141}

143/*
144 * Decide whether the current request is eligible to be mirrored at all.
145 *
146 * WHAT: run every fast-reject gate (feature off, opcode filtered, write-gate,
147 * non-replayable, oversized payload, sampling) and return 1 only when the
148 * request should proceed to per-target launch.
149 * WHY: collapses the whole guard ladder into one predicate so the launch
150 * driver expresses just "eligible? then fan out", not the policy.
151 * HOW: sequential early-returns; @opbit is returned to the caller so the launch
152 * path need not recompute it. Emits the drop metric on sample loss.
153 */
154static int
155brix_stream_mirror_eligible(brix_ctx_t *ctx,
156 ngx_stream_brix_srv_conf_t *conf, ngx_uint_t *opbit_out)
157{
158 ngx_uint_t opbit;
159
160 if (!conf->mirror.enabled || conf->mirror.targets == NULL) {
161 return 0;
162 }
163
164 opbit = brix_mirror_opcode_bit(ctx->recv.cur_reqid);
165 /* Mirror all ops in opcode_mask (default ALL) that are not de-selected via
166 * brix_mirror_exclude_opcodes. */
167 if (opbit == 0
168 || (conf->mirror.opcode_mask & opbit) == 0
169 || (conf->mirror.opcode_exclude_mask & opbit) != 0)
170 {
171 return 0;
172 }
173
174 /* Second, independent guard for write/mutation opcodes: even if an operator
175 * lists e.g. "mkdir" in brix_mirror_opcodes, it stays inert unless
176 * brix_mirror_writes is explicitly on (and the shadow is an isolated
177 * namespace). OP_WRITE (data writes) is handled by the stateful write-mirror,
178 * not this one-shot path, so it never proceeds here. */
179 if (!brix_mirror_write_op_allowed(opbit, conf)) {
180 return 0;
181 }
182
183 /* Only replay requests the shadow can answer standalone (path-based reads,
184 * read-only opens, query/Qcksum). Handle-based reads/readv/handle-stat and
185 * write opens are skipped so the mirror never spuriously diverges against an
186 * official xrootd — see brix_mirror_request_replayable(). */
187 if (!brix_mirror_request_replayable(ctx)) {
188 return 0;
189 }
190
191 /* Skip oversized payloads (write bodies); also a sanity guard. */
192 if (ctx->recv.cur_dlen > BRIX_MIRROR_MAX_PAYLOAD) {
193 return 0;
194 }
195
196 if (!brix_mirror_should_sample(conf->mirror.sample_pct)) {
197 BRIX_MIR_METRIC_INC(mirror_stream_dropped_total);
198 return 0;
199 }
200
201 *opbit_out = opbit;
202 return 1;
203}

205/*
206 * Allocate a mirror context, copy target + saved request into it, and start it.
207 *
208 * WHAT: for one resolved @target, create the mirror's private pool + ctx,
209 * snapshot the primary request frame (header, opcode, payload), and kick
210 * off brix_mir_start(). Unresolved targets and OOM are skipped silently.
211 * WHY: isolates the per-target allocation/snapshot from the fan-out loop, so
212 * the driver is a flat iteration and each launch owns its own cleanup.
213 * HOW: a fresh cycle-pool ctx (outlives the client conn); the request payload
214 * is copied now because ctx->recv.payload may be reused before the shadow
215 * exchange completes.
216 */
217static void
218brix_stream_mirror_launch_target(brix_ctx_t *ctx,
219 brix_mirror_target_t *target, ngx_stream_brix_srv_conf_t *conf,
220 int primary_ok)
221{
222 ngx_pool_t *pool;
223 brix_stream_mirror_t *mir;
224
225 if (target->socklen == 0) {
226 return; /* unresolved target */
227 }
228
229 pool = ngx_create_pool(2048, ngx_cycle->log);
230 if (pool == NULL) {
231 return;
232 }
233 mir = ngx_pcalloc(pool, sizeof(*mir));
234 if (mir == NULL) {
235 ngx_destroy_pool(pool);
236 return;
237 }
238 mir->pool = pool;
239 mir->log = ngx_cycle->log; /* outlives the client connection */
240 mir->phase = XRD_MIR_HANDSHAKE;
241 mir->primary_ok = primary_ok;
242 mir->log_diverge = conf->mirror.log_diverge;
243 mir->port = target->port;
244 mir->socklen = target->socklen;
245 ngx_memcpy(&mir->sockaddr, &target->sockaddr, target->socklen);
246 ngx_cpystrn((u_char *) mir->host,
247 target->host.data ? target->host.data
248 : (u_char *) "?",
249 sizeof(mir->host));
250
251 /* Snapshot the request now — ctx->recv.payload may be reused by the next
252 * request before the shadow exchange completes. */
253 ngx_memcpy(mir->saved_hdr, ctx->recv.hdr_buf, 24);
254 mir->saved_opcode = ctx->recv.cur_reqid;
255 if (ctx->recv.payload != NULL && ctx->recv.cur_dlen > 0) {
256 mir->saved_payload = ngx_palloc(pool, ctx->recv.cur_dlen);
257 if (mir->saved_payload != NULL) {
258 ngx_memcpy(mir->saved_payload, ctx->recv.payload, ctx->recv.cur_dlen);
259 mir->saved_dlen = ctx->recv.cur_dlen;
260 }
261 }
262
263 brix_mir_start(mir, conf->mirror.timeout_ms);
264}

266void
267brix_stream_mirror_maybe(brix_ctx_t *ctx, ngx_connection_t *c,
268 ngx_stream_brix_srv_conf_t *conf, ngx_int_t primary_rc)
269{
270 brix_mirror_target_t *targets;
271 ngx_uint_t opbit = 0, i;
272 int primary_ok;
273
274 (void) c;
275
276 if (!brix_stream_mirror_eligible(ctx, conf, &opbit)) {
277 return;
278 }
279 (void) opbit; /* eligibility owns opcode policy; driver just fans out */
280
281 primary_ok = (primary_rc != NGX_ERROR);
282 targets = conf->mirror.targets->elts;
283
284 for (i = 0; i < conf->mirror.targets->nelts; i++) {
285 brix_stream_mirror_launch_target(ctx, &targets[i], conf, primary_ok);
286 }
