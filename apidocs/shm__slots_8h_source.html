14/*
15 * WHAT: Check whether an SHM slot's expiration timestamp has passed.
16 *
17 * WHY: Slot pools (TPC key registry, cache origin connections) need to detect expired
18 * entries without scanning the entire pool. This inline helper is called at points
19 * where a slot's expires field is compared against the current time to decide if
20 * it can be reclaimed or reused.
21 *
22 * HOW: Simple comparison — returns 1 (true) when expires <= now, 0 otherwise.
23 */
24static ngx_inline ngx_flag_t
25brix_shm_slot_expired(ngx_msec_t now, ngx_msec_t expires)
26{
27 return expires <= now;
28}

30/*
31 * WHAT: Remember a candidate free slot in a pointer when no slot is known yet.
32 *
33 * WHY: SHM slot pools often track one 'free_slot' index across multiple workers or
34 * iterations. When the tracked value is still at its sentinel ('none'), this helper
35 * records the first available candidate without overwriting later discoveries.
36 *
37 * HOW: If *free_slot == none, set *free_slot = candidate. Otherwise leave it unchanged.
38 * This prevents a late-arriving worker from clobbering an earlier discovery.
39 */
40static ngx_inline void
41brix_shm_remember_free_slot(ngx_uint_t *free_slot, ngx_uint_t none,
42 ngx_uint_t candidate)
43{
44 if (*free_slot == none) {
45 *free_slot = candidate;
46 }
47}

49/*
50 * Slab-safe SHM table allocation (shm_slots.c).
51 *
52 * CRITICAL: nginx initialises every shared_memory zone as an ngx_slab_pool_t and
53 * its SIGCHLD handler (ngx_unlock_mutexes, run on EVERY child death) walks all
54 * zones, treating shm.addr as an ngx_slab_pool_t and force-unlocking sp->mutex.
55 * A zone whose init callback lays its own struct directly over shm.addr clobbers
56 * that header → the master SIGSEGVs the moment any child exits (e.g. an FRM stage
57 * copycmd). These helpers allocate the table FROM the slab pool, leaving the
58 * header intact, so any subsystem that forks children is safe.
59 *
60 * Contract: pass `mtx` to have a process-local mutex handle created, bound to
61 * the slab pool's OWN lock word (&sp->lock) — the word nginx force-unlocks on
62 * every worker death, so a worker that dies holding it is recovered rather than
63 * stranding the lock forever. Pass NULL for lock-less tables (e.g. the metrics
64 * counters). The lock is NOT taken from the table data, so the table needs no
65 * embedded lock member.
66 *
67 * brix_shm_table_alloc(): handles fresh alloc, reload (data != NULL), and
68 * re-attach (shm.exists). Publishes the table via shm_zone->data AND the slab
69 * pool's ->data (so a reload re-attaches). *fresh is set to 1 only when a brand
70 * new table was allocated (the caller must then initialise its non-lock fields;
71 * on reuse it must NOT, to preserve live state).
72 *
73 * brix_shm_zone_size(): the zone size to request from ngx_shared_memory_add so
74 * ngx_slab_alloc(table_bytes) always fits alongside the slab overhead.
75 */
76void *brix_shm_table_alloc(ngx_shm_zone_t *shm_zone, void *data,
77 size_t table_bytes, ngx_shmtx_t *mtx,
78 ngx_flag_t *fresh);
79size_t brix_shm_zone_size(size_t table_bytes);
