1/*
2 * cms/meter.c — native machine-load meter for the CMS heartbeat (Phase-89 W4).
3 *
4 * WHAT: /proc-backed load figures for the kYR_load theLoad bytes — pure
5 * parsers (unit-tested standalone) plus the sampling driver that owns the
6 * file I/O and delta bookkeeping.
7 *
8 * WHY: See meter.h. The parser/driver split keeps everything with a
9 * behavioural decision (percentage mapping, delta priming, malformed input)
10 * out of the I/O path so meter_unittest.c covers it byte-for-byte.
11 *
12 * HOW: Each parser walks one text buffer with strtoul/strtod-style scanning
13 * and rejects buffers missing the expected fields. sample() slurps each
14 * /proc file into a stack buffer (they are all small), runs the parser, and
15 * degrades any failure to a 0 byte. Deltas (net, pag) need two samples:
16 * the first call only primes the counters.
17 */

34int
35brix_cms_meter_parse_loadavg(const char *buf, long ncpu, uint8_t *pct)
36{
37 char *end;
38 double load1;
39
40 if (buf == NULL || ncpu <= 0) {
41 return -1;
42 }
43
44 load1 = strtod(buf, &end);
45 if (end == buf || load1 < 0.0) {
46 return -1;
47 }
48
49 *pct = meter_clamp_pct((unsigned long) (load1 * 100.0 / (double) ncpu));
50 return 0;
51}

54/* Find "<key> <number>" in a /proc key-value listing; the key must start a
55 * line. Returns 0 with *out set, -1 if the key is absent/malformed. */
56static int
57meter_find_counter(const char *buf, const char *key, uint64_t *out)
58{
59 const char *p = buf;
60 size_t klen = strlen(key);
61 char *end;
62
63 while (p != NULL && *p != '\0') {
64 if (strncmp(p, key, klen) == 0
65 && (p[klen] == ' ' || p[klen] == ':'))
66 {
67 const char *v = p + klen + 1;
68 unsigned long long n = strtoull(v, &end, 10);
69 if (end == v) {
70 return -1;
71 }
72 *out = (uint64_t) n;
73 return 0;
74 }
75 p = strchr(p, '\n');
76 if (p != NULL) {
77 p++;
78 }
79 }
80 return -1;
81}

84int
85brix_cms_meter_parse_meminfo(const char *buf, uint8_t *pct)
86{
87 uint64_t total = 0, avail = 0;
88
89 if (buf == NULL
90 || meter_find_counter(buf, "MemTotal:", &total) != 0
91 || meter_find_counter(buf, "MemAvailable:", &avail) != 0
92 || total == 0)
93 {
94 return -1;
95 }
96 if (avail > total) {
97 avail = total;
98 }
99
100 *pct = meter_clamp_pct((unsigned long) ((total - avail) * 100 / total));
101 return 0;
102}

105/* Parse one /proc/net/dev interface line's counters starting just past the
106 * ':' — rx_bytes first, tx_bytes as field 9 (skip fields 2-8). Returns 0
107 * with *bytes = rx + tx, -1 on a malformed field. */
108static int
109meter_netdev_line_bytes(const char *colon, uint64_t *bytes)
110{
111 char *end;
112 unsigned long long rx, tx;
113 int f;
114
115 rx = strtoull(colon + 1, &end, 10);
116 if (end == colon + 1) {
117 return -1;
118 }
119 /* tx_bytes is field 9 after the colon; skip fields 2-8. */
120 for (f = 0; f < 7; f++) {
121 const char *v = end;
122 (void) strtoull(v, &end, 10);
123 if (end == v) {
124 return -1;
125 }
126 }
127 {
128 const char *v = end;
129 tx = strtoull(v, &end, 10);
130 if (end == v) {
131 return -1;
132 }
133 }
134
135 *bytes = (uint64_t) rx + (uint64_t) tx;
136 return 0;
137}

140int
141brix_cms_meter_parse_netdev(const char *buf, uint64_t *total_bytes)
142{
143 const char *p;
144 uint64_t sum = 0;
145 int seen = 0;
146
147 if (buf == NULL) {
148 return -1;
149 }
150
151 /*
152 * Per-interface lines: " eth0: rx_bytes ... (8 fields) tx_bytes ...".
153 * The two header lines carry no ':' followed by digits, so keying on the
154 * colon is sufficient. Loopback is skipped — lo traffic is not network
155 * load and would double-count proxy-to-local-backend transfers.
156 */
157 for (p = buf; p != NULL && *p != '\0'; p = p ? strchr(p, '\n') : NULL,
158 p = p ? p + 1 : NULL)
159 {
160 const char *colon, *name;
161 uint64_t line_bytes;
162
163 colon = strchr(p, ':');
164 if (colon == NULL) {
165 break; /* no interface lines left */
166 }
167 {
168 const char *nl = strchr(p, '\n');
169 if (nl != NULL && colon > nl) {
170 continue; /* header line — colon is further down */
171 }
172 }
173
174 name = p;
175 while (*name == ' ') {
176 name++;
177 }
178 if (strncmp(name, "lo:", 3) == 0) {
179 continue;
180 }
181
182 if (meter_netdev_line_bytes(colon, &line_bytes) != 0) {
183 return -1;
184 }
185
186 sum += line_bytes;
187 seen = 1;
188 }
189
190 if (!seen) {
191 return -1;
192 }
193 *total_bytes = sum;
194 return 0;
195}

222/* Slurp a small /proc file into buf (NUL-terminated). Returns 0 on success.
223 * stdio (not raw syscalls) keeps this out of the VFS seam — /proc is host
224 * introspection, not storage. */
225static int
226meter_slurp(const char *path, char *buf, size_t bufsz)
227{
228 FILE *f;
229 size_t n;
230
231 f = fopen(path, "r");
232 if (f == NULL) {
233 return -1;
234 }
235 n = fread(buf, 1, bufsz - 1, f);
236 fclose(f);
237 if (n == 0) {
238 return -1;
239 }
240 buf[n] = '\0';
241 return 0;
242}

245/* Sample the point-in-time gauges (cpu, mem) straight into out5; any slurp
246 * or parse failure leaves that byte 0. */
247static void
248meter_sample_gauges(uint8_t out5[5])
249{
250 char buf[8192];
251 uint8_t pct;
252
253 if (meter_slurp("/proc/loadavg", buf, sizeof(buf)) == 0
254 && brix_cms_meter_parse_loadavg(buf, sysconf(_SC_NPROCESSORS_ONLN),
255 &pct) == 0)
256 {
257 out5[0] = pct; /* cpu */
258 }
259
260 /* out5[2] (xeq) stays 0 — no request queue to report. */
261
262 if (meter_slurp("/proc/meminfo", buf, sizeof(buf)) == 0
263 && brix_cms_meter_parse_meminfo(buf, &pct) == 0)
264 {
265 out5[3] = pct; /* mem */
266 }
267}

270/* Fold the delta-based figures (net, pag) into out5 and roll the previous
271 * counters forward. The first successful sample only primes the counters. */
272static void
273meter_fold_deltas(brix_cms_meter_t *m, uint64_t now_ms,
274 int have_net, uint64_t net_bytes, int have_pag, uint64_t pgmaj,
275 uint8_t out5[5])
276{
277 uint64_t elapsed;
278
279 elapsed = now_ms > m->prev_ms ? now_ms - m->prev_ms : 0;
280
281 if (m->primed && elapsed > 0) {
282 if (have_net && net_bytes >= m->prev_net_bytes) {
283 out5[1] = brix_cms_meter_rate_pct( /* net */
284 net_bytes - m->prev_net_bytes, elapsed,
285 BRIX_CMS_METER_NET_REF_BYTES_PER_SEC);
286 }
287 if (have_pag && pgmaj >= m->prev_pgmajfault) {
288 out5[4] = brix_cms_meter_rate_pct( /* pag */
289 pgmaj - m->prev_pgmajfault, elapsed,
290 BRIX_CMS_METER_PAG_REF_FAULTS_PER_SEC);
291 }
292 }
293
294 if (have_net) {
295 m->prev_net_bytes = net_bytes;
296 }
297 if (have_pag) {
298 m->prev_pgmajfault = pgmaj;
299 }
300 m->prev_ms = now_ms;
301 m->primed = (have_net || have_pag) ? 1 : m->primed;
302}

322}
