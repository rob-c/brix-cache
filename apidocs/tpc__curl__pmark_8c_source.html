8webdav_tpc_pmark_opensocket(void *clientp, curlsocktype purpose,
9 struct curl_sockaddr *address)
10{
11 webdav_tpc_pmark_rec_t *rec = clientp;
12 curl_socket_t fd;
13
14 fd = socket(address->family, address->socktype, address->protocol);
15 if (fd == CURL_SOCKET_BAD) {
16 return CURL_SOCKET_BAD;
17 }
18 /* Only the actual data connection; skip any proxy/other socket types. */
19 if (rec != NULL && rec->active && purpose == CURLSOCKTYPE_IPCXN) {
20 (void) brix_pmark_flowlabel_apply_addr((int) fd,
21 (struct sockaddr *) &address->addr, (socklen_t) address->addrlen,
22 rec->exp, rec->act, rec->log);
23 rec->fd = (int) fd;
24 }
25 return fd;
26}

30webdav_tpc_pmark_closesocket(void *clientp, curl_socket_t item)
31{
32 webdav_tpc_pmark_rec_t *rec = clientp;
33
34 if (rec != NULL && rec->active && rec->fd == (int) item) {
35 /* Socket is still connected here — emit the firefly (reads TCP_INFO)

37 brix_pmark_firefly_oneshot(rec->pm, (int) item, rec->exp, rec->act,
38 rec->peer_is_src, rec->app, rec->log);
39 rec->fd = -1;
40 }

45/* Resolve codes + attach the socket callbacks to `curl`. `rec` is caller-owned
46 * (stack) and must outlive curl_easy_perform(). No-op when pmark is off or the
47 * transfer does not map to an (experiment, activity). */
48void
49webdav_tpc_pmark_attach(CURL *curl, webdav_tpc_pmark_rec_t *rec,
50 ngx_http_brix_webdav_loc_conf_t *conf, int is_push,
51 const char *file_path, ngx_log_t *log)
52{
53 brix_pmark_conf_t *pm = &conf->common.pmark;
54 ngx_uint_t e, a;
55 size_t n;
56
57 brix_pmark_flow_id_t flow_id = {
58 .vo_csv = "", .user = "", .path = file_path, .cgi = NULL,
59 };
60
61 ngx_memzero(rec, sizeof(*rec));
62 rec->fd = -1;
63
64 if (!pm->enable
65 || brix_pmark_runtime_ensure(pm, ngx_cycle->pool, log) != NGX_OK
66 || brix_pmark_map_codes(pm, &flow_id, &e, &a) != NGX_OK)
67 {
68 return;
69 }
70
71 rec->pm = pm;
72 rec->exp = e;
73 rec->act = a;
74 rec->peer_is_src = is_push ? 0 : 1; /* pull (GET) → remote is source */
75 rec->log = log;
76 n = pm->appname.len < sizeof(rec->app) - 1
77 ? pm->appname.len : sizeof(rec->app) - 1;
78 if (n) { ngx_memcpy(rec->app, pm->appname.data, n); }
79 rec->app[n] = '\0';
80 rec->active = 1;
81
82 curl_easy_setopt(curl, CURLOPT_OPENSOCKETFUNCTION,
83 webdav_tpc_pmark_opensocket);
84 curl_easy_setopt(curl, CURLOPT_OPENSOCKETDATA, rec);
85 curl_easy_setopt(curl, CURLOPT_CLOSESOCKETFUNCTION,
86 webdav_tpc_pmark_closesocket);
87 curl_easy_setopt(curl, CURLOPT_CLOSESOCKETDATA, rec);
88}

92size_t
93ms_write_cb(char *ptr, size_t size, size_t nmemb, void *userdata)
94{
95 ms_stream_ctx_t *ctx = userdata;
96 size_t total = size * nmemb;
97 size_t done = 0;
98
99 brix_sd_obj_t obj;
100
101 brix_sd_posix_wrap(&obj, ctx->fd); /* phase-55: SD seam */
102 while (done < total) {
103 /* backend-agnostic vtable form (not the concrete brix_sd_posix_driver)
104 * so a non-POSIX backend serves this path unchanged. */
105 ssize_t n = obj.driver->pwrite(&obj, ptr + done, total - done,
106 ctx->cur_offset + (off_t) done);
107 if (n < 0) {
108 if (errno == EINTR) continue;
109 return 0; /* signal error to curl */
110 }
111 done += (size_t) n;
112 }
113
114 ctx->cur_offset += (off_t) total;
115
116 if (ctx->progress != NULL) {
117 ngx_atomic_fetch_add(
118 &ctx->progress->bytes_per_stream[ctx->stream_idx],
119 (ngx_atomic_int_t) total);
120 }
121
122 return total;
123}

126int
127webdav_tpc_curl_progress(void *clientp, curl_off_t dltotal,
128 curl_off_t dlnow, curl_off_t ultotal,
129 curl_off_t ulnow)
130{
131 webdav_tpc_curl_progress_t *progress = clientp;
132 off_t done;
133 off_t total;
134
135 if (progress == NULL || progress->transfer_id == 0) {
136 return 0;
137 }
138
139 /*
140 * Phase 39 (WS5): abort promptly if the client that requested this transfer
141 * disconnected (registry_request_cancel set the flag). Lock-free best-effort
142 * read via registry_find; a missed read just falls back to the WS4
143 * low-speed/transfer-timeout bounds. A non-zero return is
144 * CURLE_ABORTED_BY_CALLBACK.
145 */
146 {
147 const brix_tpc_transfer_t *t =
148 brix_tpc_registry_find(progress->transfer_id);
149 if (t != NULL && t->cancelled) {
150 return 1;
151 }
152 }
153
154 done = progress->is_push ? (off_t) ulnow : (off_t) dlnow;
155 total = progress->is_push ? (off_t) ultotal : (off_t) dltotal;
156
157 if (done != progress->last_done) {
158 progress->last_done = done;
159 (void) brix_tpc_progress_emit(progress->transfer_id, done, total,
160 BRIX_TPC_STATE_ACTIVE,
161 progress->log);
162 }
163
164 return 0;
165}

168/*
169 * webdav_tpc_curl_finish — release a single-transfer's libcurl resources and
170 * record the success/error metric, returning rc unchanged.
171 *
172 * Called at every exit of webdav_tpc_run_curl_core (success and each early
173 * failure). All handles are NULL-tolerant and start NULL, so invoking this at
174 * any point frees exactly what had been built so far — which lets the core
175 * function exit with a plain `return` instead of a shared goto/label.
176 */
177ngx_int_t
178webdav_tpc_curl_finish(ngx_int_t rc, CURL *curl, struct curl_slist *hdrs,
179 struct curl_slist *resolve, FILE *fp)
180{
181 if (fp != NULL) {
182 /* Cleanup-only close: rc already encodes the transfer outcome and the
183 * stream's data (if any) was consumed by libcurl; nothing actionable
184 * remains on a close error. */
185 (void) fclose(fp);
186 }
187 if (hdrs != NULL) {
188 curl_slist_free_all(hdrs);
189 }
190 if (resolve != NULL) {
191 curl_slist_free_all(resolve);
192 }
193 if (curl != NULL) {
194 curl_easy_cleanup(curl);
195 }
196
197 if (rc == NGX_OK) {
198 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_SUCCESS]);
199 } else {
200 BRIX_WEBDAV_METRIC_INC(tpc_total[BRIX_WEBDAV_TPC_CURL_ERROR]);
201 }
202 return rc;
203}
