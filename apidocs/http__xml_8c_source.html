2 * http_xml.c - shared HTTP XML chain construction and response helpers.
3 *
4 * WHAT: Builds ngx_chain_t of ngx_buf_t from formatted strings (printf-style),
5 * appends segments to a chain head/tail, and sends the completed chain
6 * as an HTTP response with status + content-type set on headers_out.
7 *
8 * WHY: S3 ListObjectsV2, CompleteMultipartUpload responses, and WebDAV PROPFIND all need

26/*
27 * brix_http_chain_vappendf - append formatted string to an existing chain via va_list.
28 *
29 * WHAT: Formats args into a buffer (stack tmp[2048] or pool-allocated heap on overflow),
30 * creates an ngx_buf_t + ngx_chain_t link, and appends it to the chain pointed
31 * to by head/tail. Returns pointer to the new buf for callers that need to inspect it.
32 *
33 * WHY: S3 XML responses build incrementally — each element (BucketName, Contents,
34 * NextContinuationToken) is appended one at a time. This helper handles va_list
35 * formatting with safe overflow into pool-allocated memory so the chain grows without
36 * stack exhaustion.
37 *
38 * HOW: va_copy → vsnprintf(tmp,2048). If n>=sizeof(tmp): ngx_palloc(n+1) + retry vsnprintf.
39 * Else use tmp. ngx_create_temp_buf(pool,n) + ngx_memcpy data. ngx_alloc_chain_link
40 * (pool) → cl->buf=b; cl->next=NULL. Append to tail, update head/tail pointers.
41 * Return b pointer.
42 */
43
44ngx_buf_t *
45brix_http_chain_vappendf(ngx_pool_t *pool, ngx_chain_t **head,
46 ngx_chain_t **tail, const char *fmt, va_list ap)
47{
48 va_list ap_copy;
49 char tmp[2048];
50 char *src;
51 int n;
52 ngx_buf_t *b;
53 ngx_chain_t *cl;
54
55 va_copy(ap_copy, ap);
56 n = vsnprintf(tmp, sizeof(tmp), fmt, ap);
57 if (n < 0) {
58 va_end(ap_copy);
59 return NULL;
60 }
61
62 if ((size_t) n >= sizeof(tmp)) {
63 src = ngx_pnalloc(pool, (size_t) n + 1);
64 if (src == NULL) {
65 va_end(ap_copy);
66 return NULL;
67 }
68 (void) vsnprintf(src, (size_t) n + 1, fmt, ap_copy);
69 } else {
70 src = tmp;
71 }
72 va_end(ap_copy);
73
74 if (n == 0) {
75 return (*tail != NULL) ? (*tail)->buf : NULL;
76 }
77
78 b = ngx_create_temp_buf(pool, (size_t) n);
79 if (b == NULL) {
80 return NULL;
81 }
82 ngx_memcpy(b->pos, src, (size_t) n);
83 b->last = b->pos + n;
84
85 cl = ngx_alloc_chain_link(pool);
86 if (cl == NULL) {
87 return NULL;
88 }
89 cl->buf = b;
90 cl->next = NULL;
91
92 if (*tail == NULL) {
93 *head = cl;
94 *tail = cl;
95 } else {
96 (*tail)->next = cl;
97 *tail = cl;
98 }
99
100 return b;
101}

103/*
104 * brix_http_chain_appendf - convenience wrapper: append formatted string via variadic args.
105 *
106 * WHAT: Wraps brix_http_chain_vappendf() with va_start/va_end so callers pass a
107 * printf-style fmt+args directly without constructing a va_list manually. Appends the
108 * resulting buffer to head/tail chain pointers. Returns buf pointer.
109 *
110 * WHY: Most callers have a simple printf-style format string and don't want to manage
111 * va_list lifecycle. This wrapper eliminates va_start/va_end boilerplate at each callsite.
112 *
113 * HOW: va_start(ap,fmt) → vappendf(pool,head,tail,fmt,ap) → va_end(ap). Return buf.
114 */
115
116ngx_buf_t *
117brix_http_chain_appendf(ngx_pool_t *pool, ngx_chain_t **head,
118 ngx_chain_t **tail, const char *fmt, ...)
119{
120 va_list ap;
121 ngx_buf_t *b;
122
123 va_start(ap, fmt);
124 b = brix_http_chain_vappendf(pool, head, tail, fmt, ap);
125 va_end(ap);
126
127 return b;
128}

130/*
131 * brix_http_send_xml_buffer - send a single ngx_buf_t as an HTTP XML response.
132 *
133 * WHAT: Sets r->headers_out.status, content_type, and content_length_n from the buffer size,
134 * then sends headers via ngx_http_send_header() followed by the buffer as a last_buf
135 * chain through ngx_http_output_filter(). Returns NGX_HTTP_INTERNAL_SERVER_ERROR if b is NULL.
136 *
137 * WHY: S3 and WebDAV XML responses follow the same pattern: set status + content-type,
138 * compute length from buffer, send header, output single-buffer chain with last_buf=1.
139 * Centralising this prevents inconsistent header setup across protocol handlers.
140 *
141 * HOW: b->last_buf=1; out.buf=b; out.next=NULL. Set headers_out fields. ngx_http_send_header;
142 * return ngx_http_output_filter(r,&out).
143 */
144
145ngx_int_t
146brix_http_send_xml_buffer(ngx_http_request_t *r, ngx_uint_t status,
147 ngx_str_t content_type, ngx_buf_t *b)
148{
149 ngx_chain_t out;
150
151 if (b == NULL) {
152 return NGX_HTTP_INTERNAL_SERVER_ERROR;
153 }
154
155 b->last_buf = 1;
156 out.buf = b;
157 out.next = NULL;
158
159 r->headers_out.status = status;
160 r->headers_out.content_type = content_type;
161 r->headers_out.content_length_n = (off_t) (b->last - b->pos);
162
163 ngx_http_send_header(r);
164 return ngx_http_output_filter(r, &out);
165}

167/*
168 * brix_http_send_xml_error - build and send a protocol-agnostic XML error response.
169 *
170 * WHAT: Formats an <Error><Code>...</Code><Message>...</Message></Error> body with
171 * XML-escaped Code and Message, then sends it with application/xml content-type.
172 *
173 * WHY: S3 uses this structure for all error responses; WebDAV REST endpoints can use it
174 * for machine-readable errors. Centralising the builder prevents each protocol from
175 * duplicating the XML escaping + buffer assembly + header setup pattern.
176 *
177 * HOW: Pre-sizes buffer with brix_xml_text_element_len(), allocates from r->pool,
178 * writes prefix + Code element + Message element + suffix, sends via
179 * brix_http_send_xml_buffer(). Returns NGX_HTTP_INTERNAL_SERVER_ERROR on OOM.
180 */
181
182ngx_int_t
183brix_http_send_xml_error(ngx_http_request_t *r, ngx_uint_t status,
184 const char *code, const char *message)
185{
186 static const char prefix[] =
187 "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n<Error>";
188 static const char suffix[] = "</Error>";
189
190 size_t codelen = strlen(code);
191 size_t msglen = strlen(message);
192 size_t total;
193 ngx_buf_t *b;
194
195 total = sizeof(prefix) - 1
196 + brix_xml_text_element_len("Code",
197 (const u_char *) code, codelen,
198 BRIX_XML_ESCAPE_APOS_ENTITY)
199 + brix_xml_text_element_len("Message",
200 (const u_char *) message, msglen,
201 BRIX_XML_ESCAPE_APOS_ENTITY)
202 + sizeof(suffix) - 1;
203
204 b = ngx_create_temp_buf(r->pool, total + 4);
205 if (b == NULL) {
206 return NGX_HTTP_INTERNAL_SERVER_ERROR;
207 }
208
209 b->last = ngx_cpymem(b->last, prefix, sizeof(prefix) - 1);
210
211 if (brix_xml_write_text_element("Code", (const u_char *) code, codelen,
212 BRIX_XML_ESCAPE_APOS_ENTITY,
213 b->last, (size_t) (b->end - b->last),
214 &codelen) != 0)
215 {
216 return NGX_HTTP_INTERNAL_SERVER_ERROR;
217 }
218 b->last += codelen;
219
220 if (brix_xml_write_text_element("Message", (const u_char *) message, msglen,
221 BRIX_XML_ESCAPE_APOS_ENTITY,
222 b->last, (size_t) (b->end - b->last),
223 &msglen) != 0)
224 {
225 return NGX_HTTP_INTERNAL_SERVER_ERROR;
226 }
227 b->last += msglen;
228 b->last = ngx_cpymem(b->last, suffix, sizeof(suffix) - 1);
229
230 return brix_http_send_xml_buffer(r, status,
231 (ngx_str_t) ngx_string("application/xml"), b);
232}
