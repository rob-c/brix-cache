8/*
9 * WHAT: kXR_Qconfig — best-effort server capability query returning known feature flags as key=value lines.
10 * Parses whitespace-separated query keys from payload, responds with supported algorithms (chksum), readv support,
11 * TPC availability (tpc=1/0 based on allow_write+thread_pool), and HTTP-TPC delegation status (tpcdlg). Unknown keys return =0.
12 *
13 * WHY: XRootD clients (xrdcp, xrdfs) query server capabilities before attempting operations like TPC transfer or readv parallel reads.
14 * This response matches reference XRootD format so client libraries can parse and decide accordingly — e.g., XrdCl parses tpc line
15 * with isdigit()+atoi() expecting just "1" or "0". Empty query returns OK with no payload for compatibility.

19 * capability, each appending its key=value line(s) via qconfig_append (vsnprintf with capacity tracking).

23/* WHAT: Advances the pointer *pp past any whitespace characters (space, tab, newline, carriage return). Used as a preamble before extracting tokens from kXR_Qconfig query payload.
24 * WHY: kXR_Qconfig accepts whitespace-separated keys in its payload; this helper ensures token extraction starts at valid non-whitespace boundaries without accidentally capturing separator characters. Standard ASCII whitespace set covers all common delimiters used in client queries.
25 * HOW: Single while loop checking **pp against ' ', '\t', '\n', '\r' — increments pointer past each whitespace character until reaching a non-whitespace byte or null terminator. */
26
27static void
28brix_qconfig_skip_ws(const char **pp)
29{
30 while (**pp == ' ' || **pp == '\t' || **pp == '\n' || **pp == '\r') {
31 (*pp)++;
32 }
33}

35/* WHAT: Extracts a single token from the payload pointer *pp, skipping leading whitespace first then reading characters until next whitespace or null terminator. Stores extracted token in tok buffer with null termination, returns 1 on success (token found), 0 on failure (end of payload). Enforces tok_sz boundary to prevent overflow.
36 * WHY: kXR_Qconfig query payloads contain whitespace-separated capability keys (e.g., "tpc tpcdlg chksum"). This helper enables sequential token extraction without allocating temporary buffers or using strchr-based splitting — efficient for single-threaded nginx event loop processing.
37 * HOW: Two-phase → first calls brix_qconfig_skip_ws() to advance past leading whitespace, then reads characters while **pp != '\0' and not whitespace, storing each char in tok[len++] with null termination at len = tok_sz - 1 or end-of-token boundary. Returns 1 if token extracted, 0 if *pp points to '\0' (end of payload). */
38
39static ngx_flag_t
40brix_qconfig_next_token(const char **pp, char *tok, size_t tok_sz)
41{
42 size_t len;
43
44 brix_qconfig_skip_ws(pp);
45 if (**pp == '\0') {
46 return 0;
47 }
48
49 len = 0;
50 while (**pp != '\0' && **pp != ' ' && **pp != '\t'
51 && **pp != '\n' && **pp != '\r')
52 {
53 if (len + 1 < tok_sz) {
54 tok[len++] = **pp;
55 }
56 (*pp)++;
57 }
58
59 tok[len] = '\0';
60 return 1;
61}

63/* WHAT: Appends formatted text to a response buffer using vsnprintf, tracking the current position via *pos parameter. Returns 1 on success (formatted output fit within remaining buffer), 0 on failure (overflow or NULL pointers). Enforces resp_sz + pos bounds to prevent response buffer overflow during query capability reporting.
64 * WHY: kXR_Qconfig builds a multi-line capability report by appending individual key=value pairs; this helper ensures each append respects the 512-byte resp buffer limit without truncating mid-response or corrupting prior output. vsnprintf with remaining capacity calculation prevents format string attacks from exceeding bounds.
65 * HOW: Calculate remaining = resp_sz - *pos, call vsnprintf(resp + *pos, remaining, fmt, ap), check n < 0 || (size_t)n >= remaining for overflow → return 0 on failure or update *pos += n and return 1 on success. NULL pointer checks prevent crashes on malformed input. */
66
67static ngx_flag_t
68brix_qconfig_append(char *resp, size_t resp_sz, size_t *pos,
69 const char *fmt, ...)
70 __attribute__((format(printf, 4, 5)));

72static ngx_flag_t
73brix_qconfig_append(char *resp, size_t resp_sz, size_t *pos,
74 const char *fmt, ...)
75{
76 va_list ap;
77 int n;
78 size_t remaining;
79
80 if (resp == NULL || pos == NULL || *pos >= resp_sz) {
81 return 0;
82 }
83
84 remaining = resp_sz - *pos;
85
86 va_start(ap, fmt);
87 n = vsnprintf(resp + *pos, remaining, fmt, ap);
88 va_end(ap);
89
90 if (n < 0 || (size_t) n >= remaining) {
91 resp[*pos] = '\0';
92 return 0;
93 }
94
95 *pos += (size_t) n;
96 return 1;
97}

108/* WHAT: Builds a comma-separated list of the inline-compression codecs actually built into this binary
109 * (zstd/lz4/... per brix_codec_by_id availability) into list[list_sz].
110 * WHY: cmpread and cmpwrite both advertise the identical built-in codec set — sharing the walk keeps the
111 * two emitters symmetric and byte-identical, and avoids duplicating the overflow-guarded snprintf loop.
112 * HOW: Iterates codec ids 1..BRIX_CODEC_MAX, skipping unavailable descriptors, appending "name" with a ","
113 * separator after the first entry; stops early on snprintf overflow. list is always null-terminated. */
114static void
115brix_qconfig_codec_list(char *list, size_t list_sz)
116{
117 size_t lp = 0;
118 brix_codec_id_t cid;
119
120 list[0] = '\0';
121 for (cid = (brix_codec_id_t) 1; cid < BRIX_CODEC_MAX; cid++) {
122 const brix_codec_desc_t *d = brix_codec_by_id(cid);
123 int n;
124
125 if (d == NULL || !d->available) {
126 continue;
127 }
128 n = snprintf(list + lp, list_sz - lp, "%s%s",
129 lp ? "," : "", d->name);
130 if (n < 0 || (size_t) n >= list_sz - lp) {
131 break;
132 }
133 lp += (size_t) n;
134 }
135}
