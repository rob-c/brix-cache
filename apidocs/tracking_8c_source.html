8/*
9 * WHAT: Track Virtual Organization (VO) activity and unique user identities in shared memory.
10 * WHY: Prometheus metrics need VO-level and user-level aggregation without storing full DNs or token subjects
11 * as label strings (INVARIANT #8: low-cardinality labels only). FNV-1a hash provides fast 32-bit identity
12 * deduplication; VO slots track bytes_tx/bytes_rx per organization; user slots track session counts and unique count.
13 * HOW: Two tracking functions share the same pattern — iterate fixed-size slot array, find match via comparison (VO=string compare, USER=hash),
14 * increment atomic counters on match or allocate new slot on miss with eviction fallback when array fills. FNV-1a hash uses 0x811c9dc5 seed + XOR-multiply loop.
15 */

17/* brix_fnv1a_hash — 32-bit FNV-1a over arbitrary data, used to dedup identities
18 * (DNs, token subjects) by hash rather than storing them verbatim (INVARIANT #8 —
19 * no DN/token label explosion in Prometheus). */
20
21
22static ngx_inline uint32_t
23brix_fnv1a_hash(const void *data, size_t len)
24{
25 const u_char *p = (const u_char *) data;
26 uint32_t h = 0x811c9dc5u;
27
28 while (len--) {
29 h ^= *p++;
30 h *= 0x01000193u;
31 }
32 return h;
33}
