1/*
2 * stage_request_registry_query.c — read/query ops of the durable stage-request
3 * registry (split from stage_request_registry.c, phase-79).
4 *
5 * WHAT: The read side of the reqid-keyed durable request store — fetch a request
6 * by reqid, find the newest active request for a path, owner-check a reqid,
7 * enumerate active requests and a request's files, and translate a pin
8 * release into a cancel. Each op takes a SHARED whole-store lock (or none,
9 * for the compositions that delegate), scans fixed slots, and unlocks.
10 * WHY: The single 956-line registry owned three concerns; isolating the read
11 * ops here keeps every translation unit under the file-size cap and
12 * focused. The durable-store substrate (file I/O, locking, slot scan)
13 * stays in stage_request_registry.c and is reached only through the seam
14 * declared in stage_request_registry_internal.h — the lock/unlock
15 * boundaries and slot math are unchanged by the split.
16 * HOW: Each op is a flat early-return sequence over the substrate helpers:
17 * srq_lock(SHARE) -> srq_offset_by_reqid / slot scan -> srq_rec_read +
18 * srq_rec_valid -> srq_unlock -> srq_rec_to_view. The composite ops
19 * (owner_check, list_files, pin_release) reuse the public read ops rather
20 * than re-scanning.
21 */

29ngx_int_t
30brix_stage_request_get(brix_stage_registry_t *reg, const char *reqid,
31 brix_stage_request_t *out, ngx_log_t *log)
32{
33 srq_rec_t rec;
34 int64_t off;
35
36 if (reg == NULL || reg->fd < 0 || reqid == NULL || out == NULL) {
37 return NGX_ERROR;
38 }
39 if (srq_lock(reg, SRQ_LK_SHARE) != NGX_OK) {
40 return NGX_ERROR;
41 }
42 off = srq_offset_by_reqid(reg, reqid, log);
43 if (off < 0 || srq_rec_read(reg, off, &rec, log) != NGX_OK
44 || !srq_rec_valid(&rec, off) || rec.status == SRQ_ST_FREE)
45 {
46 srq_unlock(reg);
47 return NGX_DECLINED;
48 }
49 srq_unlock(reg);
50 srq_rec_to_view(&rec, out);
51 return NGX_OK;
52}

54ngx_int_t
55brix_stage_request_find_by_path(brix_stage_registry_t *reg, const char *lfn,
56 char *reqid_out, size_t reqid_out_sz, ngx_log_t *log)
57{
58 srq_rec_t rec;
59 int64_t off, size, best_off = -1, best_tod = -1;
60 char best_reqid[SRQ_REQID_LEN];
61
62 if (reg == NULL || reg->fd < 0 || lfn == NULL || reqid_out == NULL
63 || reqid_out_sz < SRQ_REQID_LEN)
64 {
65 return NGX_ERROR;
66 }
67 if (srq_lock(reg, SRQ_LK_SHARE) != NGX_OK) {
68 return NGX_ERROR;
69 }
70 size = srq_file_size(reg);
71 for (off = SRQ_REC_OFF(0);
72 size > 0 && off + (int64_t) SRQ_REC_SIZE <= size;
73 off += (int64_t) SRQ_REC_SIZE)
74 {
75 if (srq_rec_read(reg, off, &rec, log) != NGX_OK) {
76 srq_unlock(reg);
77 return NGX_ERROR;
78 }
79 if (!srq_rec_valid(&rec, off) || rec.status == SRQ_ST_FREE) {
80 continue;
81 }
82 if (ngx_strcmp(rec.lfn, lfn) == 0 && rec.tod_added >= best_tod) {
83 best_tod = rec.tod_added;
84 best_off = off;
85 ngx_cpystrn((u_char *) best_reqid, (u_char *) rec.reqid,
86 sizeof(best_reqid));
87 }
88 }
89 srq_unlock(reg);
90 if (best_off < 0) {
91 return NGX_DECLINED;
92 }
93 ngx_cpystrn((u_char *) reqid_out, (u_char *) best_reqid, reqid_out_sz);
94 return NGX_OK;
95}

99 const char *requester_dn, ngx_log_t *log)
100{
101 brix_stage_request_t rec;
102
103 if (requester_dn == NULL || requester_dn[0] == '\0') {
104 return NGX_OK; /* anonymous caller: nothing to enforce */
105 }
106 if (brix_stage_request_get(reg, reqid, &rec, log) != NGX_OK) {
107 return NGX_OK; /* absent/gone: idempotent, no oracle */
108 }
109 if (rec.requester_dn[0] == '\0'
110 || ngx_strcmp(rec.requester_dn, requester_dn) == 0)
111 {
112 return NGX_OK;
113 }
114 ngx_log_error(NGX_LOG_INFO, log, 0,
115 "stage-req: cancel of reqid \"%s\" denied — owned by a "
116 "different principal", reqid);
117 return NGX_DECLINED;
