267 /* ---- root:// block-write → whole-object staged-commit adapter (phase-70) ----
268 *
269 * WHAT: When a write open resolves to a backend LEAF that advertises NO
270 * BRIX_SD_CAP_RANDOM_WRITE and has no pwrite slot (a whole-object store
271 * such as sd_http — writes are a single commit-time PUT), the block-
272 * oriented root:// write model (kXR_open-for-write → kXR_write/pgwrite at
273 * offsets → kXR_sync/close) cannot open a session-writable driver handle.
274 * Instead this handle is put in STAGED mode: `staged` holds a VFS staged
275 * handle (opened via brix_vfs_staged_open, which resolves + forwards the
276 * per-user credential exactly like the read path and self-contains its
277 * ctx on the connection pool), and every kXR_write/pgwrite APPENDS its
278 * block through brix_vfs_staged_write at `staged_expected_off`. kXR_sync
279 * and kXR_close COMMIT the whole object via brix_vfs_staged_commit (a
280 * single whole-object PUT). The bare `fd` stays -1 and sd_obj.driver stays
281 * NULL — data does NOT route through xvfs_pwrite / the driver pwrite.
282 *
283 * WHY: This is the exact block-body → staged_commit bridge WebDAV/S3 PUT
284 * already use; the root:// write path had no equivalent, so a root://
285 * upload to a whole-object backend failed with EROFS. Uploads are
286 * sequential appends, so the adapter enforces sequential-append
287 * semantics: an out-of-order offset (!= staged_expected_off) is refused
288 * cleanly (kXR_Unsupported) rather than corrupting the object — genuinely
289 * random-offset writes to a whole-object store stay unsupported.
290 *
291 * HOW: `staged` != NULL is the sole "this handle is in staged mode" flag.
292 * `staged_expected_off` is the running append offset (next byte the
293 * object expects). `staged_committed` guards against a double commit
294 * (kXR_sync followed by kXR_close, or two syncs). The staged handle is
295 * released by brix_free_fhandle (abort if not yet committed).
296 */
297 brix_vfs_staged_t *staged; /* non-NULL = whole-object staged write */
298 off_t staged_expected_off; /* next sequential append offset */
299 unsigned staged_committed:1; /* 1 = object already PUT (sync/close) */
