9 * Bearer-token (JWT/WLCG) authentication for the "ztn" credential type.
10 */
11/*
12 * WHAT: brix_handle_token_auth() — handles kXR_auth with protocol "ztn"
13 * (WLCG/SciToken bearer JWT). Extracts token from payload, validates
14 * signature against JWKS keys + issuer + audience claims.
15 *
16 * WHY: Single-round authentication — no DH exchange needed. The client
17 * submits a pre-signed JWT and the server verifies it against trusted
18 * public keys (RSA/ECDSA) configured via brix_jwks_keys directive.
19 *
20 * HOW: 1) Extract token from payload; 2) Parse macaroon secret if configured;
21 * 3) Validate JWT signature + claims; 4) Grace-period fallback with old
22 * secret for key rotation during reload; 5) On success set ctx->login.auth_done,
23 * extract DN/groups/scopes from claims, track metrics, register session.
24 *
25 * Grace-period key rotation: if the primary macaroon secret rejects a token
26 * and an old secret is configured, retry validation with the old key.
27 *
28 * WHY: Allows in-flight tokens to survive nginx -s reload during key rotation.
29 * Without this fallback, tokens signed by the previous secret would be
30 * rejected immediately after reload, breaking active client sessions.
31 *
32 * Postconditions on success:
33 *
34 * WHY: ctx->login.auth_done = 1 enables subsequent authenticated operations.
35 * ctx->bearer_token stores raw token for proxy auth-bridging to upstream.
36 * DN, vo_list, primary_vo extracted from JWT claims (sub/groups).
37 * Token scopes stored for per-path scope checks. Unique user and VO
38 * tracked in shared-memory metrics counters.
39 *
40 * Return values: NGX_OK on successful validation, kXR_NotAuthorized error
41 * on empty payload, signature failure, or issuer/audience mismatch. */
42

52 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen <= 4) {
53 brix_log_access(ctx, c, "AUTH", "-", "ztn",
54 0, kXR_NotAuthorized, "empty token payload", 0);
55 BRIX_OP_ERR(ctx, BRIX_OP_AUTH);
56 return brix_send_error(ctx, c, kXR_NotAuthorized,
57 "empty bearer token");
58 }
59
60 token = (const char *) (ctx->recv.payload + 4);
61 token_len = ctx->recv.cur_dlen - 4;
62
63 while (token_len > 0 && token[token_len - 1] == '\0') {
64 token_len--;
65 }
66
67 if (token_len == 0) {
68 brix_log_access(ctx, c, "AUTH", "-", "ztn",
69 0, kXR_NotAuthorized, "empty token payload", 0);
70 BRIX_OP_ERR(ctx, BRIX_OP_AUTH);
71 return brix_send_error(ctx, c, kXR_NotAuthorized,
72 "empty bearer token");
73 }
74
75 u_char secret[64];
76 ssize_t slen = 0;
77
78 if (conf->token_macaroon_secret.len) {
79 slen = brix_macaroon_secret_parse(
80 (const char *) conf->token_macaroon_secret.data,
81 conf->token_macaroon_secret.len, secret, sizeof(secret));
82 }
83
84 /*
85 * Token-validation caches, consulted cheapest-first so token auth does not
86 * re-run RSA/ECDSA verification on the event loop under load:
87 * L1 — always-on, per-worker, lockless (lazily created here). A hit skips
88 * both the signature verification AND the L2 spinlock.
89 * L2 — the optional cross-worker SHM cache; an L2 hit is promoted into L1.
90 * Only successfully verified claims are ever cached.
91 */
92 int cache_hit = 0;
93
94 /* phase-59 W1: a registry config validates per-issuer; bypass the
95 * token-keyed caches. The matched issuer is stored on the identity so the
96 * per-path policy check enforces its base_path. */
97 const brix_token_issuer_t *reg_issuer = NULL;
98 int via_registry = (conf->token_registry != NULL);
99
100 if (conf->token_l1 == NULL) {
101 conf->token_l1 = brix_token_l1_create(ngx_cycle->pool,
102 BRIX_TOKEN_L1_SLOTS);
103 }
104
105 if (!via_registry
106 && brix_token_l1_lookup(conf->token_l1, token, token_len, &claims))
107 {
108 rc = 0;
109 cache_hit = 1;
110 } else if (!via_registry && conf->token_cache_kv != NULL
111 && brix_token_cache_lookup(conf->token_cache_kv,
112 token, token_len, &claims))
113 {
114 rc = 0;
115 cache_hit = 1;
116 brix_token_l1_store(conf->token_l1, token, token_len, &claims);
117 } else if (via_registry) {
118 rc = brix_token_validate_registry_authn(c->log, token, token_len,
119 conf->token_registry, slen > 0 ? secret : NULL, (size_t) slen,
120 (int) conf->token_clock_skew,
121 &claims, &reg_issuer);
122 } else {
123 rc = brix_token_validate(c->log, token, token_len,
124 conf->jwks_keys, conf->jwks_key_count,
125 (const char *) conf->token_issuer.data,
126 (const char *) conf->token_audience.data,
127 slen > 0 ? secret : NULL, (size_t) slen,
128 (int) conf->token_clock_skew,
129 &claims);
130 }
131
132 /* Grace-period fallback: if the primary secret rejected a macaroon token
133 * and an old secret is configured, try validating with the old key.
134 * This lets in-flight tokens survive nginx -s reload during key rotation. */
135 if (rc != 0 && !via_registry && conf->token_macaroon_secret_old.len) {
136 u_char old_secret[64];
137 ssize_t old_slen;
138
139 old_slen = brix_macaroon_secret_parse(
140 (const char *) conf->token_macaroon_secret_old.data,
141 conf->token_macaroon_secret_old.len,
142 old_secret, sizeof(old_secret));
143
144 if (old_slen > 0) {
145 rc = brix_token_validate(c->log, token, token_len,
146 conf->jwks_keys, conf->jwks_key_count,
147 (const char *) conf->token_issuer.data,
148 (const char *) conf->token_audience.data,
149 old_secret, (size_t) old_slen,
150 (int) conf->token_clock_skew,
151 &claims);
152 if (rc == 0) {
153 ngx_log_error(NGX_LOG_INFO, c->log, 0,
154 "brix: macaroon accepted via old secret "
155 "(grace-period key rotation)");
156 }
157 }
158 }
159
160 if (rc != 0) {
161 brix_log_access(ctx, c, "AUTH", "-", "ztn",
162 0, kXR_NotAuthorized, "token validation failed", 0);
163 BRIX_OP_ERR(ctx, BRIX_OP_AUTH);
164 return brix_send_error(ctx, c, kXR_NotAuthorized,
165 "bearer token validation failed");
166 }
167
168 /* Cache the freshly verified claims for subsequent presentations (L1 always,
169 * L2 when a SHM zone is configured). Registry tokens are never cached: the
170 * per-path decision is not a function of the token alone. */
171 if (!cache_hit && !via_registry) {
172 brix_token_l1_store(conf->token_l1, token, token_len, &claims);
173 if (conf->token_cache_kv != NULL) {
174 brix_token_cache_store(conf->token_cache_kv, token, token_len,
175 &claims);
176 }
177 }
178
179 ctx->token.auth = 1;
180 ctx->login.auth_done = 1;
181 if (ctx->identity != NULL
182 && brix_identity_set_token_claims(ctx->identity, c->pool, &claims)
183 != NGX_OK)
184 {
185 return brix_send_error(ctx, c, kXR_NoMemory,
186 "identity allocation failed");
187 }
188
189 /* phase-59 W1: record the matched issuer so the per-path scope check
190 * (brix_identity_check_token_scope) enforces its base_path. */
191 if (ctx->identity != NULL && reg_issuer != NULL) {
192 ctx->identity->token_issuer = (void *) reg_issuer;
193 }
194
195 /* Save raw token so proxy auth-bridging can forward it to the upstream. */
196 if (token_len > 0 && token_len < sizeof(ctx->bearer_token) - 1) {
197 ngx_memcpy(ctx->bearer_token, token, token_len);
198 ctx->bearer_token[token_len] = '\0';
199 }
200
201 ngx_cpystrn((u_char *) ctx->login.dn, (u_char *) claims.sub, sizeof(ctx->login.dn));
202
203 if (claims.groups[0]) {
204 const char *comma;
205 size_t pvo_len;
206
207 ngx_cpystrn((u_char *) ctx->login.vo_list,
208 (u_char *) claims.groups,
209 sizeof(ctx->login.vo_list));
210
211 comma = strchr(claims.groups, ',');
212 pvo_len = comma ? (size_t) (comma - claims.groups)
213 : strlen(claims.groups);
214 if (pvo_len >= sizeof(ctx->login.primary_vo)) {
215 pvo_len = sizeof(ctx->login.primary_vo) - 1;
216 }
217 ngx_memcpy(ctx->login.primary_vo, claims.groups, pvo_len);
218 ctx->login.primary_vo[pvo_len] = '\0';
219 }
220
221 /* Track unique user and VO at auth completion. */
222 {
223 ngx_brix_metrics_t *shm = brix_metrics_shared();
224 if (shm != NULL) {
225 size_t vo_len = strlen(ctx->login.primary_vo);
226 if (vo_len > 0 && vo_len < sizeof(ctx->login.primary_vo)) {
227 brix_track_vo_activity(shm, ctx->login.primary_vo, 0, 0);
228 ngx_uint_t vi;
229 for (vi = 0; vi < BRIX_VO_MAX_TRACKED; vi++) {
230 if (ngx_strncmp(shm->vo_global.slots[vi].name, ctx->login.primary_vo,
231 BRIX_VO_NAME_LEN) == 0)
232 {
233 BRIX_ATOMIC_INC(&shm->vo_global.slots[vi].requests_total);
234 break;
235 }
236 }
237 }
238 brix_track_unique_user(shm, ctx->login.dn, strlen(ctx->login.dn));
239 }
240 }
241
242 brix_session_register(ctx->login.sessid, ctx->login.dn, ctx->login.vo_list, 1);
243
244 ctx->token.scope_count = claims.scope_count;
245 for (i = 0; i < claims.scope_count && i < BRIX_MAX_TOKEN_SCOPES; i++) {
246 ctx->token.scopes[i] = claims.scopes[i];
247 }
248
249 ngx_log_error(NGX_LOG_INFO, c->log, 0,
250 "brix: token auth ok sub=\"%s\" scopes=%d groups=\"%s\"",
251 claims.sub, claims.scope_count, claims.groups);
252
253 BRIX_RETURN_OK(ctx, c, BRIX_OP_AUTH, "AUTH", "-", claims.sub, 0);
254}
