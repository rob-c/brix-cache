8/*
9 * Bearer-token (JWT/WLCG) authentication for the "ztn" credential type.
10 *
11 * The kXR_auth "ztn" conversation is decomposed into four sequential phases,
12 * each owning one nameable concern and passing state explicitly:
13 * tokenauth_extract — locate the raw token in the auth payload
14 * tokenauth_validate — cache lookup + signature/claim verification
15 * (incl. grace-period old-secret fallback + caching)
16 * tokenauth_map_identity — stamp ctx identity/login fields from claims
17 * tokenauth_bind_session — metrics, session registry, scopes, success log
18 * The orchestrator brix_handle_token_auth() reads as a flat early-return
19 * sequence over these phases. Wire bytes, error codes, log strings, metric
20 * calls and evaluation order are preserved 1:1 with the pre-split handler.
21 */
22
23/*
24 * tokenauth_state_t — per-request state threaded between the phase helpers.
25 *
26 * WHAT: bundles the token slice extracted from the payload, the verified
27 * claims, and the registry issuer matched during validation (NULL for
28 * non-registry configs) so each phase takes/returns exactly what it needs.
29 *
30 * WHY: keeps the phase-helper signatures small (§8 explicit data flow, no new
31 * globals) — the state lives on the orchestrator's stack and is passed by
32 * pointer, never through file-scope state.
33 *
34 * HOW: extract fills token/token_len; validate fills claims/reg_issuer; the
35 * identity/session phases read them.
36 */
37typedef struct {
38 const char *token;
39 size_t token_len;
40 const u_char *secret; /* active macaroon secret, or NULL */
41 size_t secret_len; /* 0 ⇒ no secret (macaroon → NULL) */
42 brix_token_claims_t claims;
43 const brix_token_issuer_t *reg_issuer;
44} tokenauth_state_t;

47/* ---- Slice the raw token out of a kXR_auth "ztn" payload (both framings) ----
48 *
49 * WHAT: point *token / *token_len at the raw JWT inside the auth payload,
50 * handling the two ztn credential framings we accept:
51 * - stock XrdSecProtocolztn TokenResp: an 8-byte TokenHdr
52 * ("ztn\0", ver=0, opr='T', rsvd[2]), a 2-byte big-endian length, then
53 * the token followed by a trailing NUL — the token starts at offset 10.
54 * - legacy brix framing: the 4-byte credtype echo ("ztn\0") followed by
55 * the token — the token starts at offset 4.
56 * Trailing NUL padding is stripped in both cases. Returns 0 on a non-empty
57 * token, -1 on an empty/all-NUL slice.
58 *
59 * WHY: a stock XRootD client (and now brix's own cache-origin ztn client, which
60 * speaks the standard wire format so it interoperates with a stock origin)
61 * sends the TokenResp framing. Detecting it here — rather than in the caller
62 * — keeps the deny logic in tokenauth_extract untouched. This only widens
63 * what we parse; the token itself is still fully validated downstream, so no
64 * security check is weakened.
65 *
66 * HOW: identify the stock framing by its fixed header bytes (dlen big enough,
67 * id "ztn", ver byte 0 at offset 4, opr 'T' at offset 5); use the embedded
68 * length to bound the token, else fall back to the legacy payload+4 slice.
69 * Trailing NULs are stripped from whichever slice we chose.
70 */
71static int
72tokenauth_slice_token(const u_char *payload, size_t dlen,
73 const char **token, size_t *token_len)
74{
75 const char *tok;
76 size_t tlen;
77
78 /* Stock XrdSecProtocolztn TokenResp: "ztn\0" ver=0 opr='T' rsvd[2] len[2] ...
79 * The minimum is the 10-byte prefix + a 1-byte (NUL-only) token slot. */
80 if (dlen >= 11
81 && ngx_memcmp(payload, "ztn", 4) == 0
82 && payload[4] == 0
83 && payload[5] == 'T')
84 {
85 size_t claimed = ((size_t) payload[8] << 8) | (size_t) payload[9];
86 size_t avail = dlen - 10; /* bytes after the 10B prefix */
87
88 tlen = (claimed > 0 && claimed <= avail) ? claimed : avail;
89 tok = (const char *) (payload + 10);
90 } else {
91 tok = (const char *) (payload + 4); /* legacy "ztn\0" + token */
92 tlen = dlen - 4;
93 }
94
95 while (tlen > 0 && tok[tlen - 1] == '\0') {
96 tlen--;
97 }
98 if (tlen == 0) {
99 return -1;
100 }
101
102 *token = tok;
103 *token_len = tlen;
104 return 0;
105}

108/* ---- Locate the bearer token inside the kXR_auth payload ----
109 *
110 * WHAT: validates the auth payload and points st->token / st->token_len at the
111 * raw JWT (via tokenauth_slice_token, which handles both the stock
112 * XrdSecProtocolztn TokenResp framing and the legacy payload+4 framing).
113 * Returns NGX_OK on a non-empty token; on an empty payload or all-NUL
114 * token it emits the standard error triplet and returns the
115 * kXR_NotAuthorized wire response verbatim.
116 *
117 * WHY: an empty bearer token is a hard deny — this phase is the sole gate that
118 * rejects it, preserving the exact "empty token payload"/"empty bearer
119 * token" log+wire strings the pre-split handler emitted.
120 *
121 * HOW: 1) reject payload==NULL or dlen<=4; 2) slice the token (stock or legacy
122 * framing) with trailing '\0' stripped; 3) reject a now-empty token; else
123 * NGX_OK. On the deny paths *out carries the wire-response value the
124 * orchestrator must return unchanged (brix_send_error's own return can be
125 * NGX_OK, so the caller keys on NGX_DONE — not on *out — to know a response
126 * was sent).
127 */
128static ngx_int_t
129tokenauth_extract(brix_ctx_t *ctx, ngx_connection_t *c, tokenauth_state_t *st,
130 ngx_int_t *out)
131{
132 const char *token;
133 size_t token_len;
134
135 if (ctx->recv.payload == NULL || ctx->recv.cur_dlen <= 4) {
136 brix_log_access(ctx, c, "AUTH", "-", "ztn",
137 0, kXR_NotAuthorized, "empty token payload", 0);
138 BRIX_OP_ERR(ctx, BRIX_OP_AUTH);
139 *out = brix_send_error(ctx, c, kXR_NotAuthorized,
140 "empty bearer token");
141 return NGX_DONE;
142 }
143
144 if (tokenauth_slice_token(ctx->recv.payload, ctx->recv.cur_dlen,
145 &token, &token_len) != 0)
146 {
147 brix_log_access(ctx, c, "AUTH", "-", "ztn",
148 0, kXR_NotAuthorized, "empty token payload", 0);
149 BRIX_OP_ERR(ctx, BRIX_OP_AUTH);
150 *out = brix_send_error(ctx, c, kXR_NotAuthorized,
151 "empty bearer token");
152 return NGX_DONE;
153 }
154
155 st->token = token;
156 st->token_len = token_len;
157 return NGX_OK;
158}

192/* ---- Try the cheapest-first token caches, then signature verification ----
193 *
194 * WHAT: attempts L1, then L2 (promoting hits into L1), then — on a miss —
195 * registry-authn (registry mode) or JWKS validation. Returns 0 when the
196 * token verified, non-zero otherwise; sets *cache_hit on a cache hit and
197 * fills st->claims / st->reg_issuer.
198 *
199 * WHY: isolates the "did we already verify this token?" fast path from the
200 * grace-period retry so each stays a small single-purpose step. Cheapest-
201 * first order (L1 → L2 → verify) is frozen; only verified claims are cached.
202 *
203 * HOW: 1) lazily create L1; 2) L1 lookup; 3) else L2 lookup (store into L1);
204 * 4) else registry-authn; 5) else JWKS validate via tokenauth_fill_va.
205 * The active macaroon secret is carried on st (st->secret/secret_len).
206 */
207static int
208tokenauth_verify_or_cache(ngx_connection_t *c,
209 ngx_stream_brix_srv_conf_t *conf, tokenauth_state_t *st,
210 int via_registry, int *cache_hit)
211{
212 /*
213 * Token-validation caches, consulted cheapest-first so token auth does not
214 * re-run RSA/ECDSA verification on the event loop under load:
215 * L1 — always-on, per-worker, lockless (lazily created here). A hit skips
216 * both the signature verification AND the L2 spinlock.
217 * L2 — the optional cross-worker SHM cache; an L2 hit is promoted into L1.
218 * Only successfully verified claims are ever cached.
219 */
220 if (conf->token_l1 == NULL) {
221 conf->token_l1 = brix_token_l1_create(ngx_cycle->pool,
222 BRIX_TOKEN_L1_SLOTS);
223 }
224
225 if (!via_registry
226 && brix_token_l1_lookup(conf->token_l1, st->token, st->token_len,
227 &st->claims))
228 {
229 *cache_hit = 1;
230 return 0;
231 }
232
233 if (!via_registry && conf->token_cache_kv != NULL
234 && brix_token_cache_lookup(conf->token_cache_kv, st->token,
235 st->token_len, &st->claims))
236 {
237 *cache_hit = 1;
238 brix_token_l1_store(conf->token_l1, st->token, st->token_len,
239 &st->claims);
240 return 0;
241 }
242
243 if (via_registry) {
244 brix_token_registry_args_t ra;
245
246 ra.log = c->log;
247 ra.token = st->token;
248 ra.token_len = st->token_len;
249 ra.reg = conf->token_registry;
250 ra.macaroon_secret = st->secret_len > 0 ? st->secret : NULL;
251 ra.secret_len = st->secret_len;
252 ra.clock_skew = (int) conf->token_clock_skew;
253 ra.claims = &st->claims;
254
255 return brix_token_validate_registry_authn(&ra, &st->reg_issuer);
256 }
257
258 {
259 brix_token_validate_args_t va;
260
261 tokenauth_fill_va(&va, c, conf, st);
262 return brix_token_validate(&va);
263 }
264}

267/* ---- Grace-period fallback: retry a rejected macaroon with the old secret ----
268 *
269 * WHAT: re-runs JWKS validation using the configured *old* macaroon secret.
270 * Returns 0 if the token now verifies (and logs the acceptance), the
271 * passed-in prev_rc unchanged if the old secret is absent/unparseable.
272 *
273 * WHY: lets in-flight tokens signed by the previous secret survive an nginx -s
274 * reload during key rotation. Only reached when the primary attempt failed
275 * and the config is non-registry — that gate stays in the caller.
276 *
277 * HOW: 1) parse the old secret; 2) if it parses, fill a va with it and
278 * validate; 3) on success log "accepted via old secret"; 4) return rc.
279 */
280static int
281tokenauth_grace_retry(ngx_connection_t *c, ngx_stream_brix_srv_conf_t *conf,
282 tokenauth_state_t *st, int prev_rc)
283{
284 u_char old_secret[64];
285 ssize_t old_slen;
286 brix_token_validate_args_t va;
287 int rc;
288
289 old_slen = brix_macaroon_secret_parse(
290 (const char *) conf->token_macaroon_secret_old.data,
291 conf->token_macaroon_secret_old.len,
292 old_secret, sizeof(old_secret));
293
294 if (old_slen <= 0) {
295 return prev_rc;
296 }
297
298 /* Point the shared va-builder at the old secret for this retry only. */
299 st->secret = old_secret;
300 st->secret_len = (size_t) old_slen;
301
302 tokenauth_fill_va(&va, c, conf, st);
303 rc = brix_token_validate(&va);
304 if (rc == 0) {
305 ngx_log_error(NGX_LOG_INFO, c->log, 0,
306 "brix: macaroon accepted via old secret "
307 "(grace-period key rotation)");
308 }
309 return rc;
310}

313/* ---- Run macaroon/JWT validation against JWKS keys or an issuer registry ----
314 *
315 * WHAT: verifies st->token, filling st->claims and (registry mode) st->reg_issuer.
316 * Returns 0 on a verified token (from L1/L2 cache OR fresh verification),
317 * non-zero if every validation attempt rejected it. Sets *cache_hit when
318 * the claims came from a cache (so the caller skips re-caching).
319 *
320 * WHY: isolates the cheapest-first cache ladder and the signature-verification
321 * dispatch (registry vs JWKS) plus the grace-period old-secret retry into
322 * one pure verification step; keeps deny logic (rc!=0) intact for the
323 * caller to act on. Order of cache/verify/fallback is frozen.
324 *
325 * HOW: 1) parse the primary macaroon secret if configured; 2) verify-or-cache;
326 * 3) if still rejected, non-registry, and an old secret exists, grace-retry;
327 * 4) on success, cache freshly verified claims (non-registry cache miss);
328 * 5) return rc.
329 */
330static int
331tokenauth_validate(ngx_connection_t *c, ngx_stream_brix_srv_conf_t *conf,
332 tokenauth_state_t *st)
333{
334 int via_registry = (conf->token_registry != NULL);
335 int cache_hit = 0;
336 u_char secret[64];
337 ssize_t slen = 0;
338 int rc;
339
340 if (conf->token_macaroon_secret.len) {
341 slen = brix_macaroon_secret_parse(
342 (const char *) conf->token_macaroon_secret.data,
343 conf->token_macaroon_secret.len, secret, sizeof(secret));
344 }
345
346 st->secret = secret;
347 st->secret_len = (size_t) (slen > 0 ? slen : 0);
348
349 rc = tokenauth_verify_or_cache(c, conf, st, via_registry, &cache_hit);
350
351 /* Grace-period fallback: if the primary secret rejected a macaroon token
352 * and an old secret is configured, try validating with the old key. */
353 if (rc != 0 && !via_registry && conf->token_macaroon_secret_old.len) {
354 rc = tokenauth_grace_retry(c, conf, st, rc);
355 }
356
357 /* Cache the freshly verified claims for subsequent presentations (L1 always,
358 * L2 when a SHM zone is configured). Registry tokens are never cached: the
359 * per-path decision is not a function of the token alone. A cache hit
360 * already holds the claims, so re-store only on a verified cache miss. */
361 if (rc == 0 && !cache_hit && !via_registry) {
362 brix_token_l1_store(conf->token_l1, st->token, st->token_len,
363 &st->claims);
364 if (conf->token_cache_kv != NULL) {
365 brix_token_cache_store(conf->token_cache_kv, st->token,
366 st->token_len, &st->claims);
367 }
368 }
369
370 return rc;
371}

374/* ---- Stamp ctx identity + login fields from the verified claims ----
375 *
376 * WHAT: sets ctx->token.auth / ctx->login.auth_done, populates the identity
377 * (claims + matched issuer), stashes the raw token for proxy bridging,
378 * and derives DN / vo_list / primary_vo from claims. Returns NGX_OK, or
379 * NGX_DONE (with *out set) on the kXR_NoMemory identity-allocation deny.
380 *
381 * WHY: concentrates the "authenticated client → ctx" mapping in one place; the
382 * only non-OK exit is the identity-allocation ENOMEM deny, preserved
383 * verbatim. (Claim caching happens in tokenauth_validate, next to the
384 * verification that produced them.)
385 *
386 * HOW: 1) set auth flags; 2) set identity claims (ENOMEM→deny) + issuer;
387 * 3) copy raw token into ctx->bearer_token; 4) copy DN; 5) split groups
388 * into vo_list + primary_vo (first comma-delimited group). Returns NGX_OK
389 * to continue; on the ENOMEM deny returns NGX_DONE with *out set to the
390 * wire-response value (brix_send_error can itself return NGX_OK, so the
391 * caller keys on NGX_DONE, not on *out).
392 */
393static ngx_int_t
394tokenauth_map_identity(brix_ctx_t *ctx, ngx_connection_t *c,
395 tokenauth_state_t *st, ngx_int_t *out)
396{
397 const brix_token_claims_t *claims = &st->claims;
398
399 ctx->token.auth = 1;
400 ctx->login.auth_done = 1;
401 if (ctx->identity != NULL
402 && brix_identity_set_token_claims(ctx->identity, c->pool, &st->claims)
403 != NGX_OK)
404 {
405 *out = brix_send_error(ctx, c, kXR_NoMemory,

502/*
503 * WHAT: brix_handle_token_auth() — handles kXR_auth with protocol "ztn"
504 * (WLCG/SciToken bearer JWT). Extracts token from payload, validates
505 * signature against JWKS keys + issuer + audience claims.
506 *
507 * WHY: Single-round authentication — no DH exchange needed. The client
508 * submits a pre-signed JWT and the server verifies it against trusted
509 * public keys (RSA/ECDSA) configured via brix_jwks_keys directive.
510 *
511 * HOW: 1) tokenauth_extract — token from payload (empty → deny);
512 * 2) tokenauth_validate — cache/verify + grace-period fallback
513 * (rejected → error triplet + kXR_NotAuthorized deny);
514 * 3) tokenauth_map_identity — auth flags, identity, DN/groups/scopes
515 * (ENOMEM → deny); 4) tokenauth_bind_session — metrics + registry + log.
516 *
517 * Return values: NGX_OK on successful validation, kXR_NotAuthorized error
518 * on empty payload, signature failure, or issuer/audience mismatch. */
519
520ngx_int_t
521brix_handle_token_auth(brix_ctx_t *ctx, ngx_connection_t *c,
522 ngx_stream_brix_srv_conf_t *conf)
523{
524 tokenauth_state_t st;
525 int rc;
526 ngx_int_t out = NGX_OK;
527
528 ngx_memzero(&st, sizeof(st));
529
530 if (tokenauth_extract(ctx, c, &st, &out) == NGX_DONE) {
531 return out;
532 }
533
534 rc = tokenauth_validate(c, conf, &st);
535 if (rc != 0) {
536 brix_log_access(ctx, c, "AUTH", "-", "ztn",
537 0, kXR_NotAuthorized, "token validation failed", 0);
538 BRIX_OP_ERR(ctx, BRIX_OP_AUTH);
539 return brix_send_error(ctx, c, kXR_NotAuthorized,
540 "bearer token validation failed");
541 }
542
543 if (tokenauth_map_identity(ctx, c, &st, &out) == NGX_DONE) {
544 return out;
545 }
546
547 tokenauth_bind_session(ctx, c, &st);
548
549 BRIX_RETURN_OK(ctx, c, BRIX_OP_AUTH, "AUTH", "-", st.claims.sub, 0);
550}
