8/* Format an epoch as "YYYY-MM-DD HH:MM:SSZ" (UTC), or "?" when unset. */
9void
10xrd_fmt_epoch(long e, char *buf, size_t sz)
11{
12 time_t t = (time_t) e;
13 struct tm tmv;
14 if (e == 0 || gmtime_r(&t, &tmv) == NULL) { snprintf(buf, sz, "?"); return; }
15 strftime(buf, sz, "%Y-%m-%d %H:%M:%SZ", &tmv);
16}

24/* Parse an HTTP IMF-fixdate ("Sun, 06 Nov 1994 08:49:37 GMT") to epoch, locale-free.
25 * 0 / -1. */
26int
27xrd_parse_http_date(const char *s, time_t *out)
28{
29 static const char *MON = "JanFebMarAprMayJunJulAugSepOctNovDec";
30 struct tm tmv;
31 char mname[8];
32 const char *m;
33 int d = 0, y = 0, hh = 0, mm = 0, ss = 0;
34
35 memset(&tmv, 0, sizeof(tmv));
36 if (sscanf(s, "%*3s, %d %3s %d %d:%d:%d", &d, mname, &y, &hh, &mm, &ss) != 6) {
37 return -1;
38 }
39 m = strstr(MON, mname);
40 if (m == NULL || (int) ((m - MON) % 3) != 0) { return -1; }
41 tmv.tm_mday = d; tmv.tm_mon = (int) ((m - MON) / 3); tmv.tm_year = y - 1900;
42 tmv.tm_hour = hh; tmv.tm_min = mm; tmv.tm_sec = ss;
43 *out = timegm(&tmv);
44 return (*out == (time_t) -1) ? -1 : 0;
45}

48/* Clock skew via an HTTP(S)/WebDAV endpoint's Date response header (read-only). */
49int
50xrd_clockskew_http(const char *endpoint, xrd_probe *p, char *err, size_t errsz)
51{
52 brix_weburl w;
53 brix_http_resp resp;
54 brix_status st;
55 struct timespec t0, t1;
56 char date[128];
57 time_t srv;
58 double rtt_s, local_mid;
59
60 if (brix_weburl_parse(endpoint, &w) != 0) { snprintf(err, errsz, "bad URL"); return -1; }
61 clock_gettime(CLOCK_REALTIME, &t0);
62 brix_status_clear(&st);
63 if (brix_http_req(w.host, w.port, w.tls, "HEAD", w.path[0] ? w.path : "/",
64 NULL, NULL, 0, 5000, 0 /*verify off for a clock probe*/,
65 NULL, &resp, &st) != 0) {
66 snprintf(err, errsz, "%s", st.msg);
67 return -1;
68 }
69 clock_gettime(CLOCK_REALTIME, &t1);
70 if (!brix_http_header(&resp, "Date", date, sizeof(date))
71 || xrd_parse_http_date(date, &srv) != 0) {
72 snprintf(err, errsz, "no parseable Date header");
73 brix_http_resp_free(&resp);
74 return -1;
75 }
76 brix_http_resp_free(&resp);
77 rtt_s = (double) (t1.tv_sec - t0.tv_sec) + (double) (t1.tv_nsec - t0.tv_nsec) / 1e9;
78 local_mid = (double) t0.tv_sec + (double) t0.tv_nsec / 1e9 + rtt_s / 2.0;
79 p->clock_have = 1;
80 p->clock_method = "HTTP Date header (1s granularity)";
81 p->server_epoch = (long) srv;
82 p->offset_s = (double) srv - local_mid;
83 p->rtt_ms = rtt_s * 1000.0;
84 return 0;
85}

88/* Clock skew via root://: create a temp file (server stamps mtime with its wall
89 * clock), stat it, compare to the local clock, then remove it. Needs write access. */
90int
91xrd_clockskew_root(const char *endpoint, const brix_opts *o, xrd_probe *p,
92 char *err, size_t errsz)
93{
94 brix_url u;
95 brix_conn c;
96 brix_status st;
97 brix_file f;
98 brix_statinfo si;
99 char tmp[128];
100 time_t t0, t1;
101
102 brix_status_clear(&st);
103 if (brix_endpoint_parse(endpoint, &u, &st) != 0) { snprintf(err, errsz, "%s", st.msg); return -1; }
104 if (brix_connect(&c, &u, o, &st) != 0) { snprintf(err, errsz, "connect: %s", st.msg); return -1; }
105 snprintf(tmp, sizeof(tmp), "/.xrd_clockskew_%ld", (long) getpid());
106 t0 = time(NULL);
107 if (brix_file_open_write(&c, tmp, 1 /*force*/, 0, &f, &st) != 0) {
108 snprintf(err, errsz, "need an HTTP endpoint or write access (%s)", st.msg);
109 brix_close(&c);
110 return -1;
111 }
112 brix_file_close(&c, &f, &st);
113 if (brix_stat(&c, tmp, &si, &st) != 0) {
114 snprintf(err, errsz, "stat: %s", st.msg);
115 { brix_status rs; brix_status_clear(&rs); brix_rm(&c, tmp, &rs); }
116 brix_close(&c);
117 return -1;
118 }
119 t1 = time(NULL);
120 { brix_status rs; brix_status_clear(&rs); brix_rm(&c, tmp, &rs); }
121 p->clock_have = 1;
122 p->clock_method = "root:// touch+stat (1s granularity)";
123 p->server_epoch = (long) si.mtime;
124 p->offset_s = (double) si.mtime - ((double) t0 + (double) t1) / 2.0;
125 p->rtt_ms = (double) (t1 - t0) * 1000.0;
126 brix_close(&c);
127 return 0;
128}

143/* `xrd clockskew <endpoint>` — report client↔server clock offset and RTT. Warns past
144 * 60 s (token exp/nbf + GSI validity start failing); exits nonzero past 5 min. */
145int
146xrd_clockskew(int argc, char **argv)
147{
148 const char *endpoint = (argc >= 3 && argv[2][0] != '-') ? argv[2] : NULL;
149 brix_opts o;
150 xrd_probe p;
151 char err[XRDC_MSG_MAX + 64] = "", sb[32]; /* room for a prefixed msg */
152 double ao;
153
154 if (endpoint == NULL) { fprintf(stderr, "usage: xrd clockskew <endpoint>\n"); return 50; }
155 memset(&o, 0, sizeof(o)); o.verify_host = 1;
156 memset(&p, 0, sizeof(p));
157 brix_crypto_init();
158 if (xrd_measure_clock_skew(endpoint, &o, &p, err, sizeof(err)) != 0 || !p.clock_have) {
159 fprintf(stderr, "xrd clockskew: %s\n", err[0] ? err : "unavailable");
160 return 1;
161 }
162 ao = xrd_fabs(p.offset_s);
163 xrd_fmt_epoch(p.server_epoch, sb, sizeof(sb));
164 printf("server time: %s (%s)\n", sb, p.clock_method);
165 printf("clock offset: %+.1f s (server is %s the client)\n", p.offset_s,
166 p.offset_s > 0.5 ? "ahead of" : (p.offset_s < -0.5 ? "behind" : "in sync with"));
167 printf("round-trip: %.1f ms\n", p.rtt_ms);
168 if (ao > 60.0) {
169 printf("WARNING: >60s skew — bearer-token exp/nbf and GSI/proxy validity "
170 "checks may reject you\n");
171 }
172 return (ao > 300.0) ? 1 : 0;
173}
