15/*
16 * Startup PKI consistency helpers.
17 *
18 * These checks are intentionally diagnostic: they log malformed CA/CRL setups
19 * but do not fail nginx configuration parsing. Production deployments often
20 * receive CA bundles and CRLs from external grid-security packaging, so a
21 * warning is more useful than taking the service down at reload time.
22 */
23/*
24 * brix_pki_load_certs_from_path — load all PEM-format CA certificates
25 * from a directory into an STACK_OF(X509).
26 *
27 * @path: filesystem directory containing .pem certificate files.
28 * @log: nginx log context for error reporting.
29 *
30 * Returns a freshly allocated STACK_OF(X509) on success, NULL on failure
31 * (logged as warning). The caller owns the returned stack and must free it
32 * with sk_X509_pop_free(stack, X509_free).
33 */
34STACK_OF(X509) *brix_pki_load_certs_from_path(const char *path,
35 BRIX_LOG_T *log);

49/*
50 * brix_pki_verify_crls — verify that every loaded CRL has a matching
51 * CA certificate and a valid cryptographic signature.
52 *
53 * @log: nginx log context.
54 * @ca_certs: STACK_OF(X509) of trusted CA certificates.
55 * @crls: STACK_OF(X509_CRL) of loaded CRLs to verify.
56 * @log_prefix: string prefix for error log messages.
57 *
58 * Two-phase verification per CRL:
59 * 1. DN matching — the CRL's issuer DN is compared against each CA
60 * subject DN to select a candidate CA.
61 * 2. Signature check — X509_CRL_verify confirms the CRL was signed by
62 * that CA's private key, preventing forgery.
63 *
64 * Returns always NGX_OK (errors are non-fatal, logged only). The caller
65 * should monitor the error log for mismatches or signature failures.
66 */
67BRIX_PKI_STATUS_T brix_pki_verify_crls(BRIX_LOG_T *log,
68 STACK_OF(X509) *ca_certs, STACK_OF(X509_CRL) *crls,
69 const char *log_prefix);
70/*
71 * brix_pki_check_paths — load CA certificates and CRLs from disk
72 * directories during nginx configuration phase, then verify cross-signatures.
73 *
74 * @log: nginx log context.
75 * @ca_path: filesystem directory of CA certificates (NULL = skip).
76 * @crl_path: filesystem directory of CRLs (NULL = skip).
77 * @log_prefix: string prefix for error log messages.
78 *
79 * Gracefully skips checks when either path is NULL. Always returns NGX_OK;
80 * problems are logged as warnings only so the server starts even with
81 * broken CA/CRL configuration rather than taking full outage.
82 */
83BRIX_PKI_STATUS_T brix_pki_check_paths(BRIX_LOG_T *log,
