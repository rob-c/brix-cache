24static int
25sd_cache_has_prefix(const char *path, ngx_array_t *prefixes)
26{
27 ngx_str_t *p;
28 ngx_uint_t i;
29
30 if (prefixes == NULL) {
31 return 0;
32 }
33 p = prefixes->elts;
34 for (i = 0; i < prefixes->nelts; i++) {
35 if (p[i].len > 0
36 && ngx_strncmp(path, p[i].data, p[i].len) == 0)
37 {
38 return 1;
39 }
40 }
41 return 0;
42}

44/* Decide whether `path` (of `size` bytes, or -1 when not yet known) may be cached
45 * under the policy: deny-prefix wins; a non-empty allow list must match; an
46 * include regex must match; an over-size object is not cached. */
47int
48sd_cache_admit(const brix_cache_policy_t *pol, const char *path, off_t size)
49{
50 if (sd_cache_has_prefix(path, pol->deny_prefixes)) {
51 return 0;
52 }
53 if (pol->allow_prefixes != NULL && pol->allow_prefixes->nelts > 0
54 && !sd_cache_has_prefix(path, pol->allow_prefixes))
55 {
56 return 0;
57 }
58 if (pol->include_regex != NULL
59 && regexec(pol->include_regex, path, 0, NULL, 0) != 0)
60 {
61 return 0;
62 }
63 if (size >= 0 && pol->max_file_size > 0 && size > pol->max_file_size) {
64 return 0;
65 }
66 return 1;
67}

111/* phase-68 T16: per-upstream (Stratum-1) fill attribution, gated on the cvmfs
112 * personality. Resolves the http source instance beneath the cache source and
113 * records this fill attempt against the "host:port" it answered from — the
114 * RAL-vs-CERN view. `ok`/`bytes`/`dur_ms` describe the attempt; failover comes
115 * from the http instance's last-read record. */
116void
117sd_cache_note_upstream(const sd_cache_inst_state *st, int ok, off_t bytes,
118 long dur_ms)
119{
120 brix_sd_instance_t *h = st->source;
121 char host[BRIX_CVMFS_UPSTREAM_NAME_MAX];
122 int guard = 0;
123
124 if (!sd_cache_is_cvmfs(st)) {
125 return;
126 }
127 /* Descend the source chain to the http instance (usually src itself). */
128 while (h != NULL && (h->driver == NULL || h->driver->name == NULL
129 || strcmp(h->driver->name, "http") != 0)
130 && guard++ < 8)
131 {
132 h = brix_sd_cache_source_instance(h);
133 }
134 if (h == NULL || sd_http_last_origin(h, host, sizeof(host)) != 0) {
135 return;
136 }
137 brix_cvmfs_upstream_record(host, ok, bytes, dur_ms,
138 sd_http_last_was_failover(h));
139}

141/* Milliseconds elapsed since a CLOCK_MONOTONIC start, clamped non-negative. */
142long
143sd_cache_ms_since(const struct timespec *t0)
144{
145 struct timespec now;
146 long ms;
147
148 if (clock_gettime(CLOCK_MONOTONIC, &now) != 0) {
149 return 0;
150 }
151 ms = (now.tv_sec - t0->tv_sec) * 1000L
152 + (now.tv_nsec - t0->tv_nsec) / 1000000L;
153 return ms < 0 ? 0 : ms;
154}

167/* phase-68 bounded stale-if-error: 1 iff an expired-but-COMPLETE cached copy
168 * of `key` exists whose total age is inside the 10x-TTL window — the caller
169 * absorbs the refill failure and the stale copy serves. Re-arms the entry's
170 * expiry one TTL forward so the next origin retry happens a TTL from now
171 * (not on every request); the 10x bound is keyed on filled_at, which re-arms
172 * never touch. */
173int
174sd_cache_stale_serve_ok(sd_cache_inst_state *st, const char *key)
175{
176 brix_cache_cinfo_t ci;
177 time_t now = time(NULL);
178 time_t ttl = st->policy.cvmfs_manifest_ttl;
179
180 if (ttl <= 0) {
181 return 0;
182 }
183 if (brix_cstore_cinfo_load(&st->cstore, key, &ci) != NGX_OK
184 || (ci.flags & BRIX_CINFO_F_COMPLETE) == 0
185 || brix_cache_cinfo_expired(&ci, now) != 1)
186 {
187 return 0;
188 }
189 if (ci.filled_at == 0
190 || (uint64_t) now >= ci.filled_at + 10 * (uint64_t) ttl)
191 {
192 return 0; /* stale window exhausted: fail hard */
193 }
194 ngx_log_error(NGX_LOG_WARN, st->log, 0,
195 "sd_cache: refill of \"%s\" failed; serving stale copy (%uL s past "
196 "expiry)\n"
197 " cause: origin unreachable or fill error\n"
198 " fix: check Stratum-1 reachability; stale serving stops at 10x "
199 "manifest_ttl", key,
200 (uint64_t) ((uint64_t) now - ci.expires_at));
201 brix_cache_cinfo_set_expires(&ci, now + ttl); /* next retry in one TTL */
202 (void) brix_cstore_cinfo_store(&st->cstore, key, &ci);
203 return 1;
204}
