12/*
13 * ngx_http_brix_shared_conf_t — Common fields embedded at the top of every
14 * protocol location/server config struct (stream, WebDAV, S3).
15 *
16 * WHAT: A shared preamble that holds enable flags, root path, write permission,
17 * and thread pool name — fields present in all three protocol configs. Each
18 * protocol struct embeds this struct as its first member so offsetof() offsets
19 * into the protocol-specific tail remain valid after merge.
20 *
21 * WHY: Stream, WebDAV, and S3 each duplicate enable + root + allow_write in
22 * their own structs and their create/merge functions (~90 total ngx_conf_merge_*
23 * calls). Consolidating these shared fields into one struct reduces merge
24 * boilerplate to ~30 protocol-specific calls plus a single preamble merge.
25 *
26 * HOW: Protocol structs declare this as their first member (no padding needed
27 * because it starts with ngx_flag_t which aligns naturally). The create function
28 * sets all shared fields to NGX_CONF_UNSET; the merge function uses standard
29 * nginx merge macros on each field before calling protocol-specific merge logic.
30 */
31
32typedef struct {
33 ngx_flag_t enable; /* on/off toggle for protocol */
34 ngx_str_t root; /* filesystem export root path */
35 char root_canon[PATH_MAX]; /* canonicalized/confined root */
36 ngx_str_t storage_backend; /* SD backend name: "" / "posix" = the
37 * default POSIX tree; "pblock" = the
38 * block-based backend rooted at root. */
39 size_t pblock_block_size; /* pblock stripe size for new files
40 * (bytes); 0 = backend default (64 MiB) */
41 ngx_flag_t storage_staging; /* write-back: a remote (root://) backend
42 * stages uploads to the LOCAL export and
43 * promotes them on commit, vs streaming
44 * straight through (Mode A). off = Mode A */
45 ngx_str_t storage_credential; /* [brix_storage_credential <name>] —
46 * the brix_credential block (§14) the
47 * source backend authenticates with;
48 * "" = anonymous. Today threads a bearer
49 * token into sd_http. */
50 void *storage_instance; /* resolved brix_sd_instance_t* for a
51 * non-POSIX backend, built per worker at
52 * init_process. Runtime only — never
53 * merged. NULL ⇒ default POSIX path. */
54 /* ---- phase-64 composable tier grammar (additive over storage_backend) ----
55 * Raw directive values parsed + registered at finalisation (the legacy cache
56 * directives that share a name — brix_cache, _verify, _slice, _dirty_max_age
57 * — are NOT re-used here; the new cache tier uses the non-colliding names and
58 * sensible defaults until the P2 legacy-removal big-bang). */
59 ngx_str_t cache_store; /* brix_cache_store URL ("" = none) */
60 ngx_array_t *cache_store_args; /* its credential=/block_size= tokens */
61 ngx_flag_t stage_enable; /* brix_stage on|off */
62 ngx_str_t stage_store; /* brix_stage_store URL */
63 ngx_array_t *stage_store_args;
64 ngx_uint_t stage_flush_async; /* brix_stage_flush: 0 sync, 1 async */
65 off_t cache_max_object; /* brix_cache_max_object (0 = no cap) */
66 ngx_uint_t cache_evict_at; /* brix_cache_evict_at (percent) */
67 ngx_uint_t cache_evict_to; /* brix_cache_evict_to (percent) */
68 ngx_uint_t cache_meta_mode; /* brix_cache_meta (0 auto..3 sidecar)*/
69 ngx_uint_t cache_verify_mode; /* brix_cache_verify_mode_e for the

71 * 0/UNSET = off. Registered today by

74 * "" = unlink the failed part. */

77 * 0 = no expiry stamping). */

82 * single-pass fill. */

86 size_t cache_index_cache; /* brix_cache_index_cache (L1 entries) */
87 size_t cache_slice_size; /* brix_cache_slice_size (0 = whole-file) */
88 ngx_flag_t allow_write; /* write permission flag */
89 ngx_flag_t read_only; /* hard read-only switch: when on, the
90 * finaliser forces allow_write off so
91 * EVERY write op is rejected at the
92 * protocol edge (root:// require_write,
93 * WebDAV/S3 method gate, write-open)
94 * before the VFS - and before token
95 * scope, so a write token cannot bypass
96 * it. Overrides allow_write on. */
97 ngx_flag_t compress; /* phase-42: outbound GET compression
98 * (Accept-Encoding negotiated). Off by

113 brix_pmark_conf_t pmark; /* SciTags packet-marking config — see

115 * protocol; init/merge below. */

118/*
119 * ngx_http_brix_shared_create_loc_conf() — Allocates and initializes a shared
120 * preamble struct with NGX_CONF_UNSET sentinel values. Called by each protocol's
121 * create_loc_conf function to set the shared fields before returning its own
122 * full config struct.
123 *
124 * WHY: nginx merge macros detect NGX_CONF_UNSET to know which value is unset;
125 * every protocol must initialize shared fields this way so parent→child merge
126 * works correctly regardless of whether enable/root/allow_write appear in main,
127 * server, or location blocks.
128 */

133 conf->allow_write = NGX_CONF_UNSET;

138 conf->cache_verify_mode = NGX_CONF_UNSET_UINT;

168/*
169 * brix_shared_apply_read_only() — enforce the hard read-only switch. When
170 * common->read_only is on, force allow_write off so EVERY existing write gate
171 * (root:// brix_dispatch_require_write, the WebDAV/S3 write-method gate, the
172 * write-open gate) rejects writes at the protocol edge - before the VFS, and
173 * before token scope (allow_write is checked first), so a write-scoped token
174 * cannot bypass it. ngx_http_brix_shared_merge() applies it after the
175 * allow_write/read_only merges so no protocol can forget the enforcement;
176 * callers with later allow_write-dependent validations (e.g. WebDAV's
177 * "writes need auth" check) simply run them after the shared merge.
178 */
179static inline void
180brix_shared_apply_read_only(ngx_http_brix_shared_conf_t *common,
181 ngx_log_t *log)
182{
183 if (common->read_only != 1) {
184 return;
185 }
186 if (common->allow_write == 1 && log != NULL) {
187 ngx_log_error(NGX_LOG_NOTICE, log, 0,
188 "brix: read_only on - the export is read-only; all write "
189 "operations are rejected at the protocol edge (overrides allow_write)");
190 }
191 common->allow_write = 0;
192}

194/*
195 * ngx_http_brix_shared_merge() — Merges shared preamble fields from parent to
196 * child using standard nginx merge macros. Called at the top of each protocol's
197 * merge_loc_conf function before protocol-specific merge logic runs.
198 *
199 * WHY: This is the SINGLE audit point for common.* config inheritance — every
200 * HTTP protocol (WebDAV, S3, cvmfs) calls it instead of hand-merging the same
201 * ~20 fields (which drifted per protocol and dropped the read-only enforcement
202 * in cvmfs). Defaults: enable=0, allow_write=0, compress=0, ktls=1,
203 * thread_pool_name="", tier grammar defaults as before.
204 *
205 * HOW: root_default parameterizes the one deliberate per-protocol difference
206 * (WebDAV exports default to "/", S3/cvmfs to ""). Ends by applying the hard
207 * read-only switch (see brix_shared_apply_read_only above) and merging pmark;
208 * returns NGX_CONF_OK or NGX_CONF_ERROR (pmark merge failure).
209 */
210static inline char *
211ngx_http_brix_shared_merge(ngx_conf_t *cf,
212 ngx_http_brix_shared_conf_t *prev,
213 ngx_http_brix_shared_conf_t *conf,
214 const char *root_default)
215{
216 ngx_conf_merge_value(conf->enable, prev->enable, 0);
217 /* root_default is a runtime const char* parameter, not a string literal;
218 * ngx_conf_merge_str_value computes sizeof(default)-1, which yields the
219 * pointer width (7 on 64-bit) rather than the actual string length — the
220 * empty-string default used by s3/cvmfs became {len:7,data:""}, so the
221 * pure-cache-node root.len==0 → "/" fallback never fired. Hand-roll. */
222 if (conf->root.data == NULL) {
223 if (prev->root.data != NULL) {
224 conf->root = prev->root;
225 } else {
226 conf->root.data = (u_char *) root_default;
227 conf->root.len = ngx_strlen(root_default);
228 }
229 }
230 ngx_conf_merge_value(conf->allow_write, prev->allow_write, 0);
231 ngx_conf_merge_value(conf->read_only, prev->read_only, 0);
232 ngx_conf_merge_value(conf->compress, prev->compress, 0);
233 ngx_conf_merge_value(conf->ktls, prev->ktls, 1); /* default ON (offload-gated) */
234 ngx_conf_merge_value(conf->storage_staging, prev->storage_staging, 0);
235 ngx_conf_merge_str_value(conf->thread_pool_name, prev->thread_pool_name, "");
236 ngx_conf_merge_str_value(conf->storage_backend, prev->storage_backend, "");
237 ngx_conf_merge_str_value(conf->storage_credential, prev->storage_credential,
238 "");
239 ngx_conf_merge_size_value(conf->pblock_block_size, prev->pblock_block_size,
240 0);
241
242 /* phase-64 tier grammar */
243 ngx_conf_merge_str_value(conf->cache_store, prev->cache_store, "");
244 if (conf->cache_store_args == NULL) {
245 conf->cache_store_args = prev->cache_store_args;
246 }
247 ngx_conf_merge_value(conf->stage_enable, prev->stage_enable, 0);
248 ngx_conf_merge_str_value(conf->stage_store, prev->stage_store, "");
249 if (conf->stage_store_args == NULL) {
250 conf->stage_store_args = prev->stage_store_args;
251 }
252 ngx_conf_merge_uint_value(conf->stage_flush_async, prev->stage_flush_async, 0);
253 ngx_conf_merge_off_value(conf->cache_max_object, prev->cache_max_object, 0);
254 ngx_conf_merge_uint_value(conf->cache_evict_at, prev->cache_evict_at, 90);
255 ngx_conf_merge_uint_value(conf->cache_evict_to, prev->cache_evict_to, 80);
256 ngx_conf_merge_uint_value(conf->cache_meta_mode, prev->cache_meta_mode, 0);
257 ngx_conf_merge_uint_value(conf->cache_batch_cinfo, prev->cache_batch_cinfo, 2);
258 ngx_conf_merge_size_value(conf->cache_index_cache, prev->cache_index_cache, 0);
259 ngx_conf_merge_size_value(conf->cache_slice_size, prev->cache_slice_size, 0);
260 /* 0 == BRIX_CACHE_VERIFY_OFF (fs/cache/verify.h; not included here — it
261 * drags stream-typed cache internals into every HTTP module conf). */
262 ngx_conf_merge_uint_value(conf->cache_verify_mode, prev->cache_verify_mode,
263 0);
264
265 /* Hard read-only: force allow_write off HERE so no protocol merge can
266 * forget the enforcement (it must win before token-scope checks). */
267 brix_shared_apply_read_only(conf, cf->log);
268
269 return brix_pmark_conf_merge(cf, &prev->pmark, &conf->pmark);
270}
