12/*
13 * ngx_http_brix_shared_conf_t — Common fields embedded at the top of every
14 * protocol location/server config struct (stream, WebDAV, S3).
15 *
16 * WHAT: A shared preamble that holds enable flags, root path, write permission,
17 * and thread pool name — fields present in all three protocol configs. Each
18 * protocol struct embeds this struct as its first member so offsetof() offsets
19 * into the protocol-specific tail remain valid after merge.
20 *
21 * WHY: Stream, WebDAV, and S3 each duplicate enable + root + allow_write in
22 * their own structs and their create/merge functions (~90 total ngx_conf_merge_*
23 * calls). Consolidating these shared fields into one struct reduces merge
24 * boilerplate to ~30 protocol-specific calls plus a single preamble merge.
25 *
26 * HOW: Protocol structs declare this as their first member (no padding needed
27 * because it starts with ngx_flag_t which aligns naturally). The create function
28 * sets all shared fields to NGX_CONF_UNSET; the merge function uses standard
29 * nginx merge macros on each field before calling protocol-specific merge logic.
30 */

269/*
270 * brix_shared_apply_read_only() — enforce the hard read-only switch. When
271 * common->read_only is on, force allow_write off so EVERY existing write gate
272 * (root:// brix_dispatch_require_write, the WebDAV/S3 write-method gate, the
273 * write-open gate) rejects writes at the protocol edge - before the VFS, and
274 * before token scope (allow_write is checked first), so a write-scoped token
275 * cannot bypass it. ngx_http_brix_shared_merge() applies it after the
276 * allow_write/read_only merges so no protocol can forget the enforcement;
277 * callers with later allow_write-dependent validations (e.g. WebDAV's
278 * "writes need auth" check) simply run them after the shared merge.
279 */
280static inline void
281brix_shared_apply_read_only(ngx_http_brix_shared_conf_t *common,
282 ngx_log_t *log)
283{
284 if (common->read_only != 1) {
285 return;
286 }
287 if (common->allow_write == 1 && log != NULL) {
288 ngx_log_error(NGX_LOG_NOTICE, log, 0,
289 "brix: read_only on - the export is read-only; all write "
290 "operations are rejected at the protocol edge (overrides allow_write)");
291 }
292 common->allow_write = 0;
293}

295/*
296 * ngx_http_brix_shared_merge() — Merges shared preamble fields from parent to
297 * child using standard nginx merge macros. Called at the top of each protocol's
298 * merge_loc_conf function before protocol-specific merge logic runs.
299 *
300 * WHY: This is the SINGLE audit point for common.* config inheritance — every
301 * HTTP protocol (WebDAV, S3, cvmfs) calls it instead of hand-merging the same
302 * ~20 fields (which drifted per protocol and dropped the read-only enforcement
303 * in cvmfs). Defaults: enable=0, allow_write=0, compress=0, ktls=1,
304 * thread_pool_name="", tier grammar defaults as before.
305 *
306 * HOW: root_default parameterizes the one deliberate per-protocol difference
307 * (WebDAV exports default to "/", S3/cvmfs to ""). Ends by applying the hard
308 * read-only switch (see brix_shared_apply_read_only above) and merging pmark;
309 * returns NGX_CONF_OK or NGX_CONF_ERROR (pmark merge failure).
310 */
311static inline char *
312ngx_http_brix_shared_merge(ngx_conf_t *cf,
313 ngx_http_brix_shared_conf_t *prev,
314 ngx_http_brix_shared_conf_t *conf,
315 const char *root_default)
316{
317 ngx_conf_merge_value(conf->enable, prev->enable, 0);
318 /* root_default is a runtime const char* parameter, not a string literal;
319 * ngx_conf_merge_str_value computes sizeof(default)-1, which yields the
320 * pointer width (7 on 64-bit) rather than the actual string length — the
321 * empty-string default used by s3/cvmfs became {len:7,data:""}, so the
322 * pure-cache-node root.len==0 → "/" fallback never fired. Hand-roll. */
323 if (conf->root.data == NULL) {
324 if (prev->root.data != NULL) {
325 conf->root = prev->root;
326 } else {
327 conf->root.data = (u_char *) root_default;
328 conf->root.len = ngx_strlen(root_default);
329 }
330 }
331 ngx_conf_merge_value(conf->allow_write, prev->allow_write, 0);
332 ngx_conf_merge_value(conf->read_only, prev->read_only, 0);
333 ngx_conf_merge_value(conf->compress, prev->compress, 0);
334 ngx_conf_merge_str_value(conf->access_log, prev->access_log, "");
335 if (conf->access_log.len > 0
336 && ngx_strcmp(conf->access_log.data, (u_char *) "off") != 0)
337 {
338 conf->access_log_file = ngx_conf_open_file(cf->cycle,
339 &conf->access_log);
340 if (conf->access_log_file == NULL) {
341 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
342 "brix: cannot register HTTP access log \"%V\"",
343 &conf->access_log);
344 return NGX_CONF_ERROR;
345 }
346 } else {
347 conf->access_log_file = NULL;
348 }
349 ngx_conf_merge_value(conf->session_log, prev->session_log, 1);
350 ngx_conf_merge_value(conf->ktls, prev->ktls, 1); /* default ON (offload-gated) */
351 ngx_conf_merge_value(conf->storage_staging, prev->storage_staging, 0);
352 ngx_conf_merge_str_value(conf->thread_pool_name, prev->thread_pool_name, "");
353 ngx_conf_merge_str_value(conf->storage_backend, prev->storage_backend, "");
354 ngx_conf_merge_str_value(conf->storage_credential, prev->storage_credential,
355 "");
356 ngx_conf_merge_str_value(conf->storage_credential_dir,
357 prev->storage_credential_dir, "");
358 ngx_conf_merge_uint_value(conf->storage_credential_fallback,
359 prev->storage_credential_fallback, 0);
360 ngx_conf_merge_str_value(conf->storage_credential_mint_ca_cert,
361 prev->storage_credential_mint_ca_cert, "");
362 ngx_conf_merge_str_value(conf->storage_credential_mint_ca_key,
363 prev->storage_credential_mint_ca_key, "");
364 ngx_conf_merge_uint_value(conf->storage_credential_mint_ttl,
365 prev->storage_credential_mint_ttl, 3600);
366 ngx_conf_merge_uint_value(conf->backend_delegation,
367 prev->backend_delegation, 0); /* SELECT */
368 ngx_conf_merge_ptr_value(conf->backend_token_aud,
369 prev->backend_token_aud, NULL);
370 ngx_conf_merge_str_value(conf->backend_tx_endpoint,
371 prev->backend_tx_endpoint, "");
372 ngx_conf_merge_str_value(conf->backend_tx_client_id,
373 prev->backend_tx_client_id, "");
374 ngx_conf_merge_str_value(conf->backend_tx_client_secret,
375 prev->backend_tx_client_secret, "");
376 ngx_conf_merge_str_value(conf->backend_sts_endpoint,
377 prev->backend_sts_endpoint, "");
378 ngx_conf_merge_str_value(conf->backend_sts_role,
379 prev->backend_sts_role, "");
380 ngx_conf_merge_value(conf->backend_krb5_forwardable,
381 prev->backend_krb5_forwardable, 0);
382 ngx_conf_merge_value(conf->backend_passthrough_persist,
383 prev->backend_passthrough_persist, 0);
384 ngx_conf_merge_size_value(conf->pblock_block_size, prev->pblock_block_size,
385 0);
386
387 /* phase-64 tier grammar */
388 ngx_conf_merge_str_value(conf->cache_store, prev->cache_store, "");
389 if (conf->cache_store_args == NULL) {
390 conf->cache_store_args = prev->cache_store_args;
391 }
392 ngx_conf_merge_value(conf->stage_enable, prev->stage_enable, 0);
393 ngx_conf_merge_str_value(conf->stage_store, prev->stage_store, "");
394 if (conf->stage_store_args == NULL) {
395 conf->stage_store_args = prev->stage_store_args;
396 }
397 ngx_conf_merge_uint_value(conf->stage_flush_async, prev->stage_flush_async, 0);
398 ngx_conf_merge_off_value(conf->cache_max_object, prev->cache_max_object, 0);
399 ngx_conf_merge_uint_value(conf->cache_evict_at, prev->cache_evict_at, 90);
400 ngx_conf_merge_uint_value(conf->cache_evict_to, prev->cache_evict_to, 80);
401 ngx_conf_merge_uint_value(conf->cache_meta_mode, prev->cache_meta_mode, 0);
402 ngx_conf_merge_uint_value(conf->cache_batch_cinfo, prev->cache_batch_cinfo, 2);
403 ngx_conf_merge_size_value(conf->cache_index_cache, prev->cache_index_cache, 0);
404 ngx_conf_merge_size_value(conf->cache_slice_size, prev->cache_slice_size, 0);
405 /* 0 == BRIX_CACHE_VERIFY_OFF (fs/cache/verify.h; not included here — it
406 * drags stream-typed cache internals into every HTTP module conf). */
407 ngx_conf_merge_uint_value(conf->cache_verify_mode, prev->cache_verify_mode,
408 0);
409
410 /* Hard read-only: force allow_write off HERE so no protocol merge can
411 * forget the enforcement (it must win before token-scope checks). */
412 brix_shared_apply_read_only(conf, cf->log);
413
414 return brix_pmark_conf_merge(cf, &prev->pmark, &conf->pmark);
415}
