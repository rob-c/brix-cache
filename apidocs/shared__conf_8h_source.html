17/* Default per-user credential store: a RAM-backed (tmpfs) directory, so
18 * delegated private keys never persist across a reboot, never land in

115/*
116 * brix_shared_apply_read_only() — enforce the hard read-only switch. When
117 * common->read_only is on, force allow_write off so EVERY existing write gate
118 * (root:// brix_dispatch_require_write, the WebDAV/S3 write-method gate, the
119 * write-open gate) rejects writes at the protocol edge - before the VFS, and
120 * before token scope (allow_write is checked first), so a write-scoped token
121 * cannot bypass it. ngx_http_brix_shared_merge() applies it after the
122 * allow_write/read_only merges so no protocol can forget the enforcement;
123 * callers with later allow_write-dependent validations (e.g. WebDAV's
124 * "writes need auth" check) simply run them after the shared merge.
125 */
126static inline void
127brix_shared_apply_read_only(ngx_http_brix_shared_conf_t *common,
128 ngx_log_t *log)
129{
130 if (common->read_only != 1) {
131 return;
132 }
133 if (common->allow_write == 1 && log != NULL) {
134 ngx_log_error(NGX_LOG_NOTICE, log, 0,
135 "brix: read_only on - the export is read-only; all write "
136 "operations are rejected at the protocol edge (overrides allow_write)");
137 }
138 common->allow_write = 0;
139}

141/*
142 * brix_shared_credential_dir_ensure() — Config-time guarantee for
143 * brix_storage_credential_dir: create the directory 0700 (chown'd to the
144 * worker user when the master runs as root) if it is missing, and shout —
145 * NGX_LOG_WARN at parse time — when it cannot be created, is not a
146 * directory, is group/other-accessible, or is owned by a user the workers
147 * do not run as.
148 *
149 * WHY: The store holds delegated PRIVATE KEYS, and the default lives on
150 * tmpfs (/dev/shm) precisely so nothing persists across reboots or lands in
151 * backups — but /dev/shm is world-writable (1777), so the entire security
152 * boundary is this directory's 0700 mode + ownership, which must therefore
153 * be enforced here rather than left to operator setup. A broken or
154 * foreign-owned path must NOT kill startup (the store may be unused, and
155 * fallback=allow keeps requests on the service credential), but the admin
156 * must be told that credential delegation will not work until it is fixed.
157 *
158 * HOW: stat(); on ENOENT mkdir(0700) + chown to ccf->user when euid==0
159 * (mirroring ngx_create_paths). Every failure and every unsafe pre-existing
160 * state warns via ngx_conf_log_error and returns — never fatal. Called from
161 * every shared merge (per server/location), so a broken path is shouted per
162 * context; the healthy path prints nothing and costs one stat().
163 */
164/* Resolved POST-de-escalation worker identity (defined in
165 * src/auth/impersonate/lifecycle_worker.c; contract in lifecycle.h). Workers
166 * are force-dropped to brix_worker_user/nobody when root-capable, so dirs
167 * provisioned for them must be owned by THAT identity, not raw ccf->user. */
168ngx_int_t brix_imp_worker_runtime_ids(ngx_uid_t conf_uid, ngx_gid_t conf_gid,
169 uid_t *uid_out, gid_t *gid_out);
170
171/* The uid/gid worker-writable provisioned dirs must be handed to: the runtime
172 * worker identity when the master runs as root, else the invoking user. */
173static inline void
174brix_shared_worker_dir_ids(ngx_conf_t *cf, uid_t *uid_out, gid_t *gid_out)
175{
176 ngx_core_conf_t *ccf = (ngx_core_conf_t *)
177 ngx_get_conf(cf->cycle->conf_ctx, ngx_core_module);
178
179 *uid_out = geteuid();
180 *gid_out = (gid_t) -1;
181 if (geteuid() == 0) {
182 (void) brix_imp_worker_runtime_ids(
183 (ccf != NULL) ? ccf->user : (ngx_uid_t) NGX_CONF_UNSET_UINT,
184 (ccf != NULL) ? (ngx_gid_t) ccf->group
185 : (ngx_gid_t) NGX_CONF_UNSET_UINT,
186 uid_out, gid_out); /* on failure the invoking-root ids stay */
187 }
188}

190static inline void
191brix_shared_credential_dir_ensure(ngx_conf_t *cf, const ngx_str_t *dir)
192{
193 struct stat st;
194 const char *path;
195 uid_t want_uid;
196 gid_t want_gid;
197
198 if (dir == NULL || dir->len == 0) {
199 return; /* explicit "" = per-user store disabled */
200 }
201
202 path = (const char *) dir->data; /* conf tokens are NUL-terminated */
203 brix_shared_worker_dir_ids(cf, &want_uid, &want_gid);
204
205 if (stat(path, &st) != 0) {
206 if (errno != ENOENT) {
207 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
208 "brix: credential store \"%s\" is not accessible — "
209 "credential delegation will not work until "
210 "brix_storage_credential_dir is fixed", path);
211 return;
212 }
213
214 if (mkdir(path, 0700) != 0 && errno != EEXIST) {
215 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
216 "brix: cannot create credential store \"%s\" — "
217 "credential delegation will not work until "
218 "brix_storage_credential_dir is fixed", path);
219 return;
220 }
221
222 if (stat(path, &st) != 0) {
223 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
224 "brix: credential store \"%s\" vanished after create — "
225 "credential delegation will not work", path);
226 return;
227 }
228
229 /* The master parses config as root but the workers write the store
230 * as the RUNTIME worker identity (the `user` account, or the
231 * de-escalation target for a root-capable worker) — hand the fresh
232 * directory to them, as ngx_create_paths does for the temp paths. */
233 if (geteuid() == 0 && st.st_uid != want_uid
234 && chown(path, want_uid, want_gid) != 0)
235 {
236 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
237 "brix: cannot chown credential store \"%s\" to the worker "
238 "user — credential delegation will not work", path);
239 }
240 return;
241 }
242
243 if (!S_ISDIR(st.st_mode)) {
244 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
245 "brix: credential store \"%s\" is not a directory — "
246 "credential delegation will not work until "
247 "brix_storage_credential_dir is fixed", path);
248 return;
249 }
250
251 if ((st.st_mode & 0077) != 0) {
252 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
253 "brix: credential store \"%s\" is group/other-accessible "
254 "(mode %04uo) — delegated private keys may be exposed; "
255 "chmod 0700", path, (ngx_uint_t) (st.st_mode & 07777));
256 }
257
258 if (st.st_uid != want_uid) {
259 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
260 "brix: credential store \"%s\" is owned by uid %ud but the "
261 "workers run as uid %ud — credential delegation will not work "
262 "until ownership or brix_storage_credential_dir is fixed",
263 path, (ngx_uint_t) st.st_uid, (ngx_uint_t) want_uid);
264 }
265}

267/*
268 * brix_shared_security_gate() — E-1: a valid-but-dangerous config setting is
269 * loud at load and refused under strict mode.
270 *
271 * WHY: several configurations parse cleanly yet leave the export wide open —
272 * anonymous S3 (no SigV4/token verification), WebDAV writes with auth optional,
273 * an anonymous dashboard. Each is a legitimate choice for a closed lab and a
274 * foot-gun in production, so the default must be loud (an operator who never
275 * reads the config still sees the warning in the error log at every reload),
276 * and a site that wants the guarantee flips `brix_strict_security on` to turn
277 * every such setting into a hard `nginx -t` failure — fail-closed, opt-in.
278 *
279 * HOW: emit NGX_LOG_WARN (default) or NGX_LOG_EMERG (strict) naming the insecure
280 * setting `what` and the directive `remedy` that closes it. Return NGX_OK when
281 * the merge may proceed (warn-only) and NGX_ERROR when strict mode requires the
282 * caller to return NGX_CONF_ERROR. The caller owns the return so the diagnostic
283 * points at the offending location's own merge.
284 */
285static inline ngx_int_t
286brix_shared_security_gate(ngx_conf_t *cf, ngx_flag_t strict,
287 const char *what, const char *remedy)
288{
289 ngx_conf_log_error(strict ? NGX_LOG_EMERG : NGX_LOG_WARN, cf, 0,
290 "brix: insecure configuration — %s; set %s to close it%s",
291 what, remedy,
292 strict ? " (refused: brix_strict_security on)" : "");
293 return strict ? NGX_ERROR : NGX_OK;
294}

296/*
297 * ngx_http_brix_shared_merge() — Merges shared preamble fields from parent to
298 * child using standard nginx merge macros. Called at the top of each protocol's
299 * merge_loc_conf function before protocol-specific merge logic runs.
300 *
301 * WHY: This is the SINGLE audit point for common.* config inheritance — every
302 * HTTP protocol (WebDAV, S3, cvmfs) calls it instead of hand-merging the same
303 * ~20 fields (which drifted per protocol and dropped the read-only enforcement
304 * in cvmfs). Defaults: enable=0, allow_write=0, compress=0, ktls=1,
305 * thread_pool_name="", storage_credential_dir=BRIX_CREDENTIAL_DIR_DEFAULT
306 * (tmpfs store, ensured 0700 by brix_shared_credential_dir_ensure above),
307 * tier grammar defaults as before.
308 *
309 * HOW: root_default parameterizes the one deliberate per-protocol difference
310 * (WebDAV exports default to "/", S3/cvmfs to ""). Ends by applying the hard
311 * read-only switch (see brix_shared_apply_read_only above) and merging pmark;
312 * returns NGX_CONF_OK or NGX_CONF_ERROR (pmark merge failure).
313 */
314static inline char *
315ngx_http_brix_shared_merge(ngx_conf_t *cf,
316 ngx_http_brix_shared_conf_t *prev,
317 ngx_http_brix_shared_conf_t *conf,
318 const char *root_default)
319{
320 ngx_conf_merge_value(conf->enable, prev->enable, 0);
321 /* root_default is a runtime const char* parameter, not a string literal;
322 * ngx_conf_merge_str_value computes sizeof(default)-1, which yields the
323 * pointer width (7 on 64-bit) rather than the actual string length — the
324 * empty-string default used by s3/cvmfs became {len:7,data:""}, so the
325 * pure-cache-node root.len==0 → "/" fallback never fired. Hand-roll. */
326 if (conf->root.data == NULL) {
327 if (prev->root.data != NULL) {
328 conf->root = prev->root;
329 } else {
330 conf->root.data = (u_char *) root_default;
331 conf->root.len = ngx_strlen(root_default);
332 }
333 }
334 ngx_conf_merge_value(conf->allow_write, prev->allow_write, 0);
335 ngx_conf_merge_value(conf->verify_write, prev->verify_write, 0);
336 ngx_conf_merge_value(conf->require_pgwrite, prev->require_pgwrite, 0);
337 ngx_conf_merge_value(conf->read_only, prev->read_only, 0);
338 ngx_conf_merge_value(conf->compress, prev->compress, 0);
339 ngx_conf_merge_value(conf->strict_security, prev->strict_security, 0);
340 ngx_conf_merge_str_value(conf->access_log, prev->access_log, "");
341 if (conf->access_log.len > 0
342 && ngx_strcmp(conf->access_log.data, (u_char *) "off") != 0)
343 {
344 conf->access_log_file = ngx_conf_open_file(cf->cycle,
345 &conf->access_log);
346 if (conf->access_log_file == NULL) {
347 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
348 "brix: cannot register HTTP access log \"%V\"",
349 &conf->access_log);
350 return NGX_CONF_ERROR;
351 }
352 } else {
353 conf->access_log_file = NULL;
354 }
355 ngx_conf_merge_value(conf->session_log, prev->session_log, 1);
356 ngx_conf_merge_value(conf->ktls, prev->ktls, 1); /* default ON (offload-gated) */
357 /* Trusted cache-store surface: default OFF everywhere, so the reserved-name
358 * 404 guard stays in force on every normal client location (default-deny). */
359 ngx_conf_merge_value(conf->cache_store_endpoint,
360 prev->cache_store_endpoint, 0);
361 ngx_conf_merge_value(conf->storage_staging, prev->storage_staging, 0);
362 ngx_conf_merge_str_value(conf->thread_pool_name, prev->thread_pool_name, "");
363 ngx_conf_merge_str_value(conf->storage_backend, prev->storage_backend, "");
364 ngx_conf_merge_str_value(conf->storage_credential, prev->storage_credential,
365 "");
366 /* Defaults to a RAM-backed (tmpfs) store so delegated keys never touch
367 * real disk; behaviour-neutral for non-delegated deployments because a
368 * lookup miss with fallback=allow (the default) lands on the service
369 * credential exactly like an unset dir. `""` opts out entirely. */
370 ngx_conf_merge_str_value(conf->storage_credential_dir,
371 prev->storage_credential_dir,
372 BRIX_CREDENTIAL_DIR_DEFAULT);
373 brix_shared_credential_dir_ensure(cf, &conf->storage_credential_dir);
374 ngx_conf_merge_uint_value(conf->storage_credential_fallback,
375 prev->storage_credential_fallback, 0);
376 ngx_conf_merge_str_value(conf->storage_credential_mint_ca_cert,
377 prev->storage_credential_mint_ca_cert, "");
378 ngx_conf_merge_str_value(conf->storage_credential_mint_ca_key,
379 prev->storage_credential_mint_ca_key, "");
380 ngx_conf_merge_uint_value(conf->storage_credential_mint_ttl,
381 prev->storage_credential_mint_ttl, 3600);
382 ngx_conf_merge_uint_value(conf->backend_delegation,
383 prev->backend_delegation, 0); /* SELECT */
384 ngx_conf_merge_ptr_value(conf->backend_token_aud,
385 prev->backend_token_aud, NULL);
386 ngx_conf_merge_str_value(conf->backend_tx_endpoint,
387 prev->backend_tx_endpoint, "");
388 ngx_conf_merge_str_value(conf->backend_tx_client_id,
389 prev->backend_tx_client_id, "");
390 ngx_conf_merge_str_value(conf->backend_tx_client_secret,
391 prev->backend_tx_client_secret, "");
392 ngx_conf_merge_str_value(conf->backend_sts_endpoint,
393 prev->backend_sts_endpoint, "");
394 ngx_conf_merge_str_value(conf->backend_sts_role,
395 prev->backend_sts_role, "");
396 ngx_conf_merge_value(conf->backend_krb5_forwardable,
397 prev->backend_krb5_forwardable, 0);
398 ngx_conf_merge_value(conf->backend_passthrough_persist,
399 prev->backend_passthrough_persist, 0);
400 ngx_conf_merge_size_value(conf->pblock_block_size, prev->pblock_block_size,
401 0);
402
403 /* phase-64 tier grammar */
404 ngx_conf_merge_str_value(conf->cache_store, prev->cache_store, "");
405 if (conf->cache_store_args == NULL) {
406 conf->cache_store_args = prev->cache_store_args;
407 }
408 ngx_conf_merge_str_value(conf->cache_cold_store, prev->cache_cold_store, "");
409 if (conf->cache_cold_store_args == NULL) {
410 conf->cache_cold_store_args = prev->cache_cold_store_args;
411 }
412 if (conf->cache_peers == NULL) {
413 conf->cache_peers = prev->cache_peers;
414 }
415 /* stage_enable keeps UNSET through the merge: brix_tier_register_stores
416 * must tell "never configured" (may auto-provision the default gateway
417 * stage store under /tmp/staging) apart from an explicit "brix_stage off"
418 * opt-out. Its only reader tests == 1, so UNSET still means off. */
419 ngx_conf_merge_value(conf->stage_enable, prev->stage_enable, NGX_CONF_UNSET);
420 ngx_conf_merge_str_value(conf->stage_store, prev->stage_store, "");
421 if (conf->stage_store_args == NULL) {
422 conf->stage_store_args = prev->stage_store_args;
423 }
424 ngx_conf_merge_uint_value(conf->stage_flush_async, prev->stage_flush_async, 0);
425 /* Durable async backend-op queue: default OFF (mutations run inline). When on,
426 * batch defaults to 64 ops (min 1) and the time backstop to 200ms. */
427 ngx_conf_merge_value(conf->backend_async, prev->backend_async, 0);
428 ngx_conf_merge_uint_value(conf->backend_async_batch,
429 prev->backend_async_batch, 64);
430 if (conf->backend_async_batch < 1) {
431 conf->backend_async_batch = 1;
432 }
433 ngx_conf_merge_msec_value(conf->backend_async_wait,
434 prev->backend_async_wait, 200);
435 ngx_conf_merge_off_value(conf->cache_max_object, prev->cache_max_object, 0);
436 /* evict_at/evict_to stay UNSET through the merge (inherit-only): the
437 * stream reaper merge must tell an explicit percent pair (which seeds the
438 * watermark reaper) apart from the documented 90/80 defaults — those are
439 * normalised into the tier policy at brix_tier_register_cache_store. */
440 ngx_conf_merge_uint_value(conf->cache_evict_at, prev->cache_evict_at,
441 NGX_CONF_UNSET_UINT);
442 ngx_conf_merge_uint_value(conf->cache_evict_to, prev->cache_evict_to,
443 NGX_CONF_UNSET_UINT);
444 ngx_conf_merge_uint_value(conf->cache_meta_mode, prev->cache_meta_mode, 0);
445 ngx_conf_merge_uint_value(conf->cache_batch_cinfo, prev->cache_batch_cinfo, 2);
446 ngx_conf_merge_size_value(conf->cache_index_cache, prev->cache_index_cache, 0);
447 ngx_conf_merge_size_value(conf->cache_slice_size, prev->cache_slice_size, 0);
448 /* 0 == BRIX_CACHE_VERIFY_OFF (fs/cache/verify.h; not included here — it
449 * drags stream-typed cache internals into every HTTP module conf). */
450 ngx_conf_merge_uint_value(conf->cache_verify_mode, prev->cache_verify_mode,
451 0);
452
453 /* Hard read-only: force allow_write off HERE so no protocol merge can
454 * forget the enforcement (it must win before token-scope checks). */
455 brix_shared_apply_read_only(conf, cf->log);
456
457 return brix_pmark_conf_merge(cf, &prev->pmark, &conf->pmark);
458}

470/*
471 * brix_tier_register_stores() — register the export's phase-64 composable
472 * cache/stage tiers from the common preamble onto the backend registry (which
473 * composes the sd_cache / sd_stage decorators per worker). Shared by all three
474 * protocol finalisers (§4.4): each calls it with its &conf->common after the
475 * storage backend + root_canon are set. Returns NGX_OK, or NGX_ERROR after an
476 * [emerg] for an operator error (unknown scheme, bad path, stage-without-store).
477 * Defined in config/runtime_server.c.
478 */
479ngx_int_t brix_tier_register_stores(ngx_conf_t *cf,
480 ngx_http_brix_shared_conf_t *common);
481
482/* Rewrite a "posix:<path>" / "pblock://<path>" storage_backend into the export root
