17/* Default per-user credential store: a RAM-backed (tmpfs) directory, so
18 * delegated private keys never persist across a reboot, never land in
19 * backups/snapshots, and leave no blocks on real disk. /dev/shm is mounted
20 * on effectively every Linux system, so no operator setup is required —
21 * the directory itself is created 0700 at config time (see
22 * brix_shared_credential_dir_ensure below). Opt out with an explicit
23 * `brix_storage_credential_dir "";`. */
24#define BRIX_CREDENTIAL_DIR_DEFAULT "/dev/shm/brix-creds"

26/*
27 * ngx_http_brix_shared_conf_t — Common fields embedded at the top of every
28 * protocol location/server config struct (stream, WebDAV, S3).
29 *
30 * WHAT: A shared preamble that holds enable flags, root path, write permission,
31 * and thread pool name — fields present in all three protocol configs. Each
32 * protocol struct embeds this struct as its first member so offsetof() offsets
33 * into the protocol-specific tail remain valid after merge.
34 *
35 * WHY: Stream, WebDAV, and S3 each duplicate enable + root + allow_write in
36 * their own structs and their create/merge functions (~90 total ngx_conf_merge_*
37 * calls). Consolidating these shared fields into one struct reduces merge
38 * boilerplate to ~30 protocol-specific calls plus a single preamble merge.
39 *
40 * HOW: Protocol structs declare this as their first member (no padding needed
41 * because it starts with ngx_flag_t which aligns naturally). The create function
42 * sets all shared fields to NGX_CONF_UNSET; the merge function uses standard
43 * nginx merge macros on each field before calling protocol-specific merge logic.
44 */

366/*
367 * brix_shared_apply_read_only() — enforce the hard read-only switch. When
368 * common->read_only is on, force allow_write off so EVERY existing write gate
369 * (root:// brix_dispatch_require_write, the WebDAV/S3 write-method gate, the
370 * write-open gate) rejects writes at the protocol edge - before the VFS, and
371 * before token scope (allow_write is checked first), so a write-scoped token
372 * cannot bypass it. ngx_http_brix_shared_merge() applies it after the
373 * allow_write/read_only merges so no protocol can forget the enforcement;
374 * callers with later allow_write-dependent validations (e.g. WebDAV's
375 * "writes need auth" check) simply run them after the shared merge.
376 */
377static inline void
378brix_shared_apply_read_only(ngx_http_brix_shared_conf_t *common,
379 ngx_log_t *log)
380{
381 if (common->read_only != 1) {
382 return;
383 }
384 if (common->allow_write == 1 && log != NULL) {
385 ngx_log_error(NGX_LOG_NOTICE, log, 0,
386 "brix: read_only on - the export is read-only; all write "
387 "operations are rejected at the protocol edge (overrides allow_write)");
388 }
389 common->allow_write = 0;
390}

392/*
393 * brix_shared_credential_dir_ensure() — Config-time guarantee for
394 * brix_storage_credential_dir: create the directory 0700 (chown'd to the
395 * worker user when the master runs as root) if it is missing, and shout —
396 * NGX_LOG_WARN at parse time — when it cannot be created, is not a
397 * directory, is group/other-accessible, or is owned by a user the workers
398 * do not run as.
399 *
400 * WHY: The store holds delegated PRIVATE KEYS, and the default lives on
401 * tmpfs (/dev/shm) precisely so nothing persists across reboots or lands in
402 * backups — but /dev/shm is world-writable (1777), so the entire security
403 * boundary is this directory's 0700 mode + ownership, which must therefore
404 * be enforced here rather than left to operator setup. A broken or
405 * foreign-owned path must NOT kill startup (the store may be unused, and
406 * fallback=allow keeps requests on the service credential), but the admin
407 * must be told that credential delegation will not work until it is fixed.
408 *
409 * HOW: stat(); on ENOENT mkdir(0700) + chown to ccf->user when euid==0
410 * (mirroring ngx_create_paths). Every failure and every unsafe pre-existing
411 * state warns via ngx_conf_log_error and returns — never fatal. Called from
412 * every shared merge (per server/location), so a broken path is shouted per
413 * context; the healthy path prints nothing and costs one stat().
414 */
415/* Resolved POST-de-escalation worker identity (defined in
416 * src/auth/impersonate/lifecycle_worker.c; contract in lifecycle.h). Workers
417 * are force-dropped to brix_worker_user/nobody when root-capable, so dirs
418 * provisioned for them must be owned by THAT identity, not raw ccf->user. */
419ngx_int_t brix_imp_worker_runtime_ids(ngx_uid_t conf_uid, ngx_gid_t conf_gid,
420 uid_t *uid_out, gid_t *gid_out);
421
422/* The uid/gid worker-writable provisioned dirs must be handed to: the runtime
423 * worker identity when the master runs as root, else the invoking user. */
424static inline void
425brix_shared_worker_dir_ids(ngx_conf_t *cf, uid_t *uid_out, gid_t *gid_out)
426{
427 ngx_core_conf_t *ccf = (ngx_core_conf_t *)
428 ngx_get_conf(cf->cycle->conf_ctx, ngx_core_module);
429
430 *uid_out = geteuid();
431 *gid_out = (gid_t) -1;
432 if (geteuid() == 0) {
433 (void) brix_imp_worker_runtime_ids(
434 (ccf != NULL) ? ccf->user : (ngx_uid_t) NGX_CONF_UNSET_UINT,
435 (ccf != NULL) ? (ngx_gid_t) ccf->group
436 : (ngx_gid_t) NGX_CONF_UNSET_UINT,
437 uid_out, gid_out); /* on failure the invoking-root ids stay */
438 }
439}

441static inline void
442brix_shared_credential_dir_ensure(ngx_conf_t *cf, const ngx_str_t *dir)
443{
444 struct stat st;
445 const char *path;
446 uid_t want_uid;
447 gid_t want_gid;
448
449 if (dir == NULL || dir->len == 0) {
450 return; /* explicit "" = per-user store disabled */
451 }
452
453 path = (const char *) dir->data; /* conf tokens are NUL-terminated */
454 brix_shared_worker_dir_ids(cf, &want_uid, &want_gid);
455
456 if (stat(path, &st) != 0) {
457 if (errno != ENOENT) {
458 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
459 "brix: credential store \"%s\" is not accessible — "
460 "credential delegation will not work until "
461 "brix_storage_credential_dir is fixed", path);
462 return;
463 }
464
465 if (mkdir(path, 0700) != 0 && errno != EEXIST) {
466 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
467 "brix: cannot create credential store \"%s\" — "
468 "credential delegation will not work until "
469 "brix_storage_credential_dir is fixed", path);
470 return;
471 }
472
473 if (stat(path, &st) != 0) {
474 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
475 "brix: credential store \"%s\" vanished after create — "
476 "credential delegation will not work", path);
477 return;
478 }
479
480 /* The master parses config as root but the workers write the store
481 * as the RUNTIME worker identity (the `user` account, or the
482 * de-escalation target for a root-capable worker) — hand the fresh
483 * directory to them, as ngx_create_paths does for the temp paths. */
484 if (geteuid() == 0 && st.st_uid != want_uid
485 && chown(path, want_uid, want_gid) != 0)
486 {
487 ngx_conf_log_error(NGX_LOG_WARN, cf, errno,
488 "brix: cannot chown credential store \"%s\" to the worker "
489 "user — credential delegation will not work", path);
490 }
491 return;
492 }
493
494 if (!S_ISDIR(st.st_mode)) {
495 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
496 "brix: credential store \"%s\" is not a directory — "
497 "credential delegation will not work until "
498 "brix_storage_credential_dir is fixed", path);
499 return;
500 }

512 "workers run as uid %ud — credential delegation will not work "

518/*
519 * brix_shared_security_gate() — E-1: a valid-but-dangerous config setting is
520 * loud at load and refused under strict mode.
521 *
522 * WHY: several configurations parse cleanly yet leave the export wide open —
523 * anonymous S3 (no SigV4/token verification), WebDAV writes with auth optional,
524 * an anonymous dashboard. Each is a legitimate choice for a closed lab and a
525 * foot-gun in production, so the default must be loud (an operator who never
526 * reads the config still sees the warning in the error log at every reload),
527 * and a site that wants the guarantee flips `brix_strict_security on` to turn
528 * every such setting into a hard `nginx -t` failure — fail-closed, opt-in.
529 *
530 * HOW: emit NGX_LOG_WARN (default) or NGX_LOG_EMERG (strict) naming the insecure
531 * setting `what` and the directive `remedy` that closes it. Return NGX_OK when
532 * the merge may proceed (warn-only) and NGX_ERROR when strict mode requires the
533 * caller to return NGX_CONF_ERROR. The caller owns the return so the diagnostic
534 * points at the offending location's own merge.
535 */
536static inline ngx_int_t
537brix_shared_security_gate(ngx_conf_t *cf, ngx_flag_t strict,
538 const char *what, const char *remedy)
539{
540 ngx_conf_log_error(strict ? NGX_LOG_EMERG : NGX_LOG_WARN, cf, 0,
541 "brix: insecure configuration — %s; set %s to close it%s",
542 what, remedy,
543 strict ? " (refused: brix_strict_security on)" : "");
544 return strict ? NGX_ERROR : NGX_OK;
545}

547/*
548 * ngx_http_brix_shared_merge() — Merges shared preamble fields from parent to
549 * child using standard nginx merge macros. Called at the top of each protocol's
550 * merge_loc_conf function before protocol-specific merge logic runs.
551 *
552 * WHY: This is the SINGLE audit point for common.* config inheritance — every
553 * HTTP protocol (WebDAV, S3, cvmfs) calls it instead of hand-merging the same
554 * ~20 fields (which drifted per protocol and dropped the read-only enforcement
555 * in cvmfs). Defaults: enable=0, allow_write=0, compress=0, ktls=1,
556 * thread_pool_name="", storage_credential_dir=BRIX_CREDENTIAL_DIR_DEFAULT
557 * (tmpfs store, ensured 0700 by brix_shared_credential_dir_ensure above),
558 * tier grammar defaults as before.
559 *
560 * HOW: root_default parameterizes the one deliberate per-protocol difference
561 * (WebDAV exports default to "/", S3/cvmfs to ""). Ends by applying the hard
562 * read-only switch (see brix_shared_apply_read_only above) and merging pmark;
563 * returns NGX_CONF_OK or NGX_CONF_ERROR (pmark merge failure).
564 */
565static inline char *
566ngx_http_brix_shared_merge(ngx_conf_t *cf,
567 ngx_http_brix_shared_conf_t *prev,
568 ngx_http_brix_shared_conf_t *conf,
569 const char *root_default)
570{
571 ngx_conf_merge_value(conf->enable, prev->enable, 0);
572 /* root_default is a runtime const char* parameter, not a string literal;
573 * ngx_conf_merge_str_value computes sizeof(default)-1, which yields the
574 * pointer width (7 on 64-bit) rather than the actual string length — the
575 * empty-string default used by s3/cvmfs became {len:7,data:""}, so the
576 * pure-cache-node root.len==0 → "/" fallback never fired. Hand-roll. */
577 if (conf->root.data == NULL) {
578 if (prev->root.data != NULL) {
579 conf->root = prev->root;
580 } else {
581 conf->root.data = (u_char *) root_default;
582 conf->root.len = ngx_strlen(root_default);
583 }
584 }
585 ngx_conf_merge_value(conf->allow_write, prev->allow_write, 0);
586 ngx_conf_merge_value(conf->verify_write, prev->verify_write, 0);
587 ngx_conf_merge_value(conf->require_pgwrite, prev->require_pgwrite, 0);
588 ngx_conf_merge_value(conf->read_only, prev->read_only, 0);
589 ngx_conf_merge_value(conf->compress, prev->compress, 0);
590 ngx_conf_merge_value(conf->strict_security, prev->strict_security, 0);
591 ngx_conf_merge_str_value(conf->access_log, prev->access_log, "");
592 if (conf->access_log.len > 0
593 && ngx_strcmp(conf->access_log.data, (u_char *) "off") != 0)
594 {
595 conf->access_log_file = ngx_conf_open_file(cf->cycle,
596 &conf->access_log);
597 if (conf->access_log_file == NULL) {
598 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
599 "brix: cannot register HTTP access log \"%V\"",
600 &conf->access_log);
601 return NGX_CONF_ERROR;
602 }
603 } else {
604 conf->access_log_file = NULL;
605 }
606 ngx_conf_merge_value(conf->session_log, prev->session_log, 1);
607 ngx_conf_merge_value(conf->ktls, prev->ktls, 1); /* default ON (offload-gated) */
608 /* Trusted cache-store surface: default OFF everywhere, so the reserved-name
609 * 404 guard stays in force on every normal client location (default-deny). */
610 ngx_conf_merge_value(conf->cache_store_endpoint,
611 prev->cache_store_endpoint, 0);
612 ngx_conf_merge_value(conf->storage_staging, prev->storage_staging, 0);
613 ngx_conf_merge_str_value(conf->thread_pool_name, prev->thread_pool_name, "");
614 ngx_conf_merge_str_value(conf->storage_backend, prev->storage_backend, "");
615 ngx_conf_merge_str_value(conf->storage_credential, prev->storage_credential,
616 "");
617 /* Defaults to a RAM-backed (tmpfs) store so delegated keys never touch
618 * real disk; behaviour-neutral for non-delegated deployments because a
619 * lookup miss with fallback=allow (the default) lands on the service
620 * credential exactly like an unset dir. `""` opts out entirely. */
621 ngx_conf_merge_str_value(conf->storage_credential_dir,
622 prev->storage_credential_dir,
623 BRIX_CREDENTIAL_DIR_DEFAULT);
624 brix_shared_credential_dir_ensure(cf, &conf->storage_credential_dir);
625 ngx_conf_merge_uint_value(conf->storage_credential_fallback,
626 prev->storage_credential_fallback, 0);
627 ngx_conf_merge_str_value(conf->storage_credential_mint_ca_cert,
628 prev->storage_credential_mint_ca_cert, "");
629 ngx_conf_merge_str_value(conf->storage_credential_mint_ca_key,
630 prev->storage_credential_mint_ca_key, "");
631 ngx_conf_merge_uint_value(conf->storage_credential_mint_ttl,
632 prev->storage_credential_mint_ttl, 3600);
633 ngx_conf_merge_uint_value(conf->backend_delegation,
634 prev->backend_delegation, 0); /* SELECT */
635 ngx_conf_merge_ptr_value(conf->backend_token_aud,
636 prev->backend_token_aud, NULL);
637 ngx_conf_merge_str_value(conf->backend_tx_endpoint,
638 prev->backend_tx_endpoint, "");
639 ngx_conf_merge_str_value(conf->backend_tx_client_id,
640 prev->backend_tx_client_id, "");
641 ngx_conf_merge_str_value(conf->backend_tx_client_secret,
642 prev->backend_tx_client_secret, "");
643 ngx_conf_merge_str_value(conf->backend_sts_endpoint,
644 prev->backend_sts_endpoint, "");
645 ngx_conf_merge_str_value(conf->backend_sts_role,
646 prev->backend_sts_role, "");
647 ngx_conf_merge_value(conf->backend_krb5_forwardable,
648 prev->backend_krb5_forwardable, 0);
649 ngx_conf_merge_value(conf->backend_passthrough_persist,
650 prev->backend_passthrough_persist, 0);
651 ngx_conf_merge_size_value(conf->pblock_block_size, prev->pblock_block_size,
652 0);
653
654 /* phase-64 tier grammar */
655 ngx_conf_merge_str_value(conf->cache_store, prev->cache_store, "");
656 if (conf->cache_store_args == NULL) {
657 conf->cache_store_args = prev->cache_store_args;
658 }
659 ngx_conf_merge_str_value(conf->cache_cold_store, prev->cache_cold_store, "");
660 if (conf->cache_cold_store_args == NULL) {
661 conf->cache_cold_store_args = prev->cache_cold_store_args;
662 }
663 if (conf->cache_peers == NULL) {
664 conf->cache_peers = prev->cache_peers;
665 }
666 /* stage_enable keeps UNSET through the merge: brix_tier_register_stores
667 * must tell "never configured" (may auto-provision the default gateway
668 * stage store under /tmp/staging) apart from an explicit "brix_stage off"
669 * opt-out. Its only reader tests == 1, so UNSET still means off. */
670 ngx_conf_merge_value(conf->stage_enable, prev->stage_enable, NGX_CONF_UNSET);
671 ngx_conf_merge_str_value(conf->stage_store, prev->stage_store, "");
672 if (conf->stage_store_args == NULL) {
673 conf->stage_store_args = prev->stage_store_args;
674 }
675 ngx_conf_merge_uint_value(conf->stage_flush_async, prev->stage_flush_async, 0);
676 ngx_conf_merge_off_value(conf->cache_max_object, prev->cache_max_object, 0);
677 /* evict_at/evict_to stay UNSET through the merge (inherit-only): the
678 * stream reaper merge must tell an explicit percent pair (which seeds the
679 * watermark reaper) apart from the documented 90/80 defaults — those are
680 * normalised into the tier policy at brix_tier_register_cache_store. */
681 ngx_conf_merge_uint_value(conf->cache_evict_at, prev->cache_evict_at,
682 NGX_CONF_UNSET_UINT);
683 ngx_conf_merge_uint_value(conf->cache_evict_to, prev->cache_evict_to,
684 NGX_CONF_UNSET_UINT);
685 ngx_conf_merge_uint_value(conf->cache_meta_mode, prev->cache_meta_mode, 0);
686 ngx_conf_merge_uint_value(conf->cache_batch_cinfo, prev->cache_batch_cinfo, 2);
687 ngx_conf_merge_size_value(conf->cache_index_cache, prev->cache_index_cache, 0);
688 ngx_conf_merge_size_value(conf->cache_slice_size, prev->cache_slice_size, 0);
689 /* 0 == BRIX_CACHE_VERIFY_OFF (fs/cache/verify.h; not included here — it
690 * drags stream-typed cache internals into every HTTP module conf). */
691 ngx_conf_merge_uint_value(conf->cache_verify_mode, prev->cache_verify_mode,
692 0);
693
694 /* Hard read-only: force allow_write off HERE so no protocol merge can
695 * forget the enforcement (it must win before token-scope checks). */
696 brix_shared_apply_read_only(conf, cf->log);
697
698 return brix_pmark_conf_merge(cf, &prev->pmark, &conf->pmark);
699}
