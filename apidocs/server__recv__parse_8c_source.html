1/*
2 * cms/server_recv_parse.c — CMS server-side wire payload decoders.
3 *
4 * WHAT: Pure decoders for the CMS frame payloads an accepted data server sends:
5 * the XrdOucPup TLV scalar walk, the OucPup string reader, the LOGIN
6 * (CmsLoginData) parser that fills ctx->{free_mb,util_pct,port,paths}, and the
7 * LOAD / AVAIL / SPACE load-figure extractors.
8 *
9 * WHY: Split out of the former monolithic server_recv.c (Phase-79 file-size
10 * split). These functions are side-effect-free parsers (no I/O, no socket, no
11 * registry) — isolating them keeps the "how the wire bytes decode" concern in
12 * one testable place, separate from dispatch and the event loop. The three
13 * entry points invoked by the frame handlers are declared in
14 * server_recv_internal.h and are therefore non-static here; the TLV/string/
15 * LOGIN-segment helpers stay file-local.
16 *
17 * HOW: tlv_read_next / cms_srv_read_string advance a (p, end) cursor by one
18 * tagged value; cms_srv_parse_login composes the scalar prologue decode with a
19 * newline-segment Paths walk; the LOAD/AVAIL parsers read fixed scalar layouts.
20 */
21

27/*
28 * The LOGIN/LOAD/AVAIL payloads use a simple type-tagged encoding:
29 * CMS_PT_SHORT (0x80) + big-endian uint16
30 * CMS_PT_INT (0xa0) + big-endian uint32
31 *
32 * Walk *p forward by one tagged value, returning the decoded uint32.
33 * Advances *p past the tag+value bytes. Returns 0 and sets *p=end on error.
34 */
35static uint32_t
36tlv_read_next(const u_char **p, const u_char *end)
37{
38 if (*p >= end) {
39 *p = end;
40 return 0;
41 }
42
43 if (**p == CMS_PT_SHORT) {
44 if (end - *p < 3) { *p = end; return 0; }
45 uint32_t v = ngx_brix_cms_get16(*p + 1);
46 *p += 3;
47 return v;
48 }
49
50 if (**p == CMS_PT_INT) {
51 if (end - *p < 5) { *p = end; return 0; }
52 uint32_t v = ngx_brix_cms_get32(*p + 1);
53 *p += 5;
54 return v;
55 }
56
57 /* Unknown tag — treat as end of parseable data. */
58 *p = end;
59 return 0;
60}

62/*
63 * Read one XrdOucPup string from *p: a 2-byte big-endian length followed by
64 * <len> raw bytes (the length includes the trailing NUL). Sets *out / *out_len
65 * to the data span and advances *p. Returns 1 on success, 0 (and *p=end) on a
66 * short/overrun buffer. Strings carry NO type tag — the high bit of the first
67 * length byte is clear for any reasonable length, which is how the real Parser
68 * tells a string apart from a PT_short/PT_int scalar.
69 */
70static int
71cms_srv_read_string(const u_char **p, const u_char *end,
72 const u_char **out, size_t *out_len)
73{
74 uint16_t len;
75
76 if (end - *p < 2) { *p = end; return 0; }
77 len = ngx_brix_cms_get16(*p);
78 *p += 2;
79 if ((size_t) (end - *p) < len) { *p = end; return 0; }
80 *out = *p;
81 *out_len = len;
82 *p += len;
83 return 1;
84}

88/*
89 * Parse the CMS LOGIN frame payload in the real XrdCms CmsLoginData wire format
90 * (XrdOucPup), the same one a real cmsd and our own cms/send.c now emit:
91 *
92 * PT_SHORT Version PT_INT Mode PT_INT HoldTime
93 * PT_INT tSpace PT_INT fSpace ←free_mb PT_INT mSpace
94 * PT_SHORT fsNum PT_SHORT fsUtil ←util_pct PT_SHORT dPort ←port
95 * PT_SHORT sPort [Fence]
96 * string SID string Paths string ifList string envCGI
97 *
98 * The Paths string is a newline-separated list of "<type> <namespace-path>"
99 * entries (type 'r'/'w', e.g. "w /\nw /atlas"). We strip the type prefix and
100 * store the bare namespace paths colon-delimited ("/:/atlas") — the form the
101 * registry's srv_path_matches() expects.
102 */
103
104/*
105 * cms_srv_login_scalars — decode the fixed PT_SHORT/PT_INT scalar prologue of
106 * the LOGIN payload (Version..sPort), capturing free_mb/util_pct/port on ctx.
107 * Splitting the scalar block from the string/paths negotiation keeps each
108 * piece independently checkable. Advances *p past the block; tlv_read_next()
109 * self-terminates on a short buffer so missing fields decode as 0.
110 */
111static void
112cms_srv_login_scalars(brix_cms_srv_ctx_t *ctx,
113 const u_char **p, const u_char *end)
114{
115 /* version */ (void) tlv_read_next(p, end);
116 /* mode */ (void) tlv_read_next(p, end);
117 /* holdtime */ (void) tlv_read_next(p, end);
118 /* tSpace */ (void) tlv_read_next(p, end);
119 ctx->free_mb = tlv_read_next(p, end); /* fSpace */
120 /* mSpace */ (void) tlv_read_next(p, end);
121 /* fsNum */ (void) tlv_read_next(p, end);
122 ctx->util_pct = tlv_read_next(p, end); /* fsUtil */
123 ctx->port = (uint16_t) tlv_read_next(p, end); /* dPort */
124 /* sPort */ (void) tlv_read_next(p, end);
125}

127/*
128 * cms_srv_login_seg_path — reduce one "<type> <path>" Paths segment to the
129 * bare path: drop the leading type token up to and including the first run of
130 * spaces (a segment without a space is taken verbatim), then trim trailing
131 * blanks. Adjusts *tok / *tok_len in place; *tok_len may become 0.
132 */
133static void
134cms_srv_login_seg_path(const u_char **tok, size_t *tok_len)
135{
136 size_t sp = 0;
137
138 /* split off the leading "<type> " prefix, if any */
139 while (sp < *tok_len && (*tok)[sp] != ' ') {
140 sp++;
141 }
142 if (sp < *tok_len) { /* found a space → path follows it */
143 sp++;
144 while (sp < *tok_len && (*tok)[sp] == ' ') {
145 sp++;
146 }
147 *tok += sp;
148 *tok_len -= sp;
149 }
150
151 while (*tok_len > 0
152 && ((*tok)[*tok_len - 1] == ' ' || (*tok)[*tok_len - 1] == '\t'))
153 {
154 (*tok_len)--;
155 }
156}

158/*
159 * cms_srv_login_next_path — scan the newline-separated Paths string for the
160 * next segment starting at *i, skipping blank/NUL padding, and hand back the
161 * bare path span via cms_srv_login_seg_path(). Advances *i past the segment.
162 * Returns 1 with *tok / *tok_len set (len may be 0), 0 when exhausted.
163 */
164static int
165cms_srv_login_next_path(const u_char *paths, size_t paths_len, size_t *i,
166 const u_char **tok, size_t *tok_len)
167{
168 size_t seg_end;
169
170 while (*i < paths_len
171 && (paths[*i] == '\n' || paths[*i] == ' '
172 || paths[*i] == '\t' || paths[*i] == '\0'))
173 {
174 (*i)++;
175 }
176 if (*i >= paths_len) {
177 return 0;
178 }
179
180 /* find end of this segment (newline / NUL terminated) */
181 seg_end = *i;
182 while (seg_end < paths_len
183 && paths[seg_end] != '\n' && paths[seg_end] != '\0')
184 {
185 seg_end++;
186 }
187
188 *tok = paths + *i;
189 *tok_len = seg_end - *i;
190 cms_srv_login_seg_path(tok, tok_len);
191
192 *i = seg_end;
193 return 1;
194}

196/*
197 * cms_srv_login_append_path — append one bare path to the colon-delimited
198 * ctx->paths buffer, truncating at the buffer boundary exactly like the
199 * original inline copy loop. Empty tokens (type-only segments) are dropped.
200 */
201static void
202cms_srv_login_append_path(brix_cms_srv_ctx_t *ctx, u_char **dst,
203 u_char *dst_end, const u_char *tok, size_t tok_len)
204{
205 size_t cp;
206
207 if (tok_len == 0) {
208 return;
209 }
210
211 if (*dst != (u_char *) ctx->paths && *dst < dst_end) {
212 *(*dst)++ = ':';
213 }
214 cp = (size_t) (dst_end - *dst);
215 if (cp > tok_len) {
216 cp = tok_len;
217 }
218 ngx_memcpy(*dst, tok, cp);
219 *dst += cp;
220}

222int
223cms_srv_parse_login(brix_cms_srv_ctx_t *ctx,
224 const u_char *payload, size_t payload_len)
225{
226 const u_char *p = payload;
227 const u_char *end = payload + payload_len;
228 const u_char *sid;
229 const u_char *paths;
230 const u_char *tok;
231 size_t sid_len;
232 size_t paths_len;
233 size_t tok_len;
234 u_char *dst;
235 u_char *dst_end;
236 size_t i;
237
238 cms_srv_login_scalars(ctx, &p, end);
239
240 /* SID (ignored) then Paths (extracted); ifList ignored, envCGI scanned
241 * for the vnid token below. */
242 (void) cms_srv_read_string(&p, end, &sid, &sid_len);
243 if (!cms_srv_read_string(&p, end, &paths, &paths_len)) {
244 paths = NULL;
245 paths_len = 0;
246 }
247
248 /*
249 * Convert "<type> <path>\n<type> <path>..." -> colon-delimited bare paths.
250 * Each newline segment is "<type> <path>"; the path follows the first
251 * space. A segment without a space is taken verbatim as the path.
252 */
253 dst = (u_char *) ctx->paths;
254 dst_end = dst + sizeof(ctx->paths) - 1;
255 i = 0;
256 while (dst < dst_end
257 && cms_srv_login_next_path(paths, paths_len, &i, &tok, &tok_len))
258 {
259 cms_srv_login_append_path(ctx, &dst, dst_end, tok, tok_len);
260 }
261 *dst = '\0';
262
263 /*
264 * Phase-89 W9: extract the virtual network id from envCGI ("&"-separated
265 * CGI tokens; stock cmsd emits "vnid=<id>"). ifList is skipped to reach
266 * it in wire order. Absent/empty → ctx->vnid stays "".
267 */
268 ctx->vnid[0] = '\0';
269 {
270 const u_char *iflist, *env, *tok;
271 size_t iflist_len, env_len, i, vl;
272
273 if (cms_srv_read_string(&p, end, &iflist, &iflist_len)
274 && cms_srv_read_string(&p, end, &env, &env_len))
275 {
276 for (i = 0; i + 5 <= env_len; i++) {
277 if ((i == 0 || env[i - 1] == '&')
278 && ngx_strncmp(env + i, "vnid=", 5) == 0)
279 {
280 tok = env + i + 5;
281 vl = 0;
282 while (i + 5 + vl < env_len && tok[vl] != '&'
283 && tok[vl] != '\0' && vl < sizeof(ctx->vnid) - 1)
284 {
285 ctx->vnid[vl] = (char) tok[vl];
286 vl++;
287 }
288 ctx->vnid[vl] = '\0';
289 break;
290 }
291 }
292 }
293 }
294
295 /* Default XRootD port if the data server didn't advertise one. */
296 if (ctx->port == 0) {
297 ctx->port = BRIX_DEFAULT_PORT;
298 }
299
300 return 1;
301}

305/*
306 * LOAD payload (from cms/send.c):
307 * PT_SHORT count=6 (3 bytes)
308 * <6 raw bytes> (cpu load values, ignored)
309 * PT_INT free_mb (5 bytes) ← extracted
310 */
311uint32_t
312cms_srv_parse_load_free_mb(const u_char *payload, size_t payload_len)
313{
314 const u_char *p = payload;
315 const u_char *end = payload + payload_len;
316
317 /* count field */
318 (void) tlv_read_next(&p, end);
319
320 /* skip 6 raw CPU bytes */
321 if (end - p >= 6) {
322 p += 6;
323 } else {
324 return 0;
325 }
326
327 return tlv_read_next(&p, end);
328}

330/*
331 * cms_srv_parse_load_machine_pct — extract the machine-load percentage from
332 * the LOAD payload's 6 theLoad bytes: the max of cpu/net/xeq/mem/pag (the
333 * bottleneck resource; dsk is excluded — disk fill is already tracked as
334 * util_pct/free_mb). Returns 0-100; a short payload reads as 0 (idle),
335 * matching the parser's general missing-field-decodes-as-zero posture.
336 */
337uint32_t
338cms_srv_parse_load_machine_pct(const u_char *payload, size_t payload_len)
339{
340 uint32_t best = 0;
341 size_t i;
342
343 /* [2-byte blob len][6 load bytes ...] — the 5 machine bytes start at 2. */
344 if (payload_len < 2 + 5) {
345 return 0;
346 }
347 for (i = 2; i < 2 + 5; i++) {
348 if (payload[i] > best) {
349 best = payload[i];
350 }
351 }
352 return best > 100 ? 100 : best;
353}
