1/*
2 * cpool.c — a small thread-safe pool of opaque connections (generic engine).
3 *
4 * WHAT: brix_cpool_create/checkout/checkin/destroy over a caller-defined
5 * connection type (vtable {conn_size, connect, close}). Extracted from
6 * lib/net/pool.c so the binary root:// path (brix_pool adapter) and the
7 * HTTP keep-alive metadata path (brix_webmeta) share ONE pool.
8 * WHY: a connection is one-op-in-flight and NOT thread-safe; a multi-threaded
9 * consumer needs N independent connections. The pool is the concurrency
10 * primitive; the transport is a vtable parameter.
11 * HOW: one mutex guards slot bookkeeping only (never held across connect/op);
12 * a condvar wakes a waiter on checkin; the vtable connects/closes the
13 * opaque per-slot memory. No goto. Clean-room (libc + pthread only).
14 */
15#include "brix.h"
16#include "net/cpool.h"

18#include <pthread.h>
19#include <stdlib.h>

21typedef struct {
22 void *conn; /* calloc(1, vt->conn_size) — opaque to the pool */
23 int connected; /* 1 once a successful connect/reconnect has run */
24 int in_use; /* checked out by a thread */
25} cpool_slot;

36/* Bring a reserved (in_use) slot to a connected state; lock NOT held. */
37static int
38cpool_slot_connect(brix_cpool *p, cpool_slot *s, brix_status *st)
39{
40 if (s->connected) {
41 return 0;
42 }
43 if (p->vt.connect(s->conn, p->ctx, st) != 0) {
44 return -1;
45 }
46 s->connected = 1;
47 return 0;
48}

50brix_cpool *
51brix_cpool_create(const brix_cpool_vtbl *vt, void *ctx, int n, brix_status *st)
52{
53 brix_cpool *p;
54 int i;
55
56 if (vt == NULL || vt->connect == NULL || vt->close == NULL
57 || vt->conn_size == 0 || n < 1) {
58 brix_status_set(st, XRDC_EUSAGE, 0, "cpool: bad arguments");
59 return NULL;
60 }
61 if (n > 256) {
62 n = 256; /* sanity cap */
63 }
64 p = calloc(1, sizeof(*p));
65 if (p == NULL) {
66 brix_status_set(st, XRDC_ESOCK, 0, "cpool: out of memory");
67 return NULL;
68 }
69 p->slots = calloc((size_t) n, sizeof(*p->slots));
70 if (p->slots == NULL) {
71 free(p);
72 brix_status_set(st, XRDC_ESOCK, 0, "cpool: out of memory");
73 return NULL;
74 }
75 for (i = 0; i < n; i++) {
76 p->slots[i].conn = calloc(1, vt->conn_size);
77 if (p->slots[i].conn == NULL) {
78 while (--i >= 0) {
79 free(p->slots[i].conn);
80 }
81 free(p->slots);
82 free(p);
83 brix_status_set(st, XRDC_ESOCK, 0, "cpool: out of memory");
84 return NULL;
85 }
86 }
87 p->vt = *vt;
88 p->ctx = ctx;
89 p->n = n;
90 pthread_mutex_init(&p->lock, NULL);
91 pthread_cond_init(&p->avail, NULL);
92
93 /* Connect slot 0 eagerly so a bad endpoint / auth fails up front; the
94 * remaining slots connect lazily on first use. */
95 p->slots[0].in_use = 1;
96 if (cpool_slot_connect(p, &p->slots[0], st) != 0) {
97 p->slots[0].in_use = 0;
98 brix_cpool_destroy(p);
99 return NULL;
100 }
101 p->slots[0].in_use = 0;
102 return p;
103}

105void *
106brix_cpool_checkout(brix_cpool *p, brix_status *st)
107{
108 cpool_slot *s = NULL;
109 int i;
110
111 if (p == NULL) {
112 brix_status_set(st, XRDC_EUSAGE, 0, "cpool: null");
113 return NULL;
114 }
115 pthread_mutex_lock(&p->lock);
116 for (;;) {
117 for (i = 0; i < p->n; i++) {
118 if (!p->slots[i].in_use) {
119 s = &p->slots[i];
120 s->in_use = 1;
121 break;
122 }
123 }
124 if (s != NULL) {
125 break;
126 }
127 pthread_cond_wait(&p->avail, &p->lock); /* all busy: wait for checkin */
128 }
129 pthread_mutex_unlock(&p->lock);
130
131 /* Connect/reconnect outside the lock (the slot is reserved). */
132 if (cpool_slot_connect(p, s, st) != 0) {
133 pthread_mutex_lock(&p->lock);
134 s->in_use = 0;
135 pthread_cond_signal(&p->avail);
136 pthread_mutex_unlock(&p->lock);
137 return NULL;
138 }
139 return s->conn;
140}

142void
143brix_cpool_checkin(brix_cpool *p, void *conn, int healthy)
144{
145 int i;
146
147 if (p == NULL || conn == NULL) {
148 return;
149 }
150 /* A connection-level failure (healthy==0) means this conn's socket/session
151 * is suspect — close it and clear `connected` so the next checkout
152 * reconnects on a clean session rather than handing back a dead socket. */
153 if (!healthy) {
154 for (i = 0; i < p->n; i++) {
155 if (p->slots[i].conn == conn) {
156 if (p->slots[i].connected) {
157 p->vt.close(p->slots[i].conn);
158 p->slots[i].connected = 0;
159 }
160 break;
161 }
162 }
163 }
164 pthread_mutex_lock(&p->lock);
165 for (i = 0; i < p->n; i++) {
166 if (p->slots[i].conn == conn) {
167 p->slots[i].in_use = 0;
168 break;
169 }
170 }
171 pthread_cond_signal(&p->avail);
172 pthread_mutex_unlock(&p->lock);
173}

175void
176brix_cpool_destroy(brix_cpool *p)
177{
178 int i;
179
180 if (p == NULL) {
181 return;
182 }
183 for (i = 0; i < p->n; i++) {
184 if (p->slots[i].connected) {
185 p->vt.close(p->slots[i].conn);
186 }
187 free(p->slots[i].conn);
188 }
189 pthread_mutex_destroy(&p->lock);
190 pthread_cond_destroy(&p->avail);
191 free(p->slots);
192 free(p);
193}
