1/*
2 * capture.c — session capture to a portable .xrdcap bundle + offline replay.
3 *
4 * WHAT: Record every wire frame (both directions) plus negotiated session
5 * metadata into a self-describing file; decode it later offline (no
6 * server), or re-issue the recorded requests against a live server.
7 * WHY: "Paste the bytes into the bug." A capture is a deterministic artifact a
8 * server bug can be reproduced from — decode it anywhere, or replay it to
9 * reproduce the exchange — without a packet sniffer or a live cluster.
10 * HOW: Writer hooks sit beside the wire-trace hooks in frame.c (inert when the
11 * sink is NULL). The format is an 8-byte magic then a stream of records:
12 * 'M' meta : klen:1 key[klen] vlen:2BE val[vlen]
13 * 'F' frame: dir:1 isreq:1 sid:2BE code:2BE wirelen:4BE wire[wirelen]
14 * where `wire` is the exact on-the-wire bytes (header+body), so replay can
15 * both decode (code→name via trace.c) and re-issue requests verbatim.
16 *
17 * Clean-room: our own framing; reuses trace.c name decoders. No XrdCl.
18 */

66void
67brix_capture_meta(struct brix_capture *cap, const char *key, const char *val)
68{
69 size_t kl, vl;
70 if (cap == NULL || cap->fp == NULL || key == NULL) {
71 return;
72 }
73 if (val == NULL) { val = ""; }
74 kl = strlen(key);
75 vl = strlen(val);
76 if (kl > 255) { kl = 255; }
77 if (vl > 0xffff) { vl = 0xffff; }
78 fputc('M', cap->fp);
79 fputc((int) kl, cap->fp);
80 fwrite(key, 1, kl, cap->fp);
81 put_u16(cap->fp, (uint16_t) vl);
82 fwrite(val, 1, vl, cap->fp);
83}

85void
86brix_capture_frame(struct brix_capture *cap, int dir, uint16_t sid, int code,
87 int is_request, const void *hdr, uint32_t hdrlen,
88 const void *body, uint32_t blen)

91 return;

93 fputc('F', cap->fp);
94 fputc(dir, cap->fp);
95 fputc(is_request ? 1 : 0, cap->fp);

98 put_u32(cap->fp, hdrlen + blen);

102 if (blen > 0 && body != NULL) {
103 fwrite(body, 1, blen, cap->fp);

132/*
133 * skip_bytes — consume and discard `n` bytes from fp, detecting truncation.
134 *
135 * WHAT: reads n bytes into a scratch buffer, returning 0 when all n bytes were
136 * present or -1 when fewer remain (a truncated/corrupt record).
137 * WHY: fseek(SEEK_CUR) past EOF SUCCEEDS on a regular file, so seeking over a
138 * record body cannot distinguish a complete record from one truncated
139 * mid-body — the skip silently accepts a short file. An fread-based skip
140 * surfaces the short read exactly as the replay path's body reads do, so a
141 * mid-record truncation fails cleanly instead of passing as valid.
142 * HOW: loop reading up to a fixed scratch buffer until n bytes are consumed or
143 * fread returns short. */
144static int
145skip_bytes(FILE *fp, size_t n)
146{
147 char buf[512];
148 while (n > 0) {
149 size_t want = (n < sizeof(buf)) ? n : sizeof(buf);
150 if (fread(buf, 1, want, fp) != want) {
151 return -1;
152 }
153 n -= want;
154 }
155 return 0;
156}

158static uint16_t rd_u16(FILE *fp) { int a = fgetc(fp), b = fgetc(fp); return ((uint16_t)(uint8_t)a << 8) | (uint16_t)(uint8_t)b; }
159static uint32_t rd_u32(FILE *fp) {
160 uint32_t a = (uint32_t) fgetc(fp), b = (uint32_t) fgetc(fp);
161 uint32_t c = (uint32_t) fgetc(fp), d = (uint32_t) fgetc(fp);
162 return (a << 24) | (b << 16) | (c << 8) | d;
163}

165/*
166 * Per-record outcome shared by the replay/playback loops.
167 *
168 * WHAT: the three ways a single M/F record read can end.
169 * WHY: lets each per-record helper return a typed verdict so the driving loop
170 * stays a flat early-return sequence (no truncated flag threaded by hand,
171 * no goto) while preserving the exact "truncated or corrupt → exit
172 * nonzero" contract the security test depends on.
173 * HOW: OK continues the loop, END stops it cleanly, TRUNC stops it as an error.
174 */
175typedef enum {
176 XRDCAP_REC_OK = 0, /* record consumed; keep looping */
177 XRDCAP_REC_END, /* clean end of stream */
178 XRDCAP_REC_TRUNC /* short/corrupt record — caller reports the error */
179} xrdcap_rec_t;

181/*
182 * replay_meta_record — decode one 'M' metadata record and print key/value.
183 *
184 * WHAT: reads klen:1 key[klen] vlen:2BE val[vlen], prints " key val", counts it.
185 * WHY: isolates the metadata branch of the replay loop so the driver stays flat.
186 * HOW: every read is bounds-checked; any short read returns TRUNC and frees the
187 * value buffer, matching the original inline break-on-truncation behaviour.
188 */
189static xrdcap_rec_t
190replay_meta_record(FILE *fp, FILE *out, int *metas)
191{
192 int kl = fgetc(fp);
193 char key[256];
194 uint16_t vl;
195 char *val;
196
197 if (kl < 0 || fread(key, 1, (size_t) kl, fp) != (size_t) kl) {
198 return XRDCAP_REC_TRUNC;
199 }
200 key[kl] = '\0';
201 vl = rd_u16(fp);
202 val = (char *) malloc((size_t) vl + 1);
203 if (val == NULL || fread(val, 1, vl, fp) != vl) {
204 free(val);
205 return XRDCAP_REC_TRUNC;
206 }
207 val[vl] = '\0';
208 fprintf(out, " %-10s %s\n", key, val);
209 free(val);
210 (*metas)++;
211 return XRDCAP_REC_OK;
212}

214/*
215 * replay_hexdump_wire — print up to the first 64 wire bytes, 16 per line.
216 *
217 * WHAT: verbose-mode hex dump of a frame's captured on-the-wire bytes.
218 * WHY: separates the pure formatting from the frame record's read/verify logic.
219 * HOW: side-effect at the edge (fprintf only); wire bytes/length are inputs.
220 */
221static void
222replay_hexdump_wire(FILE *out, const uint8_t *w, uint32_t wlen)
223{
224 uint32_t i;
225
226 for (i = 0; i < wlen && i < 64; i++) {
227 fprintf(out, "%02x%s", w[i], (i % 16 == 15) ? "\n" : " ");
228 }
229 if (wlen > 0 && (i % 16) != 0) { fprintf(out, "\n"); }
230}

232/*
233 * replay_frame_record — decode one 'F' frame record and print (+optional hexdump).
234 *
235 * WHAT: reads dir:1 isreq:1 sid:2BE code:2BE wirelen:4BE wire[], prints a summary
236 * line and (verbose) a hex dump, counting the frame.
237 * WHY: isolates the frame branch so the driver stays a flat dispatch.
238 * HOW: rejects EOF fields or an over-long wire as truncated; the wire buffer is
239 * malloc'd (min 1 byte) and freed on every exit — same as the original.
240 */
241static xrdcap_rec_t
242replay_frame_record(FILE *fp, int verbose, FILE *out, int *frames)
243{
244 int dir = fgetc(fp), isreq = fgetc(fp);
245 uint16_t sid = rd_u16(fp);
246 uint16_t code = rd_u16(fp);
247 uint32_t wlen = rd_u32(fp);
248 uint8_t *w;
249
250 if (dir == EOF || isreq == EOF || wlen > XRDCAP_WIRE_MAX) {
251 return XRDCAP_REC_TRUNC;
252 }
253 w = (uint8_t *) malloc(wlen ? wlen : 1);
254 if (w == NULL || fread(w, 1, wlen, fp) != wlen) {
255 free(w);
256 return XRDCAP_REC_TRUNC;
257 }
258 fprintf(out, "%c sid=%u %-14s wire=%u\n", dir, sid,
259 isreq ? brix_reqid_name(code) : brix_status_name(code), wlen);
260 if (verbose >= 1) {
261 replay_hexdump_wire(out, w, wlen);
262 }
263 free(w);
264 (*frames)++;
265 return XRDCAP_REC_OK;
266}

268/*
269 * brix_capture_replay — decode a .xrdcap file offline and print a human-readable log.
270 *
271 * WHAT: reads the magic header then a stream of M/F records, printing each to out.
272 * WHY: offline decode lets engineers inspect captured sessions without a live server.
273 * HOW: dispatches each record to a per-type helper; a premature EOF or corrupt
274 * record yields XRDCAP_REC_TRUNC so the caller gets a clean error rather
275 * than silent partial output. This matters for the security test case: a
276 * truncated file must exit nonzero.
277 */
278int
279brix_capture_replay(const char *path, int verbose, FILE *out, brix_status *st)
280{
281 FILE *fp = fopen(path, "rb");
282 int type, frames = 0, metas = 0, truncated = 0;
283 xrdcap_rec_t rc = XRDCAP_REC_OK;
284
285 if (fp == NULL) {
286 brix_status_set(st, XRDC_EUSAGE, 0, "cannot open %s", path);
287 return -1;
288 }
289 if (read_magic(fp, st) != 0) {
290 fclose(fp);
291 return -1;
292 }
293 fprintf(out, "capture: %s\n", path);
294 while ((type = fgetc(fp)) != EOF) {
295 if (type == 'M') {
296 rc = replay_meta_record(fp, out, &metas);
297 } else if (type == 'F') {
298 rc = replay_frame_record(fp, verbose, out, &frames);
299 } else {
300 rc = XRDCAP_REC_TRUNC; /* unknown record type = corrupt/truncated at boundary */
301 }
302 if (rc != XRDCAP_REC_OK) {
303 truncated = (rc == XRDCAP_REC_TRUNC);
304 break;
305 }
306 }
307 fclose(fp);
308 fprintf(out, "(%d frame(s), %d metadata record(s))\n", frames, metas);
309 if (truncated) {
310 brix_status_set(st, XRDC_EPROTO, 0, "%s: truncated or corrupted capture", path);
311 return -1;
312 }
313 return 0;
314}

325/*
326 * playback_skip_meta — consume one 'M' record without decoding it.
327 *
328 * WHAT: reads klen:1 then discards key[klen] vlen:2BE val[vlen].
329 * WHY: playback ignores metadata but must still walk past it exactly, and detect
330 * a truncation inside it (a short capture must fail, not silently stop).
331 * HOW: fread-based skip (not fseek): fseek past EOF succeeds on a regular file,
332 * so it cannot tell a complete record from one truncated mid-body.
333 */
334static xrdcap_rec_t
335playback_skip_meta(FILE *fp)
336{
337 int kl = fgetc(fp);
338 uint16_t vl;
339
340 if (kl < 0) { return XRDCAP_REC_TRUNC; }
341 if (skip_bytes(fp, (size_t) kl) != 0) { return XRDCAP_REC_TRUNC; }
342 vl = rd_u16(fp);
343 if (skip_bytes(fp, vl) != 0) { return XRDCAP_REC_TRUNC; }
344 return XRDCAP_REC_OK;
345}

347/*
348 * playback_ctx — live-playback loop state passed to the per-record helpers.
349 *
350 * WHAT: the connection to re-issue against, the human-readable output stream, and
351 * the issued/ok tallies accumulated across frames.
352 * WHY: bundles the four pieces of driver state so the re-issue helper takes one
353 * context rather than a long parameter list (keeps every helper <= 5 params).
354 * HOW: the driver owns one on the stack; helpers mutate issued/ok through it.
355 */
356typedef struct {
357 brix_conn *conn;
358 FILE *out;
359 int issued;
360 int ok;
361} playback_ctx;
362
363/*
364 * playback_reissue_wire — send one recorded request frame and read its reply.
365 *
366 * WHAT: writes the exact captured wire bytes, receives the response, prints the
367 * re-issue line, and bumps the issued/ok counters.
368 * WHY: isolates the live-connection side effect (the only I/O against the server)
369 * from the frame's parse/verify, keeping the driver flat.
370 * HOW: pure write+recv over conn->io; the response body is discarded (free rb). A
371 * failed write is silent (matches the original: no line, no counter bump).
372 */
373static void
374playback_reissue_wire(playback_ctx *pc, uint16_t code, const uint8_t *w, uint32_t wlen)
375{
376 brix_status rst;
377 uint16_t rstat = 0;
378 uint8_t *rb = NULL;
379 uint32_t rl = 0;
380 int rc;
381 brix_resp_out rout = { &rstat, &rb, &rl };
382
383 brix_status_clear(&rst);
384 if (brix_write_full(&pc->conn->io, w, wlen, &rst) != 0) {
385 return;
386 }
387 rc = brix_recv(pc->conn, 0xffff, &rout, &rst);
388 free(rb);
389 pc->issued++;
390 if (rc == 0) { pc->ok++; }
391 fprintf(pc->out, " re-issue %-14s -> %s\n", brix_reqid_name(code),
392 rc == 0 ? brix_status_name(rstat) : rst.msg);
393}

395/*
396 * playback_frame_record — decode one 'F' record and re-issue it if it is a real op.
397 *
398 * WHAT: reads the frame header + wire bytes; if it is a client REQUEST for a real
399 * operation (>= 24 bytes, not a session op), re-issues it against the live
400 * connection.
401 * WHY: isolates the frame branch (parse + selective re-issue) from the driver.
402 * HOW: rejects EOF dir or an over-long wire as truncated; the wire buffer is
403 * malloc'd (min 1 byte) and freed on every exit — same as the original.
404 */
405static xrdcap_rec_t
406playback_frame_record(FILE *fp, playback_ctx *pc)
407{
408 int dir = fgetc(fp), isreq = fgetc(fp);
409 uint16_t sid = rd_u16(fp);
410 uint16_t code = rd_u16(fp);
411 uint32_t wlen = rd_u32(fp);
412 uint8_t *w;
413
414 (void) sid;
415 if (dir == EOF || wlen > XRDCAP_WIRE_MAX) { return XRDCAP_REC_TRUNC; }
416 w = (uint8_t *) malloc(wlen ? wlen : 1);
417 if (w == NULL || fread(w, 1, wlen, fp) != wlen) {
418 free(w);
419 return XRDCAP_REC_TRUNC;
420 }
421 /* Replay only client REQUEST frames that are real operations. */
422 if (isreq == 1 && dir == '>' && wlen >= 24 && !is_session_op(code)) {
423 playback_reissue_wire(pc, code, w, wlen);
424 }
425 free(w);
426 return XRDCAP_REC_OK;
427}

430brix_capture_playback(const char *path, const char *url, const brix_opts *co,
431 FILE *out, brix_status *st)
432{
433 FILE *fp;
434 brix_url u;
435 brix_conn c;
436 playback_ctx pc = {0};
437 int type, truncated = 0;
438 xrdcap_rec_t rc = XRDCAP_REC_OK;
439
440 fp = fopen(path, "rb");
441 if (fp == NULL) {
442 brix_status_set(st, XRDC_EUSAGE, 0, "cannot open %s", path);
443 return -1;

446 fclose(fp);

454

456 pc.out = out;

468 }

475 return -1;
476 }
