1/*
2 * capture.c — session capture to a portable .xrdcap bundle + offline replay.
3 *
4 * WHAT: Record every wire frame (both directions) plus negotiated session
5 * metadata into a self-describing file; decode it later offline (no
6 * server), or re-issue the recorded requests against a live server.
7 * WHY: "Paste the bytes into the bug." A capture is a deterministic artifact a
8 * server bug can be reproduced from — decode it anywhere, or replay it to
9 * reproduce the exchange — without a packet sniffer or a live cluster.
10 * HOW: Writer hooks sit beside the wire-trace hooks in frame.c (inert when the
11 * sink is NULL). The format is an 8-byte magic then a stream of records:
12 * 'M' meta : klen:1 key[klen] vlen:2BE val[vlen]
13 * 'F' frame: dir:1 isreq:1 sid:2BE code:2BE wirelen:4BE wire[wirelen]
14 * where `wire` is the exact on-the-wire bytes (header+body), so replay can
15 * both decode (code→name via trace.c) and re-issue requests verbatim.
16 *
17 * Clean-room: our own framing; reuses trace.c name decoders. No XrdCl.
18 */

66void
67brix_capture_meta(struct brix_capture *cap, const char *key, const char *val)
68{
69 size_t kl, vl;
70 if (cap == NULL || cap->fp == NULL || key == NULL) {
71 return;
72 }
73 if (val == NULL) { val = ""; }
74 kl = strlen(key);
75 vl = strlen(val);
76 if (kl > 255) { kl = 255; }
77 if (vl > 0xffff) { vl = 0xffff; }
78 fputc('M', cap->fp);
79 fputc((int) kl, cap->fp);
80 fwrite(key, 1, kl, cap->fp);
81 put_u16(cap->fp, (uint16_t) vl);
82 fwrite(val, 1, vl, cap->fp);
83}

132/*
133 * skip_bytes — consume and discard `n` bytes from fp, detecting truncation.
134 *
135 * WHAT: reads n bytes into a scratch buffer, returning 0 when all n bytes were
136 * present or -1 when fewer remain (a truncated/corrupt record).
137 * WHY: fseek(SEEK_CUR) past EOF SUCCEEDS on a regular file, so seeking over a
138 * record body cannot distinguish a complete record from one truncated
139 * mid-body — the skip silently accepts a short file. An fread-based skip
140 * surfaces the short read exactly as the replay path's body reads do, so a
141 * mid-record truncation fails cleanly instead of passing as valid.
142 * HOW: loop reading up to a fixed scratch buffer until n bytes are consumed or
143 * fread returns short. */
144static int
145skip_bytes(FILE *fp, size_t n)
146{
147 char buf[512];
148 while (n > 0) {
149 size_t want = (n < sizeof(buf)) ? n : sizeof(buf);
150 if (fread(buf, 1, want, fp) != want) {
151 return -1;
152 }
153 n -= want;
154 }
155 return 0;
156}

158static uint16_t rd_u16(FILE *fp) { int a = fgetc(fp), b = fgetc(fp); return ((uint16_t)(uint8_t)a << 8) | (uint16_t)(uint8_t)b; }
159static uint32_t rd_u32(FILE *fp) {
160 uint32_t a = (uint32_t) fgetc(fp), b = (uint32_t) fgetc(fp);
161 uint32_t c = (uint32_t) fgetc(fp), d = (uint32_t) fgetc(fp);
162 return (a << 24) | (b << 16) | (c << 8) | d;
163}

165/*
166 * brix_capture_replay — decode a .xrdcap file offline and print a human-readable log.
167 *
168 * WHAT: reads the magic header then a stream of M/F records, printing each to out.
169 * WHY: offline decode lets engineers inspect captured sessions without a live server.
170 * HOW: each inner read is bounds-checked; a premature EOF or corrupt record sets
171 * truncated=1 so the caller gets a clean error rather than silent partial output.
172 * This matters for the security test case: a truncated file must exit nonzero.
173 */
174int
175brix_capture_replay(const char *path, int verbose, FILE *out, brix_status *st)
176{
177 FILE *fp = fopen(path, "rb");
178 int type, frames = 0, metas = 0, truncated = 0;
179
180 if (fp == NULL) {
181 brix_status_set(st, XRDC_EUSAGE, 0, "cannot open %s", path);
182 return -1;
183 }
184 if (read_magic(fp, st) != 0) {
185 fclose(fp);
186 return -1;
187 }
188 fprintf(out, "capture: %s\n", path);
189 while ((type = fgetc(fp)) != EOF) {
190 if (type == 'M') {
191 int kl = fgetc(fp);
192 char key[256];
193 uint16_t vl;
194 char *val;
195 if (kl < 0 || fread(key, 1, (size_t) kl, fp) != (size_t) kl) {
196 truncated = 1;
197 break;
198 }
199 key[kl] = '\0';
200 vl = rd_u16(fp);
201 val = (char *) malloc((size_t) vl + 1);
202 if (val == NULL || fread(val, 1, vl, fp) != vl) {
203 free(val);
204 truncated = 1;
205 break;
206 }
207 val[vl] = '\0';
208 fprintf(out, " %-10s %s\n", key, val);
209 free(val);
210 metas++;
211 } else if (type == 'F') {
212 int dir = fgetc(fp), isreq = fgetc(fp);
213 uint16_t sid = rd_u16(fp);
214 uint16_t code = rd_u16(fp);
215 uint32_t wlen = rd_u32(fp);
216 uint8_t *w;
217 if (dir == EOF || isreq == EOF || wlen > XRDCAP_WIRE_MAX) {
218 truncated = 1;
219 break;
220 }
221 w = (uint8_t *) malloc(wlen ? wlen : 1);
222 if (w == NULL || fread(w, 1, wlen, fp) != wlen) {
223 free(w);
224 truncated = 1;
225 break;
226 }
227 fprintf(out, "%c sid=%u %-14s wire=%u\n", dir, sid,
228 isreq ? brix_reqid_name(code) : brix_status_name(code), wlen);
229 if (verbose >= 1) {
230 uint32_t i;
231 for (i = 0; i < wlen && i < 64; i++) {
232 fprintf(out, "%02x%s", w[i], (i % 16 == 15) ? "\n" : " ");
233 }
234 if (wlen > 0 && (i % 16) != 0) { fprintf(out, "\n"); }
235 }
236 free(w);
237 frames++;
238 } else {
239 truncated = 1; /* unknown record type = corrupted or truncated at boundary */
240 break;
241 }
242 }
243 fclose(fp);
244 fprintf(out, "(%d frame(s), %d metadata record(s))\n", frames, metas);
245 if (truncated) {
246 brix_status_set(st, XRDC_EPROTO, 0, "%s: truncated or corrupted capture", path);
247 return -1;
248 }
249 return 0;
250}

261int
262brix_capture_playback(const char *path, const char *url, const brix_opts *co,
263 FILE *out, brix_status *st)
264{
265 FILE *fp;
266 brix_url u;
267 brix_conn c;
268 int type, issued = 0, ok = 0, truncated = 0;
269
270 fp = fopen(path, "rb");
271 if (fp == NULL) {
272 brix_status_set(st, XRDC_EUSAGE, 0, "cannot open %s", path);
273 return -1;
274 }
275 if (read_magic(fp, st) != 0) {
276 fclose(fp);
277 return -1;
278 }
279 if (brix_endpoint_parse(url, &u, st) != 0 || brix_connect(&c, &u, co, st) != 0) {
280 fclose(fp);
281 return -1;
282 }
283 fprintf(out, "playback: %s -> %s:%d\n", path, u.host, u.port);
284
285 while ((type = fgetc(fp)) != EOF) {
286 if (type == 'M') { /* skip metadata */
287 int kl = fgetc(fp);
288 uint16_t vl;
289 if (kl < 0) { truncated = 1; break; }
290 /* fread-based skip (not fseek): a mid-M truncation must fail, and
291 * fseek past EOF succeeds on a regular file. */
292 if (skip_bytes(fp, (size_t) kl) != 0) { truncated = 1; break; }
293 vl = rd_u16(fp);
294 if (skip_bytes(fp, vl) != 0) { truncated = 1; break; }
295 continue;
296 }
297 if (type != 'F') {
298 truncated = 1;
299 break;
300 }
301 {
302 int dir = fgetc(fp), isreq = fgetc(fp);
303 uint16_t sid = rd_u16(fp);
304 uint16_t code = rd_u16(fp);
305 uint32_t wlen = rd_u32(fp);
306 uint8_t *w;
307 (void) sid;
308 if (dir == EOF || wlen > XRDCAP_WIRE_MAX) { truncated = 1; break; }
309 w = (uint8_t *) malloc(wlen ? wlen : 1);
310 if (w == NULL || fread(w, 1, wlen, fp) != wlen) { free(w); truncated = 1; break; }
311 /* Replay only client REQUEST frames that are real operations. */
312 if (isreq == 1 && dir == '>' && wlen >= 24 && !is_session_op(code)) {
313 brix_status rst;
314 uint16_t rstat = 0;
315 uint8_t *rb = NULL;
316 uint32_t rl = 0;
317 int rc;
318 brix_status_clear(&rst);
319 if (brix_write_full(&c.io, w, wlen, &rst) == 0) {
320 rc = brix_recv(&c, 0xffff, &rstat, &rb, &rl, &rst);
321 free(rb);
322 issued++;
323 if (rc == 0) { ok++; }
324 fprintf(out, " re-issue %-14s -> %s\n", brix_reqid_name(code),
325 rc == 0 ? brix_status_name(rstat) : rst.msg);
326 }
327 }
328 free(w);
329 }
330 }
331 fclose(fp);
332 brix_close(&c);
333 fprintf(out, "(%d request(s) re-issued, %d ok)\n", issued, ok);
334 if (truncated) {
335 brix_status_set(st, XRDC_EPROTO, 0, "truncated or corrupted capture");
336 return -1;
337 }
338 return 0;
339}
