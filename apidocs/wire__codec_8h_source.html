1/*
2 * wire_codec.h — shared per-opcode XRootD wire-body codec (xrdwire).
3 *
4 * WHAT: Decoded structs + paired pack/unpack functions for the fixed-layout
5 * request bodies (and fixed response bodies) of each XRootD opcode. The
6 * client uses the *_pack (request) / *_unpack (response) halves; the server
7 * uses the mirror halves — one definition of every wire offset, shared via
8 * libxrdproto so the two cannot drift.
9 * WHY: The 16-byte ClientRequestHdr body and the fixed response bodies were
10 * hand-marshalled twice (client builds, server parses; and vice-versa).
11 * That implicit "offsets must match" contract is the largest remaining
12 * cross-component duplication and the classic wire-bug class. This codec
13 * generalises the proven pgio / vendor_ext / fattr_codec pattern to every
14 * opcode and makes the contract round-trip-testable with no process running.
15 * HOW: Pure C, ngx-free, big-endian fixed-offset copies over a caller buffer —
16 * the codec owns ONLY the fixed-offset fields. The shared framing header
17 * (streamid/requestid/dlen) stays in frame_hdr.h; the variable tail (path,
18 * payload, dirlist/stat strings) is appended/consumed by the caller. The
19 * codec never touches a socket, nginx, or the client's poll machinery.
20 *
21 * See docs/superpowers/specs/2026-06-27-xrdwire-codec-design.md.
22 */

30/* Result codes (consumers map onto kXR_* / XRDC_* at the edge). */
31#define XRDW_OK 0
32#define XRDW_EINVAL (-1) /* NULL argument / bad field */
33#define XRDW_ETRUNC (-2) /* input shorter than the fixed region */

35/* The fixed request-body region of ClientRequestHdr (between requestid and dlen). */
36#define XRDW_BODY_LEN 16
37#define XRDW_FHANDLE_LEN 4
38
39/* Compile-time guards on the codec's foundational constants. Every pack/unpack
40 * memcpy/put/get below indexes into a XRDW_BODY_LEN-byte buffer at offsets up to
41 * 12 + XRDW_FHANDLE_LEN; if either constant drifts from the XRootD wire (the
42 * 16-byte ClientRequestHdr.body, 4-byte file handle) the offsets silently corrupt
43 * every request. These turn that drift into a build failure. The matching
44 * "codec == real ClientRequestHdr.body" tie lives at the module consumer
45 * (src/read/read.c), since this header is ngx-/XProtocol-free by design. */
46_Static_assert(XRDW_BODY_LEN == 16, "XRootD ClientRequestHdr body region is 16 bytes");
47_Static_assert(XRDW_FHANDLE_LEN == 4, "XRootD file handle is 4 bytes");
48_Static_assert(12 + XRDW_FHANDLE_LEN <= XRDW_BODY_LEN,
49 "fhandle at offset 12 must fit within the body region");

53static inline void xrdw_put_u16(uint8_t *p, uint16_t v)
54{
55 p[0] = (uint8_t) (v >> 8);
56 p[1] = (uint8_t) v;
57}
58
59static inline void xrdw_put_u32(uint8_t *p, uint32_t v)
60{
61 p[0] = (uint8_t) (v >> 24);
62 p[1] = (uint8_t) (v >> 16);
63 p[2] = (uint8_t) (v >> 8);
64 p[3] = (uint8_t) v;
65}
66
67static inline uint16_t xrdw_get_u16(const uint8_t *p)
68{
69 return (uint16_t) (((uint16_t) p[0] << 8) | p[1]);
70}
71
72static inline uint32_t xrdw_get_u32(const uint8_t *p)
73{
74 return ((uint32_t) p[0] << 24) | ((uint32_t) p[1] << 16)
75 | ((uint32_t) p[2] << 8) | (uint32_t) p[3];
76}
77
78static inline void xrdw_put_u64(uint8_t *p, uint64_t v)
79{
80 xrdw_put_u32(p, (uint32_t) (v >> 32));
81 xrdw_put_u32(p + 4, (uint32_t) v);
82}
83
84static inline uint64_t xrdw_get_u64(const uint8_t *p)
85{
86 return ((uint64_t) xrdw_get_u32(p) << 32) | xrdw_get_u32(p + 4);
87}
