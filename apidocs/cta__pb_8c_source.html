61/* Copy a length-delimited protobuf string into a fixed buffer (truncate, NUL). */
62static void
63copy_str(char *dst, size_t dst_sz, const unsigned char *src, size_t n)
64{
65 if (n >= dst_sz) {
66 n = dst_sz - 1;
67 }
68 memcpy(dst, src, n);
69 dst[n] = '\0';
70}

72/* cta.common.Service{ name=1 } → out->instance */
73static int
74decode_service(pb_reader *r, cta_request_t *out)
75{
76 uint32_t f; int wt;
77
78 while (r->p < r->end) {
79 if (pb_read_tag(r, &f, &wt) != 0) {
80 return -1;
81 }
82 if (f == F_SERVICE_NAME && wt == PB_WT_LEN) {
83 const unsigned char *d; size_t n;
84 if (pb_read_len_delim(r, &d, &n) != 0) {
85 return -1;
86 }
87 copy_str(out->instance, sizeof(out->instance), d, n);
88 } else if (pb_skip_field(r, wt) != 0) {
89 return -1;
90 }
91 }
92 return 0;
93}

95/* cta.eos.Workflow{ event=1, instance=5 } → out->op, out->instance */
96static int
97decode_workflow(pb_reader *r, cta_request_t *out)
98{
99 uint32_t f; int wt;
100
101 while (r->p < r->end) {
102 if (pb_read_tag(r, &f, &wt) != 0) {
103 return -1;
104 }
105 if (f == F_WF_EVENT && wt == PB_WT_VARINT) {
106 uint64_t ev;
107 if (pb_read_varint(r, &ev) != 0) {
108 return -1;
109 }
110 out->op = ev == EV_CLOSEW ? CTA_OP_ARCHIVE
111 : ev == EV_PREPARE ? CTA_OP_RETRIEVE
112 : ev == EV_ABORT_PREPARE ? CTA_OP_CANCEL
113 : CTA_OP_UNKNOWN;
114 } else if (f == F_WF_INSTANCE && wt == PB_WT_LEN) {
115 const unsigned char *d; size_t n; pb_reader sub;
116 if (pb_read_len_delim(r, &d, &n) != 0) {
117 return -1;
118 }
119 sub.p = d; sub.end = d + n;
120 if (decode_service(&sub, out) != 0) {
121 return -1;
122 }
123 } else if (pb_skip_field(r, wt) != 0) {
124 return -1;
125 }
126 }
127 return 0;
128}

217/* cta.eos.Notification{ wf=1, cli=2, file=4 } */
218static int
219decode_notification(pb_reader *r, cta_request_t *out)
220{
221 uint32_t f; int wt;
222
223 while (r->p < r->end) {
224 if (pb_read_tag(r, &f, &wt) != 0) {
225 return -1;
226 }
227 if (wt == PB_WT_LEN && (f == F_NOTIF_WF || f == F_NOTIF_CLI ||
228 f == F_NOTIF_FILE)) {
229 const unsigned char *d; size_t n; pb_reader sub;
230 int rc;
231 if (pb_read_len_delim(r, &d, &n) != 0) {
232 return -1;
233 }
234 sub.p = d; sub.end = d + n;
235 rc = f == F_NOTIF_WF ? decode_workflow(&sub, out)
236 : f == F_NOTIF_CLI ? decode_client(&sub, out)
237 : decode_metadata(&sub, out);
238 if (rc != 0) {
239 return -1;
240 }
241 } else if (pb_skip_field(r, wt) != 0) {
242 return -1;
243 }
244 }
245 return 0;

248int
249cta_pb_decode_request(const unsigned char *buf, size_t len, cta_request_t *out)
250{
251 pb_reader r;
252 uint32_t f;
253 int wt;
254
255 memset(out, 0, sizeof(*out));
256 out->op = CTA_OP_UNKNOWN;
257 r.p = buf; r.end = buf + len;
258
259 while (r.p < r.end) {
260 if (pb_read_tag(&r, &f, &wt) != 0) {
261 return -1;
262 }
263 if (f == F_REQ_NOTIFICATION && wt == PB_WT_LEN) {
264 const unsigned char *d; size_t n; pb_reader sub;
265 if (pb_read_len_delim(&r, &d, &n) != 0) {
266 return -1;
267 }
268 sub.p = d; sub.end = d + n;
269 if (decode_notification(&sub, out) != 0) {
270 return -1;
271 }
272 } else if (f == F_REQ_ADMINCMD && wt == PB_WT_LEN) {
273 /* Admin command (query/listing). The full AdminCmd parse is deferred;
274 * for routing it is enough to mark the op. */
275 const unsigned char *d; size_t n;
276 if (pb_read_len_delim(&r, &d, &n) != 0) {
277 return -1;
278 }
279 out->op = CTA_OP_QUERY;
280 } else if (pb_skip_field(&r, wt) != 0) {
281 return -1;
282 }
283 }
284 return 0;
285}

287int
288cta_pb_encode_response(cta_rsp_type_t type, const char *message_txt,
289 uint64_t archive_id,
290 unsigned char *out, size_t cap, size_t *out_len)
291{
292 pb_writer w = { out, 0, cap };
293
294 if (pb_write_varint_field(&w, F_RSP_TYPE, (uint64_t) type) != 0) {
295 return -1;
296 }
297 if (message_txt != NULL && message_txt[0] != '\0') {
298 if (pb_write_string(&w, F_RSP_MESSAGE_TXT, message_txt) != 0) {
299 return -1;
300 }
301 }
302 if (archive_id != 0) {
303 /* Response.archive_file_id is a string xattr; render the number. */
304 char idbuf[24];
305 int m = 0;
306 uint64_t v = archive_id;
307 char rev[24];
308 int k = 0;
309 while (v > 0) { rev[k++] = (char) ('0' + (v % 10)); v /= 10; }
310 while (k > 0) { idbuf[m++] = rev[--k]; }
311 idbuf[m] = '\0';
312 if (pb_write_string(&w, F_RSP_ARCHIVE_ID, idbuf) != 0) {
313 return -1;
314 }
315 }
316 *out_len = w.len;
317 return 0;
318}
