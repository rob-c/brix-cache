12/* Return 1 if `path` is at or beneath `prefix` (component-aligned prefix match). */
13static ngx_flag_t
14brix_path_prefix_match(const char *prefix, const char *path)
15{
16 size_t prefix_len;
17
18 if (prefix == NULL || path == NULL) {
19 return 0;
20 }
21
22 prefix_len = strlen(prefix);
23 if (strncmp(prefix, path, prefix_len) != 0) {
24 return 0;
25 }
26
27 /* Require the match to fall on a path-component boundary so that prefix
28 * "/foo" matches "/foo" (exact) and "/foo/bar" (child) but NOT "/foobar".
29 * The char immediately after the prefix in path must therefore be either
30 * end-of-string or a '/' separator. */
31 return path[prefix_len] == '\0' || path[prefix_len] == '/';
32}
33/* Compare the first `bits` of two packed addresses — the CIDR prefix-match core. */
34static ngx_flag_t
35brix_authdb_addr_prefix_match(const u_char *a, const u_char *b,
36 ngx_uint_t bits)
37{
38 ngx_uint_t full;
39 ngx_uint_t rem;
40
41 /* Compare two raw addresses (4 bytes IPv4 / 16 bytes IPv6) under a /bits
42 * CIDR mask. Split the prefix length into whole bytes plus leftover bits:
43 * the first `full` bytes must match exactly, and the next byte must match
44 * only on its top `rem` bits. */
45 full = bits / 8;
46 rem = bits % 8;
47
48 if (full > 0 && ngx_memcmp(a, b, full) != 0) {
49 return 0;
50 }
51
52 if (rem != 0) {
53 /* Top `rem` bits set: e.g. rem=3 -> 0xff << 5 -> 0b11100000. Mask both
54 * addresses' partial byte and compare only the significant high bits. */
55 u_char mask = (u_char) (0xffu << (8 - rem));
56 if ((a[full] & mask) != (b[full] & mask)) {
57 return 0;
58 }
59 }
60
61 return 1;
62}
63/* Match a peer IP against a rule's host CIDR (rule_id is host/addr[/bits]). */
64static ngx_flag_t
65brix_authdb_host_cidr_match(const char *rule_id, const char *peer_ip)
66{
67 char cidr[128];
68 char *slash;
69 char *end;
70 long bits;
71 u_char rule_addr[16];
72 u_char peer_addr[16];
73 int family;
74 ngx_uint_t max_bits;
75
76 if (rule_id == NULL || peer_ip == NULL || peer_ip[0] == '\0') {
77 return 0;
78 }
79
80 /* Reject anything that would not fit (with NUL) in the local cidr buffer;
81 * this bounds the copy below and prevents reading a truncated CIDR. */
82 if (strlen(rule_id) >= sizeof(cidr)) {
83 return 0;
84 }
85
86 /* Work on a private mutable copy so we can split "addr/bits" in place. */
87 ngx_cpystrn((u_char *) cidr, (u_char *) rule_id, sizeof(cidr));
88 slash = strchr(cidr, '/');
89 if (slash == NULL) {
90 /* No '/': rule_id is a bare address -> require an exact string match
91 * against the peer IP (no masking). */
92 return strcmp(rule_id, peer_ip) == 0;
93 }
94
95 /* Split at '/': cidr now holds the address, slash holds the bit count. */
96 *slash++ = '\0';
97 /* Infer family from the address half; a ':' can only appear in IPv6. */
98 if (strchr(cidr, ':') != NULL) {
99 family = AF_INET6;
100 max_bits = 128;
101 } else {
102 family = AF_INET;
103 max_bits = 32;
104 }
105
106 /* Parse and validate the prefix length: must be a complete integer
107 * (end consumed entirely, slash actually had digits) within [0, max_bits].
108 * Any junk or out-of-range value rejects the rule rather than matching. */
109 bits = strtol(slash, &end, 10);
110 if (end == slash || *end != '\0' || bits < 0
111 || (ngx_uint_t) bits > max_bits)
112 {
113 return 0;
114 }
115
116 if (inet_pton(family, cidr, rule_addr) != 1
117 || inet_pton(family, peer_ip, peer_addr) != 1)
118 {
119 return 0;
120 }
121
122 return brix_authdb_addr_prefix_match(rule_addr, peer_addr,
123 (ngx_uint_t) bits);
124}
125/* Match a peer IP against an authdb rule's host id (exact, hostname, or CIDR). */
126static ngx_flag_t
127brix_authdb_host_match(const ngx_str_t *rule_id, const char *peer_ip)
128{
129 if (rule_id == NULL || peer_ip == NULL || peer_ip[0] == '\0') {
130 return 0;
131 }
132
133 if (rule_id->len == 1 && rule_id->data[0] == '*') {
134 return 1;
135 }
136
137 return brix_authdb_host_cidr_match((const char *) rule_id->data,
138 peer_ip);
139}
140/* WHAT: 1 if a user-type rule's id matches this identity's DN (wildcard or exact).
141 * WHY: isolates the USER matcher (exact/'*' split) from the dispatch switch.
142 * HOW: a bare '*' id matches any DN; otherwise require an exact string compare
143 * of the rule id against dn. dn may be an empty string (no match). */
144static ngx_flag_t
145adb_user_matches(const ngx_str_t *rule_id, const char *dn)
146{
147 if (rule_id->len == 1 && rule_id->data[0] == '*') {
148 return 1;
149 }
150 return ngx_strcmp(rule_id->data, dn) == 0;
151}
152
153/* WHAT: 1 if a group-type rule's id matches this identity's VO list (wildcard or
154 * membership).
155 * WHY: isolates the GROUP matcher ('*'/VO-membership split) from the switch.
156 * HOW: a bare '*' id matches any VO list; otherwise the rule id (a VO name) must
157 * appear in the comma-separated vo_list. */
158static ngx_flag_t
159adb_group_matches(const ngx_str_t *rule_id, const char *vo_list)
160{
161 if (rule_id->len == 1 && rule_id->data[0] == '*') {
162 return 1;
163 }
164 return brix_vo_list_contains(vo_list, (const char *) rule_id->data);
165}
166
167/* WHAT: 1 if `rule` applies to this identity by its matcher type; else 0.
168 * WHY: table-flat dispatch over the four rule kinds (all/user/group/host),
169 * keeping the caller's loop body linear. Pure — no I/O, no mutation.
170 * HOW: branch on rule->type: ALL matches unconditionally; USER splits to
171 * adb_user_matches(dn); GROUP to adb_group_matches(vo_list); HOST to
172 * brix_authdb_host_match(peer_ip). Any unknown type does not match
173 * (default-deny). Semantics are frozen against the multi-user conformance
174 * suite; do not alter the per-type rules. */
175static ngx_flag_t
176adb_identity_matches(const brix_authdb_rule_t *rule, const char *dn,
177 const char *vo_list, const char *peer_ip)
178{
179 switch (rule->type) {
180 case BRIX_AUTH_ALL:
181 return 1;
182 case BRIX_AUTH_USER:
183 return adb_user_matches(&rule->id, dn);
184 case BRIX_AUTH_GROUP:
185 return adb_group_matches(&rule->id, vo_list);
186 case BRIX_AUTHDB_HOST:
187 return brix_authdb_host_match(&rule->id, peer_ip);
188 default:
189 return 0;
190 }
191}

193/* Find the authdb rule granting `needed_privs` on resolved_path for `identity`;
194 * returns the matching rule or NULL. */
195const brix_authdb_rule_t *
196brix_find_authdb_rule_identity(const char *resolved_path, ngx_array_t *rules,
197 const brix_identity_t *identity,
198 const char *peer_ip, uint32_t needed_privs)
199{
200 const brix_authdb_rule_t *best = NULL;
201 brix_authdb_rule_t *rule;
202 size_t best_len = 0;
203 ngx_uint_t i;
204 const char *dn;
205 const char *vo_list;
206
207 if (resolved_path == NULL || rules == NULL) {
208 return NULL;
209 }
210
211 dn = brix_identity_dn_cstr(identity);
212 vo_list = brix_identity_vo_csv_cstr(identity);
213
214 rule = rules->elts;
215 for (i = 0; i < rules->nelts; i++) {
216 size_t rule_len = strlen(rule[i].resolved);
217
218 if (!brix_path_prefix_match(rule[i].resolved, resolved_path)) {
219 continue;
220 }
221
222 if (!adb_identity_matches(&rule[i], dn, vo_list, peer_ip)) {
223 continue;
224 }
225
226 /* Path + identity match: now require the rule to grant ALL needed bits
227 * ((privs & needed) == needed). Only sufficient rules compete; among
228 * them the longest prefix wins, with >= letting a later equal-length
229 * rule override an earlier one (config last-wins on ties). This is the
230 * XRootD longest-prefix semantics — frozen against the conformance suite. */
231 if ((rule[i].privs & needed_privs) == needed_privs) {
232 if (rule_len >= best_len) {
233 best = &rule[i];
234 best_len = rule_len;
235 }
236 }
237 }
238
239 return best;
240}

272/* Authorize `query->identity` for query->needed_privs on query->resolved_path
273 * against query->rules. Returns NGX_OK (granted) or NGX_ERROR (denied). */
274ngx_int_t
275brix_check_authdb_identity(ngx_log_t *log, const brix_authdb_query_t *query)
276{
277 const brix_authdb_rule_t *rule;
278 char safe_path[512];
279 const char *dn;
280 const char *vo_list;
281
282 /* No rules loaded == no authdb policy == unrestricted (allow-all). */
283 if (query->rules == NULL || query->rules->nelts == 0) {
284 return NGX_OK;
285 }
286
287 rule = brix_find_authdb_rule_identity(query->resolved_path, query->rules,
288 query->identity, query->peer_ip,
289 query->needed_privs);
290 if (rule != NULL) {
291 return NGX_OK;
292 }
293
294 dn = brix_identity_dn_cstr(query->identity);
295 vo_list = brix_identity_vo_csv_cstr(query->identity);
296 brix_sanitize_log_string(query->resolved_path, safe_path, sizeof(safe_path));
297
298 ngx_log_error(NGX_LOG_WARN, log, 0,
299 "brix: authdb denied path=\"%s\" privs=0x%02xd "
300 "dn=\"%s\" vos=\"%s\" peer=\"%s\"",
301 safe_path, query->needed_privs, dn,
302 vo_list[0] ? vo_list : "-",
303 (query->peer_ip != NULL && query->peer_ip[0])
304 ? query->peer_ip : "-");
305
306 return NGX_ERROR;
307}
