24/* Parse an XrdAcc privilege string (e.g. "rwld") into an BRIX_PRIV_* bitmask. */
25static uint32_t
26brix_parse_privs(const char *p, size_t len)
27{
28 uint32_t privs = 0;
29 size_t i;
30
31 /* Each privilege char maps to one or more BRIX_AUTH_* bits; OR-accumulate
32 * across the whole string. 'r' implies 'l' (you cannot read what you cannot
33 * look up), and 'a' (append) is folded into UPDATE since the FS-level write
34 * permission is identical. Unknown chars are silently ignored. */
35 for (i = 0; i < len; i++) {
36 switch (p[i]) {
37 case 'r': privs |= BRIX_AUTH_READ | BRIX_AUTH_LOOKUP; break;
38 case 'l': privs |= BRIX_AUTH_LOOKUP; break;
39 case 'w': privs |= BRIX_AUTH_UPDATE; break;
40 case 'a': privs |= BRIX_AUTH_UPDATE; break; /* append is update */
41 case 'd': privs |= BRIX_AUTH_DELETE; break;
42 case 'm': privs |= BRIX_AUTH_MKDIR; break;
43 case 'k': privs |= BRIX_AUTH_ADMIN; break;
44 default: break;
45 }
46 }
47
48 return privs;
49}
50/* Parse the authdb file into `rules`: one rule per line (path + identity matcher
51 * + privileges). Returns NGX_CONF_OK / NGX_CONF_ERROR. */
52ngx_int_t
53brix_parse_authdb(ngx_conf_t *cf, ngx_str_t *filename, ngx_array_t *rules)
54{
55 ngx_fd_t fd;
56 ngx_file_t file;
57 ngx_file_info_t fi;
58 u_char *buf, *p, *end, *line_start, *line_end;
59 ssize_t n;
60 size_t buf_size;
61 (void) ngx_stream_conf_get_module_srv_conf(cf, ngx_stream_brix_module);
62
63 fd = ngx_open_file(filename->data, NGX_FILE_RDONLY, NGX_FILE_OPEN, 0);
64 if (fd == NGX_INVALID_FILE) {
65 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
66 ngx_open_file_n " \"%s\" failed", filename->data);
67 return NGX_ERROR;
68 }
69
70 ngx_memzero(&file, sizeof(ngx_file_t));
71 file.fd = fd;
72 file.name = *filename;
73 file.log = cf->log;
74
75 if (ngx_fd_info(fd, &fi) == NGX_FILE_ERROR) {
76 ngx_conf_log_error(NGX_LOG_EMERG, cf, ngx_errno,
77 "fstat \"%s\" failed", filename->data);
78 ngx_close_file(fd);
79 return NGX_ERROR;
80 }
81
82 buf_size = (size_t) ngx_file_size(&fi);
83 if (buf_size == 0) {
84 ngx_close_file(fd);
85 return NGX_OK;
86 }
87 if (buf_size > 1024 * 1024) {
88 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
89 "brix_authdb \"%s\" exceeds 1 MiB limit",
90 filename->data);
91 ngx_close_file(fd);
92 return NGX_ERROR;
93 }
94
95 buf = ngx_alloc(buf_size + 1, cf->log);
96 if (buf == NULL) {
97 ngx_close_file(fd);
98 return NGX_ERROR;
99 }
100
101 n = ngx_read_file(&file, buf, buf_size, 0);
102 if (n == NGX_ERROR) {
103 ngx_free(buf);
104 ngx_close_file(fd);
105 return NGX_ERROR;
106 }
107
108 p = buf;
109 end = buf + n;
110
111 /* Line loop: p is the running cursor over the whole buffer. Each iteration
112 * carves out one line [line_start, line_end), advances p past the EOL
113 * terminator, then re-tokenizes within that line. line_start/line_end are
114 * mutated below for trimming, so p (not line_end) is what carries the loop
115 * forward to the next line. */
116 while (p < end) {
117 line_start = p;
118 while (p < end && *p != '\n' && *p != '\r') {
119 p++;
120 }
121 line_end = p;
122
123 /* Consume the EOL so the next iteration starts on the following line.
124 * Handles LF, CR, and CRLF (CR then LF) without double-counting. */
125 if (p < end && *p == '\r') p++;
126 if (p < end && *p == '\n') p++;
127
128 /* Skip leading whitespace */
129 while (line_start < line_end && isspace(*line_start)) {
130 line_start++;
131 }
132
133 /* Skip comments and empty lines */
134 if (line_start == line_end || *line_start == '#') {
135 continue;
136 }
137
138 /* Format: [u|g|p|a] <id> <path> <privs>
139 *
140 * Four space-delimited fields, tokenized left-to-right with the same
141 * two-step pattern per field: (1) skip leading whitespace to the field
142 * start, (2) scan to the next whitespace for the field end. Each
143 * "if (p == line_end) continue" guards against a truncated line that
144 * lacks the field still to come — a missing trailing field aborts the
145 * whole line (the rule is never pushed). NOTE: privs is the last field,
146 * so it has no "p == line_end" guard: an empty privs string is allowed
147 * and parses to a zero (no-permission) bitmask. */
148 u_char *type_p, *id_p, *path_p, *privs_p;
149 u_char *id_end, *path_end, *privs_end;
150
151 /* Field 1: type char. Only type_p[0] is read later; the rest of the
152 * token (if any) is scanned over but ignored. */
153 type_p = line_start;
154 p = type_p;
155 while (p < line_end && !isspace(*p)) p++;
156 if (p == line_end) continue; /* no id/path/privs follow -> reject */
157
158 /* Field 2: identity id (DN, VO, host/CIDR, or '*'). */
159 id_p = p;
160 while (id_p < line_end && isspace(*id_p)) id_p++;
161 p = id_p;
162 while (p < line_end && !isspace(*p)) p++;
163 if (p == line_end) continue; /* no path/privs follow -> reject */
164 id_end = p;
165
166 /* Field 3: path prefix this rule applies to. */
167 path_p = id_end;
168 while (path_p < line_end && isspace(*path_p)) path_p++;
169 p = path_p;
170 while (p < line_end && !isspace(*p)) p++;
171 if (p == line_end) continue; /* no privs follow -> reject */
172 path_end = p;
173
174 /* Field 4: privilege chars. Last field, so reaching line_end here is
175 * fine — privs_p..privs_end may be empty. */
176 privs_p = path_end;
177 while (privs_p < line_end && isspace(*privs_p)) privs_p++;
178 p = privs_p;
179 while (p < line_end && !isspace(*p)) p++;
180 privs_end = p;
181
182 brix_authdb_rule_t *rule = ngx_array_push(rules);
183 if (rule == NULL) {
184 ngx_free(buf);
185 ngx_close_file(fd);
186 return NGX_ERROR;
187 }
188
189 /* Rule type is the first byte of field 1 only (e.g. 'u','g','p','a');
190 * a multi-char first token is effectively truncated to its lead char. */
191 rule->type = (brix_auth_type_t) type_p[0];
192
193 /* id and path are copied into the config pool (NUL-terminated) because
194 * the source buf is freed at function exit — rules must own their
195 * strings for the lifetime of the config. resolved[] is zeroed now and
196 * filled later by brix_finalize_authdb_rules() (deferred realpath). */
197
198 rule->id.len = id_end - id_p;
199 rule->id.data = ngx_palloc(cf->pool, rule->id.len + 1);
200 ngx_memcpy(rule->id.data, id_p, rule->id.len);
201 rule->id.data[rule->id.len] = '\0';
202
203 rule->path.len = path_end - path_p;
204 rule->path.data = ngx_palloc(cf->pool, rule->path.len + 1);
205 ngx_memcpy(rule->path.data, path_p, rule->path.len);
206 rule->path.data[rule->path.len] = '\0';
207
208 rule->privs = brix_parse_privs((const char *) privs_p,
209 privs_end - privs_p);
210
211 ngx_memzero(rule->resolved, sizeof(rule->resolved));
212 }
213
214 ngx_free(buf);
215 ngx_close_file(fd);
216 return NGX_OK;
217}
218/* Return 1 if `path` is at or beneath `prefix` (component-aligned prefix match). */
219static ngx_flag_t
220brix_path_prefix_match(const char *prefix, const char *path)
221{
222 size_t prefix_len;
223
224 if (prefix == NULL || path == NULL) {
225 return 0;
226 }
227
228 prefix_len = strlen(prefix);
229 if (strncmp(prefix, path, prefix_len) != 0) {
230 return 0;
231 }
232
233 /* Require the match to fall on a path-component boundary so that prefix
234 * "/foo" matches "/foo" (exact) and "/foo/bar" (child) but NOT "/foobar".
235 * The char immediately after the prefix in path must therefore be either
236 * end-of-string or a '/' separator. */
237 return path[prefix_len] == '\0' || path[prefix_len] == '/';
238}
239/* Compare the first `bits` of two packed addresses — the CIDR prefix-match core. */
240static ngx_flag_t
241brix_authdb_addr_prefix_match(const u_char *a, const u_char *b,
242 ngx_uint_t bits)
243{
244 ngx_uint_t full;
245 ngx_uint_t rem;
246
247 /* Compare two raw addresses (4 bytes IPv4 / 16 bytes IPv6) under a /bits
248 * CIDR mask. Split the prefix length into whole bytes plus leftover bits:
249 * the first `full` bytes must match exactly, and the next byte must match
250 * only on its top `rem` bits. */
251 full = bits / 8;
252 rem = bits % 8;
253
254 if (full > 0 && ngx_memcmp(a, b, full) != 0) {
255 return 0;
256 }
257
258 if (rem != 0) {
259 /* Top `rem` bits set: e.g. rem=3 -> 0xff << 5 -> 0b11100000. Mask both
260 * addresses' partial byte and compare only the significant high bits. */
261 u_char mask = (u_char) (0xffu << (8 - rem));
262 if ((a[full] & mask) != (b[full] & mask)) {
263 return 0;
264 }
265 }
266
267 return 1;
268}
269/* Match a peer IP against a rule's host CIDR (rule_id is host/addr[/bits]). */
270static ngx_flag_t
271brix_authdb_host_cidr_match(const char *rule_id, const char *peer_ip)
272{
273 char cidr[128];
274 char *slash;
275 char *end;
276 long bits;
277 u_char rule_addr[16];
278 u_char peer_addr[16];
279 int family;
280 ngx_uint_t max_bits;
281
282 if (rule_id == NULL || peer_ip == NULL || peer_ip[0] == '\0') {
283 return 0;
284 }
285
286 /* Reject anything that would not fit (with NUL) in the local cidr buffer;
287 * this bounds the copy below and prevents reading a truncated CIDR. */
288 if (strlen(rule_id) >= sizeof(cidr)) {
289 return 0;
290 }
291
292 /* Work on a private mutable copy so we can split "addr/bits" in place. */
293 ngx_cpystrn((u_char *) cidr, (u_char *) rule_id, sizeof(cidr));
294 slash = strchr(cidr, '/');
295 if (slash == NULL) {
296 /* No '/': rule_id is a bare address -> require an exact string match
297 * against the peer IP (no masking). */
298 return strcmp(rule_id, peer_ip) == 0;
299 }
300
301 /* Split at '/': cidr now holds the address, slash holds the bit count. */
302 *slash++ = '\0';
303 /* Infer family from the address half; a ':' can only appear in IPv6. */
304 if (strchr(cidr, ':') != NULL) {
305 family = AF_INET6;
306 max_bits = 128;
307 } else {
308 family = AF_INET;
309 max_bits = 32;
310 }
311
312 /* Parse and validate the prefix length: must be a complete integer
313 * (end consumed entirely, slash actually had digits) within [0, max_bits].
314 * Any junk or out-of-range value rejects the rule rather than matching. */
315 bits = strtol(slash, &end, 10);
316 if (end == slash || *end != '\0' || bits < 0
317 || (ngx_uint_t) bits > max_bits)
318 {
319 return 0;
320 }
321
322 if (inet_pton(family, cidr, rule_addr) != 1
323 || inet_pton(family, peer_ip, peer_addr) != 1)
324 {
325 return 0;
326 }
327
328 return brix_authdb_addr_prefix_match(rule_addr, peer_addr,
329 (ngx_uint_t) bits);
330}
331/* Match a peer IP against an authdb rule's host id (exact, hostname, or CIDR). */
332static ngx_flag_t
333brix_authdb_host_match(const ngx_str_t *rule_id, const char *peer_ip)
334{
335 if (rule_id == NULL || peer_ip == NULL || peer_ip[0] == '\0') {
336 return 0;
337 }
338
339 if (rule_id->len == 1 && rule_id->data[0] == '*') {
340 return 1;
341 }
342
343 return brix_authdb_host_cidr_match((const char *) rule_id->data,
344 peer_ip);
345}
346/* Find the authdb rule granting `needed_privs` on resolved_path for `identity`;
347 * returns the matching rule or NULL. */
348const brix_authdb_rule_t *
349brix_find_authdb_rule_identity(const char *resolved_path, ngx_array_t *rules,
350 const brix_identity_t *identity,
351 const char *peer_ip, uint32_t needed_privs)
352{
353 const brix_authdb_rule_t *best = NULL;
354 brix_authdb_rule_t *rule;
355 size_t best_len = 0;
356 ngx_uint_t i;
357 const char *dn;
358 const char *vo_list;
359
360 if (resolved_path == NULL || rules == NULL) {
361 return NULL;
362 }
363
364 dn = brix_identity_dn_cstr(identity);
365 vo_list = brix_identity_vo_csv_cstr(identity);
366
367 rule = rules->elts;
368 for (i = 0; i < rules->nelts; i++) {
369 size_t rule_len = strlen(rule[i].resolved);
370
371 if (!brix_path_prefix_match(rule[i].resolved, resolved_path)) {
372 continue;
373 }
374
375 /* Check identity */
376 ngx_flag_t match = 0;
377 switch (rule[i].type) {
378 case BRIX_AUTH_ALL:
379 match = 1;
380 break;
381 case BRIX_AUTH_USER:
382 if (rule[i].id.len == 1 && rule[i].id.data[0] == '*') {
383 match = 1;
384 } else if (ngx_strcmp(rule[i].id.data, dn) == 0) {
385 match = 1;
386 }
387 break;
388 case BRIX_AUTH_GROUP:
389 if (rule[i].id.len == 1 && rule[i].id.data[0] == '*') {
390 match = 1;
391 } else if (brix_vo_list_contains(vo_list,
392 (const char *) rule[i].id.data))
393 {
394 match = 1;
395 }
396 break;
397 case BRIX_AUTHDB_HOST:
398 match = brix_authdb_host_match(&rule[i].id, peer_ip);
399 break;
400 default:
401 break;
402 }
403
404 if (!match) {
405 continue;
406 }
407
408 /* If identity matches, but doesn't have enough privs, we keep looking
409 * for a more specific rule? Or does XRootD stop at the first match?
410 * XRootD usually uses longest prefix. If multiple rules have same
411 * length, it depends on order. Here we take the longest prefix that
412 * matches identity AND has the required privs. */
413
414 /* Path + identity match: now require the rule to grant ALL needed bits
415 * ((privs & needed) == needed). Only sufficient rules compete; among
416 * them the longest prefix wins, with >= letting a later equal-length
417 * rule override an earlier one (config last-wins on ties). */
418 if ((rule[i].privs & needed_privs) == needed_privs) {
419 if (rule_len >= best_len) {
420 best = &rule[i];
421 best_len = rule_len;
422 }
423 }
424 }
425
426 return best;
427}
