33size_t
34fp_ext_mutate(const unsigned char *in, size_t n,
35 unsigned char *out, size_t outcap,
36 const fp_ext_mut *m, unsigned *seed, fp_ext_stats *st)
37{
38 size_t o = 0;
39
40 /* One-shot inject: splice the payload in front of this buffer. */
41 for (size_t k = 0; k < m->inject_len; k++) {
42 if (emit(out, outcap, &o, m->inject[k])) {
43 st->injected++;
44 }
45 }
46
47 for (size_t i = 0; i < n; ) {
48 /* Substring rewrite (non-overlapping, greedy left-to-right). */
49 if (m->find_len > 0 && i + m->find_len <= n &&
50 memcmp(in + i, m->find, m->find_len) == 0) {
51 for (size_t k = 0; k < m->repl_len; k++) {
52 emit(out, outcap, &o, m->repl[k]);
53 }
54 st->replaced++;
55 i += m->find_len;
56 continue;
57 }
58 /* Probabilistic delete (framing desync — length shrinks). */
59 if (m->drop_ppm > 0 && (int) (rand_r(seed) % 1000000u) < m->drop_ppm) {
60 st->dropped++;
61 i++;
62 continue;
63 }
64 emit(out, outcap, &o, in[i]);
65 /* Probabilistic duplicate (length inflates). */
66 if (m->repeat_ppm > 0 && (int) (rand_r(seed) % 1000000u) < m->repeat_ppm) {
67 if (emit(out, outcap, &o, in[i])) {
68 st->repeated++;
69 }
70 }
71 i++;
72 }
73 return o;
74}

76/* Decode two hex nibbles at s into a byte; returns -1 on a non-hex digit. */
77static int
78hexbyte(const char *s)
79{
80 int hi = 0, lo = 0;
81 for (int pass = 0; pass < 2; pass++) {
82 char c = s[pass];
83 int v;
84 if (c >= '0' && c <= '9') v = c - '0';
85 else if (c >= 'a' && c <= 'f') v = c - 'a' + 10;
86 else if (c >= 'A' && c <= 'F') v = c - 'A' + 10;
87 else return -1;
88 if (pass == 0) hi = v; else lo = v;
89 }
90 return (hi << 4) | lo;
91}

93static int
94parse_hex(const char *p, unsigned char *out, size_t cap)
95{
96 size_t len = strlen(p);
97 if (len % 2 != 0) {
98 return -1;
99 }
100 size_t o = 0;
101 for (size_t i = 0; i < len; i += 2) {
102 int b = hexbyte(p + i);
103 if (b < 0 || o >= cap) {
104 return -1;
105 }
106 out[o++] = (unsigned char) b;
107 }
108 return (int) o;

111/* Decode a literal token honouring \r \n \t \0 \xNN \\ escapes. */
112static int
113parse_literal(const char *p, unsigned char *out, size_t cap)
114{
115 size_t o = 0;
116 for (size_t i = 0; p[i] != '\0'; i++) {
117 unsigned char c = (unsigned char) p[i];
118 if (c == '\\' && p[i + 1] != '\0') {
119 char e = p[++i];
120 switch (e) {
121 case 'r': c = '\r'; break;
122 case 'n': c = '\n'; break;
123 case 't': c = '\t'; break;
124 case '0': c = '\0'; break;
125 case '\\': c = '\\'; break;
126 case 'x': {
127 int b = hexbyte(p + i + 1);
128 if (b < 0) {
129 return -1;
130 }
131 c = (unsigned char) b;
132 i += 2;
133 break;
134 }
135 default: c = (unsigned char) e; break;
136 }
137 }
138 if (o >= cap) {
139 return -1;
140 }
141 out[o++] = c;
142 }
143 return (int) o;
144}

161/* Split "ip:port" (IPv4 dotted or IPv6 in [..]) into ip / port. Returns family
162 * AF_INET/AF_INET6 or -1. */
163static int
164split_hostport(const char *hp, char *ip, size_t ipcap, int *port)
165{
166 if (hp == NULL) {
167 return -1;
168 }
169 const char *colon;
170 int fam;
171 if (hp[0] == '[') { /* [v6]:port */
172 const char *close = strchr(hp, ']');
173 if (close == NULL || close[1] != ':') {
174 return -1;
175 }
176 size_t n = (size_t) (close - hp - 1);
177 if (n == 0 || n >= ipcap) {
178 return -1;
179 }
180 memcpy(ip, hp + 1, n);
181 ip[n] = '\0';
182 colon = close + 1;
183 fam = AF_INET6;
184 } else {
185 colon = strrchr(hp, ':');
186 if (colon == NULL) {
187 return -1;
188 }
189 size_t n = (size_t) (colon - hp);
190 if (n == 0 || n >= ipcap) {
191 return -1;
192 }
193 memcpy(ip, hp, n);
194 ip[n] = '\0';
195 fam = strchr(ip, ':') ? AF_INET6 : AF_INET;
196 }
197 char *end;
198 long p = strtol(colon + 1, &end, 10);
199 if (end == colon + 1 || p <= 0 || p > 65535) {
200 return -1;
201 }
202 *port = (int) p;
203 /* Validate the literal really parses in the claimed family. */
204 unsigned char tmp[16];
205 if (inet_pton(fam, ip, tmp) != 1) {
206 return -1;
207 }
208 return fam;
209}

217/* Parse both "ip:port" endpoints and require a common family. Returns 0 on
218 * success (fields of *hp filled), or -1 on a parse error / family mismatch. */
219static int
220parse_pair(const char *src, const char *dst, fp_hostpair *hp)
221{
222 int sf = split_hostport(src, hp->sip, sizeof(hp->sip), &hp->sport);
223 int df = split_hostport(dst, hp->dip, sizeof(hp->dip), &hp->dport);
224 if (sf < 0 || df < 0 || sf != df) {
225 return -1;
226 }
227 hp->fam = sf;
228 return 0;
229}

247int
248fp_ext_proxy_v2(unsigned char *out, size_t cap, const char *src, const char *dst)
249{
250 fp_hostpair hp;
251 if (parse_pair(src, dst, &hp) < 0) {
252 return -1;
253 }
254 static const unsigned char sig[12] = {
255 0x0D, 0x0A, 0x0D, 0x0A, 0x00, 0x0D, 0x0A, 0x51, 0x55, 0x49, 0x54, 0x0A
256 };
257 size_t addrlen = (hp.fam == AF_INET) ? 12 : 36;
258 size_t total = 16 + addrlen;
259 if (cap < total) {
260 return -1;
261 }
262 memcpy(out, sig, 12);
263 out[12] = 0x21; /* version 2, PROXY */
264 out[13] = (hp.fam == AF_INET) ? 0x11 : 0x21; /* AF_INET/6 + STREAM */
265 out[14] = (unsigned char) (addrlen >> 8);
266 out[15] = (unsigned char) (addrlen & 0xFF);
267 /* addrlen is 2*iplen + 4 port bytes; write both IPs then the port pair,
268 * family-agnostic (iplen 4 for v4, 16 for v6). */
269 size_t iplen = (hp.fam == AF_INET) ? 4 : 16;
270 unsigned char *a = out + 16;
271 inet_pton(hp.fam, hp.sip, a);
272 inet_pton(hp.fam, hp.dip, a + iplen);
273 unsigned char *p = a + 2 * iplen;
274 p[0] = (unsigned char) (hp.sport >> 8); p[1] = (unsigned char) hp.sport;
275 p[2] = (unsigned char) (hp.dport >> 8); p[3] = (unsigned char) hp.dport;
276 return (int) total;
277}
