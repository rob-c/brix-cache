7/*
8 * dashboard/history.c — rolling time-series of activity for the dashboard charts.
9 *
10 * WHAT: Maintains the SHM-backed history ring (brix_dashboard_history_t) of
11 * BRIX_DASHBOARD_HISTORY_BUCKETS fixed-width
12 * (BRIX_DASHBOARD_HISTORY_INTERVAL_MS) buckets recording active transfer
13 * counts per protocol, cumulative bytes rx/tx, error and auth-failure
14 * totals. brix_dashboard_history_sample() writes the current bucket;
15 * brix_dashboard_history_snapshot() returns the recent window in
16 * chronological order; ngx_brix_dashboard_history_shm_init() is the SHM
17 * zone init callback.
18 * WHY: The dashboard plots trends over time, which needs periodic point-in-time
19 * samples persisted across workers and reloads — hence shared memory and a
20 * bounded ring instead of per-worker state. Sampling derives values from
21 * the existing dashboard transfer table and the metrics SHM zone rather
22 * than maintaining a parallel counter set, so the series cannot drift from
23 * the live numbers.
24 * HOW: Buckets are addressed by absolute time: index =
25 * (bucket_start_ms / INTERVAL_MS) % BUCKETS, with bucket_start_ms stored in
26 * each slot so a reader can tell a current bucket from a stale wrapped one.
27 * sample() snaps now_ms down to the interval boundary, zero-fills any
28 * buckets skipped since last_bucket_start_ms (idle gaps), then tallies
29 * active transfers from ngx_brix_dashboard_shm_zone and byte/error/auth
30 * totals from ngx_brix_shm_zone (per-server, WebDAV and S3). A single
31 * static ngx_shmtx_t (re-created on reload) guards both sample and
32 * snapshot. write_stalls and cache_occupancy_ppm are reserved/unpopulated.
33 */

87void
88brix_dashboard_history_sample(int64_t now_ms)
89{
90 brix_dashboard_history_t *hist;
91 brix_dashboard_history_bucket_t *bucket;
92 ngx_brix_metrics_t *met;
93 brix_transfer_table_t *tbl;
94 int64_t bucket_start;
95 int64_t clear_start;
96 uint64_t bytes_rx = 0;
97 uint64_t bytes_tx = 0;
98 uint64_t errors = 0;
99 uint64_t auth_failures = 0;
100 ngx_uint_t active[BRIX_XFER_NPROTOS] = { 0 };
101 ngx_uint_t active_tpc = 0;
102 ngx_uint_t i, j;
103
104 hist = dashboard_history_table();
105 if (hist == NULL) {
106 return;
107 }
108
109 bucket_start = now_ms
110 - (now_ms % BRIX_DASHBOARD_HISTORY_INTERVAL_MS);
111
112 ngx_shmtx_lock(&brix_dashboard_history_mutex);
113
114 if (hist->last_bucket_start_ms == 0) {
115 hist->last_bucket_start_ms = bucket_start;
116 }
117
118 if (bucket_start > hist->last_bucket_start_ms) {
119 clear_start = hist->last_bucket_start_ms
120 + BRIX_DASHBOARD_HISTORY_INTERVAL_MS;
121 for (; clear_start <= bucket_start;
122 clear_start += BRIX_DASHBOARD_HISTORY_INTERVAL_MS)
123 {
124 j = dashboard_history_bucket_index(clear_start);
125 ngx_memzero(&hist->buckets[j], sizeof(hist->buckets[j]));
126 hist->buckets[j].bucket_start_ms = clear_start;
127 }
128 hist->last_bucket_start_ms = bucket_start;
129 }
130
131 j = dashboard_history_bucket_index(bucket_start);
132 bucket = &hist->buckets[j];
133 if (bucket->bucket_start_ms != bucket_start) {
134 ngx_memzero(bucket, sizeof(*bucket));
135 bucket->bucket_start_ms = bucket_start;
136 }
137
138 if (ngx_brix_dashboard_shm_zone != NULL
139 && ngx_brix_dashboard_shm_zone->data != NULL
140 && ngx_brix_dashboard_shm_zone->data != (void *) 1)
141 {
142 tbl = ngx_brix_dashboard_shm_zone->data;
143 for (i = 0; i < BRIX_DASHBOARD_MAX_TRANSFERS; i++) {
144 brix_transfer_slot_t *slot = &tbl->slots[i];
145
146 if (slot->in_use == 0) {
147 continue;
148 }
149
150 if (slot->direction == BRIX_XFER_DIR_TPC) {
151 active_tpc++;
152 }
153
154 /* per-proto counts index by list row (id-1); untracked/legacy
155 * slots keep the historic root bucket */
156 if (slot->proto >= 1 && slot->proto <= BRIX_XFER_NPROTOS) {
157 active[slot->proto - 1]++;
158 } else {
159 active[BRIX_XFER_PROTO_ROOT - 1]++;
160 }
161 }
162 }
163
164 if (ngx_brix_shm_zone != NULL
165 && ngx_brix_shm_zone->data != NULL
166 && ngx_brix_shm_zone->data != (void *) 1)
167 {
168 met = ngx_brix_shm_zone->data;
169 for (i = 0; i < BRIX_METRICS_MAX_SERVERS; i++) {
170 ngx_brix_srv_metrics_t *srv = &met->servers[i];
171
172 bytes_rx += (uint64_t) srv->bytes_rx_total;
173 bytes_tx += (uint64_t) srv->bytes_tx_total;
174 auth_failures += (uint64_t) srv->op_err[BRIX_OP_AUTH];
175 for (j = 0; j < BRIX_NOPS; j++) {
176 errors += (uint64_t) srv->op_err[j];
177 }
178 }
179
180 bytes_rx += (uint64_t) met->webdav.bytes_rx_total;
181 bytes_tx += (uint64_t) met->webdav.bytes_tx_total;
182 bytes_rx += (uint64_t) met->s3.bytes_rx_total;
183 bytes_tx += (uint64_t) met->s3.bytes_tx_total;
184 auth_failures +=
185 (uint64_t) met->webdav.auth_total[BRIX_WEBDAV_AUTH_RESULT_REJECTED];
186 auth_failures +=
187 (uint64_t) met->s3.auth_total[BRIX_S3_AUTH_SIG_MISMATCH]
188 + (uint64_t) met->s3.auth_total[BRIX_S3_AUTH_MALFORMED]
189 + (uint64_t) met->s3.auth_total[BRIX_S3_AUTH_BAD_KEY];
190 }
191
192 {
193 ngx_uint_t p;
194
195 for (p = 0; p < BRIX_XFER_NPROTOS; p++) {
196 bucket->active[p] = active[p];
197 }
198 }
199 bucket->active_tpc = active_tpc;
200 bucket->bytes_rx = (ngx_atomic_t) bytes_rx;
201 bucket->bytes_tx = (ngx_atomic_t) bytes_tx;
202 bucket->errors = (ngx_atomic_t) errors;
203 bucket->auth_failures = (ngx_atomic_t) auth_failures;
204 bucket->write_stalls = 0;
205 bucket->cache_occupancy_ppm = 0;
206
207 ngx_shmtx_unlock(&brix_dashboard_history_mutex);
208}

210ngx_uint_t
211brix_dashboard_history_snapshot(brix_dashboard_history_bucket_t *out,
212 ngx_uint_t max_buckets)
213{
214 brix_dashboard_history_t *hist;
215 int64_t newest;
216 int64_t oldest;
217 int64_t ts;
218 ngx_uint_t n;
219
220 if (out == NULL || max_buckets == 0) {
221 return 0;
222 }
223
224 hist = dashboard_history_table();
225 if (hist == NULL) {
226 return 0;
227 }
228
229 ngx_shmtx_lock(&brix_dashboard_history_mutex);
230
231 newest = hist->last_bucket_start_ms;
232 if (newest == 0) {
233 ngx_shmtx_unlock(&brix_dashboard_history_mutex);
234 return 0;
235 }
236
237 if (max_buckets > BRIX_DASHBOARD_HISTORY_BUCKETS) {
238 max_buckets = BRIX_DASHBOARD_HISTORY_BUCKETS;
239 }
240
241 oldest = newest - ((int64_t) max_buckets - 1)
242 * BRIX_DASHBOARD_HISTORY_INTERVAL_MS;
243 n = 0;
244
245 for (ts = oldest; ts <= newest && n < max_buckets;
246 ts += BRIX_DASHBOARD_HISTORY_INTERVAL_MS)
247 {
248 ngx_uint_t idx = dashboard_history_bucket_index(ts);
249 if (hist->buckets[idx].bucket_start_ms == ts) {
250 out[n++] = hist->buckets[idx];
251 }
252 }
253
254 ngx_shmtx_unlock(&brix_dashboard_history_mutex);
255 return n;
256}
