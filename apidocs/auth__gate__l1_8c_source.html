1/*
2 * auth_gate_l1.c — per-worker L1 auth-verdict cache (see header).
3 *
4 * Direct-mapped, lockless (single-threaded event loop), bounded memory. The key
5 * is already a SHA-256 from auth_gate.c, so the bucket index is taken straight
6 * from its leading bytes — no re-hash. Expiry uses the nginx monotonic msec
7 * clock so it inherits the L2 TTL exactly.
8 */

10#include "auth_gate_l1.h"

12#define BRIX_AUTH_L1_DEFAULT_SLOTS 2048 /* tiny entries (~48B) → ~96KB */

14typedef struct {

16 unsigned set:1; /* slot occupied */
17 ngx_msec_t expire; /* ngx_current_msec deadline */

19} brix_auth_l1_slot_t;

26brix_auth_l1_t *
27brix_auth_l1_create(ngx_pool_t *pool, ngx_uint_t slots)
28{
29 brix_auth_l1_t *cache;
30
31 if (slots == 0) {
32 slots = BRIX_AUTH_L1_DEFAULT_SLOTS;
33 }
34 if (slots < 64) {
35 slots = 64;
36 }
37
38 cache = ngx_pcalloc(pool, sizeof(*cache));
39 if (cache == NULL) {
40 return NULL;
41 }
42 cache->slots = ngx_pcalloc(pool, slots * sizeof(brix_auth_l1_slot_t));
43 if (cache->slots == NULL) {
44 return NULL;
45 }
46 cache->nslots = slots;
47 return cache;
48}

50static ngx_inline ngx_uint_t
51brix_auth_l1_index(brix_auth_l1_t *cache, const u_char key[32])
52{
53 uint32_t h = ((uint32_t) key[0] << 24) | ((uint32_t) key[1] << 16)
54 | ((uint32_t) key[2] << 8) | (uint32_t) key[3];
55 return (ngx_uint_t) (h % cache->nslots);
56}

58int
59brix_auth_l1_lookup(brix_auth_l1_t *cache, const u_char key[32],
60 brix_auth_cache_val_t *val)
61{
62 brix_auth_l1_slot_t *slot;
63
64 if (cache == NULL || key == NULL) {
65 return 0;
66 }
67
68 slot = &cache->slots[brix_auth_l1_index(cache, key)];
69 if (!slot->set || ngx_memcmp(slot->key, key, 32) != 0) {
70 return 0;
71 }
72 /* Expired? (monotonic msec; the signed delta handles wrap.) */
73 if ((ngx_msec_int_t) (ngx_current_msec - slot->expire) >= 0) {
74 return 0;
75 }
76
77 *val = slot->val;
78 return 1;
79}
