43/* Copy in -> out, escaping control bytes, quotes, backslashes and non-ASCII to
44 * \xNN, so a wire string is safe to write to the access log. */
45size_t
46brix_sanitize_log_string(const char *in, char *out, size_t outsz)
47{
48 const u_char *src;
49 size_t written;
50 u_char ch;
51
52 if (out == NULL || outsz == 0) {
53 return 0;
54 }
55
56 src = (const u_char *) ((in != NULL) ? in : "-");
57 written = 0;
58
59 /* phase74-fp: security.ArrayBound loop-widening — the only literal here is
60 * "-" (char[2]); `src` is dereferenced only while *src != '\0', so it reads
61 * indices 0 and 1 (the NUL) at most and never advances past the
62 * terminator (ch = *src++ happens strictly after the check). */
63 while (*src != '\0' && written + 1 < outsz) { /* NOLINT(clang-analyzer-security.ArrayBound) */
64 ch = *src++;
65
66 if (ch >= 0x21 && ch <= 0x7e && ch != '"' && ch != '\\') {
67 out[written++] = (char) ch;
68 continue;
69 }
70
71 if (written + 4 >= outsz) {
72 break;
73 }
74
75 out[written++] = '\\';
76 out[written++] = 'x';
77 out[written++] = (char) brix_hex_nibble((u_char) (ch >> 4));
78 out[written++] = (char) brix_hex_nibble((u_char) (ch & 0x0f));
79 }
80
81 out[written] = '\0';
82
83 return written;
84}

86// Count path components by iterating through '/' separators — O(n) string scan.
87// Returns NGX_ERROR if component count exceeds BRIX_MAX_WALK_DEPTH (32);
88// returns NGX_OK otherwise. Leading slashes skipped; trailing empty
89// components do not increment depth.
90/* Return the number of (non-empty, slash-separated) components in `path`. */
91ngx_int_t
92brix_count_path_depth(const char *path)
93{
94 const char *p = path;
95 ngx_uint_t count;
96
97 while (*p == '/') {
98 p++;
99 }
100
101 if (*p == '\0') {
102 return NGX_OK;
103 }
104
105 count = 1;
106 while (*p) {
107 if (*p == '/') {
108 p++;
109 if (*p != '\0') {
110 count++;
111 }
112 } else {
113 p++;
114 }
115 }
116
117 return (count > BRIX_MAX_WALK_DEPTH) ? NGX_ERROR : NGX_OK;
118}
