9/* Return 1 if path starts with any colon-delimited token in paths. */
10int
11srv_path_matches(const char *paths, const char *path)
12{
13 const char *p, *end;
14 size_t tok_len, path_len;
15
16 path_len = strlen(path);
17 p = paths;
18
19 while (*p) {
20 end = strchr(p, ':');
21 if (end == NULL) {
22 end = p + strlen(p);
23 }
24 tok_len = (size_t)(end - p);
25
26 if (tok_len > 0) {
27 /* "/" root token matches every path regardless of leading slash. */
28 if (tok_len == 1 && p[0] == '/') {
29 return 1;
30 }
31 /* Longest-prefix match: the token must align with a directory
32 * boundary so that token "/data" never matches "/database".
33 * After the literal prefix compares equal, accept on any of three
34 * mutually-exclusive boundary conditions:
35 * (1) token itself ends with '/' (e.g. "/data/" in the registry)
36 * -- the slash is already the boundary;
37 * (2) the path has a '/' immediately past the prefix (e.g. path
38 * "/data/file" against token "/data") -- next char is the
39 * directory separator;
40 * (3) exact match -- the path ends right at the token boundary
41 * (path[tok_len] is the NUL terminator). */
42 if (path_len >= tok_len
43 && ngx_strncmp(path, p, tok_len) == 0
44 && (p[tok_len - 1] == '/'
45 || path[tok_len] == '/' || path[tok_len] == '\0'))
46 {
47 return 1;
48 }
49 }
50
51 p = *end ? end + 1 : end;
52 }
53 return 0;
54}

58/*
59 * srv_sel_tier_t - one candidate tier in the three-tier selection ladder.
60 *
61 * Each tier tracks the single best slot seen so far by metric: `idx` is the slot
62 * index (-1 = none yet) and `val` is that slot's winning metric. The three tiers
63 * are ranked fresh > any > black; the ladder resolves to the highest non-empty
64 * tier (see srv_sel_state_winner).
65 */
66typedef struct {
67 int idx; /* best slot index in this tier, or -1 if none */
68 uint32_t val; /* metric of the best slot */
69} srv_sel_tier_t;

72/*
73 * srv_sel_state_t - full three-tier selection accumulator for one scan.
74 * fresh - live: not stale AND not blacklisted (the preferred tier)
75 * any - not blacklisted, any staleness (stale-but-live fallback)
76 * black - blacklisted last-resort candidate (only when allow_blacklisted)
77 * `for_write` fixes the metric direction for the whole scan: writes maximise
78 * free_mb, reads minimise util_pct.
79 */
80/* Phase-89 W5: cap on the fresh-tier candidate list collected for the affinity
81 * stick. Slots past the cap still compete on metric via the normal tiers; the
82 * stick just cannot land on them — with 64 co-exporting fresh servers the
83 * spread loss is negligible and the array stays a cheap stack object. */
84#define SRV_SEL_AFFINITY_MAX 64
85
86typedef struct {
87 srv_sel_tier_t fresh;
88 srv_sel_tier_t any;
89 srv_sel_tier_t black;
90 int for_write;
91 /* Phase-89 W5: every fresh-tier candidate seen this scan (not just the
92 * metric winner), so the affinity hash can pick a stable member. */
93 int fresh_cands[SRV_SEL_AFFINITY_MAX];
94 ngx_uint_t n_fresh;
95} srv_sel_state_t;

98/* WHAT: Seed one tier to "empty" with the sentinel metric for the scan direction.
99 * WHY: A write scan improves upward from 0; a read scan improves downward from
100 * UINT32_MAX. Seeding the val to the worst possible value lets the first
101 * real candidate always win without a separate "first" special-case.
102 * HOW: idx = -1 (no candidate); val = 0 for writes, (uint32_t)-1 for reads. */
103static void
104srv_sel_tier_init(srv_sel_tier_t *tier, int for_write)
105{
106 tier->idx = -1;
107 tier->val = for_write ? 0 : (uint32_t) -1;
108}

111/* WHAT: Initialise all three selection tiers and record the scan direction.
112 * WHY: Centralises the fresh/any/black seeding so the scan loop starts clean.
113 * HOW: Store for_write, then seed each tier via srv_sel_tier_init. */
114static void
115srv_sel_state_init(srv_sel_state_t *st, int for_write)
116{
117 st->for_write = for_write;
118 srv_sel_tier_init(&st->fresh, for_write);
119 srv_sel_tier_init(&st->any, for_write);
120 srv_sel_tier_init(&st->black, for_write);
121 st->n_fresh = 0;
122}

125/* WHAT: Report whether `metric` beats a tier's current best for this direction.
126 * WHY: Encapsulates the single tie-break rule shared by every tier: writes
127 * prefer a strictly greater free_mb, reads a strictly lesser util_pct.
128 * Using strict comparison preserves first-seen determinism on ties —
129 * identical to the original inlined tests.
130 * HOW: Empty tier (idx == -1) always loses; otherwise compare by direction. */
131static int
132srv_sel_metric_better(const srv_sel_tier_t *tier, uint32_t metric, int for_write)
133{
134 if (tier->idx == -1) {
135 return 1;
136 }
137 return for_write ? (metric > tier->val) : (metric < tier->val);
138}

141/* WHAT: Offer slot `idx` (metric `metric`) to a single tier, updating if better.
142 * WHY: One place owns the "is this a new tier winner?" decision, so fresh/any/
143 * black all apply exactly the same tie-break.
144 * HOW: If srv_sel_metric_better says yes, record idx+metric as the new best. */
145static void
146srv_sel_tier_offer(srv_sel_tier_t *tier, int idx, uint32_t metric, int for_write)
147{
148 if (srv_sel_metric_better(tier, metric, for_write)) {
149 tier->idx = idx;
150 tier->val = metric;
151 }
152}

155/* WHAT: Classify one matched slot and fold it into the right selection tier(s).
156 * WHY: Isolates the per-slot policy — blacklisted → black tier only; live →
157 * always the `any` tier, and additionally the `fresh` tier when not stale.
158 * Staleness uses a signed msec diff so ngx_current_msec wrap is tolerated.
159 * HOW: Compute metric by direction, then route: black slots update only black;
160 * live slots update any, and fresh when not stale. */
161/*
162 * srv_sel_load_metric — blend the heartbeat machine load into the selection
163 * metric behind brix_cms_load_weight (Phase-89 W4). Weight 0 (the default)
164 * returns the input untouched — byte-identical to the pre-W4 scoring.
165 * Reads minimise the metric: blend as ((100-w)*util + w*load)/100 so a
166 * loaded node scores worse. Writes maximise free_mb: scale it down by up to
167 * w% at full load (free - free*w*load/10000) so capacity still dominates and
168 * the direction of comparison is unchanged.
169 */
170static uint32_t
171srv_sel_load_metric(uint32_t metric, uint32_t load_pct, int for_write)
172{
173 ngx_uint_t w = brix_srv_load_weight;
174
175 if (w == 0) {
176 return metric;
177 }
178 if (for_write) {
179 return metric - (uint32_t) ((uint64_t) metric * w * load_pct / 10000);
180 }
181 return (uint32_t) (((100 - w) * (uint64_t) metric + w * (uint64_t) load_pct)
182 / 100);
183}

186static void
187srv_sel_state_consider(srv_sel_state_t *st, int idx, const brix_srv_entry_t *e,
188 int is_black)
189{
190 uint32_t metric;
191 ngx_uint_t is_stale;
192
193 metric = srv_sel_load_metric(st->for_write ? e->free_mb : e->util_pct,
194 e->load_pct, st->for_write);
195
196 if (is_black) {
197 /* allow_blacklisted only — a last-resort tier below live servers. */
198 srv_sel_tier_offer(&st->black, idx, metric, st->for_write);
199 return;
200 }
201
202 srv_sel_tier_offer(&st->any, idx, metric, st->for_write);
203
204 /*
205 * Phase 39 (WS7): a server that has not heartbeated within stale_after_ms is
206 * de-preferred but still tracked as a fallback, so an all-stale storm
207 * degrades to the freshest stale server rather than a false NotFound. The
208 * signed diff tolerates ngx_current_msec wrap.
209 */
210 is_stale = (brix_srv_stale_after_ms > 0
211 && (ngx_msec_int_t) (ngx_current_msec - e->last_seen)
212 > (ngx_msec_int_t) brix_srv_stale_after_ms);
213 if (!is_stale) {
214 srv_sel_tier_offer(&st->fresh, idx, metric, st->for_write);
215 if (st->n_fresh < SRV_SEL_AFFINITY_MAX) {
216 st->fresh_cands[st->n_fresh++] = idx;
217 }
218 }
219}

222/* WHAT: fnv1a over the NUL-terminated path (Phase-89 W5 affinity stick).
223 * WHY: The stick must be a pure function of the path so every worker on every
224 * manager picks the same member of the same candidate set; fnv1a is the
225 * design-of-record hash (loc_cache.c) for path-shaped keys.
226 * HOW: Standard XOR-multiply loop over the bytes. */
227static uint32_t
228srv_sel_path_hash(const char *path)
229{
230 uint32_t h = BRIX_FNV1A32_OFFSET_BASIS;
231
232 while (*path != '\0') {
233 h ^= (uint32_t) (u_char) *path++;
234 h *= BRIX_FNV1A32_PRIME;
235 }
236 return h;
237}

240/* WHAT: Resolve the accumulated tiers to the winning slot index (or -1).
241 * WHY: Fixes the tier priority in one place: a live-fresh server always beats a
242 * stale-live one, which always beats a blacklisted last-resort one.
243 * HOW: Prefer fresh, then any, then black. */
244static int
245srv_sel_state_winner(const srv_sel_state_t *st)
246{
247 if (st->fresh.idx >= 0) {
248 return st->fresh.idx;
249 }
250 if (st->any.idx >= 0) {
251 return st->any.idx;
252 }
253 return st->black.idx;
254}

257/* WHAT
258 * Selects the best data server for a given path from the registry table.
259 * Used by kXR_locate and kXR_open to redirect clients to optimal servers.
260
261 * WHY
262 * Selection policy: reads → lowest util_pct (least loaded); writes → highest
263 * free_mb (most available space). Path matching uses longest-prefix over colon-
264 * delimited tokens in each entry's paths field.
265 *
266 * allow_blacklisted gives the open/stat handlers a LAST-RESORT tier: when no
267 * live server matches, fall back to a currently-blacklisted one rather than a
268 * false NotFound. A CMS heartbeat drop blacklists a server for 30 s even though
269 * its data plane is almost always still serving, so under load a transient blip
270 * should still redirect to the (live) node. kXR_locate passes 0 — it must
271 * report only live servers, so a genuinely dead node is still "not found" there.
272
273 * HOW
274 * Locks mutex → scans all occupied slots → filters by srv_path_matches() →
275 * folds each match into the three-tier accumulator (srv_sel_state_consider) →
276 * resolves the winner (srv_sel_state_winner), preferring fresh→stale→
277 * (blacklisted, only if allow_blacklisted). Writes host+port to output
278 * buffers. Unlocks and returns 1/0.
279 */
280int
281srv_select_core(const char *path, int for_write, int allow_blacklisted,
282 char *host_out, size_t host_size, uint16_t *port_out)
283{
284 brix_srv_table_t *tbl;
285 brix_srv_entry_t *e;
286 ngx_uint_t i;
287 srv_sel_state_t st;
288 int best;
289
290 tbl = srv_table();
291 if (tbl == NULL) {
292 return 0;
293 }
294
295 srv_sel_state_init(&st, for_write);
296
297 ngx_shmtx_lock(&brix_srv_mutex);
298
299 for (i = 0; i < tbl->capacity; i++) {
300 int is_black;
301
302 e = &tbl->slots[i];
303 if (!e->in_use) {
304 continue;
305 }
306 if (!srv_path_matches(e->paths, path)) {
307 continue;
308 }
309
310 is_black = (e->blacklisted_until != 0
311 && e->blacklisted_until > ngx_current_msec);
312
313 if (is_black && !allow_blacklisted) {
314 continue; /* strict callers (e.g. locate) skip blacklisted */
315 }
316
317 srv_sel_state_consider(&st, (int) i, e, is_black);
318 }
319
320 best = srv_sel_state_winner(&st);
321
322 /* Phase-89 W5: with affinity on and >1 eligible FRESH candidate, the path
323 * hash — not the metric — picks the member, so repeated requests for one
324 * path stick to one server (cache locality). LOCKED precedence (phase-61
325 * note 2): the filter above already excluded blacklisted/stale slots, so a
326 * drained host is never sticky; an empty/singleton fresh tier keeps the
327 * ladder winner unchanged. */
328 if (brix_srv_affinity && st.n_fresh > 1) {
329 best = st.fresh_cands[srv_sel_path_hash(path) % st.n_fresh];
330 }
331
332 if (best >= 0) {
333 e = &tbl->slots[best];
334 ngx_cpystrn((u_char *) host_out, (u_char *) e->host, host_size);
335 *port_out = e->port;
336 }
337
338 ngx_shmtx_unlock(&brix_srv_mutex);
339 return best >= 0;
340}

367/* WHAT: Count occupied servers that export a prefix covering path — the number
368 * of distinct data servers a client could be redirected to for this path.
369 * WHY: The tried/triedrc retry protocol (see brix_manager_tried_exhausted)
370 * needs to know how many candidates exist so it can tell when the client
371 * has exhausted them all and the answer is definitively "not found".
372 * HOW: Same path scan as srv_select_core (in_use, prefix match) under the
373 * registry spinlock, returning the count. Blacklisted slots ARE counted:
374 * the open/stat path (brix_srv_select_or_blacklisted) can redirect to a
375 * blacklisted server as a last resort, so it IS a candidate the client may
376 * be sent to. Counting it keeps tried-exhausted consistent — a client
377 * that already tried that one server then converges to NotFound instead of
378 * being bounced back to it. (A server that left the cluster is
379 * unregistered → in_use == 0 → not counted.) */
380int
381brix_srv_count_matching(const char *path)
382{
383 brix_srv_table_t *tbl;
384 brix_srv_entry_t *e;
385 ngx_uint_t i;
386 int n = 0;
387
388 tbl = srv_table();
389 if (tbl == NULL) {
390 return 0;
391 }
392
393 ngx_shmtx_lock(&brix_srv_mutex);
394 for (i = 0; i < tbl->capacity; i++) {
395 e = &tbl->slots[i];
396 if (!e->in_use) {
397 continue;
398 }
399 if (!srv_path_matches(e->paths, path)) {
400 continue;
401 }
402 n++;
403 }
404 ngx_shmtx_unlock(&brix_srv_mutex);
405 return n;
406}

410/* WHAT: Honour the XRootD client's tried/triedrc retry protocol at a manager.
411 * When a data server returns an error the client re-issues the request to
412 * the manager with ?tried=h1,h2&triedrc=rc1,rc2 listing servers it already
413 * attempted. Returns 1 when every server exporting this path has already
414 * been tried — the caller must then answer kXR_NotFound instead of
415 * redirecting, otherwise the client bounces manager<->data-server until it
416 * hits its redirect limit (the divergence conformance testing caught).
417 * WHY: A conformant redirector must converge to "not found" once the client has
418 * visited every candidate; reference cmsd does this via the tried list.
419 * HOW: Extract the opaque (everything after '?') from the raw request payload,
420 * locate "tried=", count its comma-separated hosts, and compare against
421 * brix_srv_count_matching(clean_path). Conservative: a zero match count
422 * falls through to the normal CMS-locate path so this does not prematurely
423 * short-circuit hierarchical (parent-locate) clusters. */
424int
425brix_manager_tried_exhausted(const u_char *payload, size_t payload_len,
426 const char *clean_path)
427{
428 char opaque[1024];
429 const u_char *q;
430 size_t olen;
431 const char *t, *p;
432 int n_tried, n_match;
433
434 if (payload == NULL || payload_len == 0) {
435 return 0;
436 }
437 q = memchr(payload, '?', payload_len);
438 if (q == NULL) {
439 return 0; /* no opaque -> client's first attempt */
440 }
441 q++;
442 olen = (size_t) (payload + payload_len - q);
443 if (olen > 0 && q[olen - 1] == '\0') {
444 olen--; /* trim trailing NUL some payloads carry */
445 }
446 if (olen == 0 || olen >= sizeof(opaque)) {
447 return 0; /* empty or too long to inspect — be safe */
448 }
449 ngx_memcpy(opaque, q, olen);
450 opaque[olen] = '\0';
451
452 t = strstr(opaque, "tried=");
453 if (t == NULL) {
454 return 0;
455 }
456 t += 6; /* skip "tried=" */
457 if (*t == '\0' || *t == '&') {
458 return 0; /* present but empty */
459 }
460
461 n_tried = 1;
462 for (p = t; *p && *p != '&'; p++) {
463 if (*p == ',') {
464 n_tried++;
465 }
466 }
467
468 n_match = brix_srv_count_matching(clean_path);
469 return (n_match > 0 && n_tried >= n_match);
470}
