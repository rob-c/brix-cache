8/* Return 1 if path starts with any colon-delimited token in paths. */
9int
10srv_path_matches(const char *paths, const char *path)
11{
12 const char *p, *end;
13 size_t tok_len, path_len;
14
15 path_len = strlen(path);
16 p = paths;
17
18 while (*p) {
19 end = strchr(p, ':');
20 if (end == NULL) {
21 end = p + strlen(p);
22 }
23 tok_len = (size_t)(end - p);
24
25 if (tok_len > 0) {
26 /* "/" root token matches every path regardless of leading slash. */
27 if (tok_len == 1 && p[0] == '/') {
28 return 1;
29 }
30 /* Longest-prefix match: the token must align with a directory
31 * boundary so that token "/data" never matches "/database".
32 * After the literal prefix compares equal, accept on any of three
33 * mutually-exclusive boundary conditions:
34 * (1) token itself ends with '/' (e.g. "/data/" in the registry)
35 * -- the slash is already the boundary;
36 * (2) the path has a '/' immediately past the prefix (e.g. path
37 * "/data/file" against token "/data") -- next char is the
38 * directory separator;
39 * (3) exact match -- the path ends right at the token boundary
40 * (path[tok_len] is the NUL terminator). */
41 if (path_len >= tok_len
42 && ngx_strncmp(path, p, tok_len) == 0
43 && (p[tok_len - 1] == '/'
44 || path[tok_len] == '/' || path[tok_len] == '\0'))
45 {
46 return 1;
47 }
48 }
49
50 p = *end ? end + 1 : end;
51 }
52 return 0;
53}

57/* WHAT
58 * Selects the best data server for a given path from the registry table.
59 * Used by kXR_locate and kXR_open to redirect clients to optimal servers.
60
61 * WHY
62 * Selection policy: reads → lowest util_pct (least loaded); writes → highest
63 * free_mb (most available space). Path matching uses longest-prefix over colon-
64 * delimited tokens in each entry's paths field.
65 *
66 * allow_blacklisted gives the open/stat handlers a LAST-RESORT tier: when no
67 * live server matches, fall back to a currently-blacklisted one rather than a
68 * false NotFound. A CMS heartbeat drop blacklists a server for 30 s even though
69 * its data plane is almost always still serving, so under load a transient blip
70 * should still redirect to the (live) node. kXR_locate passes 0 — it must
71 * report only live servers, so a genuinely dead node is still "not found" there.
72
73 * HOW
74 * Locks mutex → scans all occupied slots → filters by srv_path_matches() →
75 * picks best by for_write (free_mb max for writes, util_pct min for reads),
76 * preferring fresh→stale→(blacklisted, only if allow_blacklisted). Writes
77 * host+port to output buffers. Unlocks and returns 1/0.
78 */
79int
80srv_select_core(const char *path, int for_write, int allow_blacklisted,
81 char *host_out, size_t host_size, uint16_t *port_out)
82{
83 brix_srv_table_t *tbl;
84 brix_srv_entry_t *e;
85 ngx_uint_t i;
86 int best_fresh; /* live: not stale AND not blacklisted */
87 int best_any; /* not blacklisted, any staleness */
88 int best_black; /* blacklisted last-resort candidate */
89 uint32_t best_fresh_val;
90 uint32_t best_any_val;
91 uint32_t best_black_val;
92 int best;
93
94 tbl = srv_table();
95 if (tbl == NULL) {
96 return 0;
97 }
98
99 best_fresh = -1;
100 best_any = -1;
101 best_black = -1;
102 best_fresh_val = for_write ? 0 : (uint32_t) -1;
103 best_any_val = for_write ? 0 : (uint32_t) -1;
104 best_black_val = for_write ? 0 : (uint32_t) -1;
105
106 ngx_shmtx_lock(&brix_srv_mutex);
107
108 for (i = 0; i < tbl->capacity; i++) {
109 ngx_uint_t is_stale;
110 ngx_uint_t is_black;
111 uint32_t metric;
112
113 e = &tbl->slots[i];
114 if (!e->in_use) {
115 continue;
116 }
117 if (!srv_path_matches(e->paths, path)) {
118 continue;
119 }
120
121 is_black = (e->blacklisted_until != 0
122 && e->blacklisted_until > ngx_current_msec);
123
124 if (is_black && !allow_blacklisted) {
125 continue; /* strict callers (e.g. locate) skip blacklisted */
126 }
127
128 /*
129 * Phase 39 (WS7): a server that has not heartbeated within
130 * stale_after_ms is de-preferred but still tracked as a fallback, so an
131 * all-stale storm degrades to the freshest stale server rather than a
132 * false NotFound. The signed diff tolerates ngx_current_msec wrap.
133 */
134 is_stale = (brix_srv_stale_after_ms > 0
135 && (ngx_msec_int_t) (ngx_current_msec - e->last_seen)
136 > (ngx_msec_int_t) brix_srv_stale_after_ms);
137
138 metric = for_write ? e->free_mb : e->util_pct;
139
140 if (is_black) {
141 /* allow_blacklisted only — a last-resort tier below live servers. */
142 if (best_black == -1
143 || (for_write ? (metric > best_black_val)
144 : (metric < best_black_val)))
145 {
146 best_black = (int) i;
147 best_black_val = metric;
148 }
149 continue;
150 }
151
152 if (best_any == -1
153 || (for_write ? (metric > best_any_val) : (metric < best_any_val)))
154 {
155 best_any = (int) i;
156 best_any_val = metric;
157 }
158 if (!is_stale
159 && (best_fresh == -1
160 || (for_write ? (metric > best_fresh_val)
161 : (metric < best_fresh_val))))
162 {
163 best_fresh = (int) i;
164 best_fresh_val = metric;
165 }
166 }
167
168 best = (best_fresh >= 0) ? best_fresh
169 : (best_any >= 0) ? best_any
170 : best_black;
171
172 if (best >= 0) {
173 e = &tbl->slots[best];
174 ngx_cpystrn((u_char *) host_out, (u_char *) e->host, host_size);
175 *port_out = e->port;
176 }
177
178 ngx_shmtx_unlock(&brix_srv_mutex);
179 return best >= 0;
180}

207/* WHAT: Count occupied servers that export a prefix covering path — the number
208 * of distinct data servers a client could be redirected to for this path.
209 * WHY: The tried/triedrc retry protocol (see brix_manager_tried_exhausted)
210 * needs to know how many candidates exist so it can tell when the client
211 * has exhausted them all and the answer is definitively "not found".
212 * HOW: Same path scan as srv_select_core (in_use, prefix match) under the
213 * registry spinlock, returning the count. Blacklisted slots ARE counted:
214 * the open/stat path (brix_srv_select_or_blacklisted) can redirect to a
215 * blacklisted server as a last resort, so it IS a candidate the client may
216 * be sent to. Counting it keeps tried-exhausted consistent — a client
217 * that already tried that one server then converges to NotFound instead of
218 * being bounced back to it. (A server that left the cluster is
219 * unregistered → in_use == 0 → not counted.) */
220int
221brix_srv_count_matching(const char *path)
222{
223 brix_srv_table_t *tbl;
224 brix_srv_entry_t *e;
225 ngx_uint_t i;
226 int n = 0;
227
228 tbl = srv_table();
229 if (tbl == NULL) {
230 return 0;
231 }
232
233 ngx_shmtx_lock(&brix_srv_mutex);
234 for (i = 0; i < tbl->capacity; i++) {
235 e = &tbl->slots[i];
236 if (!e->in_use) {
237 continue;
238 }
239 if (!srv_path_matches(e->paths, path)) {
240 continue;
241 }
242 n++;
243 }
244 ngx_shmtx_unlock(&brix_srv_mutex);
245 return n;
246}

250/* WHAT: Honour the XRootD client's tried/triedrc retry protocol at a manager.
251 * When a data server returns an error the client re-issues the request to
252 * the manager with ?tried=h1,h2&triedrc=rc1,rc2 listing servers it already
253 * attempted. Returns 1 when every server exporting this path has already
254 * been tried — the caller must then answer kXR_NotFound instead of
255 * redirecting, otherwise the client bounces manager<->data-server until it
256 * hits its redirect limit (the divergence conformance testing caught).
257 * WHY: A conformant redirector must converge to "not found" once the client has
258 * visited every candidate; reference cmsd does this via the tried list.
259 * HOW: Extract the opaque (everything after '?') from the raw request payload,
260 * locate "tried=", count its comma-separated hosts, and compare against
261 * brix_srv_count_matching(clean_path). Conservative: a zero match count
262 * falls through to the normal CMS-locate path so this does not prematurely
263 * short-circuit hierarchical (parent-locate) clusters. */
264int
265brix_manager_tried_exhausted(const u_char *payload, size_t payload_len,
266 const char *clean_path)
267{
268 char opaque[1024];
269 const u_char *q;
270 size_t olen;
271 const char *t, *p;
272 int n_tried, n_match;
273
274 if (payload == NULL || payload_len == 0) {
275 return 0;
276 }
277 q = memchr(payload, '?', payload_len);
278 if (q == NULL) {
279 return 0; /* no opaque -> client's first attempt */
280 }
281 q++;
282 olen = (size_t) (payload + payload_len - q);
283 if (olen > 0 && q[olen - 1] == '\0') {
284 olen--; /* trim trailing NUL some payloads carry */
285 }
286 if (olen == 0 || olen >= sizeof(opaque)) {
287 return 0; /* empty or too long to inspect — be safe */
288 }
289 ngx_memcpy(opaque, q, olen);
290 opaque[olen] = '\0';
291
292 t = strstr(opaque, "tried=");
293 if (t == NULL) {
294 return 0;
295 }
296 t += 6; /* skip "tried=" */
297 if (*t == '\0' || *t == '&') {
298 return 0; /* present but empty */
299 }
300
301 n_tried = 1;
302 for (p = t; *p && *p != '&'; p++) {
303 if (*p == ',') {
304 n_tried++;
305 }
306 }
307
308 n_match = brix_srv_count_matching(clean_path);
309 return (n_match > 0 && n_tried >= n_match);
310}

314/* WHAT
315 * Marks a registered server as temporarily unavailable for selection.
316 * Called from brix_cms_srv_close() when a data server's CMS connection
317 * drops. The server entry stays in the registry so its paths and metrics are
318 * preserved for the reconnect; brix_srv_select() and brix_srv_locate_all()
319 * both skip entries whose blacklisted_until is in the future.
320 *
321 * WHY
322 * A clean reconnect within the window re-registers and clears the flag,
323 * making the server immediately available again. A permanently dead server
324 * stays blacklisted until the window expires, at which point its stale metrics
325 * become visible — operators detect this via brix_cluster_server_last_seen_seconds.
326 *
327 * HOW
328 * Locks mutex → scans for host+port match → increments error_count →
329 * sets blacklisted_until = ngx_current_msec + duration_ms. Unlocks.
330 */
331void
332brix_srv_blacklist(const char *host, uint16_t port, ngx_msec_t duration_ms)
333{
334 brix_srv_table_t *tbl;
335 brix_srv_entry_t *e;
336 ngx_uint_t i;
337
338 tbl = srv_table();
339 if (tbl == NULL) {
340 return;
341 }
342
343 ngx_shmtx_lock(&brix_srv_mutex);
344
345 for (i = 0; i < tbl->capacity; i++) {
346 e = &tbl->slots[i];
347 if (!e->in_use || e->port != port
348 || ngx_strcmp(e->host, host) != 0)
349 {
350 continue;
351 }
352 e->error_count++;
353 e->blacklisted_until = ngx_current_msec + duration_ms;
354 break;
355 }
356
357 ngx_shmtx_unlock(&brix_srv_mutex);
358}
