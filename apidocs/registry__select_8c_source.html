8/* Return 1 if path starts with any colon-delimited token in paths. */
9int
10srv_path_matches(const char *paths, const char *path)
11{
12 const char *p, *end;
13 size_t tok_len, path_len;
14
15 path_len = strlen(path);
16 p = paths;
17
18 while (*p) {
19 end = strchr(p, ':');
20 if (end == NULL) {
21 end = p + strlen(p);
22 }
23 tok_len = (size_t)(end - p);
24
25 if (tok_len > 0) {
26 /* "/" root token matches every path regardless of leading slash. */
27 if (tok_len == 1 && p[0] == '/') {
28 return 1;
29 }
30 /* Longest-prefix match: the token must align with a directory
31 * boundary so that token "/data" never matches "/database".
32 * After the literal prefix compares equal, accept on any of three
33 * mutually-exclusive boundary conditions:
34 * (1) token itself ends with '/' (e.g. "/data/" in the registry)
35 * -- the slash is already the boundary;
36 * (2) the path has a '/' immediately past the prefix (e.g. path
37 * "/data/file" against token "/data") -- next char is the
38 * directory separator;
39 * (3) exact match -- the path ends right at the token boundary
40 * (path[tok_len] is the NUL terminator). */
41 if (path_len >= tok_len
42 && ngx_strncmp(path, p, tok_len) == 0
43 && (p[tok_len - 1] == '/'
44 || path[tok_len] == '/' || path[tok_len] == '\0'))
45 {
46 return 1;
47 }
48 }
49
50 p = *end ? end + 1 : end;
51 }
52 return 0;
53}

57/*
58 * srv_sel_tier_t - one candidate tier in the three-tier selection ladder.
59 *
60 * Each tier tracks the single best slot seen so far by metric: `idx` is the slot
61 * index (-1 = none yet) and `val` is that slot's winning metric. The three tiers
62 * are ranked fresh > any > black; the ladder resolves to the highest non-empty
63 * tier (see srv_sel_state_winner).
64 */
65typedef struct {
66 int idx; /* best slot index in this tier, or -1 if none */
67 uint32_t val; /* metric of the best slot */
68} srv_sel_tier_t;

71/*
72 * srv_sel_state_t - full three-tier selection accumulator for one scan.
73 * fresh - live: not stale AND not blacklisted (the preferred tier)
74 * any - not blacklisted, any staleness (stale-but-live fallback)
75 * black - blacklisted last-resort candidate (only when allow_blacklisted)
76 * `for_write` fixes the metric direction for the whole scan: writes maximise
77 * free_mb, reads minimise util_pct.
78 */
79typedef struct {
80 srv_sel_tier_t fresh;
81 srv_sel_tier_t any;
82 srv_sel_tier_t black;
83 int for_write;
84} srv_sel_state_t;

87/* WHAT: Seed one tier to "empty" with the sentinel metric for the scan direction.
88 * WHY: A write scan improves upward from 0; a read scan improves downward from
89 * UINT32_MAX. Seeding the val to the worst possible value lets the first
90 * real candidate always win without a separate "first" special-case.
91 * HOW: idx = -1 (no candidate); val = 0 for writes, (uint32_t)-1 for reads. */
92static void
93srv_sel_tier_init(srv_sel_tier_t *tier, int for_write)
94{
95 tier->idx = -1;
96 tier->val = for_write ? 0 : (uint32_t) -1;
97}

100/* WHAT: Initialise all three selection tiers and record the scan direction.
101 * WHY: Centralises the fresh/any/black seeding so the scan loop starts clean.
102 * HOW: Store for_write, then seed each tier via srv_sel_tier_init. */
103static void
104srv_sel_state_init(srv_sel_state_t *st, int for_write)
105{
106 st->for_write = for_write;
107 srv_sel_tier_init(&st->fresh, for_write);
108 srv_sel_tier_init(&st->any, for_write);
109 srv_sel_tier_init(&st->black, for_write);
110}

113/* WHAT: Report whether `metric` beats a tier's current best for this direction.
114 * WHY: Encapsulates the single tie-break rule shared by every tier: writes
115 * prefer a strictly greater free_mb, reads a strictly lesser util_pct.
116 * Using strict comparison preserves first-seen determinism on ties —
117 * identical to the original inlined tests.
118 * HOW: Empty tier (idx == -1) always loses; otherwise compare by direction. */
119static int
120srv_sel_metric_better(const srv_sel_tier_t *tier, uint32_t metric, int for_write)
121{
122 if (tier->idx == -1) {
123 return 1;
124 }
125 return for_write ? (metric > tier->val) : (metric < tier->val);
126}

129/* WHAT: Offer slot `idx` (metric `metric`) to a single tier, updating if better.
130 * WHY: One place owns the "is this a new tier winner?" decision, so fresh/any/
131 * black all apply exactly the same tie-break.
132 * HOW: If srv_sel_metric_better says yes, record idx+metric as the new best. */
133static void
134srv_sel_tier_offer(srv_sel_tier_t *tier, int idx, uint32_t metric, int for_write)
135{
136 if (srv_sel_metric_better(tier, metric, for_write)) {
137 tier->idx = idx;
138 tier->val = metric;
139 }
140}

143/* WHAT: Classify one matched slot and fold it into the right selection tier(s).
144 * WHY: Isolates the per-slot policy — blacklisted → black tier only; live →
145 * always the `any` tier, and additionally the `fresh` tier when not stale.
146 * Staleness uses a signed msec diff so ngx_current_msec wrap is tolerated.
147 * HOW: Compute metric by direction, then route: black slots update only black;
148 * live slots update any, and fresh when not stale. */
149static void
150srv_sel_state_consider(srv_sel_state_t *st, int idx, const brix_srv_entry_t *e,
151 int is_black)
152{
153 uint32_t metric;
154 ngx_uint_t is_stale;
155
156 metric = st->for_write ? e->free_mb : e->util_pct;
157
158 if (is_black) {
159 /* allow_blacklisted only — a last-resort tier below live servers. */
160 srv_sel_tier_offer(&st->black, idx, metric, st->for_write);
161 return;
162 }
163
164 srv_sel_tier_offer(&st->any, idx, metric, st->for_write);
165
166 /*
167 * Phase 39 (WS7): a server that has not heartbeated within stale_after_ms is
168 * de-preferred but still tracked as a fallback, so an all-stale storm
169 * degrades to the freshest stale server rather than a false NotFound. The
170 * signed diff tolerates ngx_current_msec wrap.
171 */
172 is_stale = (brix_srv_stale_after_ms > 0
173 && (ngx_msec_int_t) (ngx_current_msec - e->last_seen)
174 > (ngx_msec_int_t) brix_srv_stale_after_ms);
175 if (!is_stale) {
176 srv_sel_tier_offer(&st->fresh, idx, metric, st->for_write);
177 }
178}

181/* WHAT: Resolve the accumulated tiers to the winning slot index (or -1).
182 * WHY: Fixes the tier priority in one place: a live-fresh server always beats a
183 * stale-live one, which always beats a blacklisted last-resort one.
184 * HOW: Prefer fresh, then any, then black. */
185static int
186srv_sel_state_winner(const srv_sel_state_t *st)
187{
188 if (st->fresh.idx >= 0) {
189 return st->fresh.idx;
190 }
191 if (st->any.idx >= 0) {
192 return st->any.idx;
193 }
194 return st->black.idx;
195}

198/* WHAT
199 * Selects the best data server for a given path from the registry table.
200 * Used by kXR_locate and kXR_open to redirect clients to optimal servers.
201
202 * WHY
203 * Selection policy: reads → lowest util_pct (least loaded); writes → highest
204 * free_mb (most available space). Path matching uses longest-prefix over colon-
205 * delimited tokens in each entry's paths field.
206 *
207 * allow_blacklisted gives the open/stat handlers a LAST-RESORT tier: when no
208 * live server matches, fall back to a currently-blacklisted one rather than a
209 * false NotFound. A CMS heartbeat drop blacklists a server for 30 s even though
210 * its data plane is almost always still serving, so under load a transient blip
211 * should still redirect to the (live) node. kXR_locate passes 0 — it must
212 * report only live servers, so a genuinely dead node is still "not found" there.
213
214 * HOW
215 * Locks mutex → scans all occupied slots → filters by srv_path_matches() →
216 * folds each match into the three-tier accumulator (srv_sel_state_consider) →
217 * resolves the winner (srv_sel_state_winner), preferring fresh→stale→
218 * (blacklisted, only if allow_blacklisted). Writes host+port to output
219 * buffers. Unlocks and returns 1/0.
220 */
221int
222srv_select_core(const char *path, int for_write, int allow_blacklisted,
223 char *host_out, size_t host_size, uint16_t *port_out)
224{
225 brix_srv_table_t *tbl;
226 brix_srv_entry_t *e;
227 ngx_uint_t i;
228 srv_sel_state_t st;
229 int best;
230
231 tbl = srv_table();
232 if (tbl == NULL) {
233 return 0;
234 }
235
236 srv_sel_state_init(&st, for_write);
237
238 ngx_shmtx_lock(&brix_srv_mutex);
239
240 for (i = 0; i < tbl->capacity; i++) {
241 int is_black;
242
243 e = &tbl->slots[i];
244 if (!e->in_use) {
245 continue;
246 }
247 if (!srv_path_matches(e->paths, path)) {
248 continue;
249 }
250
251 is_black = (e->blacklisted_until != 0
252 && e->blacklisted_until > ngx_current_msec);
253
254 if (is_black && !allow_blacklisted) {
255 continue; /* strict callers (e.g. locate) skip blacklisted */
256 }
257
258 srv_sel_state_consider(&st, (int) i, e, is_black);
259 }
260
261 best = srv_sel_state_winner(&st);
262
263 if (best >= 0) {
264 e = &tbl->slots[best];
265 ngx_cpystrn((u_char *) host_out, (u_char *) e->host, host_size);
266 *port_out = e->port;
267 }
268
269 ngx_shmtx_unlock(&brix_srv_mutex);
270 return best >= 0;
271}

298/* WHAT: Count occupied servers that export a prefix covering path — the number
299 * of distinct data servers a client could be redirected to for this path.
300 * WHY: The tried/triedrc retry protocol (see brix_manager_tried_exhausted)
301 * needs to know how many candidates exist so it can tell when the client
302 * has exhausted them all and the answer is definitively "not found".
303 * HOW: Same path scan as srv_select_core (in_use, prefix match) under the
304 * registry spinlock, returning the count. Blacklisted slots ARE counted:
305 * the open/stat path (brix_srv_select_or_blacklisted) can redirect to a
306 * blacklisted server as a last resort, so it IS a candidate the client may
307 * be sent to. Counting it keeps tried-exhausted consistent — a client
308 * that already tried that one server then converges to NotFound instead of
309 * being bounced back to it. (A server that left the cluster is
310 * unregistered → in_use == 0 → not counted.) */
311int
312brix_srv_count_matching(const char *path)
313{
314 brix_srv_table_t *tbl;
315 brix_srv_entry_t *e;
316 ngx_uint_t i;
317 int n = 0;
318
319 tbl = srv_table();
320 if (tbl == NULL) {
321 return 0;
322 }
323
324 ngx_shmtx_lock(&brix_srv_mutex);
325 for (i = 0; i < tbl->capacity; i++) {
326 e = &tbl->slots[i];
327 if (!e->in_use) {
328 continue;
329 }
330 if (!srv_path_matches(e->paths, path)) {
331 continue;
332 }
333 n++;
334 }
335 ngx_shmtx_unlock(&brix_srv_mutex);
336 return n;
337}

341/* WHAT: Honour the XRootD client's tried/triedrc retry protocol at a manager.
342 * When a data server returns an error the client re-issues the request to
343 * the manager with ?tried=h1,h2&triedrc=rc1,rc2 listing servers it already
344 * attempted. Returns 1 when every server exporting this path has already
345 * been tried — the caller must then answer kXR_NotFound instead of
346 * redirecting, otherwise the client bounces manager<->data-server until it
347 * hits its redirect limit (the divergence conformance testing caught).
348 * WHY: A conformant redirector must converge to "not found" once the client has
349 * visited every candidate; reference cmsd does this via the tried list.
350 * HOW: Extract the opaque (everything after '?') from the raw request payload,
351 * locate "tried=", count its comma-separated hosts, and compare against
352 * brix_srv_count_matching(clean_path). Conservative: a zero match count
353 * falls through to the normal CMS-locate path so this does not prematurely
354 * short-circuit hierarchical (parent-locate) clusters. */
355int
356brix_manager_tried_exhausted(const u_char *payload, size_t payload_len,
357 const char *clean_path)
358{
359 char opaque[1024];
360 const u_char *q;
361 size_t olen;
362 const char *t, *p;
363 int n_tried, n_match;
364
365 if (payload == NULL || payload_len == 0) {
366 return 0;
367 }
368 q = memchr(payload, '?', payload_len);
369 if (q == NULL) {
370 return 0; /* no opaque -> client's first attempt */
371 }
372 q++;
373 olen = (size_t) (payload + payload_len - q);
374 if (olen > 0 && q[olen - 1] == '\0') {
375 olen--; /* trim trailing NUL some payloads carry */
376 }
377 if (olen == 0 || olen >= sizeof(opaque)) {
378 return 0; /* empty or too long to inspect — be safe */
379 }
380 ngx_memcpy(opaque, q, olen);
381 opaque[olen] = '\0';
382
383 t = strstr(opaque, "tried=");
384 if (t == NULL) {
385 return 0;
386 }
387 t += 6; /* skip "tried=" */
388 if (*t == '\0' || *t == '&') {
389 return 0; /* present but empty */
390 }
391
392 n_tried = 1;
393 for (p = t; *p && *p != '&'; p++) {
394 if (*p == ',') {
395 n_tried++;
396 }
397 }
398
399 n_match = brix_srv_count_matching(clean_path);
400 return (n_match > 0 && n_tried >= n_match);
401}

405/* WHAT
406 * Marks a registered server as temporarily unavailable for selection.
407 * Called from brix_cms_srv_close() when a data server's CMS connection
408 * drops. The server entry stays in the registry so its paths and metrics are
409 * preserved for the reconnect; brix_srv_select() and brix_srv_locate_all()
410 * both skip entries whose blacklisted_until is in the future.
411 *
412 * WHY
413 * A clean reconnect within the window re-registers and clears the flag,
414 * making the server immediately available again. A permanently dead server
415 * stays blacklisted until the window expires, at which point its stale metrics
416 * become visible — operators detect this via brix_cluster_server_last_seen_seconds.
417 *
418 * HOW
419 * Locks mutex → scans for host+port match → increments error_count →
420 * sets blacklisted_until = ngx_current_msec + duration_ms. Unlocks.
421 */
422void
423brix_srv_blacklist(const char *host, uint16_t port, ngx_msec_t duration_ms)
424{
425 brix_srv_table_t *tbl;
426 brix_srv_entry_t *e;
427 ngx_uint_t i;
428
429 tbl = srv_table();
430 if (tbl == NULL) {
431 return;
432 }
433
434 ngx_shmtx_lock(&brix_srv_mutex);
435
436 for (i = 0; i < tbl->capacity; i++) {
437 e = &tbl->slots[i];
438 if (!e->in_use || e->port != port
439 || ngx_strcmp(e->host, host) != 0)
440 {
441 continue;
442 }
443 e->error_count++;
444 e->blacklisted_until = ngx_current_msec + duration_ms;
445 break;
446 }
447
448 ngx_shmtx_unlock(&brix_srv_mutex);
449}
