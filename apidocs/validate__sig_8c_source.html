1/*
2 * validate_sig.c — JWT signature plane: JOSE-header policing and JWKS
3 * signature verification for WLCG/SciToken bearer tokens.
4 *
5 * WHAT: Owns the two trust-boundary steps that must run — and pass — before any
6 * payload claim is believed: token_check_header() base64url-decodes the JWS
7 * protected header, extracts "alg"/"kid", and rejects any algorithm other than
8 * RS256/ES256 or any asserted "crit" header (the alg:"none" forgery defence);
9 * token_verify_signature() base64url-decodes the signature segment and verifies
10 * the "header.payload" signing input against the loaded JWKS keys. The kid
11 * key-selection helper and the algorithm-dispatched verifier are file-local.
12 *
13 * WHY: validate.c exceeded the ~500-line file-size guard, so the security-
14 * critical signature plane was split out under the phase-79 guard. It shares the
15 * decoded-header carrier (token_hdr_t), the log sanitiser, and the malformed-log
16 * helper with the main pipeline via validate_internal.h. The bytes verified, the
17 * algorithm allow-list, the RFC 7515 §4.1.4 kid selection (with legacy single-
18 * key leniency and rotation-grace multi-key fallback), and the EVP dispatch are
19 * preserved exactly from the original single-file implementation.
20 *
21 * HOW: token_check_header() → b64url_decode seg[0] → json "alg"/"kid" → allow-
22 * list + "crit" rejection. token_verify_signature() → b64url_decode seg[2] →
23 * build the token_sig_input_t over "header.payload" → dispatch via token_sig_ok()
24 * (ES256/RS256) against the kid-selected key, or every JWKS key in order when kid
25 * is absent. Rejections route their untrusted inputs through token_sanitize_for_log()
26 * or brix_token_malformed() (shared with validate.c) so log lines stay uniform.
27 */

36/*
37 * token_sig_input_t — everything the algorithm-dispatched verifier needs.
38 *
39 * WHAT: The signing input ("header.payload" bytes + length) and the decoded
40 * binary signature, plus the accepted algorithm name.
41 * WHY: token_sig_ok() is called from both the kid-exact-match path and the
42 * kid-absent multi-key loop; passing one struct + the candidate key
43 * keeps the helper at 2 parameters (was 6).
44 * HOW: Filled once in token_verify_signature() after the signature segment is
45 * base64url-decoded; only the candidate EVP_PKEY varies per call.
46 */
47typedef struct {
48 const char *alg;
49 const u_char *signed_data;
50 size_t signed_len;
51 const u_char *sig;
52 size_t sig_len;
53} token_sig_input_t;

55/*
56 * token_sig_ok — dispatch signature verification by algorithm.
57 *
58 * WHAT: Calls brix_token_verify_es256() for ES256 or brix_token_verify_rs256()
59 * for RS256 (the only two accepted algorithms at this call site).
60 * WHY: DRY helper shared by the kid-present single-key path and the kid-absent
61 * multi-key fallback loop; avoids duplicating the ES256/RS256 branch.
62 * HOW: strcmp on alg (already validated non-NULL before this point).
63 * Returns 1 on success, 0 on failure, matching the underlying verify API.
64 */
65static int
66token_sig_ok(const token_sig_input_t *in, EVP_PKEY *pkey)
67{
68 if (strcmp(in->alg, "ES256") == 0) {
69 return brix_token_verify_es256(in->signed_data, in->signed_len,
70 in->sig, in->sig_len, pkey);
71 }
72 return brix_token_verify_rs256(in->signed_data, in->signed_len,
73 in->sig, in->sig_len, pkey);
74}

76/*
77 * token_check_header — decode and police the JWS protected header (RFC 7515).
78 *
79 * WHAT: Base64url-decodes segment 0, extracts "alg" and "kid" into *hdr,
80 * rejects any algorithm other than RS256/ES256, and rejects any token
81 * asserting a "crit" header. Returns 0 on acceptance, -1 on rejection
82 * (reason logged).
83 * WHY: alg:"none" bypass — a token that declares alg:"none" or any other
84 * algorithm must be rejected BEFORE signature verification, preventing
85 * an attacker from forging an unsigned token. RFC 7515 §4.1.11: we
86 * implement no `crit` extension parameters, so any token asserting
87 * critical headers we do not understand MUST be rejected.
88 * HOW: Decode failure routes through brix_token_malformed() (same log line as
89 * every structural rejection); alg/kid are zeroed before extraction so
90 * an absent kid reads as "".
91 */
92int
93token_check_header(const brix_token_validate_args_t *a, const xrdjwt_seg *seg,
94 token_hdr_t *hdr)
95{
96 u_char hdr_json[2048];
97 ssize_t hdr_len;
98
99 hdr_len = b64url_decode(seg[0].p, seg[0].n, hdr_json,
100 sizeof(hdr_json) - 1);
101 if (hdr_len < 0) {
102 return brix_token_malformed(a->log);
103 }
104 hdr_json[hdr_len] = '\0';
105
106 ngx_memzero(hdr->alg, sizeof(hdr->alg));
107 ngx_memzero(hdr->kid, sizeof(hdr->kid));
108 json_get_string((char *) hdr_json, (size_t) hdr_len, "alg", hdr->alg,
109 sizeof(hdr->alg));
110 /* alg:"none" bypass: reject any algorithm other than RS256/ES256 before
111 * touching the signature. An unsigned token (alg:"none") must be
112 * rejected here, not later, or the signature step is skipped. */
113 if (strcmp(hdr->alg, "RS256") != 0 && strcmp(hdr->alg, "ES256") != 0) {
114 char safe_alg[64];
115 token_sanitize_for_log(hdr->alg, safe_alg, sizeof(safe_alg));
116 ngx_log_error(NGX_LOG_WARN, a->log, 0,
117 "brix_token: unsupported JWT algorithm \"%s\" "
118 "(only RS256 and ES256 accepted)", safe_alg);
119 return -1;
120 }
121
122 json_get_string((char *) hdr_json, (size_t) hdr_len, "kid", hdr->kid,
123 sizeof(hdr->kid));
124
125 /* RFC 7515 §4.1.11: we implement no `crit` extension parameters, so any
126 * token asserting critical headers we do not understand MUST be rejected. */
127 if (json_has_member((char *) hdr_json, (size_t) hdr_len, "crit")) {
128 ngx_log_error(NGX_LOG_WARN, a->log, 0,
129 "brix_token: JWS 'crit' header not supported — rejecting");
130 return -1;
131 }
132 return 0;
133}

135/*
136 * token_select_key_by_kid — resolve the asserted "kid" to a trusted JWKS key.
137 *
138 * WHAT: Exact-matches hdr->kid against the loaded JWKS key ids; when no key
139 * matches but exactly one key is loaded, that key is used (legacy
140 * single-key leniency). Returns the key or NULL (mismatch logged).
141 * WHY: Key selection by kid is the RFC 7515 §4.1.4 path; the single-key
142 * fallback preserves long-standing behavior for deployments whose JWKS
143 * predates kid discipline.
144 * HOW: Linear scan (JWKS sets are tiny); the failure log sanitizes the
145 * attacker-controlled kid before it reaches the error log.
146 */
147static EVP_PKEY *
148token_select_key_by_kid(const brix_token_validate_args_t *a, const token_hdr_t *hdr)
149{
150 EVP_PKEY *pkey = NULL;
151 int i;
152
153 for (i = 0; i < a->key_count; i++) {
154 if (strcmp(a->keys[i].kid, hdr->kid) == 0) {
155 pkey = a->keys[i].pkey;
156 break;
157 }
158 }
159 if (pkey == NULL && a->key_count == 1) {
160 pkey = a->keys[0].pkey; /* preserve legacy single-key leniency */
161 }
162 if (pkey == NULL) {
163 char safe_kid[256];
164 token_sanitize_for_log(hdr->kid, safe_kid, sizeof(safe_kid));
165 ngx_log_error(NGX_LOG_WARN, a->log, 0,
166 "brix_token: no JWKS key matching kid=\"%s\"",
167 safe_kid);
168 }
169 return pkey;
170}

172/*
173 * token_verify_signature — decode the signature and verify it against JWKS.
174 *
175 * WHAT: Base64url-decodes segment 2, then verifies the "header.payload"
176 * signing input: kid asserted → the kid-selected key only; kid absent →
177 * every JWKS key in order (rotation grace, WLCG profile §3.3). Returns
178 * 0 on a valid signature, -1 otherwise (reason logged).
179 * WHY: The signature MUST be verified before any payload claim is trusted;
180 * isolating key selection + verification keeps that trust boundary in
181 * one auditable helper.
182 * HOW: signed_len spans "header.payload" (seg lengths + the joining dot);
183 * verification dispatches through token_sig_ok() per accepted algorithm.
184 */
185int
186token_verify_signature(const brix_token_validate_args_t *a, const xrdjwt_seg *seg,
187 const token_hdr_t *hdr)
188{
189 u_char sig_bin[512];
190 ssize_t sig_len;
191 token_sig_input_t in;
192 int sig_ok = 0;
193 int i;
194
195 sig_len = b64url_decode(seg[2].p, seg[2].n, sig_bin, sizeof(sig_bin));
196 if (sig_len < 0) {
197 ngx_log_error(NGX_LOG_WARN, a->log, 0,
198 "brix_token: cannot decode JWT signature");
199 return -1;
200 }
201
202 in.alg = hdr->alg;
203 in.signed_data = (const u_char *) a->token;
204 in.signed_len = seg[0].n + 1 + seg[1].n; /* "header.payload" */
205 in.sig = sig_bin;
206 in.sig_len = (size_t) sig_len;
207
208 if (hdr->kid[0] != '\0') {
209 /* kid asserted: exact-match the named key only. */
210 EVP_PKEY *pkey = token_select_key_by_kid(a, hdr);
211 if (pkey == NULL) {
212 return -1;
213 }
214 sig_ok = token_sig_ok(&in, pkey);
215 } else {
216 /* kid absent: try every JWKS key in order (rotation grace, §3.3). */
217 for (i = 0; i < a->key_count && !sig_ok; i++) {
218 sig_ok = token_sig_ok(&in, a->keys[i].pkey);
219 }
220 }
221 if (!sig_ok) {
222 ngx_log_error(NGX_LOG_WARN, a->log, 0,
223 "brix_token: JWT signature verification failed");
224 return -1;
225 }
226 return 0;
227}
