1/*
2 * access.c - WebDAV access-phase handler: auth gate, CORS, write guards,
3 * token scopes, and common WebDAV request checks.
4 *
5 * WHAT: Runs in NGX_HTTP_ACCESS_PHASE (before any content handler) for every
6 * request in an brix_webdav location. Handles: CORS headers, request
7 * metrics, authentication (GSI proxy cert then bearer token), XrdHttp opaque
8 * header extraction, allow_write gate, and token scope enforcement.
9 *
10 * WHY: Keeping auth/scope checks in the access phase means every native WebDAV
11 * content handler sees a pre-authenticated request and can focus on its
12 * method-specific path, lock, and filesystem semantics.
13 *
14 * HOW: Returns NGX_DECLINED when the location has brix_webdav disabled (not
15 * our location), NGX_OK to allow content handlers to proceed, or an HTTP error
16 * code to reject the request. webdav_metrics_return() is used for rejections
17 * so that the per-method error counters are incremented before returning.
18 * webdav_metrics_request() is called once per request at entry.
19 */

30/* Map a WebDAV HTTP method to the XrdAcc operation it requires. */
31static brix_acc_op_t
32webdav_method_aop(ngx_http_request_t *r)
33{
34 switch (r->method) {
35 case NGX_HTTP_GET:
36 case NGX_HTTP_HEAD: return BRIX_AOP_READ;
37 case NGX_HTTP_PUT: return BRIX_AOP_CREATE;
38 case NGX_HTTP_DELETE: return BRIX_AOP_DELETE;
39 case NGX_HTTP_MKCOL: return BRIX_AOP_MKDIR;
40 case NGX_HTTP_MOVE: return BRIX_AOP_RENAME;
41 case NGX_HTTP_COPY:
42 /* A TPC PULL (Source: header present) WRITES r->uri (the local
43 * destination), so it must be authorized as a CREATE — otherwise a
44 * read-only principal could pull remote data onto a path it may not
45 * write. A PUSH / plain intra-server COPY reads r->uri (the source). */
46 if (webdav_tpc_find_header(r, "Source", sizeof("Source") - 1) != NULL) {
47 return BRIX_AOP_CREATE;
48 }
49 return BRIX_AOP_READ;
50 case NGX_HTTP_PROPFIND: return BRIX_AOP_READDIR;
51 case NGX_HTTP_PROPPATCH: return BRIX_AOP_UPDATE;
52 case NGX_HTTP_LOCK:
53 case NGX_HTTP_UNLOCK: return BRIX_AOP_UPDATE;
54 case NGX_HTTP_OPTIONS: return BRIX_AOP_ANY;
55 default: return BRIX_AOP_STAT;
56 }
57}

59/*
60 * webdav_acc_check — XrdAcc tier for WebDAV (when `brix_authdb_format xrdacc`).
61 * Returns NGX_OK (allow / not selected) or NGX_HTTP_FORBIDDEN (deny).
62 */
63static ngx_int_t
64webdav_acc_check(ngx_http_request_t *r,
65 ngx_http_brix_webdav_loc_conf_t *conf)
66{
67 ngx_http_brix_webdav_req_ctx_t *mctx;
68 brix_identity_t *id = NULL;
69 const char *name = "", *vorg = "", *role = "", *grp = "";
70 char host[64], path[1024];
71 ngx_int_t rc;
72 size_t n;
73
74 if (conf->acc.format != BRIX_AUTHDB_FORMAT_XRDACC) {
75 return NGX_OK; /* engine not selected */
76 }
77
78 mctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
79 if (mctx != NULL && mctx->identity != NULL) {
80 id = mctx->identity;
81 name = brix_identity_dn_cstr(id);
82 vorg = brix_identity_acc_vorg_cstr(id);
83 role = brix_identity_acc_role_cstr(id);
84 grp = brix_identity_acc_group_cstr(id);
85 }
86
87 n = ngx_min(r->connection->addr_text.len, sizeof(host) - 1);
88 ngx_memcpy(host, r->connection->addr_text.data, n);
89 host[n] = '\0';
90
91 /* Opt-in reverse DNS for `h <host>`/`h .domain` rules (per request). */
92 if (conf->acc.resolve_hosts) {
93 char hbuf[256];
94 const char *h = brix_acc_resolve_peer(r->connection->sockaddr,
95 r->connection->socklen,
96 hbuf, sizeof(hbuf));
97 if (h != NULL) {
98 n = ngx_min(ngx_strlen(h), sizeof(host) - 1);
99 ngx_memcpy(host, h, n);
100 host[n] = '\0';
101 }
102 }
103
104 n = ngx_min(r->uri.len, sizeof(path) - 1);
105 ngx_memcpy(path, r->uri.data, n);
106 path[n] = '\0';
107
108 rc = brix_acc_http_authorize(r->pool, r->connection->log,
109 &conf->acc, name, host,
110 vorg, role, grp,
111 webdav_method_aop(r), path);
112
113 return (rc == NGX_ERROR) ? NGX_HTTP_FORBIDDEN : NGX_OK;
114}

152 *
153 * WHAT: Counts the request once (webdav_metrics_request) and increments the

176 * access_rate_limit — Phase 20 per-client-IP request rate limit.
177 *
178 * WHAT: Sheds requests from clients exceeding the configured rate.
179 *
180 * WHY: Applied before the auth burden so a flood of unauthenticated requests
181 * is rejected cheaply.
182 *
183 * HOW: Keys the shared rate-limit table by the textual client address.
184 * Returns NGX_OK to continue, or the metrics-counted 429 rejection.

186static ngx_int_t
187access_rate_limit(ngx_http_request_t *r,
188 ngx_http_brix_webdav_loc_conf_t *conf)
189{

191

195
196 ip = &r->connection->addr_text;
197
198 if (brix_rate_limit_check(&conf->rate_limit,
199 (const char *) ip->data, ip->len)

202 return webdav_metrics_return(r, NGX_HTTP_TOO_MANY_REQUESTS);
203 }

233/*
234 * access_capture_deleg_proxy — Phase-70 §5.1 delegated-proxy header capture.
235 *
236 * WHAT: When this export delegates to a backend, captures an optional
237 * user-supplied full x509 proxy (X-Brix-Delegate-Proxy).
238 *
239 * WHY: Runs only in a delegation mode (skipped for the default SELECT
240 * export). The shared parser enforces TLS-only + leaf-DN==authenticated-
241 * identity and rejects (403) a present-but-invalid header; a captured proxy
242 * is stashed on the req ctx for the VFS bind sites.
243 *
244 * HOW: Returns NGX_OK to continue or the metrics-counted rejection code from
245 * the shared parser.
246 */
247static ngx_int_t
248access_capture_deleg_proxy(ngx_http_request_t *r,
249 ngx_http_brix_webdav_loc_conf_t *conf)
250{
251 ngx_http_brix_webdav_req_ctx_t *dctx;
252 ngx_str_t proxy_pem;
253 ngx_int_t cap_rc;
254
255 if (conf->common.backend_delegation == BRIX_CRED_SELECT) {
256 return NGX_OK;
257 }
258
259 dctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
260
261 cap_rc = brix_proto_deleg_capture_proxy_header(r,
262 (dctx != NULL) ? dctx->identity : NULL, &proxy_pem);
263 if (cap_rc != NGX_OK) {
264 return webdav_metrics_return(r, cap_rc);
265 }
266 if (dctx != NULL && proxy_pem.len > 0) {
267 dctx->deleg_proxy_pem = proxy_pem;
268 }
269
270 return NGX_OK;
271}

273/*
274 * access_apply_authdb — native authdb + VO ACL read parity with root://.
275 *
276 * WHAT: Gates READ methods (GET/HEAD/PROPFIND) by the SAME per-DN/VO rules
277 * root:// enforces, so a WebDAV read — which may serve CACHED bytes — is
278 * authorized identically to a cache miss.
279 *
280 * WHY: Runs only when brix_webdav_authdb / brix_webdav_require_vo are
281 * configured (the check helpers return NGX_OK for empty rule sets, so
282 * existing deployments are unaffected). Write methods keep their
283 * allow_write + xrdacc + token-scope gates. Skips OPTIONS (CORS preflight).
284 *
285 * HOW: Resolves the confined canonical path, then applies the authdb and VO
286 * ACL identity checks. Returns NGX_OK to continue or the metrics-counted
287 * rejection (path-resolution error code, or 403 on rule denial).
288 */
289static ngx_int_t
290access_apply_authdb(ngx_http_request_t *r,
291 ngx_http_brix_webdav_loc_conf_t *conf)
292{
293 ngx_http_brix_webdav_req_ctx_t *actx;
294 const brix_identity_t *aid;
295 char resolved[PATH_MAX];
296 char peer[NGX_SOCKADDR_STRLEN];
297 size_t pn;
298 ngx_int_t pr;
299
300 if (webdav_is_write_method(r) || r->method == NGX_HTTP_OPTIONS
301 || (conf->authdb_rules == NULL && conf->vo_rules == NULL))
302 {
303 return NGX_OK;
304 }
305
306 actx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
307 aid = (actx != NULL) ? actx->identity : NULL;
308
309 pr = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon,
310 resolved, sizeof(resolved));
311 if (pr != NGX_OK) {
312 return webdav_metrics_return(r, pr);
313 }
314 pn = ngx_min(r->connection->addr_text.len, sizeof(peer) - 1);
315 ngx_memcpy(peer, r->connection->addr_text.data, pn);
316 peer[pn] = '\0';
317
318 brix_authdb_query_t query = {
319 .rules = conf->authdb_rules,
320 .identity = aid,
321 .peer_ip = peer,
322 .resolved_path = resolved,
323 .needed_privs = BRIX_AUTH_READ,
324 };
325
326 if (brix_check_authdb_identity(r->connection->log, &query) != NGX_OK
327 || brix_check_vo_acl_identity(r->connection->log, resolved,
328 conf->vo_rules, aid) != NGX_OK)
329 {
330 return webdav_metrics_return(r, NGX_HTTP_FORBIDDEN);
331 }
332
333 return NGX_OK;
334}

336/*
337 * access_is_oauth_endpoint — OAuth2 control-plane URI detection.
338 *
339 * WHAT: Returns non-zero when the request URI ends in /.oauth2/token or
340 * /.well-known/oauth-authorization-server.
341 *
342 * WHY: These control-plane endpoints are not data resources; path-scope
343 * checking the endpoint URI conflates the data-resource model with OAuth2.
344 * Authority bounding is enforced inside the issuance handler instead.
345 *
346 * HOW: Suffix comparison against the two static endpoint strings.
347 */
348static int
349access_is_oauth_endpoint(ngx_http_request_t *r)
350{
351 static const char tok_ep[] = "/.oauth2/token";
352 static const char disc_ep[] = "/.well-known/oauth-authorization-server";
353
354 return (r->uri.len >= sizeof(tok_ep) - 1
355 && ngx_memcmp(r->uri.data + r->uri.len - (sizeof(tok_ep) - 1),
356 tok_ep, sizeof(tok_ep) - 1) == 0)
357 || (r->uri.len >= sizeof(disc_ep) - 1
358 && ngx_memcmp(r->uri.data + r->uri.len - (sizeof(disc_ep) - 1),
359 disc_ep, sizeof(disc_ep) - 1) == 0);
360}

362/*
363 * access_check_scope — token scope check (read AND write).
364 *
365 * WHAT: Enforces the bearer token's path scope against the method being
366 * performed. Exempt: OPTIONS (capability query / CORS preflight, no data
367 * access), LOCK/UNLOCK (advisory locks require no scope claim,
368 * pre-existing), and the OAuth2 control-plane endpoints (see
369 * access_is_oauth_endpoint).
370 *
371 * WHY: This is the LAST gate — the global allow_write check has already run
372 * (INVARIANT 3: allow_write before token scope).
373 *
374 * HOW: op->name is a static, null-terminated string, so it is safe to log.
375 * Returns NGX_OK to continue or the metrics-counted rejection from
376 * webdav_check_token_scope.
377 */
378static ngx_int_t
379access_check_scope(ngx_http_request_t *r)
380{
381 const brix_http_operation_t *op;
382 const char *mname;
383 ngx_int_t rc;
384 int is_lock;
385
386 if (r->method == NGX_HTTP_OPTIONS) {
387 return NGX_OK;
388 }
389
390 op = brix_http_operation_find(r, brix_webdav_operations,
391 brix_webdav_operations_count);
392 is_lock = (op != NULL
393 && (ngx_strcmp(op->name, "LOCK") == 0
394 || ngx_strcmp(op->name, "UNLOCK") == 0));
395 if (is_lock || access_is_oauth_endpoint(r)) {
396 return NGX_OK;
397 }
398
399 mname = (op != NULL) ? op->name : "GET";
400 rc = webdav_check_token_scope(r, mname);
401 if (rc != NGX_OK) {
402 return webdav_metrics_return(r, rc);
403 }
404
405 return NGX_OK;
406}

408ngx_int_t
409ngx_http_brix_webdav_access_handler(ngx_http_request_t *r)
410{
411 ngx_http_brix_webdav_loc_conf_t *conf;
412 ngx_int_t rc;
413
414 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
415 if (!conf->common.enable) {
416 return NGX_DECLINED;
417 }
418
419 /* Phase 24: mirror subrequests are already authorized by the parent. */
420 if (access_is_mirror_subrequest(r)) {
421 return NGX_OK;
422 }
423
424 /* CORS headers must appear on every response. */
425 if (webdav_add_cors_headers(r) != NGX_OK) {
426 return NGX_HTTP_INTERNAL_SERVER_ERROR;
427 }
428
429 /* Count the request and track IP-version bytes. */
430 access_count_request(r);
431
432 /* Phase 20: per-client-IP rate limit, before the auth burden. */
433 rc = access_rate_limit(r, conf);
434 if (rc != NGX_OK) {
435 return rc;
436 }
437
438 /* OPTIONS pre-flight: no authentication required. */
439 if (r->method == NGX_HTTP_OPTIONS) {
440 return access_options_preflight(r);
441 }
442
443 /* Authentication gate */
444 rc = access_authenticate(r, conf);
445 if (rc != NGX_OK) {
446 return rc;
447 }
448
449 /* XrdHttp: extract client identity, UUID, opaque, and ?tpc.* params.
450 * Must run after auth so the request context exists. */
451 xrdhttp_parse_request(r);
452 xrdhttp_inject_tpc_headers(r);
453
454 /* Phase-70 §5.1: delegated-proxy header capture (delegation modes only). */
455 rc = access_capture_deleg_proxy(r, conf);
456 if (rc != NGX_OK) {
457 return rc;
458 }
459
460 /* Write-method gate — the GLOBAL allow_write check runs BEFORE the token
461 * scope check below (INVARIANT 3). */
462 if (webdav_is_write_method(r) && !conf->common.allow_write) {
463 return webdav_metrics_return(r, NGX_HTTP_FORBIDDEN);
464 }
465
466 /* XrdAcc engine (when brix_authdb_format xrdacc) */
467 if (webdav_acc_check(r, conf) != NGX_OK) {
468 return webdav_metrics_return(r, NGX_HTTP_FORBIDDEN);
469 }
470
471 /* Native authdb + VO ACL read parity with root:// */
472 rc = access_apply_authdb(r, conf);
473 if (rc != NGX_OK) {
474 return rc;
475 }
476
477 /* Token scope check (read AND write) — after allow_write (INVARIANT 3). */
478 rc = access_check_scope(r);
479 if (rc != NGX_OK) {
480 return rc;
481 }
482
483 return NGX_OK;
484}
