1/*
2 * access.c - WebDAV access-phase handler: auth gate, CORS, write guards,
3 * token scopes, and common WebDAV request checks.
4 *
5 * WHAT: Runs in NGX_HTTP_ACCESS_PHASE (before any content handler) for every
6 * request in an brix_webdav location. Handles: CORS headers, request
7 * metrics, authentication (GSI proxy cert then bearer token), XrdHttp opaque
8 * header extraction, allow_write gate, and token scope enforcement.
9 *
10 * WHY: Keeping auth/scope checks in the access phase means every native WebDAV
11 * content handler sees a pre-authenticated request and can focus on its
12 * method-specific path, lock, and filesystem semantics.
13 *
14 * HOW: Returns NGX_DECLINED when the location has brix_webdav disabled (not
15 * our location), NGX_OK to allow content handlers to proceed, or an HTTP error
16 * code to reject the request. webdav_metrics_return() is used for rejections
17 * so that the per-method error counters are incremented before returning.
18 * webdav_metrics_request() is called once per request at entry.
19 */
20

29/* Map a WebDAV HTTP method to the XrdAcc operation it requires. */
30static brix_acc_op_t
31webdav_method_aop(ngx_http_request_t *r)
32{
33 switch (r->method) {
34 case NGX_HTTP_GET:
35 case NGX_HTTP_HEAD: return BRIX_AOP_READ;
36 case NGX_HTTP_PUT: return BRIX_AOP_CREATE;
37 case NGX_HTTP_DELETE: return BRIX_AOP_DELETE;
38 case NGX_HTTP_MKCOL: return BRIX_AOP_MKDIR;
39 case NGX_HTTP_MOVE: return BRIX_AOP_RENAME;
40 case NGX_HTTP_COPY:
41 /* A TPC PULL (Source: header present) WRITES r->uri (the local
42 * destination), so it must be authorized as a CREATE — otherwise a
43 * read-only principal could pull remote data onto a path it may not
44 * write. A PUSH / plain intra-server COPY reads r->uri (the source). */
45 if (webdav_tpc_find_header(r, "Source", sizeof("Source") - 1) != NULL) {
46 return BRIX_AOP_CREATE;
47 }
48 return BRIX_AOP_READ;
49 case NGX_HTTP_PROPFIND: return BRIX_AOP_READDIR;
50 case NGX_HTTP_PROPPATCH: return BRIX_AOP_UPDATE;
51 case NGX_HTTP_LOCK:
52 case NGX_HTTP_UNLOCK: return BRIX_AOP_UPDATE;
53 case NGX_HTTP_OPTIONS: return BRIX_AOP_ANY;
54 default: return BRIX_AOP_STAT;
55 }
56}

58/*
59 * webdav_acc_check — XrdAcc tier for WebDAV (when `brix_authdb_format xrdacc`).
60 * Returns NGX_OK (allow / not selected) or NGX_HTTP_FORBIDDEN (deny).
61 */
62static ngx_int_t
63webdav_acc_check(ngx_http_request_t *r,
64 ngx_http_brix_webdav_loc_conf_t *conf)
65{
66 ngx_http_brix_webdav_req_ctx_t *mctx;
67 brix_identity_t *id = NULL;
68 const char *name = "", *vorg = "", *role = "", *grp = "";
69 char host[64], path[1024];
70 ngx_int_t rc;
71 size_t n;
72
73 if (conf->acc.format != BRIX_AUTHDB_FORMAT_XRDACC) {
74 return NGX_OK; /* engine not selected */
75 }
76
77 mctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
78 if (mctx != NULL && mctx->identity != NULL) {
79 id = mctx->identity;
80 name = brix_identity_dn_cstr(id);
81 vorg = brix_identity_acc_vorg_cstr(id);
82 role = brix_identity_acc_role_cstr(id);
83 grp = brix_identity_acc_group_cstr(id);
84 }
85
86 n = ngx_min(r->connection->addr_text.len, sizeof(host) - 1);
87 ngx_memcpy(host, r->connection->addr_text.data, n);
88 host[n] = '\0';
89
90 /* Opt-in reverse DNS for `h <host>`/`h .domain` rules (per request). */
91 if (conf->acc.resolve_hosts) {
92 char hbuf[256];
93 const char *h = brix_acc_resolve_peer(r->connection->sockaddr,
94 r->connection->socklen,
95 hbuf, sizeof(hbuf));
96 if (h != NULL) {
97 n = ngx_min(ngx_strlen(h), sizeof(host) - 1);
98 ngx_memcpy(host, h, n);
99 host[n] = '\0';
100 }
101 }
102
103 n = ngx_min(r->uri.len, sizeof(path) - 1);
104 ngx_memcpy(path, r->uri.data, n);
105 path[n] = '\0';
106
107 rc = brix_acc_http_authorize(r->pool, r->connection->log,
108 &conf->acc, name, host,
109 vorg, role, grp,
110 webdav_method_aop(r), path);
111
112 return (rc == NGX_ERROR) ? NGX_HTTP_FORBIDDEN : NGX_OK;
113}

150 * access_count_request — per-request entry metrics.
151 *
152 * WHAT: Counts the request once (webdav_metrics_request) and increments the
153 * per-IP-version received-bytes family.
154 *
155 * WHY: Every request that reaches the access phase must be counted exactly
156 * once, before any accept/reject decision, so error counters and totals agree.
157 *
158 * HOW: Inspects the connection sockaddr family; AF_INET6 goes to the v6
159 * counter, everything else to v4.
160 */
161static void
162access_count_request(ngx_http_request_t *r)
163{
164 webdav_metrics_request(r);
165 if (r->connection && r->connection->sockaddr) {
166 if (r->connection->sockaddr->sa_family == AF_INET6) {
167 BRIX_WEBDAV_METRIC_INC(bytes_rx_ipv6_total);
168 } else {
169 BRIX_WEBDAV_METRIC_INC(bytes_rx_ipv4_total);
170 }
171 }
172}

175 * access_rate_limit — Phase 20 per-client-IP request rate limit.
176 *
177 * WHAT: Sheds requests from clients exceeding the configured rate.
178 *
179 * WHY: Applied before the auth burden so a flood of unauthenticated requests
180 * is rejected cheaply.
181 *
182 * HOW: Keys the shared rate-limit table by the textual client address.
183 * Returns NGX_OK to continue, or the metrics-counted 429 rejection.

185static ngx_int_t
186access_rate_limit(ngx_http_request_t *r,
187 ngx_http_brix_webdav_loc_conf_t *conf)
188{
189 ngx_str_t *ip;
190

194
195 ip = &r->connection->addr_text;
196
197 if (brix_rate_limit_check(&conf->rate_limit,
198 (const char *) ip->data, ip->len)

200 {
201 return webdav_metrics_return(r, NGX_HTTP_TOO_MANY_REQUESTS);
202 }
203
204 return NGX_OK;

307/*
308 * access_basic_challenge — RFC 7617 Basic challenge for browser clients.
309 *
310 * WHAT: Attaches `WWW-Authenticate: Basic realm="brix"` to the response and
311 * returns NGX_HTTP_UNAUTHORIZED, or NGX_HTTP_INTERNAL_SERVER_ERROR on
312 * allocation failure.
313 *
314 * WHY: A browser only shows its login prompt (and re-prompts after a wrong
315 * password) on 401 + a challenge header; a bare 403 is a dead end. Emitted
316 * only when Basic is actually enabled on the export, so cert/token-only
317 * exports keep their historical 403 and never invite a password prompt that
318 * cannot succeed.
319 *
320 * HOW: The same headers_out wiring nginx's own auth_basic module uses — push
321 * the header and point headers_out.www_authenticate at it so the special-
322 * response path emits it.
323 */
324static ngx_int_t
325access_basic_challenge(ngx_http_request_t *r)
326{
327 ngx_table_elt_t *h = ngx_list_push(&r->headers_out.headers);
328
329 if (h == NULL) {
330 return NGX_HTTP_INTERNAL_SERVER_ERROR;
331 }
332 h->hash = 1;
333 h->next = NULL;
334 ngx_str_set(&h->key, "WWW-Authenticate");
335 ngx_str_set(&h->value, "Basic realm=\"brix\"");
336 r->headers_out.www_authenticate = h;
337 return NGX_HTTP_UNAUTHORIZED;
338}

340/*
341 * access_authenticate — the authentication gate.
342 *
343 * WHAT: Runs the credential sources in order (GSI proxy cert, bearer token,
344 * then Basic password) and applies the location's auth policy to the outcome.
345 *
346 * WHY: auth=required rejects unauthenticated requests — with a 401 Basic
347 * challenge when a pwd db is configured (so browsers prompt for credentials),
348 * else the historical 403; auth=optional lets them proceed as anonymous;
349 * auth=none skips verification entirely. Each outcome increments exactly
350 * the same metric slot as before the decomposition.
351 *
352 * HOW: Returns NGX_OK to continue (authenticated or anonymous) or the
353 * metrics-counted rejection.
354 */
355static ngx_int_t
356access_authenticate(ngx_http_request_t *r,
357 ngx_http_brix_webdav_loc_conf_t *conf)
358{
359 ngx_int_t auth_rc;
360
361 if (conf->auth == WEBDAV_AUTH_NONE) {
362 BRIX_WEBDAV_METRIC_INC(
363 auth_total[BRIX_WEBDAV_AUTH_RESULT_NONE]);
364 brix_metric_auth(BRIX_PROTO_WEBDAV, BRIX_AUTHN_NONE, 1);
365 return NGX_OK;
366 }
367
368 auth_rc = access_try_cert(r, conf);
369 if (auth_rc != NGX_OK) {
370 auth_rc = access_try_token(r, conf);
371 }
372 if (auth_rc != NGX_OK) {
373 auth_rc = access_try_basic(r, conf);
374 }
375
376 if (auth_rc != NGX_OK && conf->auth == WEBDAV_AUTH_REQUIRED) {
377 BRIX_WEBDAV_METRIC_INC(
378 auth_total[BRIX_WEBDAV_AUTH_RESULT_REJECTED]);
379 brix_metric_auth(BRIX_PROTO_WEBDAV, BRIX_AUTHN_NONE, 0);
380 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
381 "brix_webdav: unauthenticated request rejected"
382 " (auth=required)");
383 if (conf->pwd_file.len > 0) {
384 return webdav_metrics_return(r, access_basic_challenge(r));
385 }
386 return webdav_metrics_return(r, NGX_HTTP_FORBIDDEN);
387 }
388
389 if (auth_rc != NGX_OK) {
390 BRIX_WEBDAV_METRIC_INC(
391 auth_total[BRIX_WEBDAV_AUTH_RESULT_ANONYMOUS]);
392 brix_metric_auth(BRIX_PROTO_WEBDAV, BRIX_AUTHN_NONE, 1);
393 }
394
395 return NGX_OK;
396}

398/*
399 * access_capture_deleg_proxy — Phase-70 §5.1 delegated-proxy header capture.
400 *
401 * WHAT: When this export delegates to a backend, captures an optional
402 * user-supplied full x509 proxy (X-Brix-Delegate-Proxy).
403 *
404 * WHY: Runs only in a delegation mode (skipped for the default SELECT
405 * export). The shared parser enforces TLS-only + leaf-DN==authenticated-
406 * identity and rejects (403) a present-but-invalid header; a captured proxy
407 * is stashed on the req ctx for the VFS bind sites.
408 *
409 * HOW: Returns NGX_OK to continue or the metrics-counted rejection code from
410 * the shared parser.
411 */
412static ngx_int_t
413access_capture_deleg_proxy(ngx_http_request_t *r,
414 ngx_http_brix_webdav_loc_conf_t *conf)
415{
416 ngx_http_brix_webdav_req_ctx_t *dctx;
417 ngx_str_t proxy_pem;
418 ngx_int_t cap_rc;
419
420 if (conf->common.backend_delegation == BRIX_CRED_SELECT) {
421 return NGX_OK;
422 }
423
424 dctx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
425
426 cap_rc = brix_proto_deleg_capture_proxy_header(r,
427 (dctx != NULL) ? dctx->identity : NULL, &proxy_pem);
428 if (cap_rc != NGX_OK) {
429 return webdav_metrics_return(r, cap_rc);
430 }
431 if (dctx != NULL && proxy_pem.len > 0) {
432 dctx->deleg_proxy_pem = proxy_pem;
433 }
434
435 return NGX_OK;
436}

438/*
439 * access_apply_authdb — native authdb + VO ACL read parity with root://.
440 *
441 * WHAT: Gates READ methods (GET/HEAD/PROPFIND) by the SAME per-DN/VO rules
442 * root:// enforces, so a WebDAV read — which may serve CACHED bytes — is
443 * authorized identically to a cache miss.
444 *
445 * WHY: Runs only when brix_webdav_authdb / brix_webdav_require_vo are
446 * configured (the check helpers return NGX_OK for empty rule sets, so
447 * existing deployments are unaffected). Write methods keep their
448 * allow_write + xrdacc + token-scope gates. Skips OPTIONS (CORS preflight).
449 *
450 * HOW: Resolves the confined canonical path, then applies the authdb and VO
451 * ACL identity checks. Returns NGX_OK to continue or the metrics-counted
452 * rejection (path-resolution error code, or 403 on rule denial).
453 */
454static ngx_int_t
455access_apply_authdb(ngx_http_request_t *r,
456 ngx_http_brix_webdav_loc_conf_t *conf)
457{
458 ngx_http_brix_webdav_req_ctx_t *actx;
459 const brix_identity_t *aid;
460 char resolved[PATH_MAX];
461 char peer[NGX_SOCKADDR_STRLEN];
462 size_t pn;
463 ngx_int_t pr;
464
465 if (webdav_is_write_method(r) || r->method == NGX_HTTP_OPTIONS
466 || (conf->authdb_rules == NULL && conf->vo_rules == NULL))
467 {
468 return NGX_OK;
469 }
470
471 actx = ngx_http_get_module_ctx(r, ngx_http_brix_webdav_module);
472 aid = (actx != NULL) ? actx->identity : NULL;
473
474 pr = ngx_http_brix_webdav_resolve_path(r, conf->common.root_canon,
475 resolved, sizeof(resolved));
476 if (pr != NGX_OK) {
477 return webdav_metrics_return(r, pr);
478 }
479 pn = ngx_min(r->connection->addr_text.len, sizeof(peer) - 1);
480 ngx_memcpy(peer, r->connection->addr_text.data, pn);
481 peer[pn] = '\0';
482
483 brix_authdb_query_t query = {
484 .rules = conf->authdb_rules,
485 .identity = aid,
486 .peer_ip = peer,
487 .resolved_path = resolved,
488 .needed_privs = BRIX_AUTH_READ,
489 };
490
491 if (brix_check_authdb_identity(r->connection->log, &query) != NGX_OK
492 || brix_check_vo_acl_identity(r->connection->log, resolved,
493 conf->vo_rules, aid) != NGX_OK)
494 {
495 return webdav_metrics_return(r, NGX_HTTP_FORBIDDEN);
496 }
497
498 return NGX_OK;
499}

501/*
502 * access_is_oauth_endpoint — OAuth2 control-plane URI detection.
503 *
504 * WHAT: Returns non-zero when the request URI ends in /.oauth2/token or
505 * /.well-known/oauth-authorization-server.
506 *
507 * WHY: These control-plane endpoints are not data resources; path-scope
508 * checking the endpoint URI conflates the data-resource model with OAuth2.
509 * Authority bounding is enforced inside the issuance handler instead.
510 *
511 * HOW: Suffix comparison against the two static endpoint strings.
512 */
513static int
514access_is_oauth_endpoint(ngx_http_request_t *r)
515{
516 static const char tok_ep[] = "/.oauth2/token";
517 static const char disc_ep[] = "/.well-known/oauth-authorization-server";
518
519 return (r->uri.len >= sizeof(tok_ep) - 1
520 && ngx_memcmp(r->uri.data + r->uri.len - (sizeof(tok_ep) - 1),
521 tok_ep, sizeof(tok_ep) - 1) == 0)
522 || (r->uri.len >= sizeof(disc_ep) - 1
523 && ngx_memcmp(r->uri.data + r->uri.len - (sizeof(disc_ep) - 1),
524 disc_ep, sizeof(disc_ep) - 1) == 0);
525}

527/*
528 * access_check_scope — token scope check (read AND write).
529 *
530 * WHAT: Enforces the bearer token's path scope against the method being
531 * performed. Exempt: OPTIONS (capability query / CORS preflight, no data
532 * access), LOCK/UNLOCK (advisory locks require no scope claim,
533 * pre-existing), and the OAuth2 control-plane endpoints (see
534 * access_is_oauth_endpoint).
535 *
536 * WHY: This is the LAST gate — the global allow_write check has already run
537 * (INVARIANT 3: allow_write before token scope).
538 *
539 * HOW: op->name is a static, null-terminated string, so it is safe to log.
540 * Returns NGX_OK to continue or the metrics-counted rejection from
541 * webdav_check_token_scope.
542 */
543static ngx_int_t
544access_check_scope(ngx_http_request_t *r)
545{
546 const brix_http_operation_t *op;
547 const char *mname;
548 ngx_int_t rc;
549 int is_lock;
550
551 if (r->method == NGX_HTTP_OPTIONS) {
552 return NGX_OK;
553 }
554
555 op = brix_http_operation_find(r, brix_webdav_operations,
556 brix_webdav_operations_count);
557 is_lock = (op != NULL
558 && (ngx_strcmp(op->name, "LOCK") == 0
559 || ngx_strcmp(op->name, "UNLOCK") == 0));
560 if (is_lock || access_is_oauth_endpoint(r)) {
561 return NGX_OK;
562 }
563
564 mname = (op != NULL) ? op->name : "GET";
565 rc = webdav_check_token_scope(r, mname);
566 if (rc != NGX_OK) {
567 return webdav_metrics_return(r, rc);
568 }
569
570 return NGX_OK;
571}

573ngx_int_t
574ngx_http_brix_webdav_access_handler(ngx_http_request_t *r)
575{
576 ngx_http_brix_webdav_loc_conf_t *conf;
577 ngx_int_t rc;
578
579 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_webdav_module);
580 if (!conf->common.enable) {
581 return NGX_DECLINED;
582 }
583
584 /* Phase 24: mirror subrequests are already authorized by the parent. */
585 if (access_is_mirror_subrequest(r)) {
586 return NGX_OK;
587 }
588
589 /* CORS headers must appear on every response. */
590 if (webdav_add_cors_headers(r) != NGX_OK) {
591 return NGX_HTTP_INTERNAL_SERVER_ERROR;
592 }
593
594 /* Count the request and track IP-version bytes. */
595 access_count_request(r);
596
597 /* Phase 20: per-client-IP rate limit, before the auth burden. */
598 rc = access_rate_limit(r, conf);
599 if (rc != NGX_OK) {
600 return rc;
601 }
602
603 /* OPTIONS pre-flight: no authentication required. */
604 if (r->method == NGX_HTTP_OPTIONS) {
605 return access_options_preflight(r);
606 }
607
608 /* Authentication gate */
609 rc = access_authenticate(r, conf);
610 if (rc != NGX_OK) {
611 return rc;
612 }

615 * Must run after auth so the request context exists. */
616 xrdhttp_parse_request(r);
617 xrdhttp_inject_tpc_headers(r);
618

621 if (rc != NGX_OK) {
622 return rc;
623 }
624
625 /* Upstream proxy mode: auth is done; proxy content handler takes over. */
626 if (conf->upstream_proxy) {
627 return NGX_OK;
628 }
629
630 /* Write-method gate — the GLOBAL allow_write check runs BEFORE the token
631 * scope check below (INVARIANT 3). */
632 if (webdav_is_write_method(r) && !conf->common.allow_write) {
633 return webdav_metrics_return(r, NGX_HTTP_FORBIDDEN);
634 }
635
636 /* XrdAcc engine (when brix_authdb_format xrdacc) */
637 if (webdav_acc_check(r, conf) != NGX_OK) {
638 return webdav_metrics_return(r, NGX_HTTP_FORBIDDEN);
639 }
640
641 /* Native authdb + VO ACL read parity with root:// */
642 rc = access_apply_authdb(r, conf);
643 if (rc != NGX_OK) {
644 return rc;
645 }
646
647 /* Token scope check (read AND write) — after allow_write (INVARIANT 3). */
648 rc = access_check_scope(r);
649 if (rc != NGX_OK) {
650 return rc;
651 }
652
653 return NGX_OK;
654}
