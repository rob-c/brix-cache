1/*
2 * host_format.h — bracket IPv6 literal hosts for host:port wire/header strings.
3 *
4 * XRootD redirect/locate wire strings, HTTP Host: headers, and rebuilt root://
5 * URLs all carry a host[:port]. A bare IPv6 literal contains colons (e.g.
6 * "2001:db8::1"), so it MUST be bracketed ("[2001:db8::1]") or a client parsing
7 * host:port mis-reads the address. The module stores the canonical *bare*
8 * address everywhere (ngx_sock_ntop(...,0), c->addr_text, and ngx_parse_url all
9 * yield bare IPv6); these helpers bracket *on emit*. IPv4 addresses, hostnames,
10 * and already-bracketed strings pass through unchanged (none contain a bare ':').
11 *
12 * Centralises the bracket format already proven at src/read/locate.c (the
13 * AF_INET6 locate branch emits "S%c[%s]:%d") so every emit site shares one
14 * implementation. Pure C (no nginx types) so it is trivially unit-testable.
15 */
16

23/*
24 * Return 1 if `host` is a bare IPv6 literal that needs bracketing: non-NULL,
25 * non-empty, not already '['-prefixed, and contains ':'. A hostname never
26 * contains ':' and an IPv4 dotted-quad never contains ':', so the colon test is
27 * a sufficient discriminator for a host-only string (it also catches zone-id
28 * link-local forms like "fe80::1%eth0" that inet_pton would reject).
29 */
30int brix_host_is_ipv6_literal(const char *host);

32/*
33 * Write "[host]" for an IPv6 literal, else "host", into out[sz] (NUL-terminated).
34 * For the kXR_redirect body, where the port is carried in a separate field.
35 * Returns bytes written (excluding the NUL), or 0 on NULL/overflow (out is left
36 * as a valid empty string when sz > 0).
37 */
38size_t brix_format_host(const char *host, char *out, size_t sz);
