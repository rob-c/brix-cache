1/* Macaroon caveat parsing and scope narrowing — first-party caveat classification, activity→scope mapping, before: expiry, and path: intersection.
2 *
3 * WHAT: Turns the caveat bytes recovered from the HMAC packet chain into structured claims. Classifies each
4 * first-party caveat (activity:/before:/path:), maps WLCG activities to storage.* scopes, records the earliest
5 * before: expiry, collects path: caveats, and — after parsing — narrows scope paths by intersecting every path:
6 * caveat (and, via the same helper, discharge-supplied paths) against the granted scopes.
7 *
8 * WHY: Split out of macaroon.c (phase-79 file-size split). Caveat interpretation is the authorization core of a
9 * macaroon: activity mapping decides which permissions a token conveys, before: enforces expiry, and path:
10 * intersection guarantees the effective path is the most restrictive of all caveats and scopes (never widening).
11 * Isolating this logic keeps the security-critical narrowing rules in one auditable file, separate from the HMAC
12 * chain machinery (macaroon_parse.c).
13 *
14 * HOW: parse_iso8601() converts a restricted ISO8601 before: value to time_t. macaroon_parse_first_party_caveat()
15 * dispatches on the caveat prefix. macaroon_parse_activity_caveat() splits comma-separated activities and appends
16 * mapped scopes. macaroon_apply_path_caveats() walks each caveat × scope pair via macaroon_apply_path_to_scope()
17 * (narrow / keep / revoke), then rebuilds scope_raw via macaroon_rebuild_scope_raw() if anything narrowed. */

32/* WHAT: Parse ISO8601 timestamp "YYYY-MM-DDTHH:MM:SSZ" into time_t for before: caveat expiry checking.
33 * WHY: Macaroon "before:" caveats specify absolute expiry times in a restricted ISO8601 format (no timezone offsets, no fractional seconds). This helper converts the string to epoch seconds for comparison with current time.
34 * HOW: Validate len≥20 and len<32; copy into buf[32] null-terminated; memset tm struct; sscanf(buf,"%d-%d-%dT%d:%d:%dZ") extracting year/month/day/hour/min/sec (expect 6 matches); adjust tm_year-=1900, tm_mon-=1, tm_isdst=-1; return timegm(&tm) or -1 on parse failure. */
35static time_t
36parse_iso8601(const char *s, size_t len)
37{
38 struct tm tm;
39 char buf[32];
40
41 if (len < 20 || len >= sizeof(buf)) return (time_t) -1;
42 memcpy(buf, s, len);
43 buf[len] = '\0';
44
45 memset(&tm, 0, sizeof(tm));
46 /* %Y-%m-%dT%H:%M:%SZ */
47 if (sscanf(buf, "%d-%d-%dT%d:%d:%dZ",
48 &tm.tm_year, &tm.tm_mon, &tm.tm_mday,
49 &tm.tm_hour, &tm.tm_min, &tm.tm_sec) != 6)
50 {
51 return (time_t) -1;
52 }
53
54 tm.tm_year -= 1900;
55 tm.tm_mon -= 1;
56 tm.tm_isdst = -1;
57
58 return timegm(&tm);
59}

61/* WHAT: Reconstruct scope_raw from individual scope entries and re-parse to refresh claims->scopes[].
62 * WHY: After path narrowing (macaroon_apply_path_caveats) modifies individual scope paths, the raw scope string becomes stale — it still contains old paths. This function rebuilds the canonical scope_raw string from the updated scopes[] array so downstream access checks use the correct narrowed paths.
63 * HOW: Iterate each scope entry's permission bits (read/write/create/modify); for each enabled bit append "perm_name:path" to scope_raw with space separator between entries; check bounds against sizeof(claims->scope_raw) — break if overflow; re-parse the rebuilt string via brix_token_parse_scopes() to refresh claims->scopes[]. */
64/* HOW: Clear scope_raw buffer. Iterate each scope entry's permission bits (read/write/create/modify).
65 * For each enabled bit, append "perm_name:path" to scope_raw with space separator between entries.
66 * Check bounds against sizeof(claims->scope_raw) — break if overflow.
67 * Re-parse the rebuilt string via brix_token_parse_scopes() to refresh claims->scopes[]. */
68static void
69macaroon_rebuild_scope_raw(brix_token_claims_t *claims)
70{
71 static const char *perm_names[] = {
72 "storage.read", "storage.write",
73 "storage.create", "storage.modify"
74 };
75 size_t off = 0;
76 int si;
77
78 claims->scope_raw[0] = '\0';
79 for (si = 0; si < claims->scope_count; si++) {
80 brix_token_scope_t *sc = &claims->scopes[si];
81 unsigned int bits[4] = {
82 sc->read, sc->write, sc->create, sc->modify
83 };
84 int bi;
85 for (bi = 0; bi < 4; bi++) {
86 size_t plen, pathlen;
87 if (!bits[bi]) {
88 continue;
89 }
90 plen = strlen(perm_names[bi]);
91 pathlen = strlen(sc->path);
92 if (off + plen + 1 + pathlen + 2 > sizeof(claims->scope_raw)) {
93 break;
94 }
95 if (off > 0) {
96 claims->scope_raw[off++] = ' ';
97 }
98 memcpy(claims->scope_raw + off, perm_names[bi], plen);
99 off += plen;
100 claims->scope_raw[off++] = ':';
101 memcpy(claims->scope_raw + off, sc->path, pathlen);
102 off += pathlen;
103 claims->scope_raw[off] = '\0';
104 }
105 }
106
107 claims->scope_count = brix_token_parse_scopes(
108 claims->scope_raw, claims->scopes, BRIX_MAX_TOKEN_SCOPES);
109}

117/* WHAT: Apply path: caveats to restrict scope paths via intersection logic — each caveat narrows allowed paths further.
118 * WHY: WLCG macaroon "path:" caveats enforce hierarchical path restrictions on top of already-granted scope permissions. The intersection ensures the final effective path is the most restrictive among all caveats and scopes, preventing over-authorization.
119 * HOW: For each caveat path cp: strip trailing slash from scope paths for comparison; case 1 (cp equal/deeper than sc->path): narrow scope to cp if different; case 2 (sc->path deeper than cp): keep sc->path already more restrictive; case 3 (disjoint paths): revoke all permissions (read/write/create/modify=0) on this scope entry; track narrowed flag; after processing all caveats, call macaroon_rebuild_scope_raw() if any narrowing occurred. */
120static int
121macaroon_apply_path_to_scope(ngx_log_t *log, brix_token_scope_t *scope,
122 const char *caveat_path, size_t caveat_path_len)
123{
124 size_t scope_len = strlen(scope->path);
125 size_t scope_cmp_len = scope_len;
126
127 if (scope_cmp_len > 1 && scope->path[scope_cmp_len - 1] == '/') {
128 scope_cmp_len--;
129 }
130
131 if (strncmp(caveat_path, scope->path, scope_cmp_len) == 0
132 && (caveat_path[scope_cmp_len] == '/'
133 || caveat_path[scope_cmp_len] == '\0')) {
134 if (strcmp(scope->path, caveat_path) == 0) {
135 return 0;
136 }
137 ngx_log_error(NGX_LOG_INFO, log, 0,
138 "brix_macaroon: path: caveat \"%s\" narrows "
139 "scope path \"%s\" → \"%s\"", caveat_path, scope->path,
140 caveat_path);
141 memcpy(scope->path, caveat_path, caveat_path_len + 1);
142 return 1;
143 }
144
145 if (strncmp(scope->path, caveat_path, caveat_path_len) == 0
146 && (scope->path[caveat_path_len] == '/'
147 || scope->path[caveat_path_len] == '\0')) {
148 return 0;
149 }
150
151 if (scope->read || scope->write || scope->create || scope->modify) {
152 ngx_log_error(NGX_LOG_INFO, log, 0,
153 "brix_macaroon: path: caveat \"%s\" revokes "
154 "scope path \"%s\" (disjoint)", caveat_path, scope->path);
155 }
156 scope->read = scope->write = scope->create = scope->modify = 0;
157 return 1;
158}

160void
161macaroon_apply_path_caveats(ngx_log_t *log,
162 brix_token_claims_t *claims,
163 char path_caveats[][BRIX_SCOPE_PATH_MAX], int n_path_caveats)
164{
165 int ci, si;
166 int narrowed = 0;
167
168 for (ci = 0; ci < n_path_caveats; ci++) {
169 const char *cp = path_caveats[ci];
170 size_t cplen = strlen(cp);
171
172 for (si = 0; si < claims->scope_count; si++) {
173 narrowed |= macaroon_apply_path_to_scope(log, &claims->scopes[si],
174 cp, cplen);
175 }
176 }
177
178 if (narrowed) {
179 macaroon_rebuild_scope_raw(claims);
180 }
181}

183static const char *
184macaroon_scope_for_activity(const char *activity, size_t activity_len)
185{
186 if (activity_len == 8 && memcmp(activity, "DOWNLOAD", 8) == 0) {
187 return "storage.read";
188 }
189 if (activity_len == 4 && memcmp(activity, "LIST", 4) == 0) {
190 return "storage.read";
191 }
192 if (activity_len == 5 && memcmp(activity, "STAGE", 5) == 0) {
193 return "storage.stage";
194 }
195 if (activity_len == 6 && memcmp(activity, "UPLOAD", 6) == 0) {
196 return "storage.write";
197 }
198 if (activity_len == 6 && memcmp(activity, "DELETE", 6) == 0) {
199 return "storage.write";
200 }
201 if (activity_len == 6 && memcmp(activity, "MANAGE", 6) == 0) {
202 return "storage.modify";
203 }
204 return NULL;

207static void
208macaroon_append_activity_scope(brix_macaroon_parse_state_t *state,
209 const char *scope)
210{
211 size_t scope_len = strlen(scope);
212
213 /*
214 * Overflow-proof capacity check: reject first if scope_off is already
215 * at/past the buffer end, then compare against the REMAINING space by
216 * subtraction. The additive form (scope_off + scope_len + 4 >= sizeof)
217 * can wrap size_t for a corrupt/hostile scope_off and admit an
218 * out-of-bounds append; this form cannot. Semantics for all in-range
219 * offsets are identical: proceed iff scope_off + scope_len + 4 < sizeof.
220 */
221 if (state->scope_off >= sizeof(state->scope_buf)
222 || scope_len + 4 >= sizeof(state->scope_buf) - state->scope_off) {
223 return;
224 }
225 if (state->scope_off > 0) {
226 state->scope_buf[state->scope_off++] = ' ';
227 }
228 memcpy(state->scope_buf + state->scope_off, scope, scope_len);
229 state->scope_off += scope_len;
230 state->scope_buf[state->scope_off++] = ':';
231 state->scope_buf[state->scope_off++] = '/';
232 state->scope_buf[state->scope_off] = '\0';
233}

235static void
236macaroon_parse_activity_caveat(brix_macaroon_parse_state_t *state,
237 const u_char *caveat, size_t caveat_len)
238{
239 const char *activity = (const char *)(caveat + 9);
240 size_t remaining = caveat_len - 9;
241
242 while (remaining > 0) {
243 const char *comma = memchr(activity, ',', remaining);
244 size_t activity_len = (comma != NULL)
245 ? (size_t)(comma - activity) : remaining;
246 const char *scope = macaroon_scope_for_activity(activity,
247 activity_len);
248
249 if (scope != NULL) {
250 macaroon_append_activity_scope(state, scope);
251 }
252
253 activity += activity_len + (comma != NULL ? 1 : 0);
254 remaining -= activity_len + (comma != NULL ? 1 : 0);
255 }
256}
