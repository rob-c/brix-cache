8/* kXR_fa_recurse support */
9/*
10 * kXR_fa_recurse is a local extension: when fattrList targets a directory the
11 * server walks the subtree and emits one "<relpath>:<U.name>\0" entry per
12 * managed xattr found on each regular file (NOT the standard XRootD behaviour).
13 *
14 * FATTR_RECURSE_XLIST_BUF — per-file listxattr() scratch; names longer than
15 * this whole buffer are silently skipped (return value <= 0 below).
16 * FATTR_RECURSE_RESP_CAP — hard ceiling on the assembled response; entries
17 * past the cap are dropped (see the rctx->len bound check), bounding both
18 * memory and wire frame size for huge trees.
19 * FATTR_RECURSE_MAX_DEPTH — recursion guard against deep/symlink-loop trees.
20 */
21#define FATTR_RECURSE_XLIST_BUF 8192
22#define FATTR_RECURSE_RESP_CAP (256 * 1024)
23#define FATTR_RECURSE_MAX_DEPTH 16
24
25typedef struct {
26 u_char *buf;
27 size_t cap;
28 size_t len;
29 size_t root_len; /* length of root path; relpath = fullpath + root_len */
30 /* Carried so each visited node can build a transient VFS ctx (every opendir /
31 * lstat / listxattr in the walk routes through the confined VFS seam). */
32 ngx_pool_t *pool;
33 ngx_log_t *log;
34 const char *root_canon;
35} fattr_recurse_ctx_t;

37/*
38 *
39 * WHAT: Depth-first walk of dir_path. For every regular file it reads the file's
40 * xattr name list, keeps only the "user.U.*" managed keys, and appends an
41 * entry "<relpath>:<U.name>\0" (relpath = path relative to the recurse root)
42 * into rctx->buf, never exceeding rctx->cap.
43 *
44 * WHY: Lets a single fattrList on a directory enumerate all managed attributes in
45 * a subtree (a local convenience extension). lstat (not stat) is used so we
46 * descend real directories only and never follow symlinks into other trees;
47 * dotfiles are skipped to avoid . / .. recursion and hidden control files.
48 *
49 * HOW: Returns early past FATTR_RECURSE_MAX_DEPTH. Opens the dir via the VFS
50 * (brix_vfs_opendir_quiet) and iterates with brix_vfs_readdir, whose
51 * per-child lstat (nofollow) is the same "do not follow symlinks out of the
52 * subtree" guarantee — recurse into dirs, ignore non-regular files. relpath
53 * is computed by offsetting past root_len (and a leading '/'). Each file's
54 * xattr names come from brix_vfs_listxattr into a stack buffer that is
55 * NUL-separated; the inner loop strides name-by-name. Entries that would
56 * overflow rctx->cap are dropped (no error — best-effort enumeration).
57 * No return value: results accumulate in rctx. */
58static void
59fattr_recurse_dir(fattr_recurse_ctx_t *rctx, const char *dir_path, int depth)
60{
61 brix_vfs_ctx_t dvctx;
62 brix_vfs_dir_t *dir;
63 ngx_str_t name;
64 brix_vfs_stat_t vst;
65 char fpath[BRIX_PATH_MAX];
66 char xlist[FATTR_RECURSE_XLIST_BUF];
67 ssize_t list_sz;
68 char *lp, *lend;
69 size_t full_nlen;
70 const char *relpath;
71 int plen;
72 ngx_int_t rc;
73
74 if (depth > FATTR_RECURSE_MAX_DEPTH) {
75 return;
76 }
77
78 /* Confined, non-metered open of this subtree directory (the enclosing
79 * fattrList op accounts for the whole walk). */
80 brix_vfs_ctx_init(&dvctx, rctx->pool, rctx->log, BRIX_PROTO_ROOT,
81 rctx->root_canon, NULL, 0 /* allow_write */, 0 /* is_tls */, NULL,
82 dir_path);
83 dir = brix_vfs_opendir_quiet(&dvctx, NULL);
84 if (dir == NULL) {
85 return;
86 }
87
88 /* readdir with a stat_out performs a per-child lstat (nofollow) — "." and
89 * ".." are already filtered by the VFS. */
90 while ((rc = brix_vfs_readdir(dir, &name, &vst)) == NGX_OK) {
91 /* Skip any dotfile — avoids hidden control files (and self/parent). */
92 if (name.len == 0 || name.data[0] == '.') {
93 continue;
94 }
95
96 /* Build the absolute child path; drop entries that would truncate. */
97 plen = snprintf(fpath, sizeof(fpath), "%s/%s", dir_path,
98 (char *) name.data);
99 if (plen < 0 || plen >= (int) sizeof(fpath)) {
100 continue;
101 }
102
103 if (vst.is_directory) {
104 fattr_recurse_dir(rctx, fpath, depth + 1);
105 continue;
106 }
107
108 /* Only regular files carry managed xattrs we report. */
109 if (!vst.is_regular) {
110 continue;
111 }
112
113 /* relpath = path relative to the recurse root (strip root + leading '/'). */
114 relpath = fpath + rctx->root_len;
115 if (*relpath == '/') {
116 relpath++;
117 }
118
119 /* listxattr (via the VFS) fills xlist with NUL-separated names; <=0 means
120 * none/error. A transient per-file ctx re-verifies confinement. */
121 {
122 brix_vfs_ctx_t fvctx;
123
124 brix_vfs_ctx_init(&fvctx, rctx->pool, rctx->log,
125 BRIX_PROTO_ROOT, rctx->root_canon, NULL, 0, 0, NULL, fpath);
126 list_sz = brix_vfs_listxattr(&fvctx, xlist, sizeof(xlist));
127 }
128 if (list_sz <= 0) {
129 continue;
130 }
131
132 /* Stride name-by-name through the NUL-separated list (lp..lend). */
133 lp = xlist;
134 lend = xlist + list_sz;
135 while (lp < lend) {
136 full_nlen = strlen(lp);
137 /* Keep only our managed "user.U.*" keys (and require a non-empty
138 * suffix after the prefix). */
139 if (strncmp(lp, BRIX_FATTR_XKEY_PFX, BRIX_FATTR_XKEY_PFX_LEN) == 0
140 && full_nlen > BRIX_FATTR_XKEY_PFX_LEN)
141 {
142 /* Strip the full internal "user.U." prefix (7 bytes) so the
143 * client sees the bare name it set — the reference uses 'U' only
144 * as an internal SFS namespace and returns verbatim names in the
145 * List response (XeqFALsd). Returning "U.name" both diverged from
146 * stock AND broke list→get round-trips (a re-get of "U.name"
147 * would resolve to "user.U.U.name"). */
148 const char *resp_name = lp + BRIX_FATTR_XKEY_PFX_LEN;
149 size_t resp_nlen = full_nlen - BRIX_FATTR_XKEY_PFX_LEN;
150 size_t relpath_len = strlen(relpath);
151 /* entry = "relpath:name\0" — +1 for ':' separator, +1 for NUL */
152 size_t entry_len = relpath_len + 1 + resp_nlen + 1;
153
154 /* Append only if it fits; over-cap entries are silently dropped. */
155 if (rctx->len + entry_len <= rctx->cap) {
156 ngx_memcpy(rctx->buf + rctx->len, relpath, relpath_len);
157 rctx->len += relpath_len;
158 rctx->buf[rctx->len++] = ':';
159 ngx_memcpy(rctx->buf + rctx->len, resp_name, resp_nlen);
160 rctx->len += resp_nlen;
161 rctx->buf[rctx->len++] = '\0';
162 }
163 }
164 /* Advance past this name and its NUL terminator. */
165 lp += full_nlen + 1;
166 }
167 }
168
169 brix_vfs_closedir(dir, rctx->log);
170}

172/*
173 *
174 * WHAT: Allocates the response accumulator, seeds the walk context (root_len so
175 * child paths are reported relative to this directory), drives
176 * fattr_recurse_dir() over the tree, and sends the assembled
177 * "<relpath>:<U.name>\0" list as a kXR_fattr OK response.
178 *
179 * WHY: Entry point for the kXR_fa_recurse extension; isolates the one-shot buffer
180 * allocation and response framing from the recursive walker.
181 *
182 * HOW: Buffer comes from the connection pool (freed with the request). An empty
183 * result still returns OK with a NULL/0 body (valid empty list, not error). */
184static ngx_int_t
185fattr_list_recurse(brix_ctx_t *ctx, ngx_connection_t *c,
186 brix_vfs_ctx_t *vctx, const char *path)
187{
188 fattr_recurse_ctx_t rctx;
189 u_char *buf;
190
191 BRIX_PALLOC_OR_RETURN(buf, c->pool, FATTR_RECURSE_RESP_CAP, brix_send_error(ctx, c, kXR_NoMemory, "out of memory"));
192
193 rctx.buf = buf;
194 rctx.cap = FATTR_RECURSE_RESP_CAP;
195 rctx.len = 0;
196 rctx.root_len = strlen(path); /* relpath = fullpath + root_len in the walk */
197 rctx.pool = vctx->pool;
198 rctx.log = vctx->log;
199 rctx.root_canon = vctx->root_canon;
200
201 fattr_recurse_dir(&rctx, path, 0);
202
203 BRIX_OP_OK(ctx, BRIX_OP_FATTR);
204 if (rctx.len == 0) {
205 return brix_send_ok(ctx, c, NULL, 0);
206 }
207 return brix_send_ok(ctx, c, rctx.buf, (uint32_t) rctx.len);
208}

210/*
211 *
212 * WHAT: Handles kXR_fattrList by calling listxattr(path, NULL, 0) for path-based
213 * operations or flistxattr(fd, NULL, 0) for open-file-handle operations to
214 * query the size of all attribute names stored on the filesystem, then reads
215 * the full NUL-separated name string into a buffer. Filters entries to only
216 * those prefixed with BRIX_FATTR_XKEY_PFX (the user namespace prefix used by
217 * nginx-xrootd), stripping the 5-byte prefix from returned names so clients see
218 * bare attribute keys. When kXR_fa_aData flag is set, also reads each filtered
219 * attribute's value via getxattr/fgetxattr and appends it to the response.
220 *
221 * WHY: XRootD fattrList returns all user-space extended attributes on a file;
222 * nginx-xrootd uses only the user namespace prefixed with "user." so filtering
223 * ensures only managed attributes are visible. The two-phase list (query size
224 * then read) prevents over-allocation. ENOTSUP/EOPNOTSUPP handled gracefully —
225 * filesystems without xattr support return empty response rather than error.
226 * Thread safety: operates only on provided ctx, c, pool and local stack variables. */
227ngx_int_t fattr_list(brix_ctx_t *ctx, ngx_connection_t *c,
228 brix_vfs_ctx_t *vctx, const char *path, int fd,
229 int options) {
230 ngx_pool_t *pool = c->pool;
231 int aData = (options & kXR_fa_aData);
232
233 /* Directory + recurse flag → take the subtree-walk path instead. */
234 if ((options & kXR_fa_recurse) && path != NULL) {
235 brix_vfs_stat_t rvst;
236 if (brix_vfs_probe(vctx, 0 /* follow */, &rvst) == NGX_OK
237 && rvst.is_directory) {
238 return fattr_list_recurse(ctx, c, vctx, path);
239 }
240 }
241 /* Phase 1: probe (size, NULL) to learn how big the name list is. */
242 ssize_t list_sz = path ? brix_vfs_listxattr(vctx, NULL, 0)
243 : brix_vfs_flistxattr(vctx, fd, NULL, 0);
244 if (list_sz < 0) {
245 /* Filesystem with no xattr support → empty list, not an error. */
246 if (errno == ENOTSUP || errno == EOPNOTSUPP) {
247 BRIX_OP_OK(ctx, BRIX_OP_FATTR);
248 return brix_send_ok(ctx, c, NULL, 0);
249 }
250 BRIX_OP_ERR(ctx, BRIX_OP_FATTR);
251 return brix_send_error(ctx, c, kXR_FSError, "listxattr failed");
252 }
253 if (list_sz == 0) {
254 BRIX_OP_OK(ctx, BRIX_OP_FATTR);
255 return brix_send_ok(ctx, c, NULL, 0);
256 }
257 /* +4096 slack absorbs names added between phase 1 and phase 2 (TOCTOU). */
258 char *raw = ngx_palloc(pool, list_sz + 4096);
259 if (raw == NULL) {
260 return brix_send_error(ctx, c, kXR_NoMemory, "out of memory");
261 }
262 /* Phase 2: read the actual NUL-separated name list into raw. */
263 ssize_t actual = path ? brix_vfs_listxattr(vctx, raw, list_sz + 4096)
264 : brix_vfs_flistxattr(vctx, fd, raw, list_sz + 4096);
265 if (actual < 0) {
266 return brix_send_error(ctx, c, kXR_FSError, "listxattr failed");
267 }
268 if (actual == 0) {
269 BRIX_OP_OK(ctx, BRIX_OP_FATTR);
270 return brix_send_ok(ctx, c, NULL, 0);
271 }
272 /*
273 * Worst-case response size: every raw name kept, plus for aData each of up
274 * to kXR_faMaxVars attributes contributes a 4-byte length prefix + a value
275 * up to 4096 bytes; +64 header slop. Over-provisioned so the build loop need
276 * not realloc — the break below still guards against exceeding it.
277 */
278 size_t resp_cap = (size_t) actual + kXR_faMaxVars * (4 + 4096) + 64;
279 u_char *resp = ngx_palloc(pool, resp_cap);
280 if (resp == NULL) {
281 return brix_send_error(ctx, c, kXR_NoMemory, "out of memory");
282 }
283 /* wp = write cursor into resp; lp/lend stride the raw name list. */
284 u_char *wp = resp;
285 char *lp = raw;
286 char *lend = raw + actual;
287 while (lp < lend) {
288 size_t full_nlen = strlen(lp);
289 /* Emit only managed "user.U.*" keys with a non-empty suffix. */
290 if (strncmp(lp, BRIX_FATTR_XKEY_PFX, BRIX_FATTR_XKEY_PFX_LEN) == 0
291 && full_nlen > BRIX_FATTR_XKEY_PFX_LEN) {
292 /* Strip the full internal "user.U." prefix (7 bytes) so the client
293 * sees the bare name it set — matching the reference XeqFALsd, which
294 * returns verbatim names (the 'U' namespace is internal only). */
295 const char *resp_name = lp + BRIX_FATTR_XKEY_PFX_LEN;
296 size_t resp_nlen = full_nlen - BRIX_FATTR_XKEY_PFX_LEN;
297 /* name + NUL, plus (if aData) the 4-byte len prefix + value buffer. */
298 size_t space_needed = resp_nlen + 1
299 + (aData ? 4 + 4096 : 0);
300 /* Stop before overrunning resp_cap (rest of the list is dropped). */
301 if ((size_t)(wp - resp) + space_needed > resp_cap) break;
302 ngx_memcpy(wp, resp_name, resp_nlen);
303 wp += resp_nlen;
304 *wp++ = '\0';
305 if (aData) {
306 /* Append [uint32 big-endian vlen][value bytes] for this attr.
307 * Note getxattr uses the full "user.U.name" key (lp), while the
308 * emitted name above was the stripped form. */
309 char val[4096];
310 ssize_t vlen = path
311 ? brix_vfs_getxattr(vctx, lp, val, sizeof(val))
312 : brix_vfs_fgetxattr(vctx, fd, lp, val, sizeof(val));
313 if (vlen < 0) vlen = 0; /* unreadable value → zero-length */
314 uint32_t vlen_be = htonl((uint32_t) vlen);
315 ngx_memcpy(wp, &vlen_be, 4);
316 wp += 4;
317 if (vlen > 0) {
318 ngx_memcpy(wp, val, vlen);
319 wp += vlen;
320 }
321 }
322 }
323 /* Advance past this name and its NUL terminator. */
324 lp += full_nlen + 1;
325 }
326 BRIX_OP_OK(ctx, BRIX_OP_FATTR);
327 size_t resp_len = (size_t)(wp - resp);
328 if (resp_len == 0) {
329 return brix_send_ok(ctx, c, NULL, 0);
330 }
331 return brix_send_ok(ctx, c, resp, (uint32_t) resp_len);
332}
