32static void
33tpc_marker_thread_func(void *data, ngx_log_t *log)
34{
35 tpc_marker_thread_ctx_t *tt = data;
36 tpc_ms_progress_t *progress = tt->marker_ctx->progress;
37 ngx_int_t rc;
38
39 (void) log; /* use tt->log for request-level context */
40
41 /* SSRF preflight: resolve the remote URL and reject prohibited ranges. */
42 {
43 brix_net_target_t net_target;
44 brix_net_target_policy_t net_policy;
45 ngx_str_t url_str;
46 char ssrf_err[256];
47
48 url_str.data = (u_char *) tt->url;
49 url_str.len = ngx_strlen(tt->url);
50
51 ngx_memzero(&net_policy, sizeof(net_policy));
52 net_policy.require_https = 1;
53 net_policy.allow_root_scheme = 0;
54 net_policy.allow_local = tt->conf->tpc_allow_local;
55 net_policy.allow_private = tt->conf->tpc_allow_private;
56 net_policy.default_https_port = 443;
57
58 if (brix_net_target_parse(NULL, &url_str, &net_target,
59 ssrf_err, sizeof(ssrf_err)) != NGX_OK
60 || brix_net_target_check_dns(&net_target, &net_policy,
61 ssrf_err, sizeof(ssrf_err)) != NGX_OK)
62 {
63 ngx_log_error(NGX_LOG_WARN, tt->log, 0,
64 "brix_webdav: HTTP-TPC SSRF blocked: %s", ssrf_err);
65 progress->result = NGX_HTTP_FORBIDDEN;
66 ngx_memory_barrier();
67 progress->completed = 1;
68 return;
69 }
70 }
71
72 if (tt->is_push) {
73 rc = webdav_tpc_run_curl_push(tt->log, tt->conf, tt->url,
74 tt->local_path, tt->transfer_headers, 0);
75 } else if (progress->n_streams > 1) {
76 rc = webdav_tpc_run_curl_pull_multi(tt->log, tt->conf, tt->url,
77 tt->local_path, tt->transfer_headers,
78 progress->n_streams, 0, progress,
79 tt->user_cert, tt->user_key);
80 } else {
81 rc = webdav_tpc_run_curl_pull(tt->log, tt->conf, tt->url,
82 tt->local_path, tt->transfer_headers, 0,
83 tt->user_cert, tt->user_key);
84 }
85
86 progress->result = rc;
87 ngx_memory_barrier();
88 progress->completed = 1;
89}

92/*
93 * tpc_marker_start_args_t — WHAT: the immutable inputs to a 202-streaming TPC
94 * start, bundled so the setup helpers take one struct instead of the 11-parameter
95 * public argument list. WHY: the public webdav_tpc_marker_start signature is a
96 * frozen extern (declared in webdav_tpc.h, called from tpc.c) and cannot shed
97 * params, but the file-local build steps become single-argument and testable when
98 * the request-scoped inputs travel as one value. HOW: filled once from the entry
99 * point's parameters, then read-only for the remainder of setup.
100 */
101typedef struct {
102 ngx_http_request_t *r;
103 ngx_http_brix_webdav_loc_conf_t *conf;
104 ngx_uint_t n_streams;
105 const char *url;
106 const char *tmp_path;
107 const char *final_path;
108 ngx_flag_t is_pull;
109 ngx_flag_t overwrite;
110 ngx_flag_t existed;
111 ngx_array_t *transfer_headers;
112 const char *user_cert; /* per-user pull-leg cert (or NULL) */
113 const char *user_key; /* per-user pull-leg key (or NULL) */
114 uint64_t transfer_id;
115} tpc_marker_start_args_t;

117/*
118 * tpc_marker_build_ctx — WHAT: allocate and populate the event-loop marker ctx
119 * (with its shared progress counters). WHY: the ctx is the single state object
120 * the poll timer, thread-done handler and cleanup all key off; building it is one
121 * responsibility. HOW: pcalloc the progress + ctx from r->pool, copy the paths
122 * with bounded ngx_cpystrn, and for a push record the source file size for the
123 * final marker. Returns the ctx, or NULL on allocation failure.
124 */
125static tpc_marker_ctx_t *
126tpc_marker_build_ctx(const tpc_marker_start_args_t *a)
127{
128 ngx_http_request_t *r = a->r;
129 tpc_ms_progress_t *progress;
130 tpc_marker_ctx_t *ctx;
131
132 progress = ngx_pcalloc(r->pool, sizeof(*progress));
133 if (progress == NULL) {
134 return NULL;
135 }
136 progress->n_streams = a->n_streams;
137 progress->total_size = -1;
138
139 ctx = ngx_pcalloc(r->pool, sizeof(*ctx));
140 if (ctx == NULL) {
141 return NULL;
142 }
143 ctx->r = r;
144 ctx->progress = progress;
145 ctx->is_pull = a->is_pull;
146 ctx->overwrite = a->overwrite;
147 ctx->existed = a->existed;
148 ctx->root_canon = a->conf->common.root_canon;
149 ctx->transfer_id = a->transfer_id;
150 ctx->marker_interval_secs = a->conf->tpc_marker_interval;
151 ctx->last_marker_time = 0;
152
153 ngx_cpystrn((u_char *) ctx->tmp_path,
154 (u_char *) (a->tmp_path ? a->tmp_path : ""),
155 sizeof(ctx->tmp_path));
156 ngx_cpystrn((u_char *) ctx->final_path,
157 (u_char *) (a->final_path ? a->final_path : ""),
158 sizeof(ctx->final_path));
159
160 if (!a->is_pull && a->tmp_path != NULL) {
161 struct stat sb;
162 if (stat(a->tmp_path, &sb) == 0) { /* vfs-seam-allow: TPC push source temp size */
163 ctx->push_file_size = sb.st_size;
164 }
165 }
166
167 return ctx;
168}

170/*
171 * tpc_marker_build_task — WHAT: allocate and populate the thread task that runs
172 * the curl transfer. WHY: the thread reads a private copy of the request inputs;
173 * assembling it is a distinct step from the event-loop ctx. HOW: alloc the task,
174 * fill its ctx (marker back-pointer, log, conf, direction, headers, bounded url +
175 * local path copies) and bind the thread + done handlers. Returns the task, or
176 * NULL on allocation failure.
177 */
178static ngx_thread_task_t *
179tpc_marker_build_task(const tpc_marker_start_args_t *a, tpc_marker_ctx_t *ctx)
180{
181 ngx_http_request_t *r = a->r;
182 ngx_thread_task_t *task;
183 tpc_marker_thread_ctx_t *tt;
184
185 task = ngx_thread_task_alloc(r->pool, sizeof(tpc_marker_thread_ctx_t));
186 if (task == NULL) {
187 return NULL;
188 }
189 tt = task->ctx;
190 tt->marker_ctx = ctx;
191 tt->log = r->connection->log;
192 tt->conf = a->conf;
193 tt->is_push = !a->is_pull;
194 tt->transfer_headers = a->transfer_headers;
195 tt->user_cert = a->user_cert;
196 tt->user_key = a->user_key;
197 ngx_cpystrn((u_char *) tt->url,
198 (u_char *) (a->url ? a->url : ""), sizeof(tt->url));
199 ngx_cpystrn((u_char *) tt->local_path,
200 (u_char *) (a->tmp_path ? a->tmp_path : ""),
201 sizeof(tt->local_path));
202
203 brix_task_bind(task, tpc_marker_thread_func, tpc_marker_thread_done);
204 task->event.log = r->connection->log;
205
206 return task;
207}

209/*
210 * tpc_marker_arm_lifecycle — WHAT: register the request-destruction cleanup and
211 * initialise the poll timer for a marker ctx. WHY: the cleanup aborts an
212 * in-flight transfer if the client disconnects, and the timer drives marker
213 * emission; both must be wired before the 202 is sent. HOW: add a pool cleanup
214 * bound to ctx, zero and configure the timer's handler/data/log. Returns NGX_OK,
215 * or NGX_HTTP_INTERNAL_SERVER_ERROR if the cleanup slot cannot be allocated.
216 */
217static ngx_int_t
218tpc_marker_arm_lifecycle(ngx_http_request_t *r, tpc_marker_ctx_t *ctx)
219{
220 ngx_pool_cleanup_t *cln;
221
222 cln = ngx_pool_cleanup_add(r->pool, 0);
223 if (cln == NULL) {
224 return NGX_HTTP_INTERNAL_SERVER_ERROR;
225 }
226 cln->handler = tpc_marker_cleanup;
227 cln->data = ctx;
228
229 ngx_memzero(&ctx->timer, sizeof(ctx->timer));
230 ctx->timer.handler = tpc_marker_poll;
231 ctx->timer.data = ctx;
232 ctx->timer.log = r->connection->log;
233
234 return NGX_OK;
235}

237/*
238 * tpc_marker_send_accepted — WHAT: emit the 202 Accepted response head with the
239 * X-Number-Of-Streams header and a chunked text/plain body. WHY: FTS/RUCIO
240 * expect the stream count and an open chunked body before the markers stream in;
241 * this is the frozen wire shape. HOW: push the header, format n_streams into it,
242 * set the 202 status/content-type, and send the header. Returns NGX_OK, or an
243 * HTTP error code on allocation or header-send failure.
244 */
245static ngx_int_t
246tpc_marker_send_accepted(ngx_http_request_t *r, ngx_uint_t n_streams)
247{
248 ngx_table_elt_t *h;
249 ngx_int_t rc;
250
251 h = ngx_list_push(&r->headers_out.headers);
252 if (h == NULL) {
253 return NGX_HTTP_INTERNAL_SERVER_ERROR;
254 }
255 h->hash = 1;
256 ngx_str_set(&h->key, "X-Number-Of-Streams");
257 h->value.data = ngx_pnalloc(r->pool, NGX_INT_T_LEN + 1);
258 if (h->value.data == NULL) {
259 return NGX_HTTP_INTERNAL_SERVER_ERROR;
260 }
261 h->value.len = (size_t) (ngx_sprintf(h->value.data, "%ui", n_streams)
262 - h->value.data);
263
264 r->headers_out.status = 202;
265 r->headers_out.content_length_n = -1;
266 r->headers_out.content_type.data = (u_char *) "text/plain";
267 r->headers_out.content_type.len = sizeof("text/plain") - 1;
268 r->headers_out.content_type_len = sizeof("text/plain") - 1;
269
270 rc = ngx_http_send_header(r);
271 if (rc == NGX_ERROR || rc > NGX_OK) {
272 return rc != NGX_ERROR ? rc : NGX_HTTP_INTERNAL_SERVER_ERROR;
273 }
274 return NGX_OK;
275}

277/*
278 * webdav_tpc_marker_start — start a 202-streaming TPC transfer.
279 *
280 * Sends 202 Accepted immediately, posts the curl work to the thread pool, and
281 * arms a poll timer that emits Performance-Marker blocks while the thread runs.
282 * Returns NGX_DONE (202 sent) on success.
283 * Returns NGX_DECLINED when no thread pool is configured (caller uses 201 path).
284 * Returns an HTTP error code on allocation or header-send failure.
285 *
286 * For pull transfers: tmp_path is the staging file; final_path is the commit
287 * target. Both must fit within WEBDAV_MAX_PATH.
288 * For push transfers: tmp_path is the local source file; final_path is unused.
289 *
290 * Orchestrator only: builds the ctx + task, arms the lifecycle, sends the 202,
291 * posts the thread, emits the first marker and starts the poll timer — each a
292 * named helper. (11 params: frozen public extern; the inputs travel internally
293 * as a tpc_marker_start_args_t.)
294 */
295ngx_int_t
296webdav_tpc_marker_start(ngx_http_request_t *r,
297 ngx_http_brix_webdav_loc_conf_t *conf, ngx_uint_t n_streams,
298 const char *url, const char *tmp_path, const char *final_path,
299 ngx_flag_t is_pull, ngx_flag_t overwrite, ngx_flag_t existed,
300 ngx_array_t *transfer_headers, uint64_t transfer_id,
301 const char *user_cert, const char *user_key)
302{
303 tpc_marker_start_args_t args;
304 tpc_marker_ctx_t *ctx;
305 ngx_thread_task_t *task;
306 ngx_int_t rc;
307
308 if (conf->common.thread_pool == NULL) {
309 return NGX_DECLINED;
310 }
311
312 if (n_streams == 0) n_streams = 1;
313 if (n_streams > BRIX_TPC_MAX_STREAMS) n_streams = BRIX_TPC_MAX_STREAMS;
314
315 ngx_memzero(&args, sizeof(args));
316 args.r = r;
317 args.conf = conf;
318 args.n_streams = n_streams;
319 args.url = url;
320 args.tmp_path = tmp_path;
321 args.final_path = final_path;
322 args.is_pull = is_pull;
323 args.overwrite = overwrite;
324 args.existed = existed;
325 args.transfer_headers = transfer_headers;
326 args.user_cert = user_cert;
327 args.user_key = user_key;
328 args.transfer_id = transfer_id;
329
330 ctx = tpc_marker_build_ctx(&args);
331 if (ctx == NULL) {
332 return NGX_HTTP_INTERNAL_SERVER_ERROR;
333 }
334
335 task = tpc_marker_build_task(&args, ctx);
336 if (task == NULL) {
337 return NGX_HTTP_INTERNAL_SERVER_ERROR;
338 }
339
340 rc = tpc_marker_arm_lifecycle(r, ctx);
341 if (rc != NGX_OK) {
342 return rc;
343 }
344
345 rc = tpc_marker_send_accepted(r, n_streams);
346 if (rc != NGX_OK) {
347 return rc;
348 }
349
350 /* Post the thread task. */
351 if (ngx_thread_task_post(conf->common.thread_pool, task) != NGX_OK) {
352 return NGX_HTTP_INTERNAL_SERVER_ERROR;
353 }
354
355 /* Prevent request destruction until we call ngx_http_finalize_request. */
356 r->main->count++;
357
358 /* Emit the first Performance-Marker immediately. */
359 ctx->last_marker_time = time(NULL);
360 tpc_marker_send_all(r, ctx->last_marker_time, ctx);
361
362 /* Start the poll timer. */
363 ngx_add_timer(&ctx->timer, TPC_MARKER_POLL_MSEC);
364
365 return NGX_DONE;
366}
