14/* Append one byte to a bounded output cursor (NUL-termination done by caller). */
15static void
16jm_put(char *out, size_t outsz, size_t *pos, char c)
17{
18 if (*pos + 1 < outsz) { /* keep one byte for the trailing NUL */
19 out[*pos] = c;
20 }
21 (*pos)++;
22}

24/* Encode a Unicode code point as UTF-8 into the bounded output. */
25static void
26jm_put_utf8(char *out, size_t outsz, size_t *pos, uint32_t cp)
27{
28 if (cp < 0x80) {
29 jm_put(out, outsz, pos, (char) cp);
30 } else if (cp < 0x800) {
31 jm_put(out, outsz, pos, (char) (0xC0 | (cp >> 6)));
32 jm_put(out, outsz, pos, (char) (0x80 | (cp & 0x3F)));
33 } else if (cp < 0x10000) {
34 jm_put(out, outsz, pos, (char) (0xE0 | (cp >> 12)));
35 jm_put(out, outsz, pos, (char) (0x80 | ((cp >> 6) & 0x3F)));
36 jm_put(out, outsz, pos, (char) (0x80 | (cp & 0x3F)));
37 } else {
38 jm_put(out, outsz, pos, (char) (0xF0 | (cp >> 18)));
39 jm_put(out, outsz, pos, (char) (0x80 | ((cp >> 12) & 0x3F)));
40 jm_put(out, outsz, pos, (char) (0x80 | ((cp >> 6) & 0x3F)));
41 jm_put(out, outsz, pos, (char) (0x80 | (cp & 0x3F)));
42 }
43}

45/* Parse one hex nibble; -1 if not a hex digit. */
46static int
47jm_hex(char c)
48{
49 if (c >= '0' && c <= '9') { return c - '0'; }
50 if (c >= 'a' && c <= 'f') { return c - 'a' + 10; }
51 if (c >= 'A' && c <= 'F') { return c - 'A' + 10; }
52 return -1;
53}

55/* Read 4 hex digits at p (p<end-3 guaranteed by caller); -1 on a bad digit. */
56static int
57jm_u16(const char *p)
58{
59 int a = jm_hex(p[0]), b = jm_hex(p[1]), c = jm_hex(p[2]), d = jm_hex(p[3]);
60 if (a < 0 || b < 0 || c < 0 || d < 0) {
61 return -1;
62 }
63 return (a << 12) | (b << 8) | (c << 4) | d;
64}

68{

75/*
76 * Parse a JSON string starting at the opening quote `p` (*p == '"'). If `out`
77 * is non-NULL, the decoded (unescaped) content is written to out[outsz]; if NULL,
78 * the string is only skipped. Returns the position just past the closing quote,
79 * or NULL on a malformed/unterminated string.
80 */
81static const char *
82jm_string(const char *p, const char *end, char *out, size_t outsz, size_t *outlen)
83{
84 size_t pos = 0;
85
86 p++; /* skip opening quote */
87 while (p < end) {
88 char c = *p;
89
90 if (c == '"') {
91 if (out != NULL && outsz > 0) {
92 out[pos < outsz ? pos : outsz - 1] = '\0';
93 }
94 if (outlen != NULL) {
95 *outlen = pos;
96 }
97 return p + 1;
98 }
99 if (c == '\\') {
100 p++;
101 if (p >= end) {
102 return NULL;
103 }
104 switch (*p) {
105 case '"': if (out) { jm_put(out, outsz, &pos, '"'); } break;
106 case '\\': if (out) { jm_put(out, outsz, &pos, '\\'); } break;
107 case '/': if (out) { jm_put(out, outsz, &pos, '/'); } break;
108 case 'b': if (out) { jm_put(out, outsz, &pos, '\b'); } break;
109 case 'f': if (out) { jm_put(out, outsz, &pos, '\f'); } break;
110 case 'n': if (out) { jm_put(out, outsz, &pos, '\n'); } break;
111 case 'r': if (out) { jm_put(out, outsz, &pos, '\r'); } break;
112 case 't': if (out) { jm_put(out, outsz, &pos, '\t'); } break;
113 case 'u': {
114 int hi;
115 if (p + 4 >= end || (hi = jm_u16(p + 1)) < 0) {
116 return NULL;
117 }
118 p += 4;
119 if (hi >= 0xD800 && hi <= 0xDBFF) {
120 /* high surrogate — must be followed by \uDC00..\uDFFF */
121 int lo;
122 if (p + 6 < end && p[1] == '\\' && p[2] == 'u'
123 && (lo = jm_u16(p + 3)) >= 0xDC00 && lo <= 0xDFFF)
124 {
125 uint32_t cp = 0x10000u
126 + (((uint32_t) hi - 0xD800u) << 10)
127 + ((uint32_t) lo - 0xDC00u);
128 if (out) { jm_put_utf8(out, outsz, &pos, cp); }
129 p += 6;
130 } else if (out) {
131 jm_put_utf8(out, outsz, &pos, 0xFFFD); /* lone surrogate */
132 }
133 } else if (out) {
134 jm_put_utf8(out, outsz, &pos, (uint32_t) hi);
135 }
136 break;
137 }
138 default:
139 return NULL; /* invalid escape */
140 }
141 p++;
142 } else {
143 if (out) { jm_put(out, outsz, &pos, c); }
144 p++;
145 }
146 }
147 return NULL; /* unterminated */
148}

150/*
151 * Skip any JSON value at p (already ws-skipped), returning the position just past
152 * it, or NULL on malformed input. Objects/arrays are skipped with a string-aware
153 * depth counter so braces/brackets inside strings don't confuse nesting.
154 */
155static const char *
156jm_skip_value(const char *p, const char *end)
157{
158 if (p >= end) {
159 return NULL;
160 }
161 if (*p == '"') {
162 return jm_string(p, end, NULL, 0, NULL);
163 }
164 if (*p == '{' || *p == '[') {
165 int depth = 0;
166 while (p < end) {
167 char c = *p;
168 if (c == '"') {
169 p = jm_string(p, end, NULL, 0, NULL);
170 if (p == NULL) {
171 return NULL;
172 }
173 continue;
174 }
175 if (c == '{' || c == '[') {
176 depth++;
177 } else if (c == '}' || c == ']') {
178 depth--;
179 if (depth == 0) {
180 return p + 1;
181 }
182 }
183 p++;
184 }
185 return NULL; /* unbalanced */
186 }
187 /* scalar: number / true / false / null — to the next delimiter */
188 while (p < end && *p != ',' && *p != '}' && *p != ']'
189 && *p != ' ' && *p != '\t' && *p != '\n' && *p != '\r')
190 {
191 p++;
192 }
193 return p;
194}

196int
197brix_json_get_str(const char *json, size_t len, const char *key, char *out,
198 size_t outsz)
199{
200 const char *p = json;
201 const char *end = json + len;
202 size_t keylen;
203
204 if (json == NULL || key == NULL || out == NULL || outsz == 0) {
205 return 0;
206 }
207 keylen = strlen(key);
208
209 p = jm_skip_ws(p, end);
210 if (p >= end || *p != '{') {
211 return 0;
212 }
213 p++; /* enter the top-level object */
214
215 for (;;) {
216 const char *kstart;
217 char kbuf[128];
218 size_t klen = 0;
219
220 p = jm_skip_ws(p, end);
221 if (p >= end || *p == '}') {
222 return 0; /* end of object, key not found */
223 }
224 if (*p != '"') {
225 return 0; /* not a member key — malformed */
226 }
227 kstart = p;
228 /* Read the member key into kbuf (decoded) for an exact compare. */
229 p = jm_string(kstart, end, kbuf, sizeof(kbuf), &klen);
230 if (p == NULL) {
231 return 0;
232 }
233 p = jm_skip_ws(p, end);
234 if (p >= end || *p != ':') {
235 return 0;
236 }
237 p++; /* past ':' */
238 p = jm_skip_ws(p, end);
239 if (p >= end) {
240 return 0;
241 }
242
243 if (klen == keylen && memcmp(kbuf, key, keylen) == 0) {
244 /* Found it. String → decode; scalar → raw token; object/array → 0. */
245 if (*p == '"') {
246 return jm_string(p, end, out, outsz, NULL) != NULL ? 1 : 0;
247 }
248 if (*p == '{' || *p == '[') {
249 return 0; /* no flat-string form */
250 }
251 {
252 const char *vend = jm_skip_value(p, end);
253 size_t n;
254 if (vend == NULL) {
255 return 0;
256 }
257 n = (size_t) (vend - p);
258 if (n >= outsz) {
259 n = outsz - 1;
260 }
261 memcpy(out, p, n);
262 out[n] = '\0';
263 return 1;
264 }
265 }
266
267 /* Not our key — skip the value and continue to the next member. */
268 p = jm_skip_value(p, end);
269 if (p == NULL) {
270 return 0;
271 }
272 p = jm_skip_ws(p, end);
273 if (p < end && *p == ',') {
274 p++;
275 }
276 }
277}
