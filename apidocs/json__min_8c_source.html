14/* Append one byte to a bounded output cursor (NUL-termination done by caller). */
15static void
16jm_put(char *out, size_t outsz, size_t *pos, char c)
17{
18 if (*pos + 1 < outsz) { /* keep one byte for the trailing NUL */
19 out[*pos] = c;
20 }
21 (*pos)++;
22}

24/* Encode a Unicode code point as UTF-8 into the bounded output. */
25static void
26jm_put_utf8(char *out, size_t outsz, size_t *pos, uint32_t cp)
27{
28 if (cp < 0x80) {
29 jm_put(out, outsz, pos, (char) cp);
30 } else if (cp < 0x800) {
31 jm_put(out, outsz, pos, (char) (0xC0 | (cp >> 6)));
32 jm_put(out, outsz, pos, (char) (0x80 | (cp & 0x3F)));
33 } else if (cp < 0x10000) {
34 jm_put(out, outsz, pos, (char) (0xE0 | (cp >> 12)));
35 jm_put(out, outsz, pos, (char) (0x80 | ((cp >> 6) & 0x3F)));
36 jm_put(out, outsz, pos, (char) (0x80 | (cp & 0x3F)));
37 } else {
38 jm_put(out, outsz, pos, (char) (0xF0 | (cp >> 18)));
39 jm_put(out, outsz, pos, (char) (0x80 | ((cp >> 12) & 0x3F)));
40 jm_put(out, outsz, pos, (char) (0x80 | ((cp >> 6) & 0x3F)));
41 jm_put(out, outsz, pos, (char) (0x80 | (cp & 0x3F)));
42 }
43}

45/* Parse one hex nibble; -1 if not a hex digit. */
46static int
47jm_hex(char c)
48{
49 if (c >= '0' && c <= '9') { return c - '0'; }
50 if (c >= 'a' && c <= 'f') { return c - 'a' + 10; }
51 if (c >= 'A' && c <= 'F') { return c - 'A' + 10; }
52 return -1;
53}

55/* Read 4 hex digits at p (p<end-3 guaranteed by caller); -1 on a bad digit. */
56static int
57jm_u16(const char *p)
58{
59 int a = jm_hex(p[0]), b = jm_hex(p[1]), c = jm_hex(p[2]), d = jm_hex(p[3]);
60 if (a < 0 || b < 0 || c < 0 || d < 0) {
61 return -1;
62 }
63 return (a << 12) | (b << 8) | (c << 4) | d;
64}

75/* ---- Combine a UTF-16 surrogate pair into a Unicode code point ----
76 *
77 * WHAT: Returns the code point encoded by high surrogate `hi` (0xD800..0xDBFF)
78 * and low surrogate `lo` (0xDC00..0xDFFF). Pure arithmetic; no validation.
79 *
80 * WHY: Keeps the RFC 8259 surrogate-pair math in one named place instead of
81 * inlined in the escape decoder.
82 *
83 * HOW: 1. Shift the high surrogate's 10 payload bits up.
84 * 2. Add the low surrogate's 10 payload bits and the 0x10000 plane base.
85 */
86static uint32_t
87jm_utf16_pair(int hi, int lo)
88{
89 return 0x10000u + (((uint32_t) hi - 0xD800u) << 10)
90 + ((uint32_t) lo - 0xDC00u);
91}

93/* ---- Decode a \uXXXX escape (with surrogate-pair handling) ----
94 *
95 * WHAT: Decodes the \uXXXX escape whose 'u' is at **pp, emitting UTF-8 into
96 * the bounded output (if `out` is non-NULL). Advances *pp to the LAST hex
97 * digit consumed (the caller steps past it). Returns 0 on success, -1 on a
98 * truncated/non-hex escape.
99 *
100 * WHY: Splits the surrogate-pair state machine out of jm_string so the scan
101 * loop stays a flat character dispatch.
102 *
103 * HOW: 1. Require 4 hex digits after the 'u'; parse them (else -1).
104 * 2. High surrogate followed by a valid \uDC00..\uDFFF low surrogate →
105 * emit the combined code point and consume the second escape too.
106 * 3. Lone high surrogate → emit U+FFFD without consuming further input.
107 * 4. Anything else → emit the code unit as-is.
108 */
109static int
110jm_unescape_u(const char **pp, const char *end, char *out, size_t outsz,
111 size_t *pos)
112{
113 const char *p = *pp;
114 int hi;
115
116 if (p + 4 >= end || (hi = jm_u16(p + 1)) < 0) {
117 return -1;
118 }
119 p += 4;
120 if (hi >= 0xD800 && hi <= 0xDBFF) {
121 /* high surrogate — must be followed by \uDC00..\uDFFF */
122 int lo;
123 if (p + 6 < end && p[1] == '\\' && p[2] == 'u'
124 && (lo = jm_u16(p + 3)) >= 0xDC00 && lo <= 0xDFFF)
125 {
126 if (out) { jm_put_utf8(out, outsz, pos, jm_utf16_pair(hi, lo)); }
127 p += 6;
128 } else if (out) {
129 jm_put_utf8(out, outsz, pos, 0xFFFD); /* lone surrogate */
130 }
131 } else if (out) {
132 jm_put_utf8(out, outsz, pos, (uint32_t) hi);
133 }
134 *pp = p;
135 return 0;
136}

138/* ---- Decode one escape sequence after a backslash ----
139 *
140 * WHAT: Decodes the escape whose introducer character is at **pp (the byte
141 * after the backslash), emitting into the bounded output when `out` is
142 * non-NULL. Advances *pp to the LAST character consumed. Returns 0 on
143 * success, -1 on a dangling backslash or invalid/truncated escape.
144 *
145 * WHY: Pulls the per-escape dispatch out of jm_string so the string scanner
146 * only walks bytes and delegates decoding.
147 *
148 * HOW: 1. Fail if the backslash was the last input byte.
149 * 2. Map the two-character escapes to their single-byte expansions.
150 * 3. Delegate \uXXXX to jm_unescape_u.
151 * 4. Reject any other escape character.
152 */
153static int
154jm_unescape_char(const char **pp, const char *end, char *out, size_t outsz,
155 size_t *pos)
156{
157 const char *p = *pp;
158 char d;
159
160 if (p >= end) {
161 return -1;
162 }
163 switch (*p) {
164 case '"': d = '"'; break;
165 case '\\': d = '\\'; break;
166 case '/': d = '/'; break;
167 case 'b': d = '\b'; break;
168 case 'f': d = '\f'; break;
169 case 'n': d = '\n'; break;
170 case 'r': d = '\r'; break;
171 case 't': d = '\t'; break;
172 case 'u':
173 if (jm_unescape_u(&p, end, out, outsz, pos) != 0) {
174 return -1;
175 }
176 *pp = p;
177 return 0;
178 default:
179 return -1; /* invalid escape */
180 }
181 if (out) { jm_put(out, outsz, pos, d); }
182 *pp = p;
183 return 0;
184}

186/*
187 * Parse a JSON string starting at the opening quote `p` (*p == '"'). If `out`
188 * is non-NULL, the decoded (unescaped) content is written to out[outsz]; if NULL,
189 * the string is only skipped. Returns the position just past the closing quote,
190 * or NULL on a malformed/unterminated string.
191 */
192static const char *
193jm_string(const char *p, const char *end, char *out, size_t outsz, size_t *outlen)
194{
195 size_t pos = 0;
196
197 p++; /* skip opening quote */
198 while (p < end) {
199 char c = *p;
200
201 if (c == '"') {
202 if (out != NULL && outsz > 0) {
203 out[pos < outsz ? pos : outsz - 1] = '\0';
204 }
205 if (outlen != NULL) {
206 *outlen = pos;
207 }
208 return p + 1;
209 }
210 if (c == '\\') {
211 p++;
212 if (jm_unescape_char(&p, end, out, outsz, &pos) != 0) {
213 return NULL;
214 }
215 p++;
216 } else {
217 if (out) { jm_put(out, outsz, &pos, c); }
218 p++;
219 }
220 }
221 return NULL; /* unterminated */
222}

224/*
225 * Skip any JSON value at p (already ws-skipped), returning the position just past
226 * it, or NULL on malformed input. Objects/arrays are skipped with a string-aware
227 * depth counter so braces/brackets inside strings don't confuse nesting.
228 */
229/* ---- Skip an object/array with a string-aware depth counter ----
230 *
231 * WHAT: Skips the container starting at `p` (*p == '{' or '['), returning the
232 * position just past its closing brace/bracket, or NULL on an unbalanced or
233 * malformed (bad embedded string) container.
234 *
235 * WHY: Keeps the non-recursive depth-counting skip (the file's "never recurses
236 * unboundedly" guarantee) in one place, out of the value dispatcher.
237 *
238 * HOW: 1. Walk bytes, letting jm_string consume whole strings so braces and
239 * brackets inside strings never touch the depth counter.
240 * 2. Increment depth on '{'/'[', decrement on '}'/']'.
241 * 3. Depth back to zero → done; input exhausted first → NULL.
242 */
243static const char *
244jm_skip_container(const char *p, const char *end)
245{
246 int depth = 0;
247
248 while (p < end) {
249 char c = *p;
250 if (c == '"') {
251 p = jm_string(p, end, NULL, 0, NULL);
252 if (p == NULL) {
253 return NULL;
254 }
255 continue;
256 }
257 if (c == '{' || c == '[') {
258 depth++;
259 } else if (c == '}' || c == ']') {
260 depth--;
261 if (depth == 0) {
262 return p + 1;
263 }
264 }
265 p++;
266 }
267 return NULL; /* unbalanced */
268}

270/* ---- Skip a scalar token to the next delimiter ----
271 *
272 * WHAT: Advances past a number/true/false/null token starting at `p` and
273 * returns the position of the first delimiter (',', '}', ']', whitespace) or
274 * `end`. Never fails.
275 *
276 * WHY: Separates the trivial scalar scan from the container skip so
277 * jm_skip_value is a plain three-way dispatch.
278 *
279 * HOW: 1. Advance while the byte is not a value delimiter and input remains.
280 */
281static const char *
282jm_skip_scalar(const char *p, const char *end)
283{
284 while (p < end && *p != ',' && *p != '}' && *p != ']'
285 && *p != ' ' && *p != '\t' && *p != '\n' && *p != '\r')
286 {
287 p++;
288 }
289 return p;
290}

292static const char *
293jm_skip_value(const char *p, const char *end)
294{
295 if (p >= end) {
296 return NULL;
297 }
298 if (*p == '"') {
299 return jm_string(p, end, NULL, 0, NULL);
300 }
301 if (*p == '{' || *p == '[') {
302 return jm_skip_container(p, end);
303 }
304 /* scalar: number / true / false / null — to the next delimiter */
305 return jm_skip_scalar(p, end);

308/* ---- Parse one object member key and position on its value ----
309 *
310 * WHAT: At the start of an object member, decodes the member key into
311 * kbuf[kbufsz] (length in *klen), consumes the ':' separator, and leaves *pp
312 * at the first byte of the value (ws-skipped). Returns 0 on success, -1 on
313 * end-of-object, missing/unterminated key, missing ':', or truncated input.
314 *
315 * WHY: Isolates the member-header grammar (ws key ws ':' ws) so the lookup
316 * loop reads as key-compare + value-handling only.
317 *
318 * HOW: 1. Skip whitespace; '}' or end of input → no more members (-1).
319 * 2. Require and decode a quoted key via jm_string.
320 * 3. Skip whitespace; require ':'; skip whitespace.
321 * 4. Fail if the value itself is missing (input exhausted).
322 */
323static int
324jm_member_key(const char **pp, const char *end, char *kbuf, size_t kbufsz,
325 size_t *klen)
326{
327 const char *p = jm_skip_ws(*pp, end);
328
329 if (p >= end || *p == '}') {
330 return -1; /* end of object, key not found */
331 }
332 if (*p != '"') {
333 return -1; /* not a member key — malformed */
334 }
335 /* Read the member key into kbuf (decoded) for an exact compare. */
336 p = jm_string(p, end, kbuf, kbufsz, klen);
337 if (p == NULL) {
338 return -1;
339 }
340 p = jm_skip_ws(p, end);
341 if (p >= end || *p != ':') {
342 return -1;
343 }
344 p++; /* past ':' */
345 p = jm_skip_ws(p, end);
346 if (p >= end) {
347 return -1;
348 }
349 *pp = p;
350 return 0;
351}

353/* ---- Copy the matched member's value into the caller's buffer ----
354 *
355 * WHAT: Emits the value at `p` as a flat NUL-terminated string in out[outsz]:
356 * strings are unescaped, scalars copied raw (truncated to fit), objects and
357 * arrays have no flat form. Returns 1 on success, 0 on container values or
358 * malformed input.
359 *
360 * WHY: Separates the found-key output formatting from the member-scan loop in
361 * brix_json_get_str.
362 *
363 * HOW: 1. String value → decode through jm_string.
364 * 2. Object/array → 0 (no flat-string form).
365 * 3. Scalar → bound the raw token with jm_skip_value and copy it,
366 * clamped to outsz - 1, then NUL-terminate.
367 */
368static int
369jm_emit_value(const char *p, const char *end, char *out, size_t outsz)
370{
371 const char *vend;
372 size_t n;
373
374 /* Found it. String → decode; scalar → raw token; object/array → 0. */
375 if (*p == '"') {
376 return jm_string(p, end, out, outsz, NULL) != NULL ? 1 : 0;
377 }
378 if (*p == '{' || *p == '[') {
379 return 0; /* no flat-string form */
380 }
381 vend = jm_skip_value(p, end);
382 if (vend == NULL) {
383 return 0;
384 }
385 n = (size_t) (vend - p);
386 if (n >= outsz) {
387 n = outsz - 1;
388 }
389 memcpy(out, p, n);
390 out[n] = '\0';
391 return 1;
392}

395brix_json_get_str(const char *json, size_t len, const char *key, char *out,
396 size_t outsz)
397{
398 const char *p = json;
399 const char *end = json + len;
400 size_t keylen;
401
402 if (json == NULL || key == NULL || out == NULL || outsz == 0) {
403 return 0;
404 }
405 keylen = strlen(key);
406
407 p = jm_skip_ws(p, end);
408 if (p >= end || *p != '{') {
409 return 0;
410 }
411 p++; /* enter the top-level object */
412
413 for (;;) {
414 char kbuf[128];
415 size_t klen = 0;
416
417 if (jm_member_key(&p, end, kbuf, sizeof(kbuf), &klen) != 0) {
418 return 0;
419 }
420
421 if (klen == keylen && memcmp(kbuf, key, keylen) == 0) {
422 return jm_emit_value(p, end, out, outsz);
423 }
424
425 /* Not our key — skip the value and continue to the next member. */
426 p = jm_skip_value(p, end);
427 if (p == NULL) {
428 return 0;
429 }
430 p = jm_skip_ws(p, end);
431 if (p < end && *p == ',') {
432 p++;
433 }
434 }
