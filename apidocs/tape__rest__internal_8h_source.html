4/*
5 * tape_rest_internal.h - declarations shared between the two halves of the WLCG
6 * HTTP Tape REST API after the phase-79 file-size split.
7 *
8 * WHAT: Cross-declares the shared request-scoped constants, the small marshalling
9 * and authorisation helpers that live in tape_rest.c, and the per-endpoint
10 * handlers that live in tape_rest_ops.c - i.e. exactly the symbols that cross the
11 * tape_rest.c / tape_rest_ops.c boundary.
12 * WHY: tape_rest.c (the router core: response marshalling, the resolve+authz +
13 * residency helpers, JSON id/state/locality mapping, and the GET/DELETE/POST URL
14 * routing) and tape_rest_ops.c (the endpoint bodies: stage submit/status/list/
15 * delete/cancel, release/unpin, archiveinfo/fileinfo) were one 865-line unit;
16 * splitting keeps each focused and under the 500-line cap. The router calls the
17 * endpoint handlers and the handlers call the shared helpers, so exactly those
18 * symbols become non-static and are declared here.
19 * HOW: Both translation units include this header; nothing here is exported beyond
20 * the WebDAV Tape REST surface.
21 */

36/* Externally-visible stage-request id buffer. tape_mint_id() seeds a random hex
37 * placeholder, but POST /stage OVERWRITES it with the first file's DURABLE reqid
38 * ("<seq>.<pid>@<host>") — which is the key DELETE/cancel look the record up by.
39 * It MUST therefore be sized to the on-disk reqid width (stage_request_registry
40 * SRQ_REQID_LEN = 40): sized smaller (the old 33, from the retired 16-byte-hex
41 * design) a longer-hostname reqid is truncated in the response, so the client's
42 * DELETE/cancel id no longer matches the stored key — the record is "not found",
43 * the owner check is skipped as idempotent, and a foreign principal deletes it. */
44#define TAPE_ID_LEN 40
