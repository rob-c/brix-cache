2 * jwt_sign.c — ES256 JWT minting (see jwt_sign.h for the contract).
3 *
4 * The signature path is the exact inverse of brix_token_verify_es256()
5 * (signature.c): sign with EVP_DigestSign to get a DER ECDSA signature, then
6 * convert DER → IEEE P1363 raw r||s (32+32 bytes) as JWT requires. All buffers

22brix_jwt_load_ec_key(const char *pem_path)
23{
24 BIO *bio;
25 EVP_PKEY *pkey;
26
27 bio = BIO_new_file(pem_path, "r");
28 if (bio == NULL) {
29 return NULL;
30 }
31 pkey = PEM_read_bio_PrivateKey(bio, NULL, NULL, NULL);

33
34 if (pkey == NULL) {
35 return NULL;
36 }
37 if (EVP_PKEY_base_id(pkey) != EVP_PKEY_EC) {
38 EVP_PKEY_free(pkey);
39 return NULL;
40 }
41 return pkey;
42}

45/* brix_jwt_der_to_p1363 — DER ECDSA sig → fixed 64-byte r||s * Returns 0 on success (raw[0..63] populated), -1 on failure. */
46static int
47brix_jwt_der_to_p1363(const unsigned char *der, size_t der_len,
48 unsigned char raw[64])
49{
50 const unsigned char *p = der;
51 ECDSA_SIG *sig;
52 const BIGNUM *r, *s;
53
54 sig = d2i_ECDSA_SIG(NULL, &p, (long) der_len);
55 if (sig == NULL) {
56 return -1;
57 }
58 ECDSA_SIG_get0(sig, &r, &s);
59
60 /* BN_bn2binpad left-pads each component to exactly 32 bytes (P-256). */
61 if (BN_bn2binpad(r, raw, 32) != 32 || BN_bn2binpad(s, raw + 32, 32) != 32) {
62 ECDSA_SIG_free(sig);
63 return -1;
64 }
65 ECDSA_SIG_free(sig);
66 return 0;
67}

70ngx_int_t
71brix_jwt_sign_es256(EVP_PKEY *eckey, const char *header_json,
72 const char *payload_json, char *out, size_t outsz)
73{
74 EVP_MD_CTX *mdctx;
75 unsigned char *der = NULL;
76 unsigned char raw[64];
77 char signing_input[4096];
78 char sig_b64[128];
79 size_t der_len = 0;
80 int n, hlen, plen;
81
82 /* signing input = b64url(header) "." b64url(payload) */
83 hlen = (int) ngx_strlen(header_json);
84 plen = (int) ngx_strlen(payload_json);
85
86 /* Encode header. */
87 b64url_encode(header_json, (size_t) hlen, signing_input, sizeof(signing_input));
88 n = (int) ngx_strlen(signing_input);
89 if (n <= 0 || (size_t) n + 1 >= sizeof(signing_input)) {
90 return NGX_ERROR;
91 }
92 signing_input[n++] = '.';
93
94 /* Encode payload directly after the dot. */
95 b64url_encode(payload_json, (size_t) plen, signing_input + n,
96 sizeof(signing_input) - (size_t) n);
97 if (ngx_strlen(signing_input + n) == 0) {
98 return NGX_ERROR;
99 }
100 n = (int) ngx_strlen(signing_input);
101
102 /* Sign "header.payload" → DER ECDSA signature. */
103 mdctx = EVP_MD_CTX_new();
104 if (mdctx == NULL) {
105 return NGX_ERROR;
106 }
107 if (EVP_DigestSignInit(mdctx, NULL, EVP_sha256(), NULL, eckey) != 1
108 || EVP_DigestSign(mdctx, NULL, &der_len,
109 (const unsigned char *) signing_input, (size_t) n) != 1
110 || der_len == 0)
111 {
112 EVP_MD_CTX_free(mdctx);
113 return NGX_ERROR;
114 }
115 der = OPENSSL_malloc(der_len);
116 if (der == NULL) {
117 EVP_MD_CTX_free(mdctx);
118 return NGX_ERROR;
119 }
120 if (EVP_DigestSign(mdctx, der, &der_len,
121 (const unsigned char *) signing_input, (size_t) n) != 1)
122 {
123 OPENSSL_free(der);
124 EVP_MD_CTX_free(mdctx);
125 return NGX_ERROR;
126 }
127 EVP_MD_CTX_free(mdctx);
128
129 /* DER → raw r||s → base64url. */
130 if (brix_jwt_der_to_p1363(der, der_len, raw) != 0) {
131 OPENSSL_free(der);
132 return NGX_ERROR;
133 }
134 OPENSSL_free(der);
135
136 b64url_encode((const char *) raw, sizeof(raw), sig_b64, sizeof(sig_b64));
137 if (ngx_strlen(sig_b64) == 0) {
138 return NGX_ERROR;
139 }
140
141 /* out = signing_input "." signature */
142 n = snprintf(out, outsz, "%s.%s", signing_input, sig_b64);
143 if (n < 0 || (size_t) n >= outsz) {
144 return NGX_ERROR;
145 }
146 return NGX_OK;
147}
