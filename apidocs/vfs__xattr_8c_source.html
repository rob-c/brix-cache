1/*
2 * vfs_xattr.c — VFS extended-attribute family (get / set / remove / list).
3 *
4 * WHAT: Implements brix_vfs_getxattr/setxattr/removexattr/listxattr — the
5 * protocol-agnostic surface for the `user.`-namespace xattrs that S3
6 * object tagging, WebDAV dead properties, checksum sidecars, and the
7 * WebDAV lock database all store on export objects.
8 *
9 * WHY: Before this unit those callers reached the confined xattr helpers
10 * (brix_*xattr_confined_canon) directly, so the xattr touches were
11 * confined but invisible to metrics/access-logging. Routing them here
12 * gives every xattr op one BRIX_METRIC_OP_XATTR metric + access-log line
13 * and the same guard-then-syscall-then-observe shape as the rest of the
14 * VFS, while still delegating the actual syscall (and impersonation broker
15 * routing) to the confined helpers.
16 *
17 * HOW: Each entry point re-verifies confinement (brix_vfs_require_confined),
18 * calls the matching brix_*xattr_confined_canon with ctx->root_canon and
19 * the resolved path, then observes the result as OP_XATTR. set/remove are
20 * mutations but are intentionally NOT allow_write-gated: the lock-database
21 * writes happen on otherwise read-only requests and the protocol layer has
22 * already authorized the principal — matching the prior direct-call
23 * behaviour exactly (no new EACCES surface). get/list propagate the helper
24 * byte count (or ERANGE) unchanged.
25 */
26#include "vfs_internal.h"

30/* Shared observe tail for the value-returning ops (get/list): translate a
31 * helper return (>=0 ok, -1 errno) into an OP_XATTR metric + access-log line and
32 * return the count unchanged (errno preserved on error). */
33static ssize_t
34brix_vfs_xattr_observe_count(const brix_vfs_ctx_t *ctx, const char *path,
35 ssize_t n, ngx_msec_t start)
36{
37 int saved_errno = (n < 0) ? errno : 0;
38
39 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_XATTR, NULL,
40 (n > 0) ? (size_t) n : 0,
41 (n < 0) ? NGX_ERROR : NGX_OK, saved_errno, start);
42 return n;
43}

45/* Read attribute `name` on the resolved ctx path into buf[bufsz] (bufsz==0 asks
46 * for the required size). Returns the byte count, or -1 with errno set
47 * (ERANGE when the value does not fit). Metered as OP_XATTR. */
48ssize_t
49brix_vfs_getxattr(brix_vfs_ctx_t *ctx, const char *name,
50 void *buf, size_t bufsz)
51{
52 const char *path = brix_vfs_ctx_path(ctx);
53 uint64_t start = brix_vfs_now_ns();
54 ssize_t n;
55
56 if (brix_vfs_require_confined(ctx) != NGX_OK) {
57 return brix_vfs_xattr_observe_count(ctx, path, -1, start);
58 }
59
60 {
61 const brix_sd_driver_t *drv = brix_vfs_ctx_driver(ctx);
62
63 if (drv != NULL) {
64 n = (drv->getxattr != NULL)
65 ? drv->getxattr(ctx->sd, brix_vfs_export_relative(ctx, path),
66 name, buf, bufsz)
67 : (errno = ENOTSUP, (ssize_t) -1);
68 return brix_vfs_xattr_observe_count(ctx, path, n, start);
69 }
70 }
71
72 n = brix_getxattr_confined_canon(ctx->log, ctx->root_canon, path, name,
73 buf, bufsz);
74 return brix_vfs_xattr_observe_count(ctx, path, n, start);
75}

108/* Set attribute `name` to value[len] on the resolved ctx path. `flags` are the
109 * raw setxattr(2) flags (XATTR_CREATE / XATTR_REPLACE / 0). Returns NGX_OK or
110 * NGX_ERROR with errno set. Metered as OP_XATTR. Not allow_write-gated (the
111 * protocol layer authorizes; lock writes occur on read-only requests). */
112ngx_int_t
113brix_vfs_setxattr(brix_vfs_ctx_t *ctx, const char *name,
114 const void *value, size_t len, int flags)
115{
116 const char *path = brix_vfs_ctx_path(ctx);
117 uint64_t start = brix_vfs_now_ns();
118 int rc;
119 int saved_errno;
120
121 if (brix_vfs_require_confined(ctx) != NGX_OK) {
122 saved_errno = errno;
123 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_XATTR, NULL, 0,
124 NGX_ERROR, saved_errno, start);
125 return NGX_ERROR;
126 }
127
128 {
129 const brix_sd_driver_t *drv = brix_vfs_ctx_driver(ctx);
130
131 if (drv != NULL) {
132 rc = (drv->setxattr != NULL
133 && drv->setxattr(ctx->sd, brix_vfs_export_relative(ctx, path),
134 name, value, len, flags) == NGX_OK)
135 ? 0 : (errno = (drv->setxattr ? errno : ENOTSUP), -1);
136 saved_errno = (rc != 0) ? errno : 0;
137 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_XATTR, NULL,
138 (rc == 0) ? len : 0,
139 (rc != 0) ? NGX_ERROR : NGX_OK,
140 saved_errno, start);
141 return (rc != 0) ? NGX_ERROR : NGX_OK;
142 }
143 }
144
145 rc = brix_setxattr_confined_canon(ctx->log, ctx->root_canon, path, name,
146 value, len, flags);
147 saved_errno = (rc != 0) ? errno : 0;
148 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_XATTR, NULL,
149 (rc == 0) ? len : 0,
150 (rc != 0) ? NGX_ERROR : NGX_OK, saved_errno,
151 start);
152 return (rc != 0) ? NGX_ERROR : NGX_OK;
153}

198/* --- open-handle (fd) variants --------------------------------------------
199 * The path variants above re-verify confinement against ctx->resolved before
200 * each syscall. The fd variants below operate on an fd that the VFS already
201 * opened confined (via brix_vfs_open / brix_vfs_adopt_fd, or a handle-table
202 * fd that came from one), so the confinement guarantee travels with the
203 * descriptor — there is no path to re-resolve. They exist so that fattr's
204 * file-handle mode (and any other open-fd xattr caller) reaches the backend
205 * through the VFS instead of calling f*xattr(2) directly.
206 *
207 * `ctx` is optional and used only for the OP_XATTR metric + access-log line; it
208 * may be NULL (then the op is unobserved). It is NOT required to be confined —
209 * passing the request's ctx simply attributes the metric to the right proto. */
210
211ssize_t
212brix_vfs_fgetxattr(const brix_vfs_ctx_t *ctx, int fd, const char *name,
213 void *buf, size_t bufsz)
214{
215 uint64_t start = brix_vfs_now_ns();
216 ssize_t n = fgetxattr(fd, name, buf, bufsz);
217
218 return brix_vfs_xattr_observe_count(ctx, NULL, n, start);
219}
