1/*
2 * vfs_xattr.c — VFS extended-attribute family (get / set / remove / list).
3 *
4 * WHAT: Implements brix_vfs_getxattr/setxattr/removexattr/listxattr — the
5 * protocol-agnostic surface for the `user.`-namespace xattrs that S3
6 * object tagging, WebDAV dead properties, checksum sidecars, and the
7 * WebDAV lock database all store on export objects.
8 *
9 * WHY: Before this unit those callers reached the confined xattr helpers
10 * (brix_*xattr_confined_canon) directly, so the xattr touches were
11 * confined but invisible to metrics/access-logging. Routing them here
12 * gives every xattr op one BRIX_METRIC_OP_XATTR metric + access-log line
13 * and the same guard-then-syscall-then-observe shape as the rest of the
14 * VFS, while still delegating the actual syscall (and impersonation broker
15 * routing) to the confined helpers.
16 *
17 * HOW: Each entry point re-verifies confinement (brix_vfs_require_confined),
18 * calls the matching brix_*xattr_confined_canon with ctx->root_canon and
19 * the resolved path, then observes the result as OP_XATTR. set/remove are
20 * mutations but are intentionally NOT allow_write-gated: the lock-database
21 * writes happen on otherwise read-only requests and the protocol layer has
22 * already authorized the principal — matching the prior direct-call
23 * behaviour exactly (no new EACCES surface). get/list propagate the helper
24 * byte count (or ERANGE) unchanged.
25 */

30/* Shared observe tail for the value-returning ops (get/list): translate a
31 * helper return (>=0 ok, -1 errno) into an OP_XATTR metric + access-log line and
32 * return the count unchanged (errno preserved on error). ENODATA is observed as
33 * a clean zero-byte lookup, not an error: optional-attribute probes (S3
34 * usermeta/tagging on GET, WebDAV dead props) routinely miss, and logging each
35 * miss as a failed op:"xattr" line put error lines on every served GET. The
36 * caller still sees -1/ENODATA unchanged. */
37static ssize_t
38brix_vfs_xattr_observe_count(const brix_vfs_ctx_t *ctx, const char *path,
39 ssize_t n, ngx_msec_t start)
40{
41 int saved_errno = (n < 0) ? errno : 0;
42 int absent_ok = (n >= 0) || saved_errno == ENODATA;
43
44 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_XATTR, NULL,
45 (n > 0) ? (size_t) n : 0,
46 absent_ok ? NGX_OK : NGX_ERROR,
47 absent_ok ? 0 : saved_errno, start);
48 if (n < 0) {
49 errno = saved_errno;
50 }
51 return n;
52}

54/* Read attribute `name` on the resolved ctx path into buf[bufsz] (bufsz==0 asks
55 * for the required size). Returns the byte count, or -1 with errno set
56 * (ERANGE when the value does not fit). Metered as OP_XATTR. */
57ssize_t
58brix_vfs_getxattr(brix_vfs_ctx_t *ctx, const char *name,
59 void *buf, size_t bufsz)
60{
61 const char *path = brix_vfs_ctx_path(ctx);
62 uint64_t start = brix_vfs_now_ns();
63 ssize_t n;
64
65 if (brix_vfs_require_confined(ctx) != NGX_OK) {
66 return brix_vfs_xattr_observe_count(ctx, path, -1, start);
67 }
68
69 {
70 const brix_sd_driver_t *drv = brix_vfs_ctx_driver(ctx);
71
72 if (drv != NULL) {
73 brix_sd_ucred_t store;
74 brix_sd_cred_t cred;
75 int use_cred = 0, cred_err = 0;
76
77 /* Zero before the gate: it fills only the active credential kind;
78 * an unzeroed cred hands a garbage inactive pointer to the driver
79 * cred slot (bearer PASSTHROUGH would leave x509_proxy dangling). */
80 ngx_memzero(&cred, sizeof(cred));
81
82 if (brix_vfs_cred_gate_active(ctx)) {
83 if (brix_vfs_ns_cred(ctx, &store, &cred, &use_cred, &cred_err)
84 != NGX_OK)
85 {
86 errno = cred_err ? cred_err : EACCES;
87 return brix_vfs_xattr_observe_count(ctx, path, -1, start);
88 }
89 }
90 /* Dispatch on the leaf so *_maybe_cred finds the leaf driver's
91 * getxattr_cred slot (decorators have only plain relays). */
92 n = (drv->getxattr != NULL)
93 ? brix_sd_getxattr_maybe_cred(brix_vfs_ns_leaf(ctx->sd),
94 brix_vfs_export_relative(ctx, path),
95 name, buf, bufsz, use_cred ? &cred : NULL)
96 : (errno = ENOTSUP, (ssize_t) -1);
97 brix_sd_ucred_wipe(&store); /* secret consumed; erase (A-4/T4) */
98 return brix_vfs_xattr_observe_count(ctx, path, n, start);
99 }
100 }
101
102 n = brix_getxattr_confined_canon(ctx->log, ctx->root_canon, path, name,
103 buf, bufsz);
104 return brix_vfs_xattr_observe_count(ctx, path, n, start);
105}

159/* Observe a mutation (set/remove) result: translate an rc (0 ok, non-0 error
160 * with errno already set) into an OP_XATTR metric + access-log line, reporting
161 * `nbytes` on success and 0 on error, and return NGX_OK/NGX_ERROR. Shared by
162 * the set/remove entry points so the observe tail is identical for both. */
163static ngx_int_t
164brix_vfs_xattr_observe_mut(const brix_vfs_ctx_t *ctx, const char *path,
165 int rc, size_t nbytes, ngx_msec_t start)
166{
167 int saved_errno = (rc != 0) ? errno : 0;
168
169 brix_vfs_observe_ctx_op(ctx, path, BRIX_METRIC_OP_XATTR, NULL,
170 (rc == 0) ? nbytes : 0,
171 (rc != 0) ? NGX_ERROR : NGX_OK, saved_errno,
172 start);
173 return (rc != 0) ? NGX_ERROR : NGX_OK;
174}

176/* Run the shared set/remove driver-path gates: the phase-71 write-capability
177 * gate (CAP_XATTR_WRITE required) then the per-user credential gate. On success
178 * returns NGX_OK with use_cred/cred populated for the dispatch; on failure
179 * sets errno, emits the OP_XATTR error observation, and returns NGX_ERROR so the
180 * caller can early-return without duplicating the observe tail. Byte-identical
181 * to the inline gates the set/remove paths previously carried. */
182static ngx_int_t
183brix_vfs_xattr_write_gate(brix_vfs_ctx_t *ctx, const char *path,
184 const brix_sd_driver_t *drv, brix_sd_ucred_t *store, brix_sd_cred_t *cred,
185 int *use_cred, ngx_msec_t start)
186{
187 int cred_err = 0;
188
189 /* Zero before the gate: it fills only the active credential kind; an
190 * unzeroed cred hands a garbage inactive pointer to the driver cred slot
191 * (bearer PASSTHROUGH would leave x509_proxy dangling). */
192 ngx_memzero(cred, sizeof(*cred));
193
194 /* phase-71: capability gate — a backend that can read xattrs but not write
195 * them (CAP_XATTR without CAP_XATTR_WRITE) rejects set/remove uniformly,
196 * regardless of whether the vtable slot is populated. */
197 if (!(drv->caps & BRIX_SD_CAP_XATTR_WRITE)) {
198 errno = ENOTSUP;
199 (void) brix_vfs_xattr_observe_mut(ctx, path, -1, 0, start);
200 return NGX_ERROR;
201 }
202
203 if (brix_vfs_cred_gate_active(ctx)) {
204 if (brix_vfs_ns_cred(ctx, store, cred, use_cred, &cred_err) != NGX_OK) {
205 errno = cred_err ? cred_err : EACCES;
206 (void) brix_vfs_xattr_observe_mut(ctx, path, -1, 0, start);
207 return NGX_ERROR;
208 }
209 }
210
211 return NGX_OK;
212}

305/* --- open-handle (fd) variants --------------------------------------------
306 * The path variants above re-verify confinement against ctx->resolved before
307 * each syscall. The fd variants below operate on an fd that the VFS already
308 * opened confined (via brix_vfs_open / brix_vfs_adopt_fd, or a handle-table
309 * fd that came from one), so the confinement guarantee travels with the
310 * descriptor — there is no path to re-resolve. They exist so that fattr's
311 * file-handle mode (and any other open-fd xattr caller) reaches the backend
312 * through the VFS instead of calling f*xattr(2) directly.

314 * `ctx` is optional and used only for the OP_XATTR metric + access-log line; it
315 * may be NULL (then the op is unobserved). It is NOT required to be confined —
