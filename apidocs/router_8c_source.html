1/*
2 * router.c — table-driven CMS opcode routing. See router.h.
3 *
4 * The two tables mirror XrdCmsRouting.cc: the manager table merges the redirector
5 * routing (client-facing forwardable ops, initRDRrouting) with the server group
6 * (node->manager status frames, initRouter); the node table is the set of ops a
7 * data server executes when forwarded down from its manager. A data node is a
8 * leaf here (no sub-manager cascade — an explicit non-goal), so forwarded ops
9 * carry no FORWARD flag in the node table.
10 */
11

103static const brix_cms_route_t *
104route_scan(const brix_cms_route_t *tbl, size_t n, unsigned char code)
105{
106 size_t i;
107 for (i = 0; i < n; i++) {
108 if (tbl[i].code == code) {
109 return &tbl[i];
110 }
111 }
112 return NULL;
113}

115const brix_cms_route_t *
116brix_cms_route_lookup(brix_cms_role_t role, unsigned char code)
117{
118 if (role == XRDCMS_ROLE_NODE) {
119 return route_scan(node_routes,
120 sizeof(node_routes) / sizeof(node_routes[0]), code);
121 }
122 return route_scan(manager_routes,
123 sizeof(manager_routes) / sizeof(manager_routes[0]), code);
124}
