20/* Load the OpenSSL-3 "legacy" provider once (Blowfish lives there). Warm SHA2-256
21 * from the default provider first so loading legacy does not become the only
22 * active provider for an app that later wants a default-provider digest. No-op
23 * before OpenSSL 3.x. */
24static void
25sss_load_legacy_provider(void)
26{
27#if OPENSSL_VERSION_NUMBER >= 0x30000000L
28 static int done;
29
30 if (!done) {
31 EVP_MD *md = EVP_MD_fetch(NULL, "SHA2-256", NULL);
32 if (md != NULL) {
33 EVP_MD_free(md);
34 }
35 (void) OSSL_PROVIDER_load(NULL, "legacy");
36 done = 1;
37 }
38#endif
39}

42brix_sss_bf_crypt(int encrypt, const uint8_t *key, size_t key_len,
43 const uint8_t *src, size_t src_len,
44 uint8_t *dst, size_t dst_max, size_t *out_len)
45{
46 EVP_CIPHER_CTX *evp;
47 uint8_t iv[8] = { 0, 0, 0, 0, 0, 0, 0, 0 };
48 int len1 = 0, len2 = 0, ok;
49
50 if (key == NULL || src == NULL || dst == NULL || out_len == NULL) {
51 return -1;
52 }
53 if (key_len == 0 || key_len > INT_MAX || src_len > INT_MAX
54 || dst_max < src_len || dst_max > INT_MAX) {
55 return -1;
56 }
57 sss_load_legacy_provider();
58
59 evp = EVP_CIPHER_CTX_new();
60 if (evp == NULL) {
61 return -1;
62 }
63 /* CFB64 is a stream mode (no padding); the key length is variable — an SSS
64 * key is typically 32 bytes, under Blowfish's 56-byte maximum. */
65 if (encrypt) {
66 ok = EVP_EncryptInit_ex(evp, EVP_bf_cfb64(), NULL, NULL, NULL) == 1
67 && EVP_CIPHER_CTX_set_padding(evp, 0) == 1
68 && EVP_CIPHER_CTX_set_key_length(evp, (int) key_len) == 1
69 && EVP_EncryptInit_ex(evp, NULL, NULL, key, iv) == 1
70 && EVP_EncryptUpdate(evp, dst, &len1, src, (int) src_len) == 1
71 && EVP_EncryptFinal_ex(evp, dst + len1, &len2) == 1;
72 } else {
73 ok = EVP_DecryptInit_ex(evp, EVP_bf_cfb64(), NULL, NULL, NULL) == 1
74 && EVP_CIPHER_CTX_set_padding(evp, 0) == 1
75 && EVP_CIPHER_CTX_set_key_length(evp, (int) key_len) == 1
76 && EVP_DecryptInit_ex(evp, NULL, NULL, key, iv) == 1
77 && EVP_DecryptUpdate(evp, dst, &len1, src, (int) src_len) == 1
78 && EVP_DecryptFinal_ex(evp, dst + len1, &len2) == 1;
79 }
80 EVP_CIPHER_CTX_free(evp);
81 if (!ok) {
82 return -1;
83 }
84 *out_len = (size_t) (len1 + len2);
85 return 0;
86}

89brix_sss_build_credential(const uint8_t *key, size_t key_len, uint64_t key_id,
90 const char *username, const uint8_t nonce32[32], uint32_t gen_time,
91 uint8_t *out, size_t out_max, size_t *out_len)
92{
93 uint8_t clear[BRIX_SSS_DATA_HDR_LEN + 3 + 64 + 1]; /* hdr + TLV + name + NUL */
94 uint8_t plain[sizeof(clear) + 4]; /* + CRC32 */
95 uint8_t *cursor;
96 size_t ulen, clear_len, crypt_out;
97 uint32_t crc;
98
99 if (key == NULL || key_len == 0 || nonce32 == NULL || out == NULL
100 || out_len == NULL || out_max < BRIX_SSS_HDR_LEN + 64) {
101 return -1;
102 }
103 if (username == NULL || *username == '\0') {
104 username = "xrd";
105 }
106
107 /* 40-byte data header: 32 nonce + gen_time(BE) + USEDATA opt byte. */
108 memset(clear, 0, sizeof(clear));
109 memcpy(clear, nonce32, 32);
110 clear[32] = (uint8_t) (gen_time >> 24);
111 clear[33] = (uint8_t) (gen_time >> 16);
112 clear[34] = (uint8_t) (gen_time >> 8);
113 clear[35] = (uint8_t) gen_time;
114 clear[39] = BRIX_SSS_OPT_USEDATA;
115
116 /* NAME TLV: [type][0][len][username NUL-terminated]; len includes the NUL. */
117 ulen = strlen(username) + 1;
118 if (ulen > 64) {
119 ulen = 64;
120 }
121 cursor = clear + BRIX_SSS_DATA_HDR_LEN;
122 *cursor++ = BRIX_SSS_TYPE_NAME;
123 *cursor++ = 0;
124 *cursor++ = (uint8_t) ulen;
125 memcpy(cursor, username, ulen - 1);
126 cursor[ulen - 1] = '\0';
127 cursor += ulen;
128 clear_len = (size_t) (cursor - clear);
129
130 /* plain = cleartext + IEEE-CRC32 (big-endian). */
131 memcpy(plain, clear, clear_len);
132 crc = brix_crc32_ieee(plain, clear_len);
133 plain[clear_len + 0] = (uint8_t) (crc >> 24);
134 plain[clear_len + 1] = (uint8_t) (crc >> 16);
135 plain[clear_len + 2] = (uint8_t) (crc >> 8);
136 plain[clear_len + 3] = (uint8_t) crc;
137
138 if ((size_t) BRIX_SSS_HDR_LEN + clear_len + 4 > out_max) {
139 return -1;
140 }
141
142 /* BF32-encrypt the body into out[16..] (CFB64: cipher_len == plain_len). */
143 if (brix_sss_bf_crypt(1, key, key_len, plain, clear_len + 4,
144 out + BRIX_SSS_HDR_LEN,
145 out_max - BRIX_SSS_HDR_LEN, &crypt_out) != 0) {
146 return -1;
147 }
148
149 /* 16-byte outer header: magic + version + spare + kn + enc + key_id(8B BE). */
150 out[0] = 's'; out[1] = 's'; out[2] = 's'; out[3] = '\0';
151 out[4] = 1; /* version */
152 out[5] = 0; /* spare */
153 out[6] = 0; /* kn_size: no named key */
154 out[7] = BRIX_SSS_ENC_BF32;
155 out[ 8] = (uint8_t) (key_id >> 56);
156 out[ 9] = (uint8_t) (key_id >> 48);
157 out[10] = (uint8_t) (key_id >> 40);
158 out[11] = (uint8_t) (key_id >> 32);
159 out[12] = (uint8_t) (key_id >> 24);
160 out[13] = (uint8_t) (key_id >> 16);
161 out[14] = (uint8_t) (key_id >> 8);
162 out[15] = (uint8_t) key_id;
163
164 *out_len = (size_t) BRIX_SSS_HDR_LEN + crypt_out;
165 return 0;
166}
