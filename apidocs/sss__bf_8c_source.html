20/* Load the OpenSSL-3 "legacy" provider once (Blowfish lives there). Warm SHA2-256
21 * from the default provider first so loading legacy does not become the only
22 * active provider for an app that later wants a default-provider digest. No-op
23 * before OpenSSL 3.x. */
24static void
25sss_load_legacy_provider(void)
26{
27#if OPENSSL_VERSION_NUMBER >= 0x30000000L
28 static int done;
29
30 if (!done) {
31 EVP_MD *md = EVP_MD_fetch(NULL, "SHA2-256", NULL);
32 if (md != NULL) {
33 EVP_MD_free(md);
34 }
35 (void) OSSL_PROVIDER_load(NULL, "legacy");
36 done = 1;
37 }
38#endif
39}

41/* ---- Validate the crypt argument set ----
42 *
43 * WHAT: Returns 1 when every pointer is non-NULL and the lengths are in range
44 * (key 1..INT_MAX, src <= INT_MAX, dst_max both >= src_len and <= INT_MAX);
45 * returns 0 otherwise. Pure predicate — no side effects.
46 *
47 * WHY: The bounds are what let brix_sss_bf_crypt cast the size_t lengths to the
48 * int arguments OpenSSL's EVP API takes without truncation, and guarantee the
49 * output buffer can hold the CFB64 result (cipher_len == plain_len). Hoisting
50 * the two guard clauses out of the orchestrator keeps its control flow flat.
51 *
52 * HOW:
53 * 1. Reject any NULL among key/src/dst/out_len.
54 * 2. Reject a zero or over-INT_MAX key, an over-INT_MAX source, or an output
55 * cap that is smaller than the source or itself over INT_MAX.
56 * 3. Otherwise accept.
57 */
58static int
59sss_bf_args_valid(const uint8_t *key, size_t key_len,
60 const uint8_t *src, size_t src_len,
61 const uint8_t *dst, size_t dst_max, const size_t *out_len)
62{
63 if (key == NULL || src == NULL || dst == NULL || out_len == NULL) {
64 return 0;
65 }
66 if (key_len == 0 || key_len > INT_MAX || src_len > INT_MAX
67 || dst_max < src_len || dst_max > INT_MAX) {
68 return 0;
69 }
70 return 1;
71}

73/* ---- Run one Blowfish-CFB64 encrypt pass ----
74 *
75 * WHAT: Configures the caller-supplied EVP context for Blowfish-CFB64 with the
76 * given variable-length key and IV, encrypts src_len bytes of src into dst, and
77 * stores the two produced lengths through len1/len2. Returns 1 on full success,
78 * 0 if any EVP step fails (short-circuiting on the first failure).
79 *
80 * WHY: CFB64 is a stream mode (no padding); the key length is variable — an SSS
81 * key is typically 32 bytes, under Blowfish's 56-byte maximum. Encapsulating the
82 * init/update/final chain keeps the orchestrator small and mirrors the decrypt
83 * pass exactly so the two stay in lock-step.
84 *
85 * HOW:
86 * 1. Select EVP_bf_cfb64() and disable padding.
87 * 2. Set the key length, then install key + IV.
88 * 3. EncryptUpdate the whole input (setting *len1), then EncryptFinal_ex at
89 * dst + *len1 (setting *len2). The && chain guarantees *len1 is written
90 * before it is read for the Final offset.
91 */
92static int
93sss_bf_encrypt_run(EVP_CIPHER_CTX *evp, const uint8_t *key, size_t key_len,
94 const uint8_t *iv, const uint8_t *src, size_t src_len,
95 uint8_t *dst, int *len1, int *len2)
96{
97 return EVP_EncryptInit_ex(evp, EVP_bf_cfb64(), NULL, NULL, NULL) == 1
98 && EVP_CIPHER_CTX_set_padding(evp, 0) == 1
99 && EVP_CIPHER_CTX_set_key_length(evp, (int) key_len) == 1
100 && EVP_EncryptInit_ex(evp, NULL, NULL, key, iv) == 1
101 && EVP_EncryptUpdate(evp, dst, len1, src, (int) src_len) == 1
102 && EVP_EncryptFinal_ex(evp, dst + *len1, len2) == 1;
103}

105/* ---- Run one Blowfish-CFB64 decrypt pass ----
106 *
107 * WHAT: The exact inverse of sss_bf_encrypt_run — configures the EVP context for
108 * Blowfish-CFB64 with the variable-length key and IV, decrypts src_len bytes of
109 * src into dst, and stores the two produced lengths through len1/len2. Returns 1
110 * on full success, 0 on the first failing EVP step.
111 *
112 * WHY: Kept byte-for-byte parallel to the encrypt pass (same mode, same padding
113 * disable, same key-length handling) so encrypt and decrypt cannot drift; only
114 * the EVP_Decrypt* entry points differ.
115 *
116 * HOW:
117 * 1. Select EVP_bf_cfb64() and disable padding.
118 * 2. Set the key length, then install key + IV.
119 * 3. DecryptUpdate the whole input (setting *len1), then DecryptFinal_ex at
120 * dst + *len1 (setting *len2). The && chain guarantees *len1 is written
121 * before it is read for the Final offset.
122 */
123static int
124sss_bf_decrypt_run(EVP_CIPHER_CTX *evp, const uint8_t *key, size_t key_len,
125 const uint8_t *iv, const uint8_t *src, size_t src_len,
126 uint8_t *dst, int *len1, int *len2)
127{
128 return EVP_DecryptInit_ex(evp, EVP_bf_cfb64(), NULL, NULL, NULL) == 1
129 && EVP_CIPHER_CTX_set_padding(evp, 0) == 1
130 && EVP_CIPHER_CTX_set_key_length(evp, (int) key_len) == 1
131 && EVP_DecryptInit_ex(evp, NULL, NULL, key, iv) == 1
132 && EVP_DecryptUpdate(evp, dst, len1, src, (int) src_len) == 1
133 && EVP_DecryptFinal_ex(evp, dst + *len1, len2) == 1;
134}

136/* ---- Blowfish-CFB64 crypt orchestrator ----
137 *
138 * WHAT: Encrypts (encrypt != 0) or decrypts the src_len bytes of src into dst
139 * (capacity dst_max) under the variable-length key, writing the produced byte
140 * count to *out_len. Returns 0 on success, -1 on bad arguments or any EVP
141 * failure. Uses an all-zero 8-byte IV, matching the SSS wire format.
142 *
143 * WHY: Single entry point shared by the module's SSS auth and the native
144 * client's SSS mint. The heavy lifting lives in single-purpose helpers so this
145 * stays a flat validate → set-up → run → tear-down sequence.
146 *
147 * HOW:
148 * 1. Validate arguments; bail on -1 if invalid.
149 * 2. Ensure the OpenSSL-3 legacy provider (Blowfish) is loaded.
150 * 3. Allocate an EVP context; bail on allocation failure.
151 * 4. Dispatch to the encrypt or decrypt pass, capturing len1/len2.
152 * 5. Free the context; on failure return -1, else publish len1+len2.
153 */
154int
155brix_sss_bf_crypt(int encrypt, const uint8_t *key, size_t key_len,
156 const uint8_t *src, size_t src_len,
157 uint8_t *dst, size_t dst_max, size_t *out_len)
158{
159 EVP_CIPHER_CTX *evp;
160 uint8_t iv[8] = { 0, 0, 0, 0, 0, 0, 0, 0 };
161 int len1 = 0, len2 = 0, ok;
162
163 if (!sss_bf_args_valid(key, key_len, src, src_len, dst, dst_max, out_len)) {
164 return -1;
165 }
166 sss_load_legacy_provider();
167
168 evp = EVP_CIPHER_CTX_new();
169 if (evp == NULL) {
170 return -1;
171 }
172 if (encrypt) {
173 ok = sss_bf_encrypt_run(evp, key, key_len, iv, src, src_len,
174 dst, &len1, &len2);
175 } else {
176 ok = sss_bf_decrypt_run(evp, key, key_len, iv, src, src_len,
177 dst, &len1, &len2);
178 }
179 EVP_CIPHER_CTX_free(evp);
180 if (!ok) {
181 return -1;
182 }
183 *out_len = (size_t) len1 + (size_t) len2;
184 return 0;
185}

187int
188brix_sss_build_credential(const uint8_t *key, size_t key_len, uint64_t key_id,
189 const char *username, const uint8_t nonce32[32], uint32_t gen_time,
190 uint8_t *out, size_t out_max, size_t *out_len)
191{
192 uint8_t clear[BRIX_SSS_DATA_HDR_LEN + 3 + 64 + 1]; /* hdr + TLV + name + NUL */
193 uint8_t plain[sizeof(clear) + 4]; /* + CRC32 */
194 uint8_t *cursor;
195 size_t ulen, clear_len, crypt_out;
196 uint32_t crc;
197
198 if (key == NULL || key_len == 0 || nonce32 == NULL || out == NULL
199 || out_len == NULL || out_max < BRIX_SSS_HDR_LEN + 64) {
200 return -1;
201 }
202 if (username == NULL || *username == '\0') {
203 username = "xrd";
204 }
205
206 /* 40-byte data header: 32 nonce + gen_time(BE) + USEDATA opt byte. */
207 memset(clear, 0, sizeof(clear));
208 memcpy(clear, nonce32, 32);
209 clear[32] = (uint8_t) (gen_time >> 24);
210 clear[33] = (uint8_t) (gen_time >> 16);
211 clear[34] = (uint8_t) (gen_time >> 8);
212 clear[35] = (uint8_t) gen_time;
213 clear[39] = BRIX_SSS_OPT_USEDATA;
214
215 /* NAME TLV: [type][0][len][username NUL-terminated]; len includes the NUL. */
216 ulen = strlen(username) + 1;
217 if (ulen > 64) {
218 ulen = 64;
219 }
220 cursor = clear + BRIX_SSS_DATA_HDR_LEN;
221 *cursor++ = BRIX_SSS_TYPE_NAME;
222 *cursor++ = 0;
223 *cursor++ = (uint8_t) ulen;
224 memcpy(cursor, username, ulen - 1);
225 cursor[ulen - 1] = '\0';
226 cursor += ulen;
227 clear_len = (size_t) (cursor - clear);
228
229 /* plain = cleartext + IEEE-CRC32 (big-endian). */
230 memcpy(plain, clear, clear_len);
231 crc = brix_crc32_ieee(plain, clear_len);
232 plain[clear_len + 0] = (uint8_t) (crc >> 24);
233 plain[clear_len + 1] = (uint8_t) (crc >> 16);
234 plain[clear_len + 2] = (uint8_t) (crc >> 8);
235 plain[clear_len + 3] = (uint8_t) crc;
236
237 if ((size_t) BRIX_SSS_HDR_LEN + clear_len + 4 > out_max) {
238 return -1;
239 }
240
241 /* BF32-encrypt the body into out[16..] (CFB64: cipher_len == plain_len). */
242 if (brix_sss_bf_crypt(1, key, key_len, plain, clear_len + 4,
243 out + BRIX_SSS_HDR_LEN,
244 out_max - BRIX_SSS_HDR_LEN, &crypt_out) != 0) {
245 return -1;
246 }
247
248 /* 16-byte outer header: magic + version + spare + kn + enc + key_id(8B BE). */
249 out[0] = 's'; out[1] = 's'; out[2] = 's'; out[3] = '\0';
250 out[4] = 1; /* version */
251 out[5] = 0; /* spare */
252 out[6] = 0; /* kn_size: no named key */
253 out[7] = BRIX_SSS_ENC_BF32;
254 out[ 8] = (uint8_t) (key_id >> 56);
255 out[ 9] = (uint8_t) (key_id >> 48);
256 out[10] = (uint8_t) (key_id >> 40);
257 out[11] = (uint8_t) (key_id >> 32);
258 out[12] = (uint8_t) (key_id >> 24);
259 out[13] = (uint8_t) (key_id >> 16);
260 out[14] = (uint8_t) (key_id >> 8);
261 out[15] = (uint8_t) key_id;
262
263 *out_len = (size_t) BRIX_SSS_HDR_LEN + crypt_out;
264 return 0;
265}
