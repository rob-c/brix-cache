60/* ---- Apply the server's inline-compression negotiation to an open handle ----
61 *
62 * WHAT: Inspects the ServerOpenBody compression fields (cpsize[4] cptype[4] at
63 * body offsets 4..11) and, when the server confirmed inline compression, records
64 * the negotiated codec on the handle's read_codec or write_codec. Always clears
65 * both codecs first, so a plaintext reply leaves the handle in plaintext mode.
66 * Returns 0 on success (including the plaintext case); returns -1 and sets st
67 * when the server negotiated a codec this client build cannot decode. Never
68 * frees body — ownership stays with the caller.
69 *
70 * WHY: phase-42 W4/W5 inline compression. The dual-check contract in
71 * codec_core.h requires BOTH cpsize == BRIX_INLINE_CMP_MAGIC AND a valid
72 * cptype[0] codec ordinal before trusting the field: cptype is a legacy XRootD
73 * field, so a stock or non-cooperating server may leave an arbitrary small byte
74 * there; adopting it without the magic would make us inflate PLAINTEXT bytes and
75 * corrupt the transfer. If the magic confirms a codec this build lacks, the open
76 * MUST fail — silently falling back to plaintext would copy still-compressed
77 * bytes verbatim (asymmetric build) and corrupt the transfer.
78 *
79 * HOW:
80 * 1. Reset read_codec/write_codec to 0 (plaintext default).
81 * 2. If the body is shorter than 12 bytes, there is no compression tuple —
82 * return 0 (plaintext).
83 * 3. Decode the big-endian cpsize and cptype[0] codec id.
84 * 4. Require cpsize == magic AND 1 <= cid < BRIX_CODEC_MAX; otherwise plaintext.
85 * 5. Fail (-1, st set) if the confirmed codec is unavailable in this build.
86 * 6. Record the codec on write_codec (W5 write opens) or read_codec (W4 read
87 * opens) — at most one direction is negotiated.
88 */
89static int
90brix_open_apply_inline_codec(const uint8_t *body, uint32_t blen, int write,
91 brix_file *f, brix_status *st)
92{
93 uint32_t cpsize;
94 uint8_t cid;
95
96 f->read_codec = 0;
97 f->write_codec = 0;
98 if (blen < 12) {
99 return 0;
100 }
101
102 cpsize = ((uint32_t) body[4] << 24) | ((uint32_t) body[5] << 16)
103 | ((uint32_t) body[6] << 8) | (uint32_t) body[7];
104 cid = body[8]; /* cptype[0] */
105 if (cpsize != BRIX_INLINE_CMP_MAGIC || cid < 1 || cid >= BRIX_CODEC_MAX) {
106 return 0;
107 }
108
109 /* The server CONFIRMED inline compression with codec `cid` and WILL send
110 * compressed frames (read) or expect them (write). If this client build
111 * cannot handle that codec we must FAIL the open (see WHY). */
112 if (!brix_codec_available(cid)) {
113 brix_status_set(st, XRDC_EUNSUPPORTED, 0,
114 "server negotiated inline-compression codec %u that this "
115 "client build cannot decode", (unsigned) cid);
116 return -1;
117 }
118
119 /* W4 read opens compress responses; W5 write opens compress the payloads
120 * this client sends. At most one direction is negotiated. */
121 if (write) {
122 f->write_codec = cid;
123 } else {
124 f->read_codec = cid;
125 }
126 return 0;
127}

131int
132brix_file_open_opaque(brix_conn *c, const char *path, const char *opaque,
133 int write, int force, int posc, brix_file *f,
134 brix_status *st)
135{
136 ClientOpenRequest req;
137 uint16_t status, options;
138 uint8_t *body = NULL;
139 uint32_t blen = 0;
140 char *payload;
141 size_t need;
142 int plen;
143
144 /* kXR_open option bits come from the shared builder (protocol/open_flags.h)
145 * so this request and the server's POSIX-flag decode share one definition of
146 * the create/truncate/in-place (`force`) semantics. Writes always make parent
147 * dirs (mkpath). */
148 options = brix_open_options_build(write, force, posc, /*mkpath=*/1);
149
150 /* The server splits "<path>?<opaque>" — open_extract_opaque (src/protocols/root/read).
151 * Heap-size to the actual lengths; no opaque ⇒ bare path (open_read/write). */
152 need = strlen(path) + 1 + (opaque ? strlen(opaque) : 0) + 1;
153 payload = (char *) malloc(need);
154 if (payload == NULL) {
155 brix_status_set(st, XRDC_EPROTO, 0, "out of memory");
156 return -1;
157 }
158 plen = (opaque != NULL && opaque[0] != '\0')
159 ? snprintf(payload, need, "%s?%s", path, opaque)
160 : snprintf(payload, need, "%s", path);
161
162 memset(&req, 0, sizeof(req));
163 req.requestid = htons(kXR_open);
164 /* Marshal the fixed body (mode/options/optiont/fhtemplt) through the shared
165 * codec so the client and server agree on the offsets + byte order. */
166 {
167 xrdw_open_req_t o = { .mode = (uint16_t) (write ? 0644 : 0),
168 .options = options, .optiont = 0 };
169 xrdw_open_req_pack(&o, ((ClientRequestHdr *) &req)->body);
170 }
171
172 {
173 brix_payload pl = { payload, (uint32_t) plen };
174 brix_resp_out out = { &status, &body, &blen };
175 if (brix_roundtrip(c, &req, &pl, &out, st) != 0) {
176 free(payload);
177 return -1;
178 }
179 }
180 free(payload);
181
182 /* TPC coordinator open deferred: brix_recv surfaced the source's kXR_waitresp
183 * (rendezvous key registered, real reply pending) instead of blocking. There is
184 * no fhandle yet — the orchestrator never does I/O on this handle (the dest pulls
185 * the bytes); it only keeps the connection open so the registration stays live,
186 * then drains the deferred reply once the pull is triggered. Report success. */
187 if (c->tpc_coord_defer && status == kXR_waitresp) {
188 memset(f->fhandle, 0, XRD_FHANDLE_LEN);
189 f->read_codec = 0;
190 f->write_codec = 0;
191 free(body); /* NULL on the deferred path */
192 return 0;
193 }
194
195 if (blen < XRD_FHANDLE_LEN) {
196 brix_status_set(st, XRDC_EPROTO, 0, "open reply too short (%u bytes)", blen);
197 free(body);
198 return -1;
199 }
200 memcpy(f->fhandle, body, XRD_FHANDLE_LEN);
201
202 /* phase-42 W4/W5 — inline read/write compression. The ServerOpenBody is
203 * fhandle[4] cpsize[4] cptype[4]; the codec-negotiation decode + dual-check
204 * contract live in brix_open_apply_inline_codec (body ownership stays here). */
205 if (brix_open_apply_inline_codec(body, blen, write, f, st) != 0) {
206 free(body);
207 return -1;
208 }
209
210 free(body);
211 return 0;
212}
