48/*
49 * brix_open_attach_csi — attach the CSI block-checksum integrity engine to a
50 * freshly-opened handle (xmeta P3). A write handle folds block CRCs and merges
51 * them into the file's xmeta record at close; a read handle verifies spanned
52 * blocks against the record. A pure read on a trust_fs backing fs skips the
53 * engine; a require=on read with no verifiable record is refused (fd closed).
54 * Returns NGX_OK, or the error-response value on refusal.
55 */

57brix_open_attach_csi(brix_open_args_t *a)
58{

62 const char *resolved = a->resolved;
63 int idx = a->idx;
64 int fd = a->fd;
65 const struct stat *st = a->st;
66 ngx_flag_t is_write = a->is_write;
67
68 ctx->files[idx].csi = NULL;

91 kXR_ChkSumErr, "integrity record missing");
92 }

94 * proceed without CSI for this handle (fail-open). */
95 }
96 }
97 }

143 /* Write-through decision evaluation (mirrors XrdPfc::Cache::Decide())
144 * WHAT: Evaluate write-through policy at kXR_open time and cache the result on the handle.
145 * This is called once per open — the cached wt_policy determines close-time flush behavior.
146 *
147 * WHY: Mirrors XrdPfcDecision::Decide() from official XRootD PFC module (Cache::Attach()).
148 * Caching at open time avoids repeated policy evaluation for every write operation,
149 * reduces latency, and ensures consistent close-time behavior across the session.
150 *
151 * HOW: Policy callback flow (src/cache/writethrough_decision.h):
152 * 1. conf->wt.decision.fn(resolved, options, &conf->wt.decision) — default is brix_wt_default_decide()
153 * 2. Default engine checks: size filter → deny prefixes → allow prefixes → ALLOW_ASYNC (default)
154 * 3. Cache result on handle: ctx->files[idx].wt_policy = decision, wt_enabled = (decision != DENY),
155 * wt_dirty_offset = -1 (clean state), wt_bytes_written = 0
156 *
157 * Decision outcomes are cached for a future close-time write-back implementation:
158 * DENY → no write-back; local-only writes, handle treated as non-WT
159 * ALLOW_SYNC → synchronous flush to origin before closing handle (blocks)
160 * ALLOW_ASYNC→ schedule async thread-pool flush, return immediately to client */
161
162/* WT decision policy engine (default: prefix-based)
163 * WHAT: brix_wt_default_decide() — built-in prefix-based policy engine.
164 * External plugins can provide their own fn pointer for custom policies.
165 *
166 * WHY: Provides sensible defaults for most deployments without requiring external plugin setup.
167 * Prefix-based matching is O(n) where n = prefix length — acceptable for typical paths (/data/, /atlas/).
168 *
169 * HOW: Decision logic order (src/cache/writethrough_decision.c):
170 * 1. Size filter: if file > max_write_through_bytes AND no include regex match → DENY
171 * 2. Deny prefixes: any deny_prefix matches → DENY (deny takes precedence)
172 * 3. Allow prefixes: if allow list configured AND none match → DENY (whitelist mode)
173 * 4. Default: ALLOW_ASYNC (mirrors XrdPfcAllowDecision, sync preferred for local origins) */
174static void
175brix_open_decide_writethrough(brix_open_args_t *a)
176{
177 brix_ctx_t *ctx = a->ctx;
178 ngx_connection_t *c = a->c;
179 ngx_stream_brix_srv_conf_t *conf = a->conf;
180 const char *resolved = a->resolved;
181 uint16_t options = a->options;
182 int idx = a->idx;
183 mode_t create_mode = a->create_mode;
184 ngx_flag_t is_write = a->is_write;
185 ngx_flag_t wt_via_stage = (ngx_flag_t) a->wt_via_stage;
186
187 if (is_write && conf->wt.enable) {
188 brix_wt_decision_t decision = BRIX_WT_DECISION_DENY;
189
190 if (conf->wt.decision.fn != NULL) {
191 decision = conf->wt.decision.fn(resolved, options, &conf->wt.decision);
192 }

195 * the close-time run_flush — so leave wt_enabled clear for them. */
196 ctx->files[idx].wt_enabled = (!wt_via_stage
197 && decision != BRIX_WT_DECISION_DENY) ? 1 : 0;
198 ctx->files[idx].wt_policy = decision;
199 ctx->files[idx].wt_mode_bits = create_mode;
200 ctx->files[idx].wt_dirty_offset = -1; /* no dirty writes yet */
201 ctx->files[idx].wt_bytes_written = 0;
202
203 ctx->files[idx].wt_flush_task = NULL;
204 ctx->files[idx].wt_flush_pending = 0;

206 if (c->log->log_level & NGX_LOG_DEBUG_STREAM) {
207 char wt_log_path[512];

209 brix_sanitize_log_string(resolved, wt_log_path,
210 sizeof(wt_log_path));
211 ngx_log_debug2(NGX_LOG_DEBUG_STREAM, c->log, 0,
212 "brix: wt decision=%s path=%s",
213 decision == BRIX_WT_DECISION_DENY ? "DENY" :
214 decision == BRIX_WT_DECISION_ALLOW_SYNC ? "ALLOW_SYNC" : "ALLOW_ASYNC",
215 wt_log_path);
216 }

280 /* The POSIX-fd path stats the fd to validate type and clear O_NONBLOCK. A
281 * driver-backed open already synthesized `st` from the driver snapshot and
282 * its directory/type rejection happens at the driver (EISDIR mapped above)
283 * and the pre-flight VFS probe, so this fd-specific block is skipped. */
284static ngx_int_t
285brix_open_validate_fd(brix_open_args_t *a)
286{
287 brix_ctx_t *ctx = a->ctx;
288 ngx_connection_t *c = a->c;
289 const char *resolved = a->resolved;
290 int fd = a->fd;
291 struct stat *st = a->st;
292 int op = a->is_write ? BRIX_OP_OPEN_WR
293 : BRIX_OP_OPEN_RD;
294 const char *mode_str = a->is_write ? "wr" : "rd";
295
296 if (!a->driver_backed) {
297 /* A no-fd handle must never reach the POSIX validation path — the
298 * driver-backed branch owns those (st was synthesized there). */
299 if (fd < 0) {
300 BRIX_RETURN_ERR(ctx, c, op,
301 "OPEN", resolved, mode_str,
302 kXR_IOError, "open produced no file descriptor");
303 }
304 if (fstat(fd, st) != 0) {
305 int err = errno;
306
307 close(fd);
308 BRIX_RETURN_ERR(ctx, c, op,
309 "OPEN", resolved, mode_str,
310 kXR_IOError, strerror(err));
311 }
312
313 if (S_ISDIR(st->st_mode)) {
314 close(fd);
315 BRIX_RETURN_ERR(ctx, c, op,
316 "OPEN", resolved, mode_str,
317 kXR_isDirectory, "is a directory");
318 }
319
320 /* Only regular files are servable byte streams. A FIFO, socket, device
321 * or other special file was opened O_NONBLOCK (so the open could not
322 * wedge the worker); refuse to serve it rather than let a read/write spin
323 * on EAGAIN. A staged write always lands on a freshly O_CREAT'd regular
324 * temp, so this only ever rejects a pre-existing special file. */
325 if (!S_ISREG(st->st_mode)) {
326 close(fd);
327 BRIX_RETURN_ERR(ctx, c, op,
328 "OPEN", resolved, mode_str,
329 kXR_IOError, "not a regular file");
330 }
331
332 /* The fd is a confirmed regular file: drop O_NONBLOCK so every downstream
333 * read/write/sendfile sees ordinary blocking semantics (a no-op for local
334 * regular files, but it keeps the fd's flags unsurprising for callers). */
335 {
336 int fl = fcntl(fd, F_GETFL);
337 if (fl != -1 && (fl & O_NONBLOCK)) {
338 (void) fcntl(fd, F_SETFL, fl & ~O_NONBLOCK);
339 }
340 }
341 }
342 return NGX_OK;
343}

350/* Post-fd handle setup: validate the POSIX fd (type + clear O_NONBLOCK), init
351 * the handle bookkeeping, attach the CSI integrity engine, enforce the open-files
352 * cap, register the live-transfer monitor slot, record the on-disk path, apply
353 * the parent group policy, build the kXR_retstat string (may clear *want_stat),
354 * emit the debug line, and cache the write-through decision. Returns NGX_DECLINED
355 * to proceed; any rejection (fd validate, CSI, throttle, path-alloc) has already
356 * torn down the fd/sent its reply and its rc is returned. */
357ngx_int_t
358brix_open_finalize_handle(brix_open_args_t *a)
359{
360 ngx_connection_t *c = a->c;
361 ngx_int_t rc;
362
363 /* Stat + validate the POSIX fd (regular file, clear O_NONBLOCK) (split out). */
364 rc = brix_open_validate_fd(a);
365 if (rc != NGX_OK) {
366 return rc;
367 }
368
369 brix_open_init_handle(a);
370
371 /* Attach the CSI integrity engine to the handle (xmeta P3, split out). */
372 rc = brix_open_attach_csi(a);
373 if (rc != NGX_OK) {
374 return rc;
375 }
376
377 /* Enforce the per-user open-files cap (phase-59 W3a, split out). */
378 rc = brix_open_apply_throttle(a);
379 if (rc != NGX_OK) {
380 return rc;
381 }
382
383 /* Register the open file with the live transfer monitor (split out). */
384 brix_open_register_monitor(a->ctx, a->resolved, a->idx, a->is_write);
385
386 /* Record the handle's on-disk path (POSC temp vs final) for close-time
387 * rename/unlink (split out). */
388 rc = brix_open_set_handle_path(a);
389 if (rc != NGX_OK) {
390 return rc;
391 }
392
393 if (a->is_write && a->conf->group_rules != NULL) {
394 brix_apply_parent_group_policy_fd(c->log, a->fd, a->resolved,
395 a->conf->group_rules);
396 }
397
398 /* Build the kXR_retstat metadata string when requested (split out). */
399 brix_open_build_retstat(a);
400
401 if (c->log->log_level & NGX_LOG_DEBUG_STREAM) {
402 char log_path[512];
403
404 brix_sanitize_log_string(a->resolved, log_path, sizeof(log_path));
405 ngx_log_debug4(NGX_LOG_DEBUG_STREAM, c->log, 0,
406 "brix: kXR_open handle=%d path=%s mode=%s retstat=%d",
407 a->idx, log_path, a->is_write ? "wr" : "rd",
408 (int) a->want_stat);
409 }
410
411 /* Evaluate + cache the write-through policy on the handle (split out). */
412 brix_open_decide_writethrough(a);
413
414 return NGX_DECLINED;
415}

417/* Initialise the freshly-opened handle's bookkeeping fields from the fd + stat:
418 * fd/direction/from_cache/type/device/inode/size/read-ahead state, the inline
419 * read/write codec slots (regular files only), the write-through fields (cleared
420 * to a clean state), and the kXR_recoverWrts journal (armed for a recover-writes
421 * write open, else zeroed). Leaves the dashboard_slot at -1 for the monitor. */

445 * the open opaque (0 = none). Honour it only for a regular file, and store it
446 * in the direction-appropriate slot: read_codec for a read open (W4, compress

471 if (is_write && wrts_conf->caps.recover_writes) {
472 brix_wrts_open(&ctx->files[idx]);
473 } else {
474 ctx->files[idx].wrts_enabled = 0;
475 ctx->files[idx].wrts_head = 0;
476 ctx->files[idx].wrts_count = 0;
477 ctx->files[idx].wrts_gen = 0;
478 }
479 }
480
481 ctx->files[idx].dashboard_slot = -1;
