97 * WHAT: the symptom→cause rule table. Each row maps a (probe, kXR-code) pair to a
98 * verdict + a PII-free root cause + a remediation.
99 * WHY: turns a raw server status ("kXR_NotAuthorized on write") into an actionable
100 * diagnosis ("export is read-only → enable allow_write"). The mapping is keyed on
101 * BOTH the probe and the code because the same code means different things per
102 * subsystem (NotAuthorized on read = ACL; on write = read-only/scope).
103 * HOW: dx_classify scans top-to-bottom for the first row whose probe matches (or is the
104 * NULL wildcard) and whose kxr matches (or is the DX_ANY wildcard). Unmatched codes
105 * fall back to a generic finding so we never invent guidance we are unsure of.

110/* Append a classified finding; escalate the endpoint's status to its severity. */
111void
112dx_record(doctor_ep *e, const dx_note *n)
113{
114 dx_finding *f;
115 if (e->ndx >= DOC_MAXDX) {
116 return;
117 }
118 f = &e->dx[e->ndx++];
119 snprintf(f->probe, sizeof(f->probe), "%s", n->probe);
120 f->verdict = n->verdict;
121 f->kxr = n->kxr;
122 snprintf(f->cause, sizeof(f->cause), "%s", n->cause ? n->cause : "");
123 snprintf(f->remedy, sizeof(f->remedy), "%s", n->remedy ? n->remedy : "");
124 if (n->verdict > e->status) {
125 e->status = n->verdict;
126 }
127}

130/*
131 * Classify a failed probe (st->kxr) into a finding via DX_RULES, with a graceful
132 * generic fallback for codes we have no specific rule for. A successful probe
133 * (kxr==0, return 0 → caller records DX_OK) is handled by the caller.
134 */
135void
136dx_record_status(doctor_ep *e, const char *probe, const brix_status *st)
137{
138 size_t i;
139 int kxr = st ? st->kxr : 0;
140 for (i = 0; i < sizeof(DX_RULES) / sizeof(DX_RULES[0]); i++) {
141 const dx_rule *r = &DX_RULES[i];
142 if (r->probe != NULL && strcmp(r->probe, probe) != 0) {
143 continue;
144 }
145 if (r->kxr != DX_ANY && r->kxr != kxr) {
146 continue;
147 }
148 dx_record(e, &(dx_note){ probe, r->sev, kxr, r->cause, r->remedy });
149 return;
150 }
151 {
152 /* generic fallback — name the code, give conservative guidance. */
153 char cause[160];
154 if (kxr > 0) {
155 snprintf(cause, sizeof(cause), "%s probe failed: server returned %s",
156 probe, brix_kxr_name(kxr));
157 } else {
158 /* PII: never echo st->msg — server wire text may carry a path. */
159 snprintf(cause, sizeof(cause), "%s probe failed (local/transport error)",
160 probe);
161 }
162 dx_record(e, &(dx_note){ probe, DX_FAIL, kxr, cause,
163 "inspect the server logs for this operation" });
164 }
165}

168/* Is the target a loopback host? Write-mutation probes run unconditionally only here;
169 * for any other host they additionally require the explicit --i-am-authorized gate. */
170int
171dx_is_loopback(const char *host)
172{
173 /* Exact match only — a prefix like "127." would match "127.attacker.com" and
174 * wrongly enable mutating probes on a remote host. Anything not exactly a
175 * loopback literal must pass --i-am-authorized instead. */
176 return host != NULL && (strcmp(host, "127.0.0.1") == 0
177 || strcmp(host, "localhost") == 0 || strcmp(host, "::1") == 0);
178}

181/* auth/permissions suite (--auth-suite) — differential authZ testing */

183/* base64url-encode (no padding) into out[outsz] — to assemble synthetic JWTs for
184 * negative tests. Returns 0, or -1 if the encoded form would not fit (the caller
185 * then skips the test rather than emitting a truncated token). */
186int
187dx_b64url_enc(const unsigned char *in, size_t n, char *out, size_t outsz)
188{
189 static const char T[] =
190 "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-_";
191 size_t i, o = 0;
192 if ((n + 2) / 3 * 4 + 1 > outsz) { /* encoded length + NUL must fit */
193 return -1;
194 }
195 for (i = 0; i + 3 <= n; i += 3) {
196 uint32_t v = ((uint32_t) in[i] << 16) | ((uint32_t) in[i+1] << 8) | in[i+2];
197 out[o++] = T[(v >> 18) & 63]; out[o++] = T[(v >> 12) & 63];
198 out[o++] = T[(v >> 6) & 63]; out[o++] = T[v & 63];
199 }
200 if (n - i == 1) {
201 uint32_t v = (uint32_t) in[i] << 16;
202 out[o++] = T[(v >> 18) & 63]; out[o++] = T[(v >> 12) & 63];
203 } else if (n - i == 2) {
204 uint32_t v = ((uint32_t) in[i] << 16) | ((uint32_t) in[i+1] << 8);
205 out[o++] = T[(v >> 18) & 63]; out[o++] = T[(v >> 12) & 63];
206 out[o++] = T[(v >> 6) & 63];
207 }
208 out[o] = '\0';
209 return 0;
210}

213/* Build "<b64url(header)>.<b64url(payload)>.<sig>" into out. 0 / -1 on overflow. */
214int
215dx_make_jwt(const char *header, const char *payload, const char *sig,
216 char *out, size_t outsz)
217{
218 char h[192], p[320]; /* sized for the short fixed probe header/payload below */
219 if (dx_b64url_enc((const unsigned char *) header, strlen(header), h, sizeof(h)) != 0
220 || dx_b64url_enc((const unsigned char *) payload, strlen(payload), p, sizeof(p)) != 0) {
221 return -1;
222 }
223 if (strlen(h) + strlen(p) + strlen(sig) + 3 > outsz) {
224 return -1;
225 }
226 snprintf(out, outsz, "%s.%s.%s", h, p, sig);
227 return 0;
228}

231/*
232 * Open a scoped diagnostic connection: a copy of the base opts with the
233 * credential selection in *sel (force_anon, a specific bearer token, an
234 * optional forced auth protocol). Saves/restores $BEARER_TOKEN around the call
235 * so the credential matrix never leaks between probes. 0 on connect, -1 + *st.
236 */
237int
238dx_connect_as(const diag_args *a, const brix_url *u, const dx_cred_sel *sel,
239 brix_conn *c, brix_status *st)
240{
241 brix_opts opts = a->conn;
242 char *saved = NULL;
243 int had = 0;
244 int rc;
245
246 opts.force_anon = sel->force_anon;
247 if (sel->auth_force != NULL) {
248 opts.auth_force = sel->auth_force;
249 }
250 if (sel->token_override != NULL) {
251 const char *cur = getenv("BEARER_TOKEN");
252 if (cur != NULL) { saved = strdup(cur); had = (saved != NULL); }
253 setenv("BEARER_TOKEN", sel->token_override, 1); /* checked first in discovery */
254 }
255 brix_status_clear(st);
256 rc = brix_connect(c, u, &opts, st);
257 if (sel->token_override != NULL) {
258 if (had) { setenv("BEARER_TOKEN", saved, 1); free(saved); }
259 else { unsetenv("BEARER_TOKEN"); }
260 }
261 return rc;
262}

283/*
284 * WHAT: the scheme table for the deep-dive URL parser — one row per recognised
285 * scheme prefix with its protocol, TLS flag, and default port.
286 * WHY: replaces a per-scheme strncmp ladder; adding a scheme is one row.
287 * HOW: dx_scheme_match scans top-to-bottom, first prefix match wins — longer
288 * prefixes ("roots://") MUST precede their proper prefixes ("root://").
289 */
290typedef struct {
291 const char *prefix; /* scheme prefix including "://" */
292 size_t len; /* strlen(prefix), for strncmp + cursor advance */
293 dx_proto proto; /* protocol battery to route to */
294 int tls; /* scheme implies TLS */
295 int defport; /* default port when the URL names none */
296} dx_scheme_t;
297
298static const dx_scheme_t DX_SCHEMES[] = {
299 { "roots://", 8, DXP_ROOT, 1, 1094 },
300 { "xroots://", 9, DXP_ROOT, 1, 1094 },
301 { "root://", 7, DXP_ROOT, 0, 1094 },
302 { "xroot://", 8, DXP_ROOT, 0, 1094 },
303 { "https://", 8, DXP_HTTPS, 1, 8443 },
304 { "http://", 7, DXP_HTTP, 0, 8080 },
305 { "davs://", 7, DXP_DAVS, 1, 8443 },
306 { "dav://", 6, DXP_DAVS, 0, 8080 },
307 { "s3s://", 6, DXP_S3, 1, 443 },
308 { "s3://", 5, DXP_S3, 0, 9000 },
309 { "cms://", 6, DXP_CMS, 0, 1213 },
310};

313/*
314 * WHAT: match + consume the URL's scheme prefix against DX_SCHEMES.
315 * WHY: isolates scheme recognition so dx_url_parse stays a small pipeline.
316 * HOW: on the first prefix match, fill proto/tls/port (the scheme default)
317 * and advance *p past the prefix. 0 on match, -1 for an unknown scheme.
318 */
319static int
320dx_scheme_match(const char **p, dx_proto *proto, int *tls, int *port)
321{
322 size_t k;
323
324 for (k = 0; k < sizeof(DX_SCHEMES) / sizeof(DX_SCHEMES[0]); k++) {
325 const dx_scheme_t *s = &DX_SCHEMES[k];
326 if (strncmp(*p, s->prefix, s->len) == 0) {
327 *proto = s->proto;
328 *tls = s->tls;
329 *port = s->defport;
330 *p += s->len;
331 return 0;
332 }
333 }
334 return -1;
335}

338/*
339 * WHAT: parse a bracketed [IPv6] host literal (with optional :port) at *p.
340 * WHY: IPv6 literals embed ':' so they need their own delimiter logic.
341 * HOW: copy the bytes between the brackets into host (must be non-empty and
342 * fit hsz), advance *p past ']', and read a port if ':' follows.
343 * 0 on success, -1 on a malformed or over-long literal.
344 */
345static int
346dx_host6_parse(const char **p, char *host, size_t hsz, int *port)
347{
348 const char *rb = strchr(*p, ']');
349 size_t n;
350
351 if (rb == NULL) {
352 return -1;
353 }
354 n = (size_t) (rb - (*p + 1));
355 if (n == 0 || n >= hsz) {
356 return -1;
357 }
358 memcpy(host, *p + 1, n);
359 host[n] = '\0';
360 *p = rb + 1;
361 if (**p == ':') {
362 *port = atoi(*p + 1);
363 }
364 return 0;
365}
366
367
368/*
369 * WHAT: parse a plain host name (with optional :port) at *p, up to '/' or NUL.
370 * WHY: companion to dx_host6_parse — the non-bracketed authority form.
371 * HOW: scan to the end of the authority remembering the last ':' (port
372 * separator), copy the host bytes (non-empty, must fit hsz), read the
373 * port if present, and leave *p at the authority end. 0 / -1.
374 */
375static int
376dx_hostname_parse(const char **p, char *host, size_t hsz, int *port)
377{
378 const char *colon = NULL, *e, *hend;
379 size_t n;
380
381 for (e = *p; *e != '\0' && *e != '/'; e++) {
382 if (*e == ':') {
383 colon = e;
384 }
385 }
386 hend = colon ? colon : e;
387 n = (size_t) (hend - *p);
388 if (n == 0 || n >= hsz) {
389 return -1;
390 }
391 memcpy(host, *p, n);
392 host[n] = '\0';
393 if (colon != NULL) {
394 *port = atoi(colon + 1);
395 }
396 *p = e;
397 return 0;
398}

401/*
402 * Parse a scheme://host[:port][/path] URL for the deep-dive router. Recognizes
403 * root[s]/xroot[s], http/https, dav/davs, s3/s3s, cms. Fills *u (proto, tls,
404 * host, port — a per-scheme default if absent — and path). Returns 0, or -1 if
405 * the scheme is unknown. IPv6 literals in [..] are accepted.
406 */
407int
408dx_url_parse(const char *url, dx_url_t *u)
409{
410 const char *p = url, *slash;
411 int rc;
412
413 u->tls = 0;
414 if (dx_scheme_match(&p, &u->proto, &u->tls, &u->port) != 0) {
415 return -1;
416 }
417 rc = (*p == '[') ? dx_host6_parse(&p, u->host, sizeof(u->host), &u->port)
418 : dx_hostname_parse(&p, u->host, sizeof(u->host), &u->port);
419 if (rc != 0) {
420 return -1;
421 }
422 if (u->port <= 0 || u->port > 65535) {
423 return -1;
424 }
425 slash = strchr(p, '/');
426 snprintf(u->path, sizeof(u->path), "%s", slash ? slash : "/");
427 return 0;
428}

431/* Classify an HTTP status into an "http" finding on e. */
432void
433dx_http_status(doctor_ep *e, const char *probe, int status)
434{
435 if (status >= 200 && status < 300) {
436 dx_record(e, &(dx_note){ probe, DX_OK, status, "request succeeded", "" });
437 } else if (status == 401 || status == 403) {
438 dx_record(e, &(dx_note){ probe, DX_WARN, status,
439 "access requires authentication/authorization (401/403)",
440 "provide a credential (Bearer token / cert) if this object should be reachable" });
441 } else if (status == 404 || status == 410) {
442 dx_record(e, &(dx_note){ probe, DX_WARN, status, "object not found (404/410)",
443 "verify the path/bucket/key exists on the server" });
444 } else if (status >= 300 && status < 400) {
445 dx_record(e, &(dx_note){ probe, DX_WARN, status, "server returned a redirect (3xx)",
446 "follow the Location target; check it is intended" });
447 } else if (status >= 500) {
448 dx_record(e, &(dx_note){ probe, DX_FAIL, status, "server error (5xx) on the request",
449 "check the server logs for this operation" });
450 } else if (status == 0) {
451 dx_record(e, &(dx_note){ probe, DX_FAIL, 0, "no HTTP status parsed (malformed/partial response)",
452 "the endpoint may not be an HTTP server on this port" });
453 } else {
454 dx_record(e, &(dx_note){ probe, DX_WARN, status, "unexpected HTTP status", "" });
455 }
456}

459/* Classify an HTTP-family transport failure (connect / TLS) on e. */
460void
461dx_http_fail(doctor_ep *e, int tls, const brix_status *st)
462{
463 const char *cause = "connection setup failed";
464 const char *remedy = "check the endpoint is up and the port is correct";
465 if (tls && st->kxr == XRDC_EAUTH) {
466 cause = "TLS verification failed (cert untrusted/expired/wrong host)";
467 remedy = "fix the server certificate chain, or pass --no-verify-tls for a self-signed test endpoint";
468 } else if (st->sys_errno == ECONNREFUSED) {
469 cause = "no listener on host:port (service down or wrong port)";
470 remedy = "start the gateway / verify the port and any firewall";
471 } else if (st->sys_errno == ETIMEDOUT || st->sys_errno == EHOSTUNREACH
472 || st->sys_errno == ENETUNREACH) {
473 cause = "host/network unreachable";
474 remedy = "check routing/firewall and that the host is up";
475 }
476 doc_issue(e, DOC_RED, "%s", cause);
477 dx_record(e, &(dx_note){ tls ? "tls" : "reachability", DX_FAIL, st->kxr, cause, remedy });
478}

509/* srr + tape — HTTP/JSON consumers over the general HTTP client */
510
511/* Scalar JSON scan (flat fields; no nesting awareness — sufficient for the SRR /
512 * Tape-REST documents). Extract the string value of "key":"value" into out. 1/0. */
513int
514js_str(const char *json, const char *key, char *out, size_t osz)
515{
516 char pat[64];
517 const char *p, *v, *e;
518 if (json == NULL) { return 0; }
519 snprintf(pat, sizeof(pat), "\"%s\"", key);
520 p = strstr(json, pat);
521 if (p == NULL) { return 0; }
522 p = strchr(p + strlen(pat), ':');
523 if (p == NULL) { return 0; }
524 p++;
525 while (*p == ' ' || *p == '\t') { p++; }
526 if (*p != '"') { return 0; }
527 v = p + 1;
528 /* Find the closing quote, honouring backslash escapes: a '"' preceded by an
529 * even number of backslashes is the delimiter; an odd number means it's \". */
530 e = v;
531 while (*e != '\0') {
532 if (*e == '"') {
533 const char *bs = e;
534 int nb = 0;
535 while (bs > v && bs[-1] == '\\') { nb++; bs--; }
536 if ((nb & 1) == 0) { break; }
537 }
538 e++;
539 }
540 if (*e != '"') { return 0; }
541 {
542 size_t n = (size_t) (e - v);
543 if (n >= osz) { n = osz - 1; }
544 memcpy(out, v, n);
545 out[n] = '\0';
546 }
547 return 1;
548}

551/* Sum every numeric "key": N occurrence in the document (e.g. all shares' sizes). */
552long long
553js_sum(const char *json, const char *key)
554{
555 char pat[64];
556 const char *p;
557 long long sum = 0;
558 if (json == NULL) { return 0; }
559 snprintf(pat, sizeof(pat), "\"%s\"", key);
560 p = json;
561 while ((p = strstr(p, pat)) != NULL) {
562 const char *c = strchr(p + strlen(pat), ':');
563 p += strlen(pat);
564 if (c != NULL) {
565 const char *n = c + 1;
566 long long v;
567 while (*n == ' ' || *n == '\t') { n++; }
568 v = strtoll(n, NULL, 10);
569 /* saturating add — a diagnostic total must never wrap to nonsense. */
570 if (v > 0 && sum > LLONG_MAX - v) { sum = LLONG_MAX; }
571 else if (v < 0 && sum < LLONG_MIN - v) { sum = LLONG_MIN; }
572 else { sum += v; }
573 }
574 }
575 return sum;
576}

646/*
647 * WHAT: the xrddiag option table — every non-connection flag/option as a
648 * {name, kind, diag_args-field-offset} descriptor.
649 * WHY: replaces main()'s 19-arm strcmp ladder; adding an option is one row.
650 * HOW: dx_opt_parse strcmp-matches a row, dx_opt_apply writes the field:
651 * FLAG sets the int to 1, CLEAR to 0, INT/STR consume the next argv
652 * word (a missing value fails the match, so the caller reports the
653 * same "unknown option" error the old fall-through ladder produced).
654 */
655typedef enum {
656 DXO_FLAG, /* presence flag: int field = 1 */
657 DXO_CLEAR, /* negative flag: int field = 0 */
658 DXO_INT, /* atoi(next argv) into an int field */
659 DXO_STR, /* next argv into a const char * field */
660} dx_opt_kind;
661
662typedef struct {
663 const char *name; /* exact option spelling */
664 dx_opt_kind kind; /* how the value is taken and stored */
665 size_t off; /* offsetof(diag_args, <target field>) */
666} dx_opt_t;

672 { "--metrics-port", DXO_INT, offsetof(diag_args, metrics_port) },
673 { "--cluster-url", DXO_STR, offsetof(diag_args, cluster_url) },
674 { "--i-am-authorized", DXO_FLAG, offsetof(diag_args, authorized) },
675 { "--i-am-authorised", DXO_FLAG, offsetof(diag_args, authorized) },

691/*
692 * WHAT: apply one matched option descriptor to the args struct.
693 * WHY: keeps the field write (and value consumption) in one place so the
694 * descriptor table stays declarative.
695 * HOW: flags write 1/0 in place; INT/STR advance *i and consume argv[*i].
696 * Returns 0, or -1 when a value-taking option has no next word (the
697 * caller then reports it exactly like an unknown option — the same
698 * behavior the original fall-through ladder had).
699 */
700static int
701dx_opt_apply(diag_args *a, const dx_opt_t *o, int argc, char **argv, int *i)
702{
703 char *base = (char *) a;
704
705 if (o->kind == DXO_FLAG) {
706 *(int *) (base + o->off) = 1;
707 return 0;
708 }
709 if (o->kind == DXO_CLEAR) {
710 *(int *) (base + o->off) = 0;
711 return 0;
712 }
713 if (*i + 1 >= argc) {
714 return -1;
715 }
716 ++*i;
717 if (o->kind == DXO_INT) {
718 *(int *) (base + o->off) = atoi(argv[*i]);
719 } else {
720 *(const char **) (base + o->off) = argv[*i];
721 }
722 return 0;
723}

726/*
727 * WHAT: parse one xrddiag-specific option at argv[*i].
728 * WHY: main()'s per-option knowledge lives in DX_OPTS; this is just lookup.
729 * HOW: exact match against the table, then the one prefix-form option
730 * (--prometheus=PATH). Returns 0 on success (advancing *i past any
731 * consumed value), -1 for an unknown option or a missing value.
732 */
733static int
734dx_opt_parse(diag_args *a, int argc, char **argv, int *i)
735{
736 const char *p = argv[*i];
737 size_t k;
738
739 for (k = 0; k < sizeof(DX_OPTS) / sizeof(DX_OPTS[0]); k++) {
740 if (strcmp(p, DX_OPTS[k].name) == 0) {
741 return dx_opt_apply(a, &DX_OPTS[k], argc, argv, i);
742 }
743 }
744 if (strncmp(p, "--prometheus=", 13) == 0) {
745 a->watch_prom = 1;
746 a->prom_path = p + 13;
747 return 0;
748 }
749 return -1;
750}

753/*
754 * WHAT: collected positional (non-option) arguments — the URL list.
755 * WHY: bundles the array + count so the argv walker stays under the
756 * parameter cap while main keeps ownership of the storage.
757 */
758typedef struct {
759 const char *v[8]; /* positional args in order (URL cap) */
760 int n; /* how many of v[] are filled */
761} dx_pos_t;

764/*
765 * WHAT: walk argv[2..] splitting options from positional URLs.
766 * WHY: isolates the whole option/positional scan (including the shared
767 * connection-option parser and its --help path) out of main.
768 * HOW: option-shaped words go to brix_opts_parse_arg first, then the
769 * DX_OPTS table; everything else lands in pos (capped). Returns 0 to
770 * continue, or 1 with *rc set when main should exit immediately
771 * (--help → 0, unknown option / too many URLs → 50).
772 */
773static int
774dx_parse_argv(diag_args *a, int argc, char **argv, dx_pos_t *pos, int *rc)
775{
776 int i;
777
778 for (i = 2; i < argc; i++) {
779 const char *p = argv[i];
780 if (p[0] == '-' && p[1] != '\0' && strcmp(p, "-") != 0) {
781 int pr = brix_opts_parse_arg(&a->conn, argc, argv, &i);
782 if (pr == 2) { /* --help */
783 usage_fp(stdout, argv[0]);
784 *rc = 0;
785 return 1;
786 }
787 if (pr) {
788 continue;
789 }
790 if (dx_opt_parse(a, argc, argv, &i) != 0) {
791 fprintf(stderr, "xrddiag: unknown option '%s'\n", p);
792 usage(argv[0]);
793 *rc = 50;
794 return 1;
795 }
796 } else if (pos->n < (int) (sizeof(pos->v) / sizeof(pos->v[0]))) {
797 pos->v[pos->n++] = p;
798 } else {
799 fprintf(stderr, "xrddiag: too many arguments (max %zu URLs)\n",
800 sizeof(pos->v) / sizeof(pos->v[0]));
801 *rc = 50;
802 return 1;
803 }
804 }
805 return 0;
806}

809/*
810 * WHAT: route the multi-call personalities before normal parsing.
811 * WHY: the absorbed micro-tools stay invocable via symlinks (argv[0]) and
812 * double as xrddiag subcommands (argv[1]). The installed link names
813 * carry a -brix suffix (wait41-brix, mpxstats-brix) so the client RPM
814 * co-installs with the stock xrootd server RPMs, which own the bare
815 * names; the bare spellings remain reachable as subcommands.
816 * HOW: match argv[0]'s basename against the installed link names, then
817 * argv[1] against the subcommand aliases (shifting argv by one).
818 * Returns 1 with *rc set when a personality ran, 0 to continue as

821static int
822dx_personality_route(int argc, char **argv, int *rc)

828 /* The bare-name personalities carry a -brix suffix in the stock package
829 * (wait41-brix); the compat package spells them brix-wait41, which strips to
830 * the bare "wait41". Accept both so either install layout routes correctly. */
831 if (strcmp(base, "xrdqstats") == 0) { *rc = brix_qstats_main(argc, argv); return 1; }
832 if (strcmp(base, "wait41-brix") == 0 || strcmp(base, "wait41") == 0) { *rc = brix_wait41_main(argc, argv); return 1; }
833 if (strcmp(base, "mpxstats-brix") == 0 || strcmp(base, "mpxstats") == 0) { *rc = brix_mpxstats_main(argc, argv); return 1; }
834 if (argc >= 2) {
835 if (strcmp(argv[1], "qstats") == 0) { *rc = brix_qstats_main(argc - 1, argv + 1); return 1; }
836 if (strcmp(argv[1], "wait41") == 0) { *rc = brix_wait41_main(argc - 1, argv + 1); return 1; }
837 if (strcmp(argv[1], "mpxstats") == 0) { *rc = brix_mpxstats_main(argc - 1, argv + 1); return 1; }
838 }
839 return 0;
840}
