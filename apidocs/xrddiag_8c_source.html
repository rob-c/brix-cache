94/*
95 * WHAT: the symptom→cause rule table. Each row maps a (probe, kXR-code) pair to a
96 * verdict + a PII-free root cause + a remediation.
97 * WHY: turns a raw server status ("kXR_NotAuthorized on write") into an actionable
98 * diagnosis ("export is read-only → enable allow_write"). The mapping is keyed on
99 * BOTH the probe and the code because the same code means different things per
100 * subsystem (NotAuthorized on read = ACL; on write = read-only/scope).
101 * HOW: dx_classify scans top-to-bottom for the first row whose probe matches (or is the
102 * NULL wildcard) and whose kxr matches (or is the DX_ANY wildcard). Unmatched codes
103 * fall back to a generic finding so we never invent guidance we are unsure of.
104 */

108/* Append a classified finding; escalate the endpoint's status to its severity. */
109void
110dx_record(doctor_ep *e, const char *probe, int verdict, int kxr,
111 const char *cause, const char *remedy)
112{
113 dx_finding *f;
114 if (e->ndx >= DOC_MAXDX) {
115 return;
116 }
117 f = &e->dx[e->ndx++];
118 snprintf(f->probe, sizeof(f->probe), "%s", probe);
119 f->verdict = verdict;
120 f->kxr = kxr;
121 snprintf(f->cause, sizeof(f->cause), "%s", cause ? cause : "");
122 snprintf(f->remedy, sizeof(f->remedy), "%s", remedy ? remedy : "");
123 if (verdict > e->status) {
124 e->status = verdict;
125 }
126}

129/*
130 * Classify a failed probe (st->kxr) into a finding via DX_RULES, with a graceful
131 * generic fallback for codes we have no specific rule for. A successful probe
132 * (kxr==0, return 0 → caller records DX_OK) is handled by the caller.
133 */
134void
135dx_record_status(doctor_ep *e, const char *probe, const brix_status *st)
136{
137 size_t i;
138 int kxr = st ? st->kxr : 0;
139 for (i = 0; i < sizeof(DX_RULES) / sizeof(DX_RULES[0]); i++) {
140 const dx_rule *r = &DX_RULES[i];
141 if (r->probe != NULL && strcmp(r->probe, probe) != 0) {
142 continue;
143 }
144 if (r->kxr != DX_ANY && r->kxr != kxr) {
145 continue;
146 }
147 dx_record(e, probe, r->sev, kxr, r->cause, r->remedy);
148 return;
149 }
150 {
151 /* generic fallback — name the code, give conservative guidance. */
152 char cause[160];
153 if (kxr > 0) {
154 snprintf(cause, sizeof(cause), "%s probe failed: server returned %s",
155 probe, brix_kxr_name(kxr));
156 } else {
157 /* PII: never echo st->msg — server wire text may carry a path. */
158 snprintf(cause, sizeof(cause), "%s probe failed (local/transport error)",
159 probe);
160 }
161 dx_record(e, probe, DX_FAIL, kxr, cause,
162 "inspect the server logs for this operation");
163 }
164}

167/* Is the target a loopback host? Write-mutation probes run unconditionally only here;
168 * for any other host they additionally require the explicit --i-am-authorized gate. */
169int
170dx_is_loopback(const char *host)
171{
172 /* Exact match only — a prefix like "127." would match "127.attacker.com" and
173 * wrongly enable mutating probes on a remote host. Anything not exactly a
174 * loopback literal must pass --i-am-authorized instead. */
175 return host != NULL && (strcmp(host, "127.0.0.1") == 0
176 || strcmp(host, "localhost") == 0 || strcmp(host, "::1") == 0);
177}

182/* base64url-encode (no padding) into out[outsz] — to assemble synthetic JWTs for
183 * negative tests. Returns 0, or -1 if the encoded form would not fit (the caller
184 * then skips the test rather than emitting a truncated token). */
185int
186dx_b64url_enc(const unsigned char *in, size_t n, char *out, size_t outsz)
187{
188 static const char T[] =
189 "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-_";
190 size_t i, o = 0;
191 if ((n + 2) / 3 * 4 + 1 > outsz) { /* encoded length + NUL must fit */
192 return -1;
193 }
194 for (i = 0; i + 3 <= n; i += 3) {
195 uint32_t v = ((uint32_t) in[i] << 16) | ((uint32_t) in[i+1] << 8) | in[i+2];
196 out[o++] = T[(v >> 18) & 63]; out[o++] = T[(v >> 12) & 63];
197 out[o++] = T[(v >> 6) & 63]; out[o++] = T[v & 63];
198 }
199 if (n - i == 1) {
200 uint32_t v = (uint32_t) in[i] << 16;
201 out[o++] = T[(v >> 18) & 63]; out[o++] = T[(v >> 12) & 63];
202 } else if (n - i == 2) {
203 uint32_t v = ((uint32_t) in[i] << 16) | ((uint32_t) in[i+1] << 8);
204 out[o++] = T[(v >> 18) & 63]; out[o++] = T[(v >> 12) & 63];
205 out[o++] = T[(v >> 6) & 63];
206 }
207 out[o] = '\0';
208 return 0;
209}

212/* Build "<b64url(header)>.<b64url(payload)>.<sig>" into out. 0 / -1 on overflow. */
213int
214dx_make_jwt(const char *header, const char *payload, const char *sig,
215 char *out, size_t outsz)
216{
217 char h[192], p[320]; /* sized for the short fixed probe header/payload below */
218 if (dx_b64url_enc((const unsigned char *) header, strlen(header), h, sizeof(h)) != 0
219 || dx_b64url_enc((const unsigned char *) payload, strlen(payload), p, sizeof(p)) != 0) {
220 return -1;
221 }
222 if (strlen(h) + strlen(p) + strlen(sig) + 3 > outsz) {
223 return -1;
224 }
225 snprintf(out, outsz, "%s.%s.%s", h, p, sig);
226 return 0;
227}

230/*
231 * Open a scoped diagnostic connection: a copy of the base opts with force_anon
232 * and/or a specific bearer token (token_override, NULL = use the env as-is) and
233 * an optional forced auth protocol. Saves/restores $BEARER_TOKEN around the call
234 * so the credential matrix never leaks between probes. 0 on connect, -1 + *st.
235 */
236int
237dx_connect_as(const diag_args *a, const brix_url *u, int force_anon,
238 const char *token_override, const char *auth_force,
239 brix_conn *c, brix_status *st)
240{
241 brix_opts opts = a->conn;
242 char *saved = NULL;
243 int had = 0;
244 int rc;
245
246 opts.force_anon = force_anon;
247 if (auth_force != NULL) {
248 opts.auth_force = auth_force;
249 }
250 if (token_override != NULL) {
251 const char *cur = getenv("BEARER_TOKEN");
252 if (cur != NULL) { saved = strdup(cur); had = (saved != NULL); }
253 setenv("BEARER_TOKEN", token_override, 1); /* checked first in discovery */
254 }
255 brix_status_clear(st);
256 rc = brix_connect(c, u, &opts, st);
257 if (token_override != NULL) {
258 if (had) { setenv("BEARER_TOKEN", saved, 1); free(saved); }
259 else { unsetenv("BEARER_TOKEN"); }
260 }
261 return rc;
262}

283/*
284 * Parse a scheme://host[:port][/path] URL for the deep-dive router. Recognizes
285 * root[s]/xroot[s], http/https, dav/davs, s3/s3s, cms. Fills proto, *tls, host,
286 * *port (a per-scheme default if absent), and path. Returns 0, or -1 if the scheme
287 * is unknown. IPv6 literals in [..] are accepted.
288 */
289int
290dx_url_parse(const char *url, dx_proto *proto, int *tls, char *host, size_t hsz,
291 int *port, char *path, size_t psz)
292{
293 const char *p = url, *hoststart, *slash;
294 int defport;
295
296 *tls = 0;
297 if (strncmp(p, "roots://", 8) == 0) { *proto = DXP_ROOT; *tls = 1; defport = 1094; p += 8; }
298 else if (strncmp(p, "xroots://", 9) == 0) { *proto = DXP_ROOT; *tls = 1; defport = 1094; p += 9; }
299 else if (strncmp(p, "root://", 7) == 0) { *proto = DXP_ROOT; defport = 1094; p += 7; }
300 else if (strncmp(p, "xroot://", 8) == 0) { *proto = DXP_ROOT; defport = 1094; p += 8; }
301 else if (strncmp(p, "https://", 8) == 0) { *proto = DXP_HTTPS; *tls = 1; defport = 8443; p += 8; }
302 else if (strncmp(p, "http://", 7) == 0) { *proto = DXP_HTTP; defport = 8080; p += 7; }
303 else if (strncmp(p, "davs://", 7) == 0) { *proto = DXP_DAVS; *tls = 1; defport = 8443; p += 7; }
304 else if (strncmp(p, "dav://", 6) == 0) { *proto = DXP_DAVS; defport = 8080; p += 6; }
305 else if (strncmp(p, "s3s://", 6) == 0) { *proto = DXP_S3; *tls = 1; defport = 443; p += 6; }
306 else if (strncmp(p, "s3://", 5) == 0) { *proto = DXP_S3; defport = 9000; p += 5; }
307 else if (strncmp(p, "cms://", 6) == 0) { *proto = DXP_CMS; defport = 1213; p += 6; }
308 else { return -1; }
309
310 *port = defport;
311 hoststart = p;
312 if (*p == '[') { /* [IPv6] */
313 const char *rb = strchr(p, ']');
314 if (rb == NULL) { return -1; }
315 {
316 size_t n = (size_t) (rb - (p + 1));
317 if (n == 0 || n >= hsz) { return -1; }
318 memcpy(host, p + 1, n); host[n] = '\0';
319 }
320 p = rb + 1;
321 if (*p == ':') { *port = atoi(p + 1); }
322 } else {
323 const char *colon = NULL, *e;
324 for (e = hoststart; *e != '\0' && *e != '/'; e++) {
325 if (*e == ':') { colon = e; }
326 }
327 {
328 const char *hend = colon ? colon : e;
329 size_t n = (size_t) (hend - hoststart);
330 if (n == 0 || n >= hsz) { return -1; }
331 memcpy(host, hoststart, n); host[n] = '\0';
332 if (colon != NULL) { *port = atoi(colon + 1); }
333 }
334 }
335 if (*port <= 0 || *port > 65535) { return -1; }
336 slash = strchr(p, '/');
337 snprintf(path, psz, "%s", slash ? slash : "/");
338 return 0;
339}

342/* Classify an HTTP status into an "http" finding on e. */
343void
344dx_http_status(doctor_ep *e, const char *probe, int status)
345{
346 if (status >= 200 && status < 300) {
347 dx_record(e, probe, DX_OK, status, "request succeeded", "");
348 } else if (status == 401 || status == 403) {
349 dx_record(e, probe, DX_WARN, status,
350 "access requires authentication/authorization (401/403)",
351 "provide a credential (Bearer token / cert) if this object should be reachable");
352 } else if (status == 404 || status == 410) {
353 dx_record(e, probe, DX_WARN, status, "object not found (404/410)",
354 "verify the path/bucket/key exists on the server");
355 } else if (status >= 300 && status < 400) {
356 dx_record(e, probe, DX_WARN, status, "server returned a redirect (3xx)",
357 "follow the Location target; check it is intended");
358 } else if (status >= 500) {
359 dx_record(e, probe, DX_FAIL, status, "server error (5xx) on the request",
360 "check the server logs for this operation");
361 } else if (status == 0) {
362 dx_record(e, probe, DX_FAIL, 0, "no HTTP status parsed (malformed/partial response)",
363 "the endpoint may not be an HTTP server on this port");
364 } else {
365 dx_record(e, probe, DX_WARN, status, "unexpected HTTP status", "");
366 }
367}

370/* Classify an HTTP-family transport failure (connect / TLS) on e. */
371void
372dx_http_fail(doctor_ep *e, int tls, const brix_status *st)
373{
374 const char *cause = "connection setup failed";
375 const char *remedy = "check the endpoint is up and the port is correct";
376 if (tls && st->kxr == XRDC_EAUTH) {
377 cause = "TLS verification failed (cert untrusted/expired/wrong host)";
378 remedy = "fix the server certificate chain, or pass --no-verify-tls for a self-signed test endpoint";
379 } else if (st->sys_errno == ECONNREFUSED) {
380 cause = "no listener on host:port (service down or wrong port)";
381 remedy = "start the gateway / verify the port and any firewall";
382 } else if (st->sys_errno == ETIMEDOUT || st->sys_errno == EHOSTUNREACH
383 || st->sys_errno == ENETUNREACH) {
384 cause = "host/network unreachable";
385 remedy = "check routing/firewall and that the host is up";
386 }
387 doc_issue(e, DOC_RED, "%s", cause);
388 dx_record(e, tls ? "tls" : "reachability", DX_FAIL, st->kxr, cause, remedy);
389}

422/* srr + tape — HTTP/JSON consumers over the general HTTP client */
423
424/* Scalar JSON scan (flat fields; no nesting awareness — sufficient for the SRR /
425 * Tape-REST documents). Extract the string value of "key":"value" into out. 1/0. */
426int
427js_str(const char *json, const char *key, char *out, size_t osz)
428{
429 char pat[64];
430 const char *p, *v, *e;
431 if (json == NULL) { return 0; }
432 snprintf(pat, sizeof(pat), "\"%s\"", key);
433 p = strstr(json, pat);
434 if (p == NULL) { return 0; }
435 p = strchr(p + strlen(pat), ':');
436 if (p == NULL) { return 0; }
437 p++;
438 while (*p == ' ' || *p == '\t') { p++; }
439 if (*p != '"') { return 0; }
440 v = p + 1;
441 /* Find the closing quote, honouring backslash escapes: a '"' preceded by an
442 * even number of backslashes is the delimiter; an odd number means it's \". */
443 e = v;
444 while (*e != '\0') {
445 if (*e == '"') {
446 const char *bs = e;
447 int nb = 0;
448 while (bs > v && bs[-1] == '\\') { nb++; bs--; }
449 if ((nb & 1) == 0) { break; }
450 }
451 e++;
452 }
453 if (*e != '"') { return 0; }
454 {
455 size_t n = (size_t) (e - v);
456 if (n >= osz) { n = osz - 1; }
457 memcpy(out, v, n);
458 out[n] = '\0';
459 }
460 return 1;
461}

464/* Sum every numeric "key": N occurrence in the document (e.g. all shares' sizes). */
465long long
466js_sum(const char *json, const char *key)
467{
468 char pat[64];
469 const char *p;
470 long long sum = 0;
471 if (json == NULL) { return 0; }
472 snprintf(pat, sizeof(pat), "\"%s\"", key);
473 p = json;
474 while ((p = strstr(p, pat)) != NULL) {
475 const char *c = strchr(p + strlen(pat), ':');
476 p += strlen(pat);
477 if (c != NULL) {
478 const char *n = c + 1;
479 long long v;
480 while (*n == ' ' || *n == '\t') { n++; }
481 v = strtoll(n, NULL, 10);
482 /* saturating add — a diagnostic total must never wrap to nonsense. */
483 if (v > 0 && sum > LLONG_MAX - v) { sum = LLONG_MAX; }
484 else if (v < 0 && sum < LLONG_MIN - v) { sum = LLONG_MIN; }
485 else { sum += v; }
486 }
487 }
488 return sum;
489}
