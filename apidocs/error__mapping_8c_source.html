1/*
2 * error_mapping.c — unified errno → protocol status mapping service.
3 *
4 * WHAT: Consolidates three complementary error-mapping domains into one file:
5 * 1. POSIX errno → XRootD kXR error codes (kxr_errno.c)
6 * 2. POSIX errno → HTTP status codes (http_errno.c)
7 * 3. Namespace service status → HTTP status codes (result_mapper.c)
8 *
9 * WHY: All three modules serve the same purpose — translating filesystem errors
10 * into protocol-specific response codes. Centralising here ensures consistency,
11 * eliminates duplication between http_errno.h and result_mapper.h, and reduces
12 * the compat directory's file count by three files with zero functional change.
13 *
14 * HOW: Each domain lives in its own section with WHAT/WHY/HOW documentation blocks.
15 * No cross-dependencies between sections — each function is self-contained.
16 */
17

23 * Section 1: errno → kXR (XRootD wire protocol error codes)

26/*
27 * WHAT: Maps a single POSIX errno to the kXR error code used in wire responses.
28 *
29 * WHY: write-side opcodes (rm, chmod, mv, rmdir, truncate) each had their own
30 * inline chain; centralising here ensures consistent error codes across
31 * all handlers and makes future adjustments apply everywhere at once.
32 */
33
34uint16_t
35brix_kxr_from_errno(int err)
36{
37 switch (err) {
38 case ENOENT:
39 return kXR_NotFound;
40
41 case EACCES:
42 case EPERM:
43 case EXDEV: /* openat2 RESOLVE_BENEATH ".." path-escape */
44 case ELOOP: /* RESOLVE_BENEATH/NO_MAGICLINKS rejecting an escaping symlink */
45 return kXR_NotAuthorized;
46
47 case ENOTEMPTY:
48 case EEXIST:
49 /* Reference mapError() returns kXR_ItExists for BOTH EEXIST and
50 * ENOTEMPTY (XProtocol.hh: a non-empty directory removal reports
51 * kXR_ItExists "until the next major release", not a generic FS error).
52 * Stock `xrdfs rm/rmdir` of a populated directory returns 3018 — match. */
53 return kXR_ItExists;
54
55 case ENOTDIR:
56 /* A non-directory in a path prefix (e.g. stat "/file/under/it") — the
57 * reference maps ENOTDIR to kXR_FSError ("Unable to locate ...; not a
58 * directory"), NOT kXR_NotFile. Match it for stat/statx/rmdir parity. */
59 return kXR_FSError;
60
61 case ENOMEM:
62 return kXR_NoMemory;
63
64 case ENOSPC:
65#ifdef EDQUOT
66 case EDQUOT:
67#endif
68 return kXR_NoSpace;
69
70 case EINVAL:
71 return kXR_ArgInvalid;
72
73 case EBUSY:
74 /* A mandatory lease/lock refusal (pblock F15, cache fill lock): the
75 * honest "try again once the holder releases" code. NOT EAGAIN — on a
76 * read open EAGAIN means "nearline recall in flight" and is answered
77 * with kXR_wait, which a lock refusal must never be (the client would
78 * retry a lock that may never clear on its own schedule). */
79 return kXR_FileLocked;
80
81 case ENOTSUP: /* backend has no such op (VFS NULL-slot, object stores) */
82 case ENOSYS:
83#if defined(EOPNOTSUPP) && (EOPNOTSUPP != ENOTSUP)
84 case EOPNOTSUPP:
85#endif
86 /* Honest "this server cannot do that" instead of a generic I/O error —
87 * the reverse table (kXR_Unsupported → ENOSYS) already existed, this
88 * makes the pair symmetric. */
89 return kXR_Unsupported;
90
91 default:
92 return kXR_IOError;
93 }
94}

96/*
97 * WHAT: Maps a kXR wire error code back to a POSIX errno — the inverse of
98 * brix_kxr_from_errno(), co-located so the project's canonical errno↔kXR
99 * table (CLAUDE.md invariant) lives in ONE shared, ngx-free place rather than
100 * the forward direction here and the reverse direction in the native client.
101 * Returns a POSITIVE errno, or 0 when the code is not a recognised kXR error
102 * (the caller chooses the fallback — e.g. the client's POSIX layers negate the
103 * result for the kernel and substitute a captured sys_errno on 0).
104 *
105 * WHY: the native client's FUSE/preload layers must hand the kernel a -errno for
106 * a server kXR_error; keeping this beside errno→kXR prevents the two directions
107 * drifting out of sync as error codes are added.
108 */
109/*
110 * WHAT: static kXR-code -> POSIX-errno lookup table backing brix_errno_from_kxr().
111 *
112 * WHY: a flat data table keeps the canonical reverse mapping in one auditable
113 * place and collapses what was a long branch ladder into a trivial scan — the
114 * pairs (and the "unrecognised => 0" fallback) are the ONLY behavioural
115 * contract, unchanged from the former switch.
116 */
117typedef struct {
118 uint16_t kxr; /* XRootD wire error code */
119 int err; /* positive POSIX errno */
120} brix_kxr_errno_entry_t;
121
122static const brix_kxr_errno_entry_t brix_kxr_errno_table[] = {
123 { kXR_NotFound, ENOENT },
124 { kXR_NotAuthorized, EACCES },
125 { kXR_AuthFailed, EACCES },
126 { kXR_isDirectory, EISDIR },
127 { kXR_NotFile, EISDIR },
128 { kXR_FSError, EEXIST },
129 { kXR_ItExists, EEXIST },
130 { kXR_Conflict, EEXIST },
131 { kXR_NoSpace, ENOSPC },
132 { kXR_overQuota, EDQUOT },
133 { kXR_Unsupported, ENOSYS },
134 { kXR_fsReadOnly, EROFS },
135 { kXR_FileLocked, EAGAIN },
136 { kXR_inProgress, EINPROGRESS },
137 { kXR_ArgInvalid, EINVAL },
138 { kXR_ArgMissing, EINVAL },
139 { kXR_ArgTooLong, ENAMETOOLONG },
140 { kXR_InvalidRequest, EINVAL },
141 { kXR_FileNotOpen, EBADF },
142 { kXR_NoMemory, ENOMEM },
143 { kXR_ChkSumErr, EIO },
144 { kXR_IOError, EIO },
145 { kXR_AttrNotFound, ENODATA },
146 { kXR_TLSRequired, EACCES },
147 { kXR_Overloaded, EBUSY },
148 { kXR_noserver, EHOSTUNREACH },
149};
150
151int
152brix_errno_from_kxr(uint16_t kxr)
153{
154 size_t i;
155
156 for (i = 0; i < sizeof(brix_kxr_errno_table) / sizeof(brix_kxr_errno_table[0]); i++) {
157 if (brix_kxr_errno_table[i].kxr == kxr) {
158 return brix_kxr_errno_table[i].err;
159 }
160 }
161
162 return 0; /* not a recognised kXR error code */
163}

165/*
166 * WHAT: Maps an brix_ns_status_t to a kXR error code for stream protocol
167 * responses. BRIX_NS_IO_ERROR delegates to brix_kxr_from_errno(sys_errno).
168 */
169uint16_t
170brix_kxr_map_ns_status(brix_ns_status_t status, int sys_errno)
171{
172 switch (status) {
173 case BRIX_NS_OK: return kXR_ok;
174 case BRIX_NS_NOT_FOUND: return kXR_NotFound;
175 case BRIX_NS_DENIED: return kXR_NotAuthorized;
176 case BRIX_NS_EXISTS: return kXR_FSError;
177 case BRIX_NS_CONFLICT: return kXR_FSError;
178 case BRIX_NS_NOT_EMPTY: return kXR_ItExists; /* ENOTEMPTY → 3018 (mapError) */
179 case BRIX_NS_NO_SPACE: return kXR_NoSpace; /* ENOSPC → 3017, not NoMemory */
180 case BRIX_NS_TOO_LONG: return kXR_ArgTooLong;
181 case BRIX_NS_IO_ERROR: return brix_kxr_from_errno(sys_errno);
182 }
183 return kXR_IOError;
184}

190/*
191 * WHAT: Converts filesystem errors (ENOENT, EACCES, ENOSPC, etc.) into standard
192 * HTTP 4xx/5xx status codes. Returns 404 for not-found, 403 for permission,
193 * 507 for storage-full, 409 for conflict, 414 for path-too-long, 500 default.
194 *
195 * WHY: WebDAV and S3 modules both receive errno from POSIX syscalls but need HTTP
196 * status codes for nginx response dispatch. This centralised map ensures both
197 * modules return identical codes for the same error (fixes ENOSPC→507,
198 * ENAMETOOLONG→414 divergences documented in file comment).
199 *
200 * HOW: Direct switch-case mapping per AGENTS.md errno→HTTP table. EDQUOT grouped
201 * with ENOSPC under 507 (storage quota exceeded). Default case returns 500.
202 */
203
204int
205brix_http_errno_to_status(int err)
206{
207 switch (err) {
208 case ENOENT:
209 case ENOTDIR:
210 return 404;
211
212 case EACCES:
213 case EPERM:
214 case EROFS:
215 case EXDEV: /* openat2 RESOLVE_BENEATH path-escape via ".." out of the
216 * export root. */
217 case ELOOP: /* RESOLVE_BENEATH/RESOLVE_NO_MAGICLINKS rejecting a symlink
218 * (or magic-link) that would escape the export root. */
219 /*
220 * Both reach here now that confinement is enforced by the kernel at the
221 * op instead of by an upstream realpath() that used to fail earlier. A
222 * blocked traversal is forbidden, not a server fault — never a 500.
223 */
224 return 403;
225
226 case ENOSPC:
227#ifdef EDQUOT
228 case EDQUOT:
229#endif
230 return 507;
231
232 case EEXIST:
233 case ENOTEMPTY:
234 return 409;
235
236 case ENAMETOOLONG:
237 return 414;
238
239 default:
240 return 500;
241 }
242}
