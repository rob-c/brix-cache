4 * WHAT: GET /xrootd/api/v1/scan?mode=dump|verify|fill|compare&path=<rel>&alg=<a>

34/*
35 * WHAT: one query-arg read request — the name to fetch, the caller's output
36 * buffer + size, and a default to apply when the arg is absent/empty.
37 * WHY: scan_arg previously took these five as loose parameters (6 params with
38 * `r`), tripping the param-count gate; grouping them keeps every call site
39 * self-documenting and drops scan_arg to two params.
40 * HOW: a plain value struct; the request handle `r` stays a separate parameter
41 * because it is request context, not part of the arg specification.
42 */
43typedef struct {
44 const char *name; /* query key */
45 size_t namelen; /* length of `name` */
46 char *out; /* caller output buffer */
47 size_t outsz; /* capacity of `out` */
48 const char *dflt; /* value when arg absent/empty (NULL ⇒ leave empty) */
49} scan_arg_spec_t;
50
51/*
52 * WHAT: read query arg `spec->name` (URL-decoded) into `spec->out[spec->outsz]`.
53 * WHY: every query field the scan endpoint accepts flows through one bounded,
54 * NUL-rejecting reader so no field can overflow or smuggle an embedded NUL.
55 * HOW: absent/empty ⇒ apply `spec->dflt`; otherwise URL-unescape into the
56 * buffer. NGX_ERROR only on overflow / embedded NUL.
57 */
58static ngx_int_t
59scan_arg(ngx_http_request_t *r, const scan_arg_spec_t *spec)
60{
61 ngx_str_t raw;
62 u_char *dst, *src;
63 size_t n;
64
65 if (ngx_http_arg(r, (u_char *) spec->name, spec->namelen, &raw) != NGX_OK
66 || raw.len == 0)
67 {
68 if (spec->dflt != NULL) {
69 if (ngx_strlen(spec->dflt) >= spec->outsz) {
70 return NGX_ERROR;
71 }
72 ngx_memcpy(spec->out, spec->dflt, ngx_strlen(spec->dflt) + 1);
73 } else {
74 spec->out[0] = '\0';
75 }
76 return NGX_OK;
77 }
78 if (raw.len >= spec->outsz) {
79 return NGX_ERROR;
80 }
81 dst = (u_char *) spec->out;
82 src = raw.data;
83 ngx_unescape_uri(&dst, &src, raw.len, 0);
84 n = (size_t) (dst - (u_char *) spec->out);
85 spec->out[n] = '\0';
86 if (ngx_strlchr((u_char *) spec->out,
87 (u_char *) spec->out + n, '\0') != NULL)
88 {
89 return NGX_ERROR;
90 }
91 return NGX_OK;
92}

94/* The ?path arg, normalized relative to the scan root ("/" ⇒ "." anchor). */
95static ngx_int_t
96scan_get_path(ngx_http_request_t *r, char *out, size_t outsz)
97{
98 scan_arg_spec_t spec = { "path", 4, out, outsz, "." };
99 char *p;
100
101 if (scan_arg(r, &spec) != NGX_OK) {
102 return NGX_ERROR;
103 }
104 p = out;
105 while (*p == '/') {
106 p++;
107 }
108 if (*p == '\0') {
109 ngx_memcpy(out, ".", 2);
110 } else if (p != out) {
111 ngx_memmove(out, p, strlen(p) + 1);
112 }
113 return NGX_OK;
114}

129/* Send the assembled NDJSON body (len bytes from `body`, pool-owned). */
130static ngx_int_t
131scan_send(ngx_http_request_t *r, u_char *body, size_t len)
132{
133 ngx_buf_t *b;
134 ngx_chain_t out;
135 ngx_int_t rc;
136
137 r->headers_out.status = NGX_HTTP_OK;
138 r->headers_out.content_length_n = (off_t) len;
139 r->headers_out.content_type = (ngx_str_t) ngx_string("application/x-ndjson");
140 r->headers_out.content_type_len = r->headers_out.content_type.len;
141
142 rc = ngx_http_send_header(r);
143 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only) {
144 return rc;
145 }
146
147 b = ngx_pcalloc(r->pool, sizeof(*b));
148 if (b == NULL) {
149 return NGX_HTTP_INTERNAL_SERVER_ERROR;
150 }
151 b->pos = b->start = body;
152 b->last = b->end = body + len;
153 b->memory = 1;
154 b->last_buf = 1;
155 out.buf = b;
156 out.next = NULL;
157 return ngx_http_output_filter(r, &out);
158}

160/* health (C1): one record with the scan root's filesystem capacity. The scan
161 * endpoint sees the POSIX export root (not the SD driver), so backend is
162 * "posix"; driver/cluster facts (Ceph HEALTH_OK, OSDs) are a Phase-4 add. */
163static ngx_int_t
164scan_health(ngx_http_request_t *r, const char *root_canon)
165{
166 struct statvfs vfs;
167 char line[256];
168 u_char *body;
169 uint64_t total, freeb, used;
170 int n;
171
172 if (statvfs(root_canon, &vfs) != 0) {
173 return NGX_HTTP_INTERNAL_SERVER_ERROR;
174 }
175 total = (uint64_t) vfs.f_blocks * vfs.f_frsize;
176 freeb = (uint64_t) vfs.f_bavail * vfs.f_frsize;
177 used = total - (uint64_t) vfs.f_bfree * vfs.f_frsize;
178
179 n = brix_scan_record_health(line, sizeof(line), "posix", total, freeb, used);
180 if (n < 0) {
181 return NGX_HTTP_INTERNAL_SERVER_ERROR;
182 }
183 body = ngx_palloc(r->pool, (size_t) n + 1);
184 if (body == NULL) {
185 return NGX_HTTP_INTERNAL_SERVER_ERROR;
186 }
187 ngx_memcpy(body, line, (size_t) n);
188 body[n] = '\n';
189 return scan_send(r, body, (size_t) n + 1);
190}

234/*
235 * WHAT: read + validate the ?mode arg into `req->mode_raw`; returns an HTTP
236 * status. Does NOT map the mode string to an enum — the caller inspects
237 * `mode_raw` for the `health` point-query first.
238 * WHY: `health` must be recognised before mode→enum parsing (it is not a walk
239 * mode), matching the pre-refactor order exactly.
240 * HOW: NGX_OK ⇒ `mode_raw` is a non-empty, bounds-checked string.
241 */
242static ngx_int_t
243scan_read_mode(ngx_http_request_t *r, scan_request_t *req)
244{
245 scan_arg_spec_t spec = { "mode", 4, req->mode_raw,
246 sizeof(req->mode_raw), NULL };
247
248 ngx_memzero(&req->opts, sizeof(req->opts));
249 if (scan_arg(r, &spec) != NGX_OK || req->mode_raw[0] == '\0') {
250 return NGX_HTTP_BAD_REQUEST;
251 }
252 return NGX_OK;
253}

255/*
256 * WHAT: parse + validate the remaining query args (mode enum, alg, path,
257 * max_files) into `req`; returns an HTTP status.
258 * WHY: isolates request-shape validation from the walk so each rejection maps
259 * to one early return with a clear code.
260 * HOW: requires `req->mode_raw` already read by scan_read_mode. NGX_OK ⇒ req
261 * fully populated; any bad field ⇒ the matching 4xx. Frozen behavior:
262 * same fields, same defaults, same bounds as before.
263 */
264static ngx_int_t
265scan_parse_query(ngx_http_request_t *r,
266 ngx_http_brix_dashboard_loc_conf_t *conf,
267 scan_request_t *req)
268{
269 scan_arg_spec_t alg_spec = { "alg", 3, req->opts.alg,
270 sizeof(req->opts.alg), "adler32" };
271 char maxbuf[24];
272 scan_arg_spec_t max_spec = { "max_files", 9, maxbuf, sizeof(maxbuf), NULL };
273
274 if (brix_scan_mode_parse(req->mode_raw, &req->opts.mode) != NGX_OK) {
275 return NGX_HTTP_BAD_REQUEST;
276 }
277 if (scan_arg(r, &alg_spec) != NGX_OK) {
278 return NGX_HTTP_BAD_REQUEST;
279 }
280 if (scan_get_path(r, req->relpath, sizeof(req->relpath)) != NGX_OK) {
281 return NGX_HTTP_BAD_REQUEST;
282 }
283
284 /* max_files: per-request value lowers but never raises the operator cap. */
285 req->opts.max_files = conf->scan_max_files;
286 req->opts.max_depth = 1024; /* effectively-full recursion (small depth) */
287 if (scan_arg(r, &max_spec) == NGX_OK && maxbuf[0] != '\0')
288 {
289 ngx_uint_t rq = (ngx_uint_t) ngx_atoi((u_char *) maxbuf,
290 ngx_strlen(maxbuf));
291 if (rq != (ngx_uint_t) NGX_ERROR && rq > 0 && rq < req->opts.max_files) {
292 req->opts.max_files = rq;
293 }
294 }
295 return NGX_OK;
296}

298/*
299 * WHAT: run the scan, choosing catalog-native enumeration vs the confined POSIX
300 * walk, and fill `res` with the NDJSON buffer + summary.
301 * WHY: a catalog backend (Ceph/RADOS) answers inventory/verify through its OWN
302 * object catalog (no rootfd, no namespace walk); every other export/mode
303 * runs the confined POSIX walk. Keeping the dispatch here leaves the
304 * orchestrator agnostic to backend shape.
305 * HOW: returns the engine rc; on error `*err_code`/`err_msg` carry the kXR
306 * reason. `res->buf` may be set even on error and is freed by the caller.
307 */
308static ngx_int_t
309scan_run_walk(ngx_http_request_t *r,
310 ngx_http_brix_dashboard_loc_conf_t *conf,
311 scan_request_t *req, scan_result_t *res, scan_err_t *err)
312{
313 brix_sd_instance_t *sd;
314 int rootfd;
315 ngx_int_t rc;
316
317 sd = brix_vfs_backend_resolve(conf->scan_root_canon, r->connection->log);
318 if (req->opts.mode == BRIX_SCAN_INVENTORY && sd != NULL
319 && (brix_sd_caps(sd) & BRIX_SD_CAP_CATALOG))
320 {
321 return brix_scan_run_inventory(r->connection->log, sd, &req->opts,
322 &res->buf, &res->cap, &res->used,
323 &res->summary, &err->code,
324 err->msg, sizeof(err->msg));
325 }
326 if (req->opts.mode == BRIX_SCAN_VERIFY && sd != NULL
327 && (brix_sd_caps(sd) & BRIX_SD_CAP_CATALOG))
328 {
329 /* Verify checksums of every object the backend physically holds, reading
330 * bytes through the driver (Ceph: libradosstriper-reassembled). */
331 return brix_scan_run_verify_catalog(r->connection->log, sd, &req->opts,
332 &res->buf, &res->cap, &res->used,
333 &res->summary, &err->code,
334 err->msg, sizeof(err->msg));
335 }
336
337 rootfd = brix_beneath_open_root(conf->scan_root_canon);
338 if (rootfd < 0) {
339 return NGX_HTTP_INTERNAL_SERVER_ERROR;
340 }
341 rc = brix_scan_run(r->connection->log, rootfd, req->relpath, &req->opts,
342 &res->buf, &res->cap, &res->used, &res->summary,
343 &err->code, err->msg, sizeof(err->msg));
344 close(rootfd);
345 return rc;
346}

348/*
349 * WHAT: assemble the pool-owned response body (file records + summary line) and
350 * stream it; frees the engine's heap buffer either way.
351 * WHY: keeps all body-assembly + free bookkeeping in one place so the walk
352 * phase never has to think about the response encoding.
353 * HOW: copy `res->buf[0..used]`, append the summary record + newline, send.
354 * NDJSON output bytes are frozen.
355 */
356static ngx_int_t
357scan_render(ngx_http_request_t *r, scan_result_t *res)
358{
359 char sumline[256];
360 int sumlen;
361 u_char *body;
362
363 sumlen = brix_scan_record_summary(sumline, sizeof(sumline), &res->summary);
364 if (sumlen < 0) {
365 if (res->buf != NULL) {
366 ngx_free(res->buf);
367 }
368 return NGX_HTTP_INTERNAL_SERVER_ERROR;
369 }
370
371 body = ngx_palloc(r->pool, res->used + (size_t) sumlen + 1);
372 if (body == NULL) {
373 if (res->buf != NULL) {
374 ngx_free(res->buf);
375 }
376 return NGX_HTTP_INTERNAL_SERVER_ERROR;
377 }
378 if (res->used > 0) {
379 ngx_memcpy(body, res->buf, res->used);
380 }
381 ngx_memcpy(body + res->used, sumline, (size_t) sumlen);
382 body[res->used + (size_t) sumlen] = '\n';
383 if (res->buf != NULL) {
384 ngx_free(res->buf);
385 }
386
387 return scan_send(r, body, res->used + (size_t) sumlen + 1);
388}

390/* GET /xrootd/api/v1/scan */
391ngx_int_t
392ngx_http_brix_dashboard_scan_handler(ngx_http_request_t *r)
393{
394 ngx_http_brix_dashboard_loc_conf_t *conf;
395 scan_request_t req;
396 scan_result_t res;
397 scan_err_t err;
398 ngx_int_t rc;
399 uint64_t t0;
400
401 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
402 if (conf->scan_root_canon[0] == '\0') {
403 return NGX_HTTP_NOT_FOUND; /* feature disabled */
404 }
405 rc = ngx_http_brix_dashboard_check_auth(r, conf, 0);
406 if (rc != NGX_OK) {
407 return rc;
408 }
409 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
410 return NGX_HTTP_NOT_ALLOWED;
411 }
412 brix_http_source_offer(r);
413
414 rc = scan_read_mode(r, &req);
415 if (rc != NGX_OK) {
416 return rc;
417 }
418 /* health is a point query about the root, not a walk — recognised before
419 * mode→enum parsing (it is not a walk mode). */
420 if (strcmp(req.mode_raw, "health") == 0) {
421 return scan_health(r, conf->scan_root_canon);
422 }
423 rc = scan_parse_query(r, conf, &req);
424 if (rc != NGX_OK) {
425 return rc;
426 }
427
428 ngx_memzero(&res, sizeof(res));
429 ngx_memzero(&err, sizeof(err));
430 t0 = ngx_current_msec;
431
432 rc = scan_run_walk(r, conf, &req, &res, &err);
433 if (rc != NGX_OK) {
434 if (res.buf != NULL) {
435 ngx_free(res.buf);
436 }
437 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
438 "scan %s %s: %s", req.mode_raw, req.relpath, err.msg);
439 return scan_err_http(err.code);
440 }
441
442 res.summary.elapsed_s = (double) (ngx_current_msec - t0) / 1000.0;
443 return scan_render(r, &res);
444}
