1/*
2 * scan_http.c — admin HTTP endpoint for the storage-scan engine.
3 *
4 * WHAT: GET /xrootd/api/v1/scan?mode=dump|verify|fill|compare&path=<rel>&alg=<a>
5 * Streams one NDJSON record per object plus a final summary record.
6 * WHY: gives an admin a confined, auth-gated way to dump/verify/backfill
7 * checksums across a subtree on any backend (POSIX/pblock/Ceph) — the
8 * engine rides the VFS seam so no backend-specific code lives here.
9 * HOW: mirrors the admin file-browser (dashboard/files.c): admin-auth +
10 * openat2 RESOLVE_BENEATH confinement under brix_scan_root. v1 runs the
11 * walk synchronously and emits one buffered NDJSON body (bounded by
12 * brix_scan_max_files). Off-loading to the thread pool + chunked
13 * streaming + the byte-rate throttle is the next increment (the throttle
14 * math and ordered-emit buffer already exist in scan_throttle/scan_emit).
15 *
16 * NOTE: this is the ONLY nginx-coupled file in src/scan/.
17 */

34/* Read query arg `name` (URL-decoded) into out[outsz]; falls back to `dflt`
35 * (may be NULL ⇒ leaves out empty). NGX_ERROR only on overflow / embedded NUL. */
36static ngx_int_t
37scan_arg(ngx_http_request_t *r, const char *name, size_t namelen,
38 char *out, size_t outsz, const char *dflt)
39{
40 ngx_str_t raw;
41 u_char *dst, *src;
42 size_t n;
43
44 if (ngx_http_arg(r, (u_char *) name, namelen, &raw) != NGX_OK
45 || raw.len == 0)
46 {
47 if (dflt != NULL) {
48 if (ngx_strlen(dflt) >= outsz) {
49 return NGX_ERROR;
50 }
51 ngx_memcpy(out, dflt, ngx_strlen(dflt) + 1);
52 } else {
53 out[0] = '\0';
54 }
55 return NGX_OK;
56 }
57 if (raw.len >= outsz) {
58 return NGX_ERROR;
59 }
60 dst = (u_char *) out;
61 src = raw.data;
62 ngx_unescape_uri(&dst, &src, raw.len, 0);
63 n = (size_t) (dst - (u_char *) out);
64 out[n] = '\0';
65 if (ngx_strlchr((u_char *) out, (u_char *) out + n, '\0') != NULL) {
66 return NGX_ERROR;
67 }
68 return NGX_OK;
69}

71/* The ?path arg, normalized relative to the scan root ("/" ⇒ "." anchor). */
72static ngx_int_t
73scan_get_path(ngx_http_request_t *r, char *out, size_t outsz)
74{
75 char *p;
76
77 if (scan_arg(r, "path", 4, out, outsz, ".") != NGX_OK) {
78 return NGX_ERROR;
79 }
80 p = out;
81 while (*p == '/') {
82 p++;
83 }
84 if (*p == '\0') {
85 ngx_memcpy(out, ".", 2);
86 } else if (p != out) {
87 ngx_memmove(out, p, strlen(p) + 1);
88 }
89 return NGX_OK;
90}

105/* Send the assembled NDJSON body (len bytes from `body`, pool-owned). */
106static ngx_int_t
107scan_send(ngx_http_request_t *r, u_char *body, size_t len)
108{
109 ngx_buf_t *b;
110 ngx_chain_t out;
111 ngx_int_t rc;
112
113 r->headers_out.status = NGX_HTTP_OK;
114 r->headers_out.content_length_n = (off_t) len;
115 r->headers_out.content_type = (ngx_str_t) ngx_string("application/x-ndjson");
116 r->headers_out.content_type_len = r->headers_out.content_type.len;
117
118 rc = ngx_http_send_header(r);
119 if (rc == NGX_ERROR || rc > NGX_OK || r->header_only) {
120 return rc;
121 }
122
123 b = ngx_pcalloc(r->pool, sizeof(*b));
124 if (b == NULL) {
125 return NGX_HTTP_INTERNAL_SERVER_ERROR;
126 }
127 b->pos = b->start = body;
128 b->last = b->end = body + len;
129 b->memory = 1;
130 b->last_buf = 1;
131 out.buf = b;
132 out.next = NULL;
133 return ngx_http_output_filter(r, &out);
134}

136/* health (C1): one record with the scan root's filesystem capacity. The scan
137 * endpoint sees the POSIX export root (not the SD driver), so backend is
138 * "posix"; driver/cluster facts (Ceph HEALTH_OK, OSDs) are a Phase-4 add. */
139static ngx_int_t
140scan_health(ngx_http_request_t *r, const char *root_canon)
141{
142 struct statvfs vfs;
143 char line[256];
144 u_char *body;
145 uint64_t total, freeb, used;
146 int n;
147
148 if (statvfs(root_canon, &vfs) != 0) {
149 return NGX_HTTP_INTERNAL_SERVER_ERROR;
150 }
151 total = (uint64_t) vfs.f_blocks * vfs.f_frsize;
152 freeb = (uint64_t) vfs.f_bavail * vfs.f_frsize;
153 used = total - (uint64_t) vfs.f_bfree * vfs.f_frsize;
154
155 n = brix_scan_record_health(line, sizeof(line), "posix", total, freeb, used);
156 if (n < 0) {
157 return NGX_HTTP_INTERNAL_SERVER_ERROR;
158 }
159 body = ngx_palloc(r->pool, (size_t) n + 1);
160 if (body == NULL) {
161 return NGX_HTTP_INTERNAL_SERVER_ERROR;
162 }
163 ngx_memcpy(body, line, (size_t) n);
164 body[n] = '\n';
165 return scan_send(r, body, (size_t) n + 1);
166}

168/* GET /xrootd/api/v1/scan */
169ngx_int_t
170ngx_http_brix_dashboard_scan_handler(ngx_http_request_t *r)
171{
172 ngx_http_brix_dashboard_loc_conf_t *conf;
173 brix_scan_opts_t opts;
174 brix_scan_summary_t summary;
175 char relpath[PATH_MAX];
176 char modebuf[16];
177 char maxbuf[24];
178 int rootfd;
179 brix_sd_instance_t *sd;
180 u_char *buf = NULL;
181 size_t cap = 0, used = 0;
182 u_char *body;
183 char sumline[256];
184 int sumlen;
185 uint16_t err_code = 0;
186 char err_msg[160] = "";
187 ngx_int_t rc;
188 uint64_t t0;
189
190 conf = ngx_http_get_module_loc_conf(r, ngx_http_brix_dashboard_module);
191 if (conf->scan_root_canon[0] == '\0') {
192 return NGX_HTTP_NOT_FOUND; /* feature disabled */
193 }
194 rc = ngx_http_brix_dashboard_check_auth(r, conf, 0);
195 if (rc != NGX_OK) {
196 return rc;
197 }
198 if (r->method != NGX_HTTP_GET && r->method != NGX_HTTP_HEAD) {
199 return NGX_HTTP_NOT_ALLOWED;
200 }
201 brix_http_source_offer(r);
202
203 ngx_memzero(&opts, sizeof(opts));
204 if (scan_arg(r, "mode", 4, modebuf, sizeof(modebuf), NULL) != NGX_OK
205 || modebuf[0] == '\0')
206 {
207 return NGX_HTTP_BAD_REQUEST;
208 }
209 /* health is a point query about the root, not a walk. */
210 if (strcmp(modebuf, "health") == 0) {
211 return scan_health(r, conf->scan_root_canon);
212 }
213 if (brix_scan_mode_parse(modebuf, &opts.mode) != NGX_OK) {
214 return NGX_HTTP_BAD_REQUEST;
215 }
216 if (scan_arg(r, "alg", 3, opts.alg, sizeof(opts.alg), "adler32") != NGX_OK) {
217 return NGX_HTTP_BAD_REQUEST;
218 }
219 if (scan_get_path(r, relpath, sizeof(relpath)) != NGX_OK) {
220 return NGX_HTTP_BAD_REQUEST;
221 }
222
223 /* max_files: per-request value lowers but never raises the operator cap. */
224 opts.max_files = conf->scan_max_files;
225 opts.max_depth = 1024; /* effectively-full recursion (export depth is small) */
226 if (scan_arg(r, "max_files", 9, maxbuf, sizeof(maxbuf), NULL) == NGX_OK
227 && maxbuf[0] != '\0')
228 {
229 ngx_uint_t req = (ngx_uint_t) ngx_atoi((u_char *) maxbuf,
230 ngx_strlen(maxbuf));
231 if (req != (ngx_uint_t) NGX_ERROR && req > 0 && req < opts.max_files) {
232 opts.max_files = req;
233 }
234 }
235
236 ngx_memzero(&summary, sizeof(summary));
237 t0 = ngx_current_msec;
238
239 /* A catalog-native backend (e.g. Ceph/RADOS) answers `inventory` by
240 * enumerating its OWN object catalog through the SD `enumerate` verb — no
241 * POSIX rootfd, no namespace walk. Every other export (and every other mode)
242 * runs the confined POSIX walk. */
243 sd = brix_vfs_backend_resolve(conf->scan_root_canon, r->connection->log);
244 if (opts.mode == BRIX_SCAN_INVENTORY && sd != NULL
245 && (brix_sd_caps(sd) & BRIX_SD_CAP_CATALOG))
246 {
247 rc = brix_scan_run_inventory(r->connection->log, sd, &opts,
248 &buf, &cap, &used, &summary,
249 &err_code, err_msg, sizeof(err_msg));
250 } else if (opts.mode == BRIX_SCAN_VERIFY && sd != NULL
251 && (brix_sd_caps(sd) & BRIX_SD_CAP_CATALOG))
252 {
253 /* Verify checksums of every object the backend physically holds, reading
254 * bytes through the driver (Ceph: libradosstriper-reassembled). */
255 rc = brix_scan_run_verify_catalog(r->connection->log, sd, &opts,
256 &buf, &cap, &used, &summary,
257 &err_code, err_msg, sizeof(err_msg));
258 } else {
259 rootfd = brix_beneath_open_root(conf->scan_root_canon);
260 if (rootfd < 0) {
261 return NGX_HTTP_INTERNAL_SERVER_ERROR;
262 }
263 rc = brix_scan_run(r->connection->log, rootfd, relpath, &opts,
264 &buf, &cap, &used, &summary,
265 &err_code, err_msg, sizeof(err_msg));
266 close(rootfd);
267 }
268
269 if (rc != NGX_OK) {
270 if (buf != NULL) {
271 ngx_free(buf);
272 }
273 ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
274 "scan %s %s: %s", modebuf, relpath, err_msg);
275 return scan_err_http(err_code);
276 }
277
278 summary.elapsed_s = (double) (ngx_current_msec - t0) / 1000.0;
279 sumlen = brix_scan_record_summary(sumline, sizeof(sumline), &summary);
280 if (sumlen < 0) {
281 if (buf != NULL) {
282 ngx_free(buf);
283 }
284 return NGX_HTTP_INTERNAL_SERVER_ERROR;
285 }
286
287 /* Assemble the pool-owned response = file records + summary line. */
288 body = ngx_palloc(r->pool, used + (size_t) sumlen + 1);
289 if (body == NULL) {
290 if (buf != NULL) {
291 ngx_free(buf);
292 }
293 return NGX_HTTP_INTERNAL_SERVER_ERROR;
294 }
295 if (used > 0) {
296 ngx_memcpy(body, buf, used);
297 }
298 ngx_memcpy(body + used, sumline, (size_t) sumlen);
299 body[used + (size_t) sumlen] = '\n';
300 if (buf != NULL) {
301 ngx_free(buf);
302 }
303
304 return scan_send(r, body, used + (size_t) sumlen + 1);
305}
