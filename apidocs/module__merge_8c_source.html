1/*
2 * module_merge.c — per-concern S3 location-merge helpers.
3 *
4 * WHAT: The four merge concerns invoked by the ngx_http_s3_merge_loc_conf
5 * orchestrator (which stays in module.c as an nginx module-context callback):
6 * preamble (unified-directive adopt + shared common merge), scalars
7 * (protocol-local flag/int/str merges), token (WLCG bearer merge + JWKS load),
8 * and export (the enabled-only export/backend/tier/credential build block),
9 * plus the export's private credential-attach sub-helper.
10 *
11 * WHY: Split out of module.c VERBATIM for file-size governance. Behavior,
12 * ordering, defaults and diagnostics are byte-frozen — the only change is that
13 * the four orchestrator-called helpers are now non-static (declared in
14 * module_internal.h); s3_export_attach_credential stays file-local static.
15 *
16 * HOW: Mirrors module.c's include block; the orchestrator in module.c calls
17 * these in order and early-returns NGX_CONF_ERROR on the first failure.
18 */

33/*
34 * s3_merge_preamble() — adopt the unified brix_* directives, then run the
35 * shared common.* preamble merge.
36 *
37 * WHAT: Pulls the location's merged values for the unified directives
38 * (brix_export, brix_cache_store, ...) out of the common module into the
39 * embedded common preamble, then applies the shared common.* merge (hard
40 * read-only enforcement + pmark) with parent-inheritance.
41 *
42 * WHY: These two steps must run first and in this order — the adopt seeds
43 * conf->common from the common module before the protocol merge applies its
44 * own defaults, and the shared merge folds parent→child on the same struct.
45 * Isolating them keeps the ordering dependency visible and the orchestrator
46 * flat. Merge order + defaults are byte-frozen.
47 *
48 * HOW: brix_http_common_adopt() copies the common-module values in place;
49 * ngx_http_brix_shared_merge() returns NGX_CONF_OK/ERROR — propagated 1:1.
50 */
51char *
52s3_merge_preamble(ngx_conf_t *cf, ngx_http_s3_loc_conf_t *prev,
53 ngx_http_s3_loc_conf_t *conf)
54{
55 /* Unified directives (brix_export, brix_cache_store, ...) live in the
56 * common module; pull the merged values for this location into our
57 * embedded preamble before protocol merge applies defaults. */
58 brix_http_common_adopt(cf, &conf->common);
59
60 /* Shared common.* preamble (incl. hard read-only enforcement + pmark). */
61 if (ngx_http_brix_shared_merge(cf, &prev->common, &conf->common, "")
62 != NGX_CONF_OK)
63 {
64 return NGX_CONF_ERROR;
65 }
66
67 return NGX_CONF_OK;
68}

70/*
71 * s3_merge_scalars() — merge the protocol-local scalar/string directive group.
72 *
73 * WHAT: Folds parent→child for the S3 flags, ints, msec, size and string
74 * directives (session-token/chunk-sig/list-cache/max-keys/mpu/zip/bucket/
75 * keys/region), plus the XrdAcc engine merge.
76 *
77 * WHY: These are independent value merges with fixed defaults and no failure
78 * path; grouping them separates the pure-merge concern from the token and
79 * export-build concerns that can fail. Defaults are byte-frozen.
80 *
81 * HOW: ngx_conf_merge_* macros apply the frozen default when a value is unset;
82 * brix_acc_http_merge_conf() folds the embedded XrdAcc conf.
83 */
84void
85s3_merge_scalars(ngx_http_s3_loc_conf_t *prev, ngx_http_s3_loc_conf_t *conf)
86{
87 ngx_conf_merge_value(conf->allow_unsigned_session_token,
88 prev->allow_unsigned_session_token, 0);
89 ngx_conf_merge_value(conf->verify_chunk_signatures,
90 prev->verify_chunk_signatures, 0);
91 ngx_conf_merge_value(conf->list_cache, prev->list_cache, 0);
92 ngx_conf_merge_msec_value(conf->list_cache_ttl, prev->list_cache_ttl,
93 10000); /* 10s default staleness bound */
94 ngx_conf_merge_value(conf->max_keys, prev->max_keys, 1000);
95 ngx_conf_merge_value(conf->mpu_max_age, prev->mpu_max_age, 0);
96 ngx_conf_merge_value(conf->zip_access, prev->zip_access, 0);
97 ngx_conf_merge_size_value(conf->zip_cd_max_bytes, prev->zip_cd_max_bytes,
98 16 * 1024 * 1024);
99 brix_acc_http_merge_conf(&conf->acc, &prev->acc);
100 ngx_conf_merge_str_value(conf->cache_root, prev->cache_root, "");
101 ngx_conf_merge_str_value(conf->bucket, prev->bucket, "");
102 ngx_conf_merge_str_value(conf->access_key, prev->access_key, "");
103 ngx_conf_merge_str_value(conf->secret_key, prev->secret_key, "");
104 ngx_conf_merge_str_value(conf->region, prev->region, "us-east-1");
105}

107/*
108 * s3_merge_token() — merge the WLCG bearer-token group and load the JWKS.
109 *
110 * WHAT: Folds parent→child for the token enable/jwks/issuer/audience/skew
111 * directives, requires a JWKS path when token auth is on, and when a path is
112 * present loads its keys and registers a pool cleanup for them.
113 *
114 * WHY: Token auth carries config-time validation and side effects (key load +
115 * cleanup registration) that the pure scalar merge does not; keeping it
116 * separate keeps each concern under the complexity gate. Error strings and
117 * defaults are byte-frozen.
118 *
119 * HOW: The merges apply frozen defaults; the required-check and key-load emit
120 * the same NGX_LOG_EMERG diagnostics as before and early-return
121 * NGX_CONF_ERROR on the first failure.
122 */
123char *
124s3_merge_token(ngx_conf_t *cf, ngx_http_s3_loc_conf_t *prev,
125 ngx_http_s3_loc_conf_t *conf)
126{
127 /* WLCG bearer-token auth merge */
128 ngx_conf_merge_value(conf->token_enable, prev->token_enable, 0);
129 ngx_conf_merge_str_value(conf->token_jwks, prev->token_jwks, "");
130 ngx_conf_merge_str_value(conf->token_issuer, prev->token_issuer, "");
131 ngx_conf_merge_str_value(conf->token_audience, prev->token_audience, "");
132 ngx_conf_merge_value(conf->token_clock_skew, prev->token_clock_skew, 60);
133
134 if (conf->token_enable && conf->token_jwks.len == 0) {
135 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
136 "brix_s3_token: brix_s3_token_jwks is required when "
137 "brix_s3_token is on");
138 return NGX_CONF_ERROR;
139 }
140
141 if (conf->token_enable && conf->token_jwks.len > 0) {
142 int key_rc;
143
144 key_rc = brix_jwks_load(cf->log,
145 (const char *) conf->token_jwks.data,
146 conf->jwks_keys, BRIX_MAX_JWKS_KEYS);
147 if (key_rc <= 0) {
148 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
149 "brix_s3_token_jwks: no usable keys in \"%V\"",
150 &conf->token_jwks);
151 return NGX_CONF_ERROR;
152 }
153 conf->jwks_key_count = key_rc;
154
155 if (brix_jwks_register_cleanup(cf->pool, conf->jwks_keys,
156 &conf->jwks_key_count) != NGX_OK)
157 {
158 return NGX_CONF_ERROR;
159 }
160 }
161
162 return NGX_CONF_OK;
163}

165/*
166 * s3_export_attach_credential() — attach a named brix_credential's secrets to
167 * the export's source backend (§14).
168 *
169 * WHAT: When the export names a brix_credential, resolves it, derives its
170 * bearer token, and hands the bearer plus any x509/CA/S3/SSS secrets to the
171 * VFS backend registered for this export root.
172 *
173 * WHY: This is the densest sub-concern of the enabled-export build — a lookup,
174 * a bearer derivation, and a fully-populated backend-cred struct. Extracting
175 * it keeps the export orchestrator readable while preserving the exact
176 * deny/error diagnostics.
177 *
178 * HOW: Copies the credential name into a bounded buffer, looks it up
179 * (missing → NGX_LOG_EMERG + NGX_CONF_ERROR), maps it through the shared
180 * brix_credential_to_backend_cred() (P80.1 — the ONE mapper), then calls
181 * brix_vfs_backend_set_credential().
182 */
183static char *
184s3_export_attach_credential(ngx_conf_t *cf, ngx_http_s3_loc_conf_t *conf)
185{
186 char cred_z[256];
187 char bearer[4096];
188 const brix_credential_t *cred;
189 brix_vfs_backend_cred_t bcred;
190
191 ngx_cpystrn((u_char *) cred_z, conf->common.storage_credential.data,
192 ngx_min(conf->common.storage_credential.len + 1,
193 sizeof(cred_z)));
194 cred = brix_credential_lookup(cred_z);
195 if (cred == NULL) {
196 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
197 "brix_s3_storage_credential: no brix_credential \"%V\"",
198 &conf->common.storage_credential);
199 return NGX_CONF_ERROR;
200 }
201 if (brix_credential_to_backend_cred(cred, bearer, sizeof(bearer),
202 &bcred, cf->log) != NGX_OK)
203 {
204 return NGX_CONF_ERROR;
205 }
206 brix_vfs_backend_set_credential(conf->common.root_canon, &bcred);
207
208 return NGX_CONF_OK;
209}

211/*
212 * s3_merge_export() — the enabled-export config build block.
213 *
214 * WHAT: For an S3-enabled location: pins the posix backend to the export tree,
215 * canonicalizes the export root, registers the temp reaper + confinement
216 * rootfd, registers the composable storage backend, attaches a named
217 * credential (if any), registers the cache/stage tiers, canonicalizes the
218 * read-through cache root, and enforces the cache-outside-export guard.
219 *
220 * WHY: This whole block only runs when common.enable is true and chains a set
221 * of config-time build steps that each canonicalize/register/guard, any of
222 * which can fail. Grouping it under one helper keeps the orchestrator flat
223 * and the step order — which is byte-frozen — in one place.
224 *
225 * HOW: Each step early-returns NGX_CONF_ERROR on failure; the credential
226 * attach is delegated when storage_credential is named; the final guard runs
227 * unconditionally (cache_root_canon empty when no cache root was set).
228 */
229char *
230s3_merge_export(ngx_conf_t *cf, ngx_http_s3_loc_conf_t *conf)
231{
232 brix_export_root_opts_t root_opts;
233
234 /* posix:<path> backend → the local export tree (composable brix_export). */
235 brix_storage_backend_posix_root(&conf->common);
236
237 root_opts.directive_name = "brix_export";
238 root_opts.allow_write = conf->common.allow_write
239 && !brix_storage_backend_is_remote(&conf->common);
240 root_opts.required = 0;
241 root_opts.canon_size = sizeof(conf->common.root_canon);
242 if (brix_prepare_export_root(cf, &conf->common.root, &root_opts,
243 conf->common.root_canon) != NGX_CONF_OK)
244 {
245 return NGX_CONF_ERROR;
246 }
247 /* SP4: reap interrupted NON-staged direct-write temps under this root. */
248 if (conf->common.root_canon[0] != '\0') {
249 brix_tmp_reap_register(conf->common.root_canon);
250 }
251
252 /* Open the persistent confinement rootfd (kernel openat2
253 * RESOLVE_BENEATH anchor); no-op when no brix_s3_root is set. */
254 if (brix_http_open_rootfd(cf, &conf->common) != NGX_CONF_OK) {
255 return NGX_CONF_ERROR;
256 }
257
258 /* Register the export's composable storage backend (phase-63): a
259 * "root://"/"http://" URL or a driver name routes every VFS op (S3 GET
260 * goes through brix_vfs_open) to the source backend; default POSIX is a
261 * no-op. Mirrors the stream/webdav config paths. */
262 if (brix_vfs_backend_config_str(cf, conf->common.root_canon,
263 &conf->common.storage_backend, conf->common.pblock_block_size,
264 BRIX_AF_AUTO)
265 != NGX_OK)
266 {
267 return NGX_CONF_ERROR;
268 }
269
270 /* §14: attach the named brix_credential's bearer to the source backend. */
271 if (conf->common.storage_credential.len > 0) {
272 if (s3_export_attach_credential(cf, conf) != NGX_CONF_OK) {
273 return NGX_CONF_ERROR;
274 }
275 }
276
277 /* Phase-64: register the composable cache/stage tiers (§4.4 mirror). */
278 if (brix_tier_register_stores(cf, &conf->common) != NGX_OK) {
279 return NGX_CONF_ERROR;
280 }
281
282 if (conf->cache_root.len > 0) {
283 brix_export_root_opts_t cache_opts;
284 cache_opts.directive_name = "brix_s3_cache_root";
285 cache_opts.allow_write = 0;
286 cache_opts.required = 0;
287 cache_opts.canon_size = sizeof(conf->cache_root_canon);
288 if (brix_prepare_export_root(cf, &conf->cache_root, &cache_opts,
289 conf->cache_root_canon) != NGX_CONF_OK)
290 {
291 return NGX_CONF_ERROR;
292 }
293 }
294
295 /* HARD config guard: the read-through cache root must live OUTSIDE the
296 * export, or cache sidecars would be exposed in the client namespace. */
297 if (brix_assert_dir_outside_export(cf, "brix_s3_cache_root",
298 conf->common.root_canon, conf->cache_root_canon) != NGX_OK)
299 {
300 return NGX_CONF_ERROR;
301 }
302
303 return NGX_CONF_OK;
304}
