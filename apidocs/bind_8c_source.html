9/*
10 * kXR_bind — attach a secondary TCP data channel to an existing session.
11 *
12 * Secondary connections are used by xrdcp for parallel data transfer. The
13 * client establishes a primary connection (handshake + kXR_login + auth),
14 * then opens additional TCP connections that skip login and send kXR_bind
15 * with the primary session's sessid.
16 *
17 * The server assigns a pathid (1–253) to the secondary. The client then
18 * tags kXR_read/kXR_write payloads with this pathid to indicate which data
19 * channel the request arrived on (allowing the server to load-balance or
20 * multiplex across channels).
21 *
22 * Bound connections are intentionally narrower than full sessions: they may
23 * read primary-published handles, but they may not open, close, write, stat,
24 * or otherwise manipulate files independently. The primary remains the
25 * authority that decides which handles exist; secondaries are data channels.
26 */
27
28static ngx_uint_t brix_next_pathid = 0;
29
30/* Handle kXR_bind — attach a secondary TCP data channel to an existing primary
31 * session: look the session up in the registry and assign a pathid (1-253; 0 is
32 * the primary). Bound connections are capability-restricted (e.g. read primary-
33 * published handles via pathid-tagged kXR_read). */
34ngx_int_t
35brix_handle_bind(brix_ctx_t *ctx, ngx_connection_t *c,
36 ngx_stream_brix_srv_conf_t *conf)
37{
38 xrdw_sessid_req_t req;
39 u_char pathid;
40 u_char *buf;
41 size_t total;
42 ngx_uint_t token_auth = 0;
43
44 (void) conf;
45
46 xrdw_sessid_req_unpack(((ClientRequestHdr *) ctx->recv.hdr_buf)->body, &req);
47
48 /*
49 * The primary connection inserts its session registry entry at the point
50 * where the session is allowed to perform file I/O: anonymous login for
51 * auth=none, or successful kXR_auth for authenticated deployments.
52 */
53 if (!brix_session_lookup(req.sessid,
54 ctx->login.dn, sizeof(ctx->login.dn),
55 ctx->login.vo_list, sizeof(ctx->login.vo_list),
56 &token_auth))
57 {
58 brix_log_access(ctx, c, "BIND", "-", "-",
59 0, kXR_NotAuthorized,
60 "sessid not found in session registry", 0);
61 return brix_send_error(ctx, c, kXR_NotAuthorized,
62 "bind sessid not recognised");
63 }
64 ctx->token.auth = (int) token_auth;
65 if (ctx->identity != NULL) {
66 if (token_auth) {
67 if (brix_identity_set_subject(ctx->identity, c->pool, ctx->login.dn,
68 BRIX_AUTHN_TOKEN) != NGX_OK
69 || brix_identity_set_cstr(c->pool, &ctx->identity->dn,
70 ctx->login.dn) != NGX_OK)
71 {
72 return NGX_ERROR;
73 }
74 } else if (brix_identity_set_dn(ctx->identity, c->pool, ctx->login.dn,
75 BRIX_AUTHN_GSI) != NGX_OK)
76 {
77 return NGX_ERROR;
78 }
79 if (brix_identity_set_vos_csv(ctx->identity, c->pool,
80 ctx->login.vo_list) != NGX_OK)
81 {
82 return NGX_ERROR;
83 }
84 }
85
86 /* Assign a path ID. Pathid 0 is reserved for the primary; we cycle 1–253. */
87 if (++brix_next_pathid > 253) {
88 brix_next_pathid = 1;
89 }
90 pathid = (u_char) brix_next_pathid;
91
92 /* Bind the session: record the primary's sessid and inherit auth state. */
93 ngx_memcpy(ctx->bound_sessid, req.sessid, BRIX_SESSION_ID_LEN);
94 ctx->is_bound = 1;
95 ctx->pathid = (int) pathid;
96 ctx->login.logged_in = 1; /* secondary skips kXR_login */
97 ctx->login.auth_done = 1; /* identity inherited from registry lookup above */
98
99 ngx_log_debug3(NGX_LOG_DEBUG_STREAM, c->log, 0,
100 "brix: kXR_bind: pathid=%d sessid=%02xd%02xd...",
101 (int) pathid,
102 (unsigned) req.sessid[0],
103 (unsigned) req.sessid[1]);
104
105 total = XRD_RESPONSE_HDR_LEN + 1;
106 BRIX_PALLOC_OR_RETURN(buf, c->pool, total, NGX_ERROR);
107
108 brix_build_resp_hdr(ctx->recv.cur_streamid, kXR_ok, 1,
109 (ServerResponseHdr *) buf);
110 buf[XRD_RESPONSE_HDR_LEN] = pathid;
111
112 brix_log_access(ctx, c, "BIND", "-", "-", 1, 0, NULL, 0);
113
114 return brix_queue_response(ctx, c, buf, total);
115}
