1/*
2 * registry.c — SHM-backed session registry for cross-worker coordination.
3 *
4 * Stores each session's metadata (sessid, DN, VO list, token_auth) in a shared-
5 * memory table so any worker can resolve a session it did not itself create —
6 * needed because kXR_bind secondaries and proxied requests may land on a
7 * different worker than the primary login. All slot access is mutex-protected
8 * (spin+yield; see shm_slots). The published-handle table lives in handles.c.
9 */

39/* Shared-memory zone init callback: lay the session table out in the zone and
40 * create its spin+yield mutex. Returns NGX_OK / NGX_ERROR. */
41ngx_int_t
42brix_session_shm_init_zone(ngx_shm_zone_t *shm_zone, void *data)
43{
44 ngx_flag_t fresh;
45 brix_session_table_t *tbl;
46
47 /*
48 * Allocate the table FROM the slab pool (never lay it over shm.addr) so
49 * nginx's ngx_unlock_mutexes() — which treats every zone's shm.addr as an
50 * ngx_slab_pool_t header on every child death — does not get clobbered and
51 * SIGSEGV the master. The helper handles fresh-alloc, reload (data != NULL),
52 * and re-attach, creates brix_session_mutex from the table's leading
53 * ngx_shmtx_sh_t lock, and publishes the table via shm_zone->data.
54 */
55 tbl = brix_shm_table_alloc(shm_zone, data,
56 sizeof(brix_session_table_t)
57 + (size_t) brix_session_registry_nslots
58 * sizeof(brix_session_entry_t),
59 &brix_session_mutex, &fresh);
60 if (tbl == NULL) {
61 return NGX_ERROR;
62 }
63
64 if (fresh) {
65 tbl->capacity = brix_session_registry_nslots;
66 }
67
68 return NGX_OK;
69}

114/* ---- Scan the session table for a sessid, tracking free and LRU slots ----
115 *
116 * WHAT: Walks every occupied slot looking for `sessid`. On a hit it refreshes
117 * that slot's last_seen to `now` and returns 1. On a miss it returns 0 and, via
118 * the out-params, reports the first free slot index (or tbl->capacity if none)
119 * and the least-recently-seen occupied slot index + its last_seen (used for
120 * reap-on-full); lru_slot is tbl->capacity when the table holds no occupied slot.
121 *
122 * WHY: Isolates the single linear pass over the SHM table so the caller stays a
123 * flat sequence of decisions. Must run with brix_session_mutex held — it reads
124 * and (on a hit) writes shared slot state.
125 *
126 * HOW:
127 * 1. Seed all three out-params to their empty-table defaults.
128 * 2. For each slot: record the first free one and skip it.
129 * 3. On a sessid match, stamp last_seen and return 1.
130 * 4. Otherwise fold the occupied slot into the running LRU minimum.
131 * 5. Return 0 if no slot matched.
132 */
133static int
134brix_session_scan(brix_session_table_t *tbl,
135 const u_char sessid[BRIX_SESSION_ID_LEN], ngx_msec_t now,
136 ngx_uint_t *free_slot_out, ngx_uint_t *lru_slot_out,
137 ngx_msec_t *lru_seen_out)
138{
139 brix_session_entry_t *e;
140 ngx_uint_t i;
141
142 *free_slot_out = tbl->capacity;
143 *lru_slot_out = tbl->capacity;
144 *lru_seen_out = 0;
145
146 for (i = 0; i < tbl->capacity; i++) {
147 e = &tbl->slots[i];
148 if (!e->in_use) {
149 if (*free_slot_out == tbl->capacity) {
150 *free_slot_out = i;
151 }
152 continue;
153 }
154 if (ngx_memcmp(e->sessid, sessid, BRIX_SESSION_ID_LEN) == 0) {
155 e->last_seen = now; /* refresh activity on re-register */
156 return 1;
157 }
158 /* Track the global-LRU occupied slot for reap-on-full (F4). */
159 if (*lru_slot_out == tbl->capacity || e->last_seen < *lru_seen_out) {
160 *lru_slot_out = i;
161 *lru_seen_out = e->last_seen;
162 }
163 }
164
165 return 0;
166}

168/* ---- Reap the LRU slot when the table is full, else count the rejection ----
169 *
170 * WHAT: Phase 27 F4 slot-exhaustion defence, invoked only when no free slot and
171 * no matching session were found. If the LRU slot is older than the minimum reap
172 * age it copies that slot's sessid into `victim`, clears the slot, publishes its
173 * index via `free_slot_out`, and returns 1. Otherwise it increments the
174 * registry-full metric and returns 0 (leaving `victim`/`free_slot_out` untouched).
175 *
176 * WHY: A slot-exhaustion attacker must not be able to permanently deny new
177 * logins; reaping the least-recently-seen aged session bounds that. Must run
178 * with brix_session_mutex held — it mutates shared slot state.
179 *
180 * HOW:
181 * 1. If an aged LRU slot exists, snapshot its sessid, zero the slot, hand its
182 * index back as the free slot, and return 1.
183 * 2. Otherwise bump session_registry_full_total (if metrics are up) and
184 * return 0.
185 */
186static int
187brix_session_reap_lru(brix_session_table_t *tbl, ngx_msec_t now,
188 ngx_uint_t lru_slot, ngx_msec_t lru_seen,
189 ngx_uint_t *free_slot_out, u_char victim[BRIX_SESSION_ID_LEN])
190{
191 ngx_brix_metrics_t *m;
192
193 if (lru_slot < tbl->capacity
194 && (now - lru_seen) >= BRIX_SESSION_REAP_MIN_AGE_MS)
195 {
196 ngx_memcpy(victim, tbl->slots[lru_slot].sessid, BRIX_SESSION_ID_LEN);
197 ngx_memzero(&tbl->slots[lru_slot], sizeof(tbl->slots[lru_slot]));
198 *free_slot_out = lru_slot;
199 return 1;
200 }
201
202 m = brix_metrics_shared();
203 if (m != NULL) {
204 (void) ngx_atomic_fetch_add(&m->session_registry_full_total, 1);
205 }
206 return 0;
207}

209/* ---- Populate a free session slot with a login's metadata ----
210 *
211 * WHAT: Writes sessid, DN, VO list and token_auth into slot `slot` and marks it
212 * in_use with last_seen = now. No return value.
213 *
214 * WHY: Keeps the field-by-field copy (with NULL-string coalescing and bounded
215 * ngx_cpystrn) in one named place off the register orchestrator. Must run with
216 * brix_session_mutex held — it writes shared slot state.
217 *
218 * HOW:
219 * 1. Copy the fixed-length sessid.
220 * 2. Bounded-copy DN and VO list, substituting "" for NULL inputs.
221 * 3. Store token_auth and last_seen, then flag the slot in_use.
222 */
223static void
224brix_session_fill_slot(brix_session_table_t *tbl, ngx_uint_t slot,
225 const u_char sessid[BRIX_SESSION_ID_LEN],
226 const char *dn, const char *vo_list, ngx_uint_t token_auth,
227 ngx_msec_t now)
228{
229 brix_session_entry_t *e = &tbl->slots[slot];
230
231 ngx_memcpy(e->sessid, sessid, BRIX_SESSION_ID_LEN);
232 ngx_cpystrn((u_char *) e->dn, (u_char *) (dn ? dn : ""), sizeof(e->dn));
233 ngx_cpystrn((u_char *) e->vo_list, (u_char *) (vo_list ? vo_list : ""),
234 sizeof(e->vo_list));
235 e->token_auth = token_auth;
236 e->last_seen = now;
237 e->in_use = 1;
238}

240/* ---- Finish an eviction after the session mutex is released ----
241 *
242 * WHAT: Increments the eviction metric and unpublishes every handle owned by the
243 * reaped `victim` session. No return value.
244 *
245 * WHY: Handle unpublish MUST happen AFTER releasing brix_session_mutex to mirror
246 * brix_session_unregister's lock order (session then handle) and avoid a lock
247 * inversion. Keeping it in its own helper makes that ordering contract explicit.
248 *
249 * HOW:
250 * 1. Bump session_evict_total (if metrics are up).
251 * 2. Unpublish all of the victim's handles via the handle table.
252 */
253static void
254brix_session_finish_eviction(const u_char victim[BRIX_SESSION_ID_LEN])
255{
256 ngx_brix_metrics_t *m = brix_metrics_shared();
257
258 if (m != NULL) {
259 (void) ngx_atomic_fetch_add(&m->session_evict_total, 1);
260 }
261 brix_session_handle_unpublish_all(victim);
262}

264/* Store a session's metadata (sessid, DN, VO list, token_auth) in the first free
265 * SHM slot at login completion; a no-op if the sessid is already present.
266 * Mutex-protected (cross-worker). */
267void
268brix_session_register(const u_char sessid[BRIX_SESSION_ID_LEN],
269 const char *dn, const char *vo_list, ngx_uint_t token_auth)
270{
271 brix_session_table_t *tbl;
272 ngx_uint_t free_slot, lru_slot;
273 ngx_msec_t now, lru_seen;
274 int found, reaped = 0;
275 u_char victim[BRIX_SESSION_ID_LEN];
276
277 tbl = session_table();
278 if (tbl == NULL) {
279 return;
280 }
281
282 now = ngx_current_msec;
283
284 ngx_shmtx_lock(&brix_session_mutex);
285
286 found = brix_session_scan(tbl, sessid, now,
287 &free_slot, &lru_slot, &lru_seen);
288
289 /* Table full and no match: try the Phase 27 F4 reap-on-full defence, which
290 * may free the LRU slot (reaped) or reject and count the attempt. */
291 if (!found && free_slot == tbl->capacity) {
292 reaped = brix_session_reap_lru(tbl, now, lru_slot, lru_seen,
293 &free_slot, victim);
294 }
295
296 if (!found && free_slot < tbl->capacity) {
297 brix_session_fill_slot(tbl, free_slot, sessid, dn, vo_list,
298 token_auth, now);
299 }
300
301 ngx_shmtx_unlock(&brix_session_mutex);
302
303 /* Unpublish the reaped victim's handles AFTER releasing the session mutex
304 * (mirrors brix_session_unregister's lock order: session then handle). */
305 if (reaped) {
306 brix_session_finish_eviction(victim);
307 }
308}

310/* Look a session up by sessid, copying its DN / VO list / token_auth out (used
311 * by kXR_bind secondaries and proxy mode). Returns 1 on hit, 0 on miss.
312 * Mutex-protected. */
313int
314brix_session_lookup(const u_char sessid[BRIX_SESSION_ID_LEN],
315 char *dn_out, size_t dn_size,
316 char *vo_out, size_t vo_size,
317 ngx_uint_t *token_auth_out)
318{
319 brix_session_table_t *tbl;
320 brix_session_entry_t *e;
321 ngx_uint_t i;
322 int found = 0;
323
324 tbl = session_table();
325 if (tbl == NULL) {
326 return 0;
327 }
328
329 ngx_shmtx_lock(&brix_session_mutex);
330
331 for (i = 0; i < tbl->capacity; i++) {
332 e = &tbl->slots[i];
333 if (!e->in_use) {
334 continue;
335 }
336 if (ngx_memcmp(e->sessid, sessid, BRIX_SESSION_ID_LEN) == 0) {
337 ngx_cpystrn((u_char *) dn_out, (u_char *) e->dn, dn_size);
338 ngx_cpystrn((u_char *) vo_out, (u_char *) e->vo_list, vo_size);
339 *token_auth_out = e->token_auth;
340 e->last_seen = ngx_current_msec; /* F4: activity keeps it off the LRU */
341 found = 1;
342 break;
343 }
344 }
345
346 ngx_shmtx_unlock(&brix_session_mutex);
347 return found;
348}
