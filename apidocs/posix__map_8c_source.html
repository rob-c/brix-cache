15void
16brix_statinfo_to_stat(const brix_statinfo *si, int allow_symlink,
17 struct stat *stbuf)
18{
19 memset(stbuf, 0, sizeof(*stbuf));
20
21 /* File-type + permission bits come from the shared stat-flags semantics, the
22 * inverse-by-spec of the server's st_mode -> kXR flags encode. The FUSE-facing
23 * link count / size policy stays here: a directory advertises nlink 2 and no
24 * size; a symlink (lstat) or regular file advertises nlink 1 and the wire
25 * size (for a symlink that is the target length, which drives readlink). */
26 stbuf->st_mode = brix_stat_mode_from_flags(si->flags, allow_symlink);
27 if (S_ISDIR(stbuf->st_mode)) {
28 stbuf->st_nlink = 2;
29 } else {
30 stbuf->st_nlink = 1;
31 stbuf->st_size = (off_t) si->size;
32 }
33 stbuf->st_mtime = (time_t) si->mtime;
34 stbuf->st_atime = stbuf->st_mtime;
35 stbuf->st_ctime = stbuf->st_mtime;
36 /* Stable inode from the server's file id so inode-tracking tools (find
37 * -samefile, rsync, tar) see a consistent identity across calls. */
38 stbuf->st_ino = (ino_t) si->id;
39 stbuf->st_blksize = 1024 * 1024; /* hint a large I/O unit for cp/dd */
40 stbuf->st_blocks = (blkcnt_t) ((stbuf->st_size + 511) / 512);
41}

51int
52brix_fattr_listxattr_xlate(const char *raw, size_t rawlen, char *list, size_t size)
53{
54 size_t total = 0, i = 0;
55
56 while (i < rawlen) {
57 const char *nm = raw + i;
58 size_t nl = strnlen(nm, rawlen - i);
59 const char *body = nm;
60 size_t blen = nl;
61 size_t need;
62
63 if (nl == 0) {
64 i += 1;
65 continue;
66 }
67 if (blen >= 2 && body[0] == 'U' && body[1] == '.') {
68 body += 2;
69 blen -= 2;
70 }
71 need = 5 + blen + 1; /* "user." + body + NUL */
72 if (size > 0) {
73 if (total + need > size) {
74 return -ERANGE;
75 }
76 memcpy(list + total, "user.", 5);
77 memcpy(list + total + 5, body, blen);
78 list[total + 5 + blen] = '\0';
79 }
80 total += need;
81 i += nl + 1; /* advance past the name and its NUL */
82 }
83 return (int) total;
84}
