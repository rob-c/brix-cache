1/*
2 * sd_ceph_dir.c — Ceph/RADOS driver: directory iteration over the flat key
3 * namespace (phase-89 §B.1, stripe-collapse listing).
4 *
5 * WHAT: The opendir/readdir/closedir vtable slots. RADOS has no directory
6 * tree, so opendir synthesizes one level of a hierarchical listing from
7 * the pool's object catalog: it drives sd_ceph_enumerate (which already
8 * collapses libradosstriper stripes to one entry per logical file and
9 * recovers export-relative logical paths), keeps only the entries under
10 * the requested directory, and collapses each to its first path
11 * component — a component with a '/' remainder becomes a synthetic
12 * subdirectory entry, emitted once. The collapsed set is stored sorted
13 * and deduplicated in the dir handle; readdir walks the snapshot.
14 *
15 * WHY: Gives kXR_dirlist / WebDAV PROPFIND depth-1 / the cache eviction scan
16 * a real listing on a rados export with zero protocol-layer changes —
17 * they all ride the VFS dir seam, which dispatches on these slots.
18 * Directories are SYNTHETIC (phase-89 ADR-1: no marker objects): a
19 * directory exists iff at least one object lives under its prefix, so
20 * opendir of an empty non-root prefix reports ENOENT, exactly the S3
21 * model.
22 *
23 * HOW: One bounded enumerate pass per opendir. The snapshot is capped
24 * (SD_CEPH_DIR_MAX_ENTRIES) and the enumeration aborts with E2BIG when
25 * the cap is hit — a truncated-but-"complete" listing is never returned
26 * (fail-closed, no silent caps). Entry names live in one malloc'd set
27 * freed by closedir (a listing is transient; the instance pool must not
28 * grow per opendir). The classification helper (sd_ceph_path_child) is
29 * pure and unit-tested standalone in sd_ceph_unittest.c.
30 */

44/* Ceiling on one directory snapshot. Exceeding it fails the opendir (E2BIG)
45 * rather than returning a silently truncated listing. */
46#define SD_CEPH_DIR_MAX_ENTRIES 65536

48/* Driver-private dir state: the sorted, deduplicated one-level snapshot. */

63static void
64sd_ceph_dir_state_free(sd_ceph_dir_state_t *ds)
65{
66 ngx_uint_t i;
67
68 if (ds->names != NULL) {
69 for (i = 0; i < ds->count; i++) {
70 free(ds->names[i]);
71 }
72 free(ds->names);
73 ds->names = NULL;
74 }
75 ds->count = 0;

78/* sd_ceph_dir_insert — insert `name` into the sorted snapshot, dropping
79 * duplicates (the same first component recurs once per object under it).
80 * Returns 0 on success/duplicate, or an errno (E2BIG at the snapshot ceiling,
81 * ENOMEM on allocation failure). */
82static int
83sd_ceph_dir_insert(sd_ceph_dir_state_t *ds, const char *name)
84{
85 ngx_uint_t lo = 0, hi = ds->count;
86 size_t n;
87 char *dup;
88
89 while (lo < hi) {
90 ngx_uint_t mid = lo + (hi - lo) / 2;
91 int cmp = strcmp(name, ds->names[mid]);
92
93 if (cmp == 0) {
94 return 0; /* duplicate — already listed */
95 }
96 if (cmp < 0) {
97 hi = mid;
98 } else {
99 lo = mid + 1;
100 }
101 }
102
103 if (ds->count >= SD_CEPH_DIR_MAX_ENTRIES) {
104 return E2BIG;
105 }
106 if (ds->count == ds->cap) {
107 ngx_uint_t ncap = (ds->cap == 0) ? 64 : ds->cap * 2;
108 char **nn = realloc(ds->names, ncap * sizeof(char *));
109
110 if (nn == NULL) {
111 return ENOMEM;
112 }
113 ds->names = nn;
114 ds->cap = ncap;
115 }
116
117 n = strlen(name) + 1;
118 dup = malloc(n);
119 if (dup == NULL) {
120 return ENOMEM;
121 }
122 memcpy(dup, name, n);
123
124 memmove(ds->names + lo + 1, ds->names + lo,
125 (ds->count - lo) * sizeof(char *));
126 ds->names[lo] = dup;
127 ds->count++;
128 return 0;
129}

131/* sd_ceph_dir_collect — per-object enumerate callback: keep one-level children
132 * of the listing directory. Orphans (path == NULL: keys outside the export's
133 * key prefix) are not listable and never leak into a listing. Returns non-zero
134 * only to abort the pass on a hard bound (recorded in ctx->err). */
135static int
136sd_ceph_dir_collect(void *ctx, const brix_sd_catalog_ent_t *ent)
137{
138 sd_ceph_dir_collect_t *c = ctx;
139 char name[sizeof(((brix_sd_dirent_t *) 0)->name)];
140 int rc;
141
142 if (ent->path == NULL) {
143 return 0;
144 }
145 if (sd_ceph_path_child(c->dir, ent->path, name, sizeof(name)) == 0) {
146 return 0; /* not under this directory */
147 }
148 rc = sd_ceph_dir_insert(c->ds, name);
149 if (rc != 0) {
150 c->err = rc;
151 return 1; /* abort — fail the opendir */
152 }
153 return 0;
154}

156brix_sd_dir_t *
157sd_ceph_opendir(brix_sd_instance_t *inst, const char *path, int *err_out)
158{
159 char norm[1024];
160 sd_ceph_dir_state_t snap;
161 sd_ceph_dir_collect_t c;
162 brix_sd_dir_t *dir;
163 sd_ceph_dir_state_t *ds;
164
165 if (sd_ceph_normalize(path, norm, sizeof(norm)) != 0) {
166 if (err_out != NULL) { *err_out = errno; }
167 return NULL;
168 }
169
170 ngx_memzero(&snap, sizeof(snap));
171 c.dir = norm;
172 c.ds = &snap;
173 c.err = 0;
174
175 if (sd_ceph_enumerate(inst, 0, sd_ceph_dir_collect, &c) != NGX_OK) {
176 if (err_out != NULL) { *err_out = errno; }
177 sd_ceph_dir_state_free(&snap);
178 return NULL;
179 }
180 if (c.err != 0) {
181 if (err_out != NULL) { *err_out = c.err; }
182 errno = c.err;
183 sd_ceph_dir_state_free(&snap);
184 return NULL;
185 }
186
187 /* Synthetic directories exist only while non-empty (ADR-1); the export
188 * root always exists. */
189 if (snap.count == 0 && !(norm[0] == '/' && norm[1] == '\0')) {
190 if (err_out != NULL) { *err_out = ENOENT; }
191 errno = ENOENT;
192 sd_ceph_dir_state_free(&snap);
193 return NULL;
194 }
195
196 dir = ngx_pcalloc(inst->pool, sizeof(*dir));
197 ds = ngx_pcalloc(inst->pool, sizeof(*ds));
198 if (dir == NULL || ds == NULL) {
199 if (err_out != NULL) { *err_out = ENOMEM; }
200 errno = ENOMEM;
201 sd_ceph_dir_state_free(&snap);
202 return NULL;
203 }
204
205 *ds = snap; /* snapshot ownership → handle */
206 dir->inst = inst;
207 dir->state = ds;
208 return dir;
209}

211ngx_int_t
212sd_ceph_readdir(brix_sd_dir_t *d, brix_sd_dirent_t *out)
213{
214 sd_ceph_dir_state_t *ds = d->state;
215 size_t n;
216
217 if (ds == NULL || ds->pos >= ds->count) {
218 return NGX_DONE;
219 }
220 n = strlen(ds->names[ds->pos]) + 1; /* insert bounded to name[256] */
221 memcpy(out->name, ds->names[ds->pos], n);
222 ds->pos++;
223 return NGX_OK;
224}
