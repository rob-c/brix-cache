1/*
2 * lifecycle.c — nginx integration glue for per-request UNIX impersonation
3 * (phase 40). See lifecycle.h for the contract.
4 *
5 * This is the ONLY impersonation file that knows about nginx config/lifecycle
6 * types. It owns one process-global settings block (there is at most one broker
7 * per nginx instance), turns the `brix_impersonation*` directives into that
8 * block, validates the chosen mode, spawns the privileged broker in the master
9 * (FRM double-fork, reparented to init), connects the worker client, and sets the
10 * broker's target principal per request. Everything is inert unless the mode is
11 * `map`; `off` and `single` add no process, socket, or capability.
12 */

57/* Duplicate a conf token into a NUL-terminated string on cf->pool. */
58static char *
59imp_dup(ngx_conf_t *cf, ngx_str_t *src, ngx_str_t *dst)
60{
61 dst->data = ngx_pnalloc(cf->pool, src->len + 1);
62 if (dst->data == NULL) {
63 return NGX_CONF_ERROR;
64 }
65 ngx_memcpy(dst->data, src->data, src->len);
66 dst->data[src->len] = '\0';
67 dst->len = src->len;
68 return NGX_CONF_OK;
69}

71char *
72brix_imp_conf_mode(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
73{
74 ngx_str_t *v = cf->args->elts;
75
76 (void) cmd;
77 (void) conf;
78 imp_settings.configured = 1;
79
80 if (ngx_strcmp(v[1].data, "off") == 0) {
81 imp_settings.mode = BRIX_IMP_OFF;
82 } else if (ngx_strcmp(v[1].data, "single") == 0) {
83 imp_settings.mode = BRIX_IMP_SINGLE;
84 } else if (ngx_strcmp(v[1].data, "map") == 0) {
85 imp_settings.mode = BRIX_IMP_MAP;
86 } else {
87 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
88 "invalid brix_impersonation mode \"%V\" "
89 "(expected off|single|map)", &v[1]);
90 return NGX_CONF_ERROR;
91 }
92 return NGX_CONF_OK;
93}

95char *
96brix_imp_conf_str(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
97{
98 ngx_str_t *v = cf->args->elts;
99 ngx_str_t *dst;
100
101 (void) conf;
102 imp_settings.configured = 1;
103
104 switch (cmd->offset) {
105 case BRIX_IMP_F_SOCKET: dst = &imp_settings.socket; break;
106 case BRIX_IMP_F_EXPORT_ROOT: dst = &imp_settings.export_root; break;
107 case BRIX_IMP_F_GRIDMAP: dst = &imp_settings.gridmap; break;
108 case BRIX_IMP_F_DEFAULT_USER: dst = &imp_settings.default_user; break;
109 case BRIX_IMP_F_SINGLE_USER: dst = &imp_settings.single_user; break;
110 case BRIX_IMP_F_BROKER_USER: dst = &imp_settings.broker_user; break;
111 case BRIX_IMP_F_FORBIDDEN_USERS: dst = &imp_settings.forbidden_users; break;
112 case BRIX_IMP_F_FORBIDDEN_GROUPS: dst = &imp_settings.forbidden_groups; break;
113 default:
114 return "has an unknown target field";
115 }
116 return imp_dup(cf, &v[1], dst);
117}

120brix_imp_conf_num(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
121{
122 ngx_str_t *v = cf->args->elts;
123 ngx_int_t n;
124
125 (void) conf;
126 imp_settings.configured = 1;
127
128 n = ngx_atoi(v[1].data, v[1].len);
129 if (n == NGX_ERROR || n < 0) {
130 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
131 "invalid number \"%V\"", &v[1]);
132 return NGX_CONF_ERROR;
133 }
134 switch (cmd->offset) {
135 case BRIX_IMP_F_MIN_UID: imp_settings.min_uid = n; break;
136 case BRIX_IMP_F_CACHE_TTL: imp_settings.cache_ttl = n; break;
137 default:
138 return "has an unknown target field";
139 }
140 return NGX_CONF_OK;
141}

144ngx_int_t
145brix_imp_validate(ngx_conf_t *cf, const char *derived_export_root)
146{
147 if (!imp_settings.configured || imp_settings.mode == BRIX_IMP_OFF) {
148 imp_settings.mode = imp_settings.configured ? imp_settings.mode
149 : BRIX_IMP_OFF;
150 return NGX_OK;
151 }
152
153 if (imp_settings.mode == BRIX_IMP_SINGLE) {
154 if (imp_settings.single_user.len == 0) {
155 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
156 "brix_impersonation single requires "
157 "brix_impersonation_user <name>");
158 return NGX_ERROR;
159 }
160 if (imp_settings.gridmap.len || imp_settings.default_user.len) {
161 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
162 "brix_impersonation single ignores gridmap/default_user "
163 "(all identities squash to \"%V\")", &imp_settings.single_user);
164 }
165 return NGX_OK;
166 }
167
168 /* mode == map */
169 if (geteuid() != 0) {
170 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
171 "brix_impersonation map requires the nginx master to run as root "
172 "(needed to spawn the privileged identity broker)");
173 return NGX_ERROR;
174 }
175 if (imp_settings.socket.len == 0) {
176 ngx_str_t def = ngx_string(IMP_DEFAULT_SOCKET);
177 if (imp_dup(cf, &def, &imp_settings.socket) != NGX_CONF_OK) {
178 return NGX_ERROR;
179 }
180 }
181 if (imp_settings.export_root.len == 0) {
182 if (derived_export_root == NULL || derived_export_root[0] == '\0') {
183 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
184 "brix_impersonation map needs an export root: set "
185 "brix_impersonation_export <path> (no data server with a "
186 "local root was found to derive it from)");
187 return NGX_ERROR;
188 }
189 {
190 ngx_str_t s;
191 s.data = (u_char *) derived_export_root;
192 s.len = ngx_strlen(derived_export_root);
193 if (imp_dup(cf, &s, &imp_settings.export_root) != NGX_CONF_OK) {
194 return NGX_ERROR;
195 }
196 }
197 }
198 return NGX_OK;
199}
