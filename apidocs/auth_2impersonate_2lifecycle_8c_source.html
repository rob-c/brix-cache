1/*
2 * lifecycle.c — nginx integration glue for per-request UNIX impersonation
3 * (phase 40). See lifecycle.h for the contract.
4 *
5 * This is the ONLY impersonation file that knows about nginx config/lifecycle
6 * types. It owns one process-global settings block (there is at most one broker
7 * per nginx instance), turns the `brix_impersonation*` directives into that
8 * block, validates the chosen mode, spawns the privileged broker in the master
9 * (FRM double-fork, reparented to init), connects the worker client, and sets the
10 * broker's target principal per request. Everything is inert unless the mode is
11 * `map`; `off` and `single` add no process, socket, or capability.
12 */

69/* Duplicate a conf token into a NUL-terminated string on cf->pool. */
70static char *
71imp_dup(ngx_conf_t *cf, ngx_str_t *src, ngx_str_t *dst)
72{
73 dst->data = ngx_pnalloc(cf->pool, src->len + 1);
74 if (dst->data == NULL) {
75 return NGX_CONF_ERROR;
76 }
77 ngx_memcpy(dst->data, src->data, src->len);
78 dst->data[src->len] = '\0';
79 dst->len = src->len;
80 return NGX_CONF_OK;
81}

100 "invalid brix_impersonation mode \"%V\" "

102 return NGX_CONF_ERROR;
103 }
104 return NGX_CONF_OK;

107char *
108brix_imp_conf_str(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
109{
110 ngx_str_t *v = cf->args->elts;
111 ngx_str_t *dst;
112
113 (void) conf;
114 imp_settings.configured = 1;
115
116 switch (cmd->offset) {
117 case BRIX_IMP_F_SOCKET: dst = &imp_settings.socket; break;
118 case BRIX_IMP_F_EXPORT_ROOT: dst = &imp_settings.export_root; break;
119 case BRIX_IMP_F_GRIDMAP: dst = &imp_settings.gridmap; break;
120 case BRIX_IMP_F_DEFAULT_USER: dst = &imp_settings.default_user; break;
121 case BRIX_IMP_F_SINGLE_USER: dst = &imp_settings.single_user; break;
122 case BRIX_IMP_F_BROKER_USER: dst = &imp_settings.broker_user; break;
123 case BRIX_IMP_F_FORBIDDEN_USERS: dst = &imp_settings.forbidden_users; break;
124 case BRIX_IMP_F_FORBIDDEN_GROUPS: dst = &imp_settings.forbidden_groups; break;
125 default:
126 return "has an unknown target field";
127 }
128 return imp_dup(cf, &v[1], dst);

174ngx_int_t
175brix_imp_validate(ngx_conf_t *cf, const char *derived_export_root)
176{
177 if (!imp_settings.configured || imp_settings.mode == BRIX_IMP_OFF) {
178 imp_settings.mode = imp_settings.configured ? imp_settings.mode
179 : BRIX_IMP_OFF;
180 return NGX_OK;
181 }
182
183 if (imp_settings.mode == BRIX_IMP_SINGLE) {
184 if (imp_settings.single_user.len == 0) {
185 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
186 "brix_impersonation single requires "
187 "brix_impersonation_user <name>");
188 return NGX_ERROR;
189 }
190 if (imp_settings.gridmap.len || imp_settings.default_user.len) {
191 ngx_conf_log_error(NGX_LOG_WARN, cf, 0,
192 "brix_impersonation single ignores gridmap/default_user "
193 "(all identities squash to \"%V\")", &imp_settings.single_user);
194 }
195 return NGX_OK;
196 }
197
198 /* mode == map */
199 if (geteuid() != 0) {
200 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
201 "brix_impersonation map requires the nginx master to run as root "
202 "(needed to spawn the privileged identity broker)");
203 return NGX_ERROR;
204 }
205 if (imp_settings.socket.len == 0) {
206 ngx_str_t def = ngx_string(IMP_DEFAULT_SOCKET);
207 if (imp_dup(cf, &def, &imp_settings.socket) != NGX_CONF_OK) {
208 return NGX_ERROR;
209 }
210 }
211 if (imp_settings.export_root.len == 0) {
212 if (derived_export_root == NULL || derived_export_root[0] == '\0') {
213 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
214 "brix_impersonation map needs an export root: set "
215 "brix_impersonation_export <path> (no data server with a "
216 "local root was found to derive it from)");
217 return NGX_ERROR;
218 }
219 {
220 ngx_str_t s;
221 s.data = (u_char *) derived_export_root;
222 s.len = ngx_strlen(derived_export_root);
223 if (imp_dup(cf, &s, &imp_settings.export_root) != NGX_CONF_OK) {
224 return NGX_ERROR;
225 }
226 }
227 }
228 return NGX_OK;
229}

232/* Bind a 0600 listening AF_UNIX socket at `path`, owned by `wuid`. */
233static int
234imp_make_listen(const char *path, uid_t wuid, ngx_log_t *log)
235{
236 struct sockaddr_un addr;
237 int lfd;
238
239 if (ngx_strlen(path) >= sizeof(addr.sun_path)) {
240 ngx_log_error(NGX_LOG_EMERG, log, 0,
241 "impersonate: socket path too long \"%s\"", path);
242 return -1;
243 }
244 lfd = socket(AF_UNIX, SOCK_STREAM | SOCK_CLOEXEC, 0);
245 if (lfd < 0) {
246 return -1;
247 }
248 unlink(path);
249 ngx_memzero(&addr, sizeof(addr));
250 addr.sun_family = AF_UNIX;
251 ngx_memcpy(addr.sun_path, path, ngx_strlen(path) + 1);
252 if (bind(lfd, (struct sockaddr *) &addr, sizeof(addr)) != 0
253 || listen(lfd, 64) != 0)
254 {
255 BRIX_DIAG_EMERG(log, ngx_errno,
256 "impersonate: cannot bind broker socket \"%s\"",
257 "the directory is not writable by the master, or a stale socket "
258 "is held by another process",
259 "ensure the socket's parent directory exists and is writable; the "
260 "OS reason is appended below",
261 path);
262 close(lfd);
263 return -1;
264 }
265 chmod(path, 0600);
266 if (wuid != (uid_t) -1) {
267 if (chown(path, wuid, (gid_t) -1) != 0) {
268 ngx_log_error(NGX_LOG_WARN, log, ngx_errno,
269 "impersonate: chown socket to uid %d failed",
270 (int) wuid);
271 }
272 }
273 return lfd;
274}

283/* Terminate a previously-spawned broker (across reload) if still alive. */
284static void
285imp_kill_stale_broker(ngx_log_t *log)
286{
287 char pf[256];
288 FILE *fp;
289 long pid = 0;
290
291 imp_pidfile(pf, sizeof(pf));
292 fp = fopen(pf, "re");
293 if (fp == NULL) {
294 return;
295 }
296 if (fscanf(fp, "%ld", &pid) == 1 && pid > 1 && kill((pid_t) pid, 0) == 0) {
297 kill((pid_t) pid, SIGTERM);
298 ngx_log_error(NGX_LOG_NOTICE, log, 0,
299 "impersonate: terminated stale broker pid %ld", pid);
300 }
301 fclose(fp);
302 unlink(pf);
303}

305/* The broker child body (post double-fork): record pid, run the serve loop. */
306static void
307imp_broker_child(int lfd, int rootfd, ngx_log_t *log)
308{
309 char pf[256];
310 FILE *fp;
311
312 setsid();
313 imp_pidfile(pf, sizeof(pf));
314 fp = fopen(pf, "we");
315 if (fp != NULL) {
316 fprintf(fp, "%ld\n", (long) getpid());
317 fclose(fp);
318 }
319 /* broker_run drops caps to {SETUID,SETGID} then serves until killed. */
320 _exit(brix_imp_broker_run(lfd, rootfd, NULL, log) == 0 ? 0 : 1);
321}

323ngx_int_t
324brix_imp_init_module(ngx_cycle_t *cycle)
325{
326 ngx_core_conf_t *ccf;
327 brix_idmap_conf_t idc;
328 sigset_t block, prev;
329 char sockbuf[256], rootbuf[1024];
330 uid_t wuid;
331 int lfd, rootfd;
332 pid_t inter;
333
334 /*
335 * Publish the config/reload fingerprint on every real config load (this is
336 * the module's single init_module hook, so it runs once per cycle in the
337 * master regardless of impersonation mode). Skipped under `nginx -t`, which
338 * only validates the config and must not bump the live generation counter.
339 */
340 if (!ngx_test_config) {
341 brix_config_version_publish(cycle);
342 }
343
344 if (imp_settings.mode != BRIX_IMP_MAP || ngx_test_config) {
345 return NGX_OK;
346 }
347 if (geteuid() != 0) {
348 BRIX_DIAG_EMERG(cycle->log, 0,
349 "impersonate: map mode requires a root master process",
350 "brix_impersonation is set to 'map' but nginx was not started "
351 "as root, so it cannot set up the privileged uid-mapping broker",
352 "start nginx as root (workers still drop to the configured user), "
353 "or change brix_impersonation away from 'map'");
354 return NGX_ERROR;
355 }
356
357 ngx_snprintf((u_char *) sockbuf, sizeof(sockbuf), "%V%Z",
358 &imp_settings.socket);
359 ngx_snprintf((u_char *) rootbuf, sizeof(rootbuf), "%V%Z",
360 &imp_settings.export_root);
361
362 ccf = (ngx_core_conf_t *) ngx_get_conf(cycle->conf_ctx, ngx_core_module);
363 wuid = (ccf && ccf->user != (uid_t) NGX_CONF_UNSET_UINT) ? ccf->user
364 : (uid_t) -1;
365
366 /*
367 * Resolve the optional non-root broker service account. When configured, the
368 * broker drops its real uid/gid to it (keeping only CAP_SETUID/CAP_SETGID), so
369 * nothing runs as root after startup. Must be a real, non-root, non-worker
370 * account. Set the broker globals BEFORE the fork so the broker inherits them.
371 */
372 brix_imp_broker_user_uid = (uid_t) -1;
373 brix_imp_broker_user_gid = (gid_t) -1;
374 if (imp_settings.broker_user.len > 0) {
375 char nm[256];
376 struct passwd *pw;
377 ngx_snprintf((u_char *) nm, sizeof(nm), "%V%Z", &imp_settings.broker_user);
378 pw = getpwnam(nm);
379 if (pw == NULL) {
380 BRIX_DIAG_EMERG(cycle->log, 0,
381 "impersonate: broker user \"%s\" does not exist",
382 "brix_impersonation_broker_user names a local account that "
383 "is not present in /etc/passwd (or NSS)",
384 "create the dedicated service account first, or correct the "
385 "name in the directive",
386 nm);
387 return NGX_ERROR;
388 }
389 if (pw->pw_uid == 0 || (wuid != (uid_t) -1 && pw->pw_uid == wuid)) {
390 BRIX_DIAG_EMERG(cycle->log, 0,
391 "impersonate: broker user \"%s\" is not a safe choice",
392 "the broker account must NOT be root and must differ from the "
393 "nginx worker user, so a compromise cannot escalate",
394 "point brix_impersonation_broker_user at a dedicated, "
395 "unprivileged account used for nothing else",
396 nm);
397 return NGX_ERROR;
398 }
399 brix_imp_broker_user_uid = pw->pw_uid;
400 brix_imp_broker_user_gid = pw->pw_gid;
401 }
402
403 imp_kill_stale_broker(cycle->log);
404
405 /* Install mapping config in THIS (master) process; the broker forks from
406 * here and inherits the parsed gridmap + policy. */
407 imp_fill_idmap_conf(&idc, wuid);
408 if (brix_idmap_init(&idc, cycle->log) != NGX_OK) {
409 BRIX_DIAG_EMERG(cycle->log, 0,
410 "impersonate: identity map failed to load",
411 "the grid-mapfile is missing, unreadable, or malformed",
412 "check brix_impersonation_gridmap points at a readable "
413 "grid-mapfile with valid \"<DN>\" <user> lines");
414 return NGX_ERROR;
415 }
416
417 lfd = imp_make_listen(sockbuf, wuid, cycle->log);
418 if (lfd < 0) {
419 return NGX_ERROR;
420 }
421 rootfd = open(rootbuf, O_PATH | O_DIRECTORY | O_CLOEXEC);
422 if (rootfd < 0) {
423 BRIX_DIAG_EMERG(cycle->log, ngx_errno,
424 "impersonate: cannot open export root \"%s\"",
425 "the export root directory is missing or unreadable by the master",
426 "create the directory and ensure the master can open it; the OS "
427 "reason is appended below",
428 rootbuf);
429 close(lfd);
430 return NGX_ERROR;
431 }
432
433 /* Gate the broker to the worker uid (defence in depth atop confinement). */
434 brix_imp_broker_allow_uid = (wuid != (uid_t) -1) ? wuid : 0;
435
436 /* Double-fork so nginx never reaps the broker (SHM-safe; FRM pattern). */
437 sigemptyset(&block);
438 sigaddset(&block, SIGCHLD);
439 sigprocmask(SIG_BLOCK, &block, &prev);
440
441 inter = fork();
442 if (inter < 0) {
443 sigprocmask(SIG_SETMASK, &prev, NULL);
444 close(lfd);
445 close(rootfd);
446 ngx_log_error(NGX_LOG_EMERG, cycle->log, ngx_errno,
447 "impersonate: broker fork failed");
448 return NGX_ERROR;
449 }
450 if (inter == 0) {
451 if (fork() == 0) {
452 imp_broker_child(lfd, rootfd, cycle->log); /* never returns */
453 }
454 _exit(0); /* intermediate -> broker reparents */
455 }
456
457 close(lfd);
458 close(rootfd);
459 while (waitpid(inter, NULL, 0) < 0 && errno == EINTR) { }
460 sigprocmask(SIG_SETMASK, &prev, NULL);
461
462 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
463 "impersonate: started privileged broker on \"%s\" "
464 "(export root \"%s\", worker uid %d)",
465 sockbuf, rootbuf, (int) wuid);
466 return NGX_OK;
467}

470/*
471 * Hyper-harden the worker: shed the privilege-escalation / DAC-bypass / identity
472 * capabilities so a compromised worker can never setfsuid/setgid or override DAC,
473 * even if nginx was granted them via file/ambient capabilities to feed the broker.
474 * Best-effort: clearing effective+permitted always succeeds; bounding-set drops
475 * need CAP_SETPCAP and are skipped (with no error) when already unprivileged. The
476 * worker is a pure broker CLIENT and needs none of these.
477 */
478static void
479imp_worker_drop_caps(ngx_log_t *log)
480{
481 struct __user_cap_header_struct hdr;
482 struct __user_cap_data_struct data[2];
483 static const int kill_caps[] = {
484 CAP_SETUID, CAP_SETGID, CAP_SETPCAP, CAP_DAC_OVERRIDE,
485 CAP_DAC_READ_SEARCH, CAP_FOWNER, CAP_CHOWN, CAP_FSETID,
486 CAP_SYS_ADMIN, CAP_SYS_PTRACE, CAP_MKNOD, CAP_SETFCAP,
487 };
488 unsigned i;
489
490 (void) prctl(PR_SET_NO_NEW_PRIVS, 1, 0, 0, 0);
491
492 /* Clear ALL effective + permitted + inheritable capabilities. */
493 ngx_memzero(&hdr, sizeof(hdr));
494 ngx_memzero(data, sizeof(data));
495 hdr.version = _LINUX_CAPABILITY_VERSION_3;
496 hdr.pid = 0;
497 if (syscall(SYS_capset, &hdr, data) != 0) {
498 if (log) ngx_log_error(NGX_LOG_WARN, log, ngx_errno,
499 "impersonate: worker capset(clear) failed "
500 "(continuing — worker is an unprivileged client)");
501 }
502 /* Best-effort bounding-set drop of the dangerous caps (needs CAP_SETPCAP). */
503 for (i = 0; i < sizeof(kill_caps) / sizeof(kill_caps[0]); i++) {
504 (void) prctl(PR_CAPBSET_DROP, kill_caps[i], 0, 0, 0);
505 }
506}

508ngx_int_t
509brix_imp_init_worker(ngx_cycle_t *cycle)
510{
511 char sockbuf[256];
512
513 if (imp_settings.mode != BRIX_IMP_MAP) {
514 return NGX_OK; /* off/single: no client */
515 }
516
517 /* Shed privileged caps first — the worker is a pure broker client. */
518 imp_worker_drop_caps(cycle->log);
519
520 ngx_snprintf((u_char *) sockbuf, sizeof(sockbuf), "%V%Z",
521 &imp_settings.socket);
522
523 if (brix_imp_client_connect(sockbuf, cycle->log) != NGX_OK) {
524 /* Broker may still be starting; the client retries lazily per op. */
525 ngx_log_error(NGX_LOG_WARN, cycle->log, 0,
526 "impersonate: broker not reachable yet at \"%s\" "
527 "(will reconnect on first op)", sockbuf);
528 }
529 return NGX_OK;
530}
