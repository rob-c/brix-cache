1/*
2 * cks_verify.c — verify a file ON DISK against its recorded checksum.
3 *
4 * WHAT: brix_cks_verify_file() recomputes a local file's checksum (using the
5 * same shared engine as the server / xrdcrc32c) and compares it to the
6 * checksum already recorded for that file, looking in every place this
7 * project records one:
8 * storage endpoint — the "user.XrdCks.<alg>" extended attribute (both
9 * our text form "<hex> <mtime> <nsec> <size>" and the stock binary
10 * XrdCksData record), with a "<path>.cks" sidecar fallback;
11 * proxy cache — the "<path>.cinfo" block-bitmap record or the
12 * "<path>.meta" sidecar (their cks_alg/cks_hex fields).
13 *
14 * WHY: An operator needs a single command to answer "is this cached/stored file
15 * still bit-for-bit what its checksum says" without going through the
16 * running server. The front-end is apps/xrdckverify.c.
17 *
18 * HOW: Collect the recorded (algorithm, hex) records for the selected sources,
19 * then recompute each distinct algorithm once over the file (brix_cksum_fd)
20 * and compare. The on-disk binary records (XrdCksData, .cinfo, .meta) are
21 * mirrored here as fixed-layout structs — kept byte-compatible with their
22 * canonical definitions in src/core/compat/integrity_info.c, src/fs/cache/cinfo.h
23 * and src/fs/cache/meta.h (same x86-64 ABI, read verbatim).
24 */

99static void
100ckv_bin_to_hex(const unsigned char *bin, int len, char *out, size_t outsz)
101{
102 static const char hx[] = "0123456789abcdef";
103 int i;
104
105 if ((size_t) (2 * len + 1) > outsz) {
106 len = (int) ((outsz - 1) / 2);
107 }
108 for (i = 0; i < len; i++) {
109 out[2 * i] = hx[(bin[i] >> 4) & 0xf];
110 out[2 * i + 1] = hx[bin[i] & 0xf];
111 }
112 out[2 * len] = '\0';
113}

115/* Decode one "user.XrdCks.<algo>" xattr value into rec.hex. The xattr is either
116 * our text form "<hex> <mtime> <nsec> <size>" or a binary XrdCksData record. */
117static int
118ckv_decode_xattr_value(const char *algo, const char *val, size_t vlen,
119 ckv_record *rec)
120{
121 snprintf(rec->src, sizeof(rec->src), "xattr");
122 snprintf(rec->algo, sizeof(rec->algo), "%.15s", algo);
123
124 if (vlen == sizeof(struct ckv_cksdata)) {
125 struct ckv_cksdata d;
126 memcpy(&d, val, sizeof(d));
127 if (d.Length <= 0 || d.Length > (char) sizeof(d.Value)) {
128 return 0;
129 }
130 ckv_bin_to_hex((const unsigned char *) d.Value, d.Length,
131 rec->hex, sizeof(rec->hex));
132 return 1;
133 }
134 /* text form: the leading token is the lowercase hex digest. */
135 {
136 char hex[XRDC_CKV_HEX_MAX];
137 if (sscanf(val, "%128s", hex) != 1 || hex[0] == '\0') {
138 return 0;
139 }
140 snprintf(rec->hex, sizeof(rec->hex), "%s", hex);
141 return 1;
142 }
143}

145/* Collect every "user.XrdCks.*" xattr on `path`. */
146static void
147ckv_collect_xattr(const char *path, ckv_record *recs, size_t max, size_t *n)
148{
149 char names[4096];
150 ssize_t ln = listxattr(path, names, sizeof(names));
151 ssize_t off;
152
153 if (ln <= 0) {
154 return;
155 }
156 for (off = 0; off < ln && *n < max; off += (ssize_t) strlen(names + off) + 1) {
157 const char *key = names + off;
158 char val[1024];
159 ssize_t vl;
160
161 if (strncmp(key, CKV_XATTR_PREFIX, sizeof(CKV_XATTR_PREFIX) - 1) != 0) {
162 continue;
163 }
164 vl = getxattr(path, key, val, sizeof(val));
165 if (vl <= 0) {
166 continue;
167 }
168 if (ckv_decode_xattr_value(key + sizeof(CKV_XATTR_PREFIX) - 1, val,
169 (size_t) vl, &recs[*n])) {
170 (*n)++;
171 }
172 }
173}

175/* Collect records from the "<path>.cks" sidecar (one line per algorithm). */
176static void
177ckv_collect_sidecar(const char *path, ckv_record *recs, size_t max, size_t *n)
178{
179 char scpath[XRDC_PATH_MAX];
180 char line[1024];
181 FILE *fp;
182
183 if ((size_t) snprintf(scpath, sizeof(scpath), "%s.cks", path) >= sizeof(scpath)) {
184 return;
185 }
186 fp = fopen(scpath, "re");
187 if (fp == NULL) {
188 return;
189 }
190 while (*n < max && fgets(line, sizeof(line), fp) != NULL) {
191 char ralgo[32], rhex[XRDC_CKV_HEX_MAX];
192 long ms, mn;
193 long long sz;
194 if (sscanf(line, "%31s %128s %ld %ld %lld", ralgo, rhex, &ms, &mn, &sz) >= 2
195 && ralgo[0] && rhex[0]) {
196 snprintf(recs[*n].src, sizeof(recs[*n].src), "cks");
197 snprintf(recs[*n].algo, sizeof(recs[*n].algo), "%.15s", ralgo);
198 snprintf(recs[*n].hex, sizeof(recs[*n].hex), "%.128s", rhex);
199 (*n)++;
200 }
201 }
202 fclose(fp);
203}

205/* Read a fixed-size record from "<path><suffix>" into buf (zeroed first). */
206static int
207ckv_read_sidecar_blob(const char *path, const char *suffix, void *buf, size_t sz)
208{
209 char p[XRDC_PATH_MAX];
210 int fd;
211 ssize_t got;
212
213 if ((size_t) snprintf(p, sizeof(p), "%s%s", path, suffix) >= sizeof(p)) {
214 return -1;
215 }
216 fd = open(p, O_RDONLY | O_CLOEXEC | O_NOFOLLOW | O_NOCTTY);
217 if (fd < 0) {
218 return -1;
219 }
220 memset(buf, 0, sz);
221 got = pread(fd, buf, sz, 0);
222 close(fd);
223 return (got >= 0) ? (int) got : -1;
224}

226/* Append a cache record (cks_alg/cks_hex) if it carries a checksum. */
227static void
228ckv_add_cache_record(const char *src, const char *cks_alg, uint8_t cks_alg_len,
229 const char *cks_hex, uint8_t cks_len, ckv_record *recs, size_t max, size_t *n)
230{
231 if (cks_len == 0 || cks_alg_len == 0 || *n >= max) {
232 return;
233 }
234 snprintf(recs[*n].src, sizeof(recs[*n].src), "%.15s", src);
235 snprintf(recs[*n].algo, sizeof(recs[*n].algo), "%.*s",
236 cks_alg_len > 15 ? 15 : (int) cks_alg_len, cks_alg);
237 snprintf(recs[*n].hex, sizeof(recs[*n].hex), "%.*s",
238 cks_len > 128 ? 128 : (int) cks_len, cks_hex);
239 (*n)++;
240}

242static void
243ckv_collect_cache(const char *path, ckv_record *recs, size_t max, size_t *n)
244{
245 struct ckv_cinfo ci;
246 struct ckv_meta mt;
247
248 if (ckv_read_sidecar_blob(path, ".cinfo", &ci, sizeof(ci)) >= (int) sizeof(ci)
249 && ci.magic == CKV_CINFO_MAGIC) {
250 ckv_add_cache_record("cinfo", ci.cks_alg, ci.cks_alg_len,
251 ci.cks_hex, ci.cks_len, recs, max, n);
252 }
253 if (ckv_read_sidecar_blob(path, ".meta", &mt, sizeof(mt)) >= (int) sizeof(mt)
254 && mt.version >= 1) {
255 ckv_add_cache_record("meta", mt.cks_alg, mt.cks_alg_len,
256 mt.cks_hex, mt.cks_len, recs, max, n);
257 }
258}

260/* public entry point */
261brix_ckv_result
262brix_cks_verify_file(const char *path, const char *want_algo,
263 brix_ckv_mode mode, brix_ckv_report *rep, brix_status *st)
264{
265 ckv_record recs[CKV_MAX_RECORDS];
266 size_t nrec = 0, i;
267 int fd;
268 int matched = 0, unsupported = 0;
269 char comp[XRDC_CK_ZCRC32 + 1][XRDC_CKV_HEX_MAX];
270 int comp_done[XRDC_CK_ZCRC32 + 1];
271
272 if (rep != NULL) {
273 memset(rep, 0, sizeof(*rep));
274 }
275 memset(comp_done, 0, sizeof(comp_done));
276
277 if (mode == XRDC_CKV_CACHE || mode == XRDC_CKV_AUTO) {
278 ckv_collect_cache(path, recs, CKV_MAX_RECORDS, &nrec);
279 }
280 if (mode == XRDC_CKV_STORAGE || mode == XRDC_CKV_AUTO) {
281 ckv_collect_xattr(path, recs, CKV_MAX_RECORDS, &nrec);
282 ckv_collect_sidecar(path, recs, CKV_MAX_RECORDS, &nrec);
283 }
284
285 if (nrec == 0) {
286 brix_status_set(st, XRDC_EUSAGE, 0,
287 "no recorded checksum found for %s", path);
288 return XRDC_CKV_NO_RECORD;
289 }
290
291 fd = open(path, O_RDONLY | O_CLOEXEC | O_NOCTTY);
292 if (fd < 0) {
293 brix_status_set(st, XRDC_ESOCK, errno, "open %s: %s",
294 path, strerror(errno));
295 return XRDC_CKV_ERROR;
296 }
297
298 for (i = 0; i < nrec; i++) {
299 brix_cksum_algo alg;
300
301 if (want_algo != NULL && strcmp(recs[i].algo, want_algo) != 0) {
302 continue;
303 }
304 if (brix_cksum_algo_parse(recs[i].algo, &alg) != 0
305 || (int) alg > XRDC_CK_ZCRC32) {
306 unsupported = 1;
307 continue; /* recorded with an engine we cannot compute */
308 }
309 if (!comp_done[alg]) {
310 if (lseek(fd, 0, SEEK_SET) < 0
311 || brix_cksum_fd(fd, alg, comp[alg], XRDC_CKV_HEX_MAX, st) != 0) {
312 close(fd);
313 return XRDC_CKV_ERROR;
314 }
315 comp_done[alg] = 1;
316 }
317
318 if (rep != NULL) {
319 snprintf(rep->source, sizeof(rep->source), "%.15s", recs[i].src);
320 snprintf(rep->algo, sizeof(rep->algo), "%.15s", recs[i].algo);
321 snprintf(rep->recorded, sizeof(rep->recorded), "%.128s", recs[i].hex);
322 snprintf(rep->computed, sizeof(rep->computed), "%.128s", comp[alg]);
323 }
324 if (strcasecmp(recs[i].hex, comp[alg]) != 0) {
325 close(fd);
326 brix_status_set(st, XRDC_EINTEGRITY, 0,
327 "%s checksum mismatch on %s: recorded %s != computed %s",
328 recs[i].algo, path, recs[i].hex, comp[alg]);
329 return XRDC_CKV_MISMATCH;
330 }
331 matched = 1;
332 }
333
334 close(fd);
335
336 if (matched) {
337 return XRDC_CKV_OK;
338 }
339 if (unsupported) {
340 brix_status_set(st, XRDC_EUSAGE, 0,
341 "recorded checksum uses an algorithm this tool cannot compute");
342 return XRDC_CKV_UNSUPPORTED;
343 }
344 brix_status_set(st, XRDC_EUSAGE, 0,
345 "no recorded %s checksum for %s",
346 want_algo ? want_algo : "matching", path);
347 return XRDC_CKV_NO_RECORD;
348}
