1/*
2 * cks_verify.c — verify a file ON DISK against its recorded checksum.
3 *
4 * WHAT: brix_cks_verify_file() recomputes a local file's checksum (using the
5 * same shared engine as the server / xrdcrc32c) and compares it to the
6 * checksum already recorded for that file, looking in every place this
7 * project records one:
8 * storage endpoint — the "user.XrdCks.<alg>" extended attribute (both
9 * our text form "<hex> <mtime> <nsec> <size>" and the stock binary
10 * XrdCksData record), with a "<path>.cks" sidecar fallback;
11 * proxy cache — the unified xmeta record ("user.xrd.cinfo" xattr,
12 * else the "<path>.cinfo" sidecar) ORIGIN section, with a legacy
13 * "<path>.meta" POD fallback for pre-xmeta caches.
14 *
15 * WHY: An operator needs a single command to answer "is this cached/stored file
16 * still bit-for-bit what its checksum says" without going through the
17 * running server. The front-end is apps/xrdckverify.c.
18 *
19 * HOW: Collect the recorded (algorithm, hex) records for the selected sources,
20 * then recompute each distinct algorithm once over the file (brix_cksum_fd)
21 * and compare. XrdCksData and the legacy .meta POD are mirrored here as
22 * fixed-layout structs (byte-compatible with src/core/compat/integrity_info.c
23 * and the legacy reader in src/fs/cache/meta.c); the xmeta cache record is
24 * parsed from its little-endian wire bytes (same walk as
25 * apps/cksum/xrdcinfo.c and src/fs/meta/xmeta_decode.c).
26 */

74 * cinfo prefix + "XCX1" TLV extension block whose ORIGIN section (type 0x0004)
75 * holds {u8 etag_len, u8 alg_len, u8 cks_len, u8 pad} + etag/alg/hex strings.

91static void
92ckv_bin_to_hex(const unsigned char *bin, int len, char *out, size_t outsz)
93{
94 static const char hx[] = "0123456789abcdef";
95 int i;
96
97 if ((size_t) (2 * len + 1) > outsz) {
98 len = (int) ((outsz - 1) / 2);
99 }
100 for (i = 0; i < len; i++) {
101 out[2 * i] = hx[(bin[i] >> 4) & 0xf];
102 out[2 * i + 1] = hx[bin[i] & 0xf];
103 }
104 out[2 * len] = '\0';
105}

107/* Decode one "user.XrdCks.<algo>" xattr value into rec.hex. The xattr is either
108 * our text form "<hex> <mtime> <nsec> <size>" or a binary XrdCksData record. */
109static int
110ckv_decode_xattr_value(const char *algo, const char *val, size_t vlen,
111 ckv_record *rec)
112{
113 snprintf(rec->src, sizeof(rec->src), "xattr");
114 snprintf(rec->algo, sizeof(rec->algo), "%.15s", algo);
115
116 if (vlen == sizeof(struct ckv_cksdata)) {
117 struct ckv_cksdata d;
118 memcpy(&d, val, sizeof(d));
119 if (d.Length <= 0 || d.Length > (char) sizeof(d.Value)) {
120 return 0;
121 }
122 ckv_bin_to_hex((const unsigned char *) d.Value, d.Length,
123 rec->hex, sizeof(rec->hex));
124 return 1;
125 }
126 /* text form: the leading token is the lowercase hex digest. */
127 {
128 char hex[XRDC_CKV_HEX_MAX];
129 if (sscanf(val, "%128s", hex) != 1 || hex[0] == '\0') {
130 return 0;
131 }
132 snprintf(rec->hex, sizeof(rec->hex), "%s", hex);
133 return 1;
134 }
135}

137/* Collect every "user.XrdCks.*" xattr on `path`. */
138static void
139ckv_collect_xattr(const char *path, ckv_record *recs, size_t max, size_t *n)
140{
141 char names[4096];
142 ssize_t ln = listxattr(path, names, sizeof(names));
143 ssize_t off;
144
145 if (ln <= 0) {
146 return;
147 }
148 for (off = 0; off < ln && *n < max; off += (ssize_t) strlen(names + off) + 1) {
149 const char *key = names + off;
150 char val[1024];
151 ssize_t vl;
152
153 if (strncmp(key, CKV_XATTR_PREFIX, sizeof(CKV_XATTR_PREFIX) - 1) != 0) {
154 continue;
155 }
156 vl = getxattr(path, key, val, sizeof(val));
157 if (vl <= 0) {
158 continue;
159 }
160 if (ckv_decode_xattr_value(key + sizeof(CKV_XATTR_PREFIX) - 1, val,
161 (size_t) vl, &recs[*n])) {
162 (*n)++;
163 }
164 }
165}

167/* Collect records from the "<path>.cks" sidecar (one line per algorithm). */
168static void
169ckv_collect_sidecar(const char *path, ckv_record *recs, size_t max, size_t *n)
170{
171 char scpath[XRDC_PATH_MAX];
172 char line[1024];
173 FILE *fp;
174
175 if ((size_t) snprintf(scpath, sizeof(scpath), "%s.cks", path) >= sizeof(scpath)) {
176 return;
177 }
178 fp = fopen(scpath, "re");
179 if (fp == NULL) {
180 return;
181 }
182 while (*n < max && fgets(line, sizeof(line), fp) != NULL) {
183 char ralgo[32], rhex[XRDC_CKV_HEX_MAX];
184 long ms, mn;
185 long long sz;
186 if (sscanf(line, "%31s %128s %ld %ld %lld", ralgo, rhex, &ms, &mn, &sz) >= 2
187 && ralgo[0] && rhex[0]) {
188 snprintf(recs[*n].src, sizeof(recs[*n].src), "cks");
189 snprintf(recs[*n].algo, sizeof(recs[*n].algo), "%.15s", ralgo);
190 snprintf(recs[*n].hex, sizeof(recs[*n].hex), "%.128s", rhex);
191 (*n)++;
192 }
193 }
194 fclose(fp);
195}

197/* Read a fixed-size record from "<path><suffix>" into buf (zeroed first). */
198static int
199ckv_read_sidecar_blob(const char *path, const char *suffix, void *buf, size_t sz)
200{
201 char p[XRDC_PATH_MAX];
202 int fd;
203 ssize_t got;
204
205 if ((size_t) snprintf(p, sizeof(p), "%s%s", path, suffix) >= sizeof(p)) {
206 return -1;
207 }
208 fd = open(p, O_RDONLY | O_CLOEXEC | O_NOFOLLOW | O_NOCTTY);
209 if (fd < 0) {
210 return -1;

235static uint16_t
236ckv_rd_u16(const uint8_t *b, size_t off)
237{
238 return (uint16_t) (b[off] | (b[off + 1] << 8));

241static uint32_t
242ckv_rd_u32(const uint8_t *b, size_t off)
243{
244 return (uint32_t) b[off] | ((uint32_t) b[off + 1] << 8)
245 | ((uint32_t) b[off + 2] << 16) | ((uint32_t) b[off + 3] << 24);
246}

248static uint64_t
249ckv_rd_u64(const uint8_t *b, size_t off)
250{
251 uint64_t v = 0;
252 int i;
253
254 for (i = 0; i < 8; i++) {
255 v |= (uint64_t) b[off + i] << (8 * i);

260/* ---- Validate the v4 xmeta prefix and locate the "XCX1" extension block ----
261 *
262 * WHAT: Checks the stock v4 cinfo prefix of the raw xmeta record `buf`
263 * (`n` bytes) — version + 48-byte Store + crc + present bitmap +
264 * AStat[] + crc — then verifies the "XCX1" extension magic that follows.
265 * Returns 1 with `*sec_off` set to the first extension section and
266 * `*nsec` to the section count; returns 0 on any malformed/truncated
267 * prefix.
268 *
269 * WHY: Isolates the fixed-prefix arithmetic from the section scan so the
270 * ORIGIN extraction reads as prefix-check → scan → decode; the inline
271 * walk drove ckv_xmeta_origin over the complexity cap.
272 *
273 * HOW: 1. Require >= 56 bytes and version CKV_XMETA_VERSION.
274 * 2. Read bsize/size/astatn and reject non-positive/negative values.
275 * 3. Compute the block-bitmap length from size/bsize and derive the
276 * extension-block offset.
277 * 4. Bounds-check the 8-byte extension header and require "XCX1".
278 * 5. Publish the section count and the offset just past the header.
279 */
280static int
281ckv_xmeta_ext_locate(const uint8_t *buf, size_t n, size_t *sec_off,
282 uint16_t *nsec)
283{
284 int64_t bsize, size;
285 int32_t astatn;
286 uint64_t nblocks;
287 size_t off;
288
289 if (n < 56 || (int32_t) ckv_rd_u32(buf, 0) != CKV_XMETA_VERSION) {
290 return 0;
291 }
292 bsize = (int64_t) ckv_rd_u64(buf, 4);
293 size = (int64_t) ckv_rd_u64(buf, 12);
294 astatn = (int32_t) ckv_rd_u32(buf, 48);
295 if (bsize <= 0 || size < 0 || astatn < 0) {
296 return 0;
297 }
298 nblocks = (size > 0)
299 ? ((uint64_t) size + (uint64_t) bsize - 1) / (uint64_t) bsize
300 : 0;
301 off = 4 + 48 + 4 + (size_t) ((nblocks + 7) / 8)
302 + (size_t) astatn * 56 + 4;
303 if (off + 8 > n || ckv_rd_u32(buf, off) != CKV_XMETA_EXT_MAGIC) {
304 return 0;
305 }
306 *nsec = ckv_rd_u16(buf, off + 6);
307 *sec_off = off + 8;
308 return 1;
309}

311/* ---- Decode one ORIGIN-section payload into *rec ----
312 *
313 * WHAT: Parses an ORIGIN (0x0004) section payload `p` of `plen` bytes —
314 * {u8 etag_len, u8 alg_len, u8 cks_len, u8 pad} + etag/alg/hex strings —
315 * into rec->{src,algo,hex}. Returns 1 on success, 0 when the embedded
316 * lengths are inconsistent with the payload (corrupt record).
317 *
318 * WHY: Keeps the length-sanity checks and string extraction next to each
319 * other and out of the section-scan loop; a 0 here means the whole
320 * record is untrustworthy, matching the pre-split behaviour.
321 *
322 * HOW: 1. Read the etag/alg/cks length bytes.
323 * 2. Reject zero or oversized alg/cks lengths and payload under-runs.
324 * 3. Copy source tag, algorithm, and hex digest into *rec.
325 */
326static int
327ckv_xmeta_origin_decode(const uint8_t *p, uint32_t plen, ckv_record *rec)
328{
329 uint8_t el = p[0], al = p[1], cl = p[2];
330
331 if (al == 0 || cl == 0 || al > 15 || cl > 128
332 || plen < 4u + el + al + cl)
333 {
334 return 0;
335 }
336 snprintf(rec->src, sizeof(rec->src), "cinfo");
337 snprintf(rec->algo, sizeof(rec->algo), "%.*s",
338 (int) al, (const char *) p + 4 + el);
339 snprintf(rec->hex, sizeof(rec->hex), "%.*s",
340 (int) cl, (const char *) p + 4 + el + al);
341 return 1;
342}

344/* Extract the ORIGIN-section checksum from one raw xmeta record into *rec.
345 * Walk: v4 stock prefix (ckv_xmeta_ext_locate) -> "XCX1" extension block ->
346 * section type 0x0004 (ckv_xmeta_origin_decode). Returns 1 when an (alg, hex)
347 * pair was found; every offset is bounds-checked against n so a truncated or
348 * corrupt record yields 0, never an over-read. */
349static int
350ckv_xmeta_origin(const uint8_t *buf, size_t n, ckv_record *rec)
351{
352 size_t off;
353 uint16_t nsec, i;
354
355 if (!ckv_xmeta_ext_locate(buf, n, &off, &nsec)) {
356 return 0;
357 }
358 for (i = 0; i < nsec && off + 12 <= n; i++) {
359 uint16_t type = ckv_rd_u16(buf, off);
360 uint32_t plen = ckv_rd_u32(buf, off + 4);
361 const uint8_t *p = buf + off + 8;
362
363 if (off + 8 + (size_t) plen + 4 > n) {
364 return 0;
365 }
366 if (type == CKV_XMETA_SEC_ORIGIN && plen >= 4) {
367 return ckv_xmeta_origin_decode(p, plen, rec);
368 }
369 off += 8 + (size_t) plen + 4;
370 }
371 return 0;
372}

432 * (up to `max`, advancing `*n`), pulling from the cache sidecars

458/* ---- Verify one recorded checksum against the open file ----
459 *
460 * WHAT: Compares record `rec` against the file behind `fd`. Returns 0 to keep
461 * iterating (record skipped, unsupported, or matched) and 1 for a terminal
462 * outcome, writing that outcome into `*out` (XRDC_CKV_ERROR on a read/hash
463 * failure, XRDC_CKV_MISMATCH on a digest mismatch). Sets `*matched` on a
464 * match and `*unsupported` when the algorithm cannot be computed. Computed
465 * digests are memoised in `comp`/`comp_done` keyed by algorithm so each
466 * distinct algorithm is hashed over the file at most once.
467 *
468 * WHY: Extracts the per-record body of the verification loop so the orchestrator
469 * reads as a short scan; the compare/hash logic is integrity-critical and
470 * preserved verbatim, just relocated behind a return-value contract that
471 * replaces the loop's early `return`/`continue` control flow.
472 *
473 * HOW: 1. Skip (return 0) if `want_algo` is set and this record's algo differs.
474 * 2. Mark unsupported and skip if the algo does not parse to a computable
475 * engine.
476 * 3. Compute the digest once per algorithm (rewind + brix_cksum_fd); on any
477 * read/hash failure set `*out` = ERROR and return 1.
478 * 4. Fill the caller's report (when non-NULL).
479 * 5. Case-insensitively compare recorded vs computed hex; on a mismatch set
480 * the integrity status, `*out` = MISMATCH, and return 1; otherwise flag
481 * `*matched` and return 0.
482 */
483static int
484ckv_verify_one(int fd, const ckv_record *rec, const char *path,
485 const char *want_algo, char comp[][XRDC_CKV_HEX_MAX], int *comp_done,
486 brix_ckv_report *rep, brix_status *st, int *matched, int *unsupported,
487 brix_ckv_result *out)
488{
489 brix_cksum_algo alg;
490
491 if (want_algo != NULL && strcmp(rec->algo, want_algo) != 0) {
492 return 0;
493 }
494 if (brix_cksum_algo_parse(rec->algo, &alg) != 0
495 || (int) alg > XRDC_CK_ZCRC32) {
496 *unsupported = 1; /* recorded with an engine we cannot compute */
497 return 0;
498 }
499 if (!comp_done[alg]) {
500 if (lseek(fd, 0, SEEK_SET) < 0
501 || brix_cksum_fd(fd, alg, comp[alg], XRDC_CKV_HEX_MAX, st) != 0) {
502 *out = XRDC_CKV_ERROR;
503 return 1;
504 }
505 comp_done[alg] = 1;
506 }
507
508 if (rep != NULL) {
509 snprintf(rep->source, sizeof(rep->source), "%.15s", rec->src);
510 snprintf(rep->algo, sizeof(rep->algo), "%.15s", rec->algo);
511 snprintf(rep->recorded, sizeof(rep->recorded), "%.128s", rec->hex);
512 snprintf(rep->computed, sizeof(rep->computed), "%.128s", comp[alg]);
513 }
514 if (strcasecmp(rec->hex, comp[alg]) != 0) {
515 brix_status_set(st, XRDC_EINTEGRITY, 0,
516 "%s checksum mismatch on %s: recorded %s != computed %s",
517 rec->algo, path, rec->hex, comp[alg]);
518 *out = XRDC_CKV_MISMATCH;
519 return 1;
520 }
521 *matched = 1;
522 return 0;
523}

525/* public entry point */
526brix_ckv_result
527brix_cks_verify_file(const char *path, const char *want_algo,
528 brix_ckv_mode mode, brix_ckv_report *rep, brix_status *st)
529{
530 ckv_record recs[CKV_MAX_RECORDS];
531 size_t nrec = 0, i;
532 int fd;
533 int matched = 0, unsupported = 0;
534 char comp[XRDC_CK_ZCRC32 + 1][XRDC_CKV_HEX_MAX];
535 int comp_done[XRDC_CK_ZCRC32 + 1];
536
537 if (rep != NULL) {
538 memset(rep, 0, sizeof(*rep));
539 }
540 memset(comp_done, 0, sizeof(comp_done));
541
542 ckv_collect_records(path, mode, recs, CKV_MAX_RECORDS, &nrec);
543
544 if (nrec == 0) {
545 brix_status_set(st, XRDC_EUSAGE, 0,
546 "no recorded checksum found for %s", path);
547 return XRDC_CKV_NO_RECORD;
548 }
549
550 fd = open(path, O_RDONLY | O_CLOEXEC | O_NOCTTY);
551 if (fd < 0) {
552 brix_status_set(st, XRDC_ESOCK, errno, "open %s: %s",
553 path, strerror(errno));
554 return XRDC_CKV_ERROR;
555 }
556
557 for (i = 0; i < nrec; i++) {
558 brix_ckv_result outcome = XRDC_CKV_ERROR;
559
560 if (ckv_verify_one(fd, &recs[i], path, want_algo, comp, comp_done,
561 rep, st, &matched, &unsupported, &outcome) != 0) {
562 close(fd);
563 return outcome;
564 }
565 }
566
567 close(fd);
568
569 if (matched) {
570 return XRDC_CKV_OK;
571 }
572 if (unsupported) {
573 brix_status_set(st, XRDC_EUSAGE, 0,
574 "recorded checksum uses an algorithm this tool cannot compute");
575 return XRDC_CKV_UNSUPPORTED;
576 }
577 brix_status_set(st, XRDC_EUSAGE, 0,
578 "no recorded %s checksum for %s",
579 want_algo ? want_algo : "matching", path);
580 return XRDC_CKV_NO_RECORD;
581}
