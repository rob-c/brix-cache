9/* Choose a remote regular file to operate on: if the URL carried an explicit
10 * path (not "/") use it; otherwise list "/" and pick the largest regular file.
11 * Fills target[tsz] with the absolute path and *sti with its stat. 0 / -1. */
12int
13resolve_target(brix_conn *c, const brix_url *u, char *target, size_t tsz,
14 brix_statinfo *sti, brix_status *st)
15{
16 brix_dirent *ents = NULL;
17 size_t n = 0, i;
18 int64_t best = -1;
19 int found = 0;
20
21 if (u->path[0] != '\0' && strcmp(u->path, "/") != 0) {
22 if (brix_stat(c, u->path, sti, st) != 0) {
23 return -1;
24 }
25 if (sti->flags & kXR_isDir) {
26 brix_status_set(st, XRDC_EUSAGE, 0, "%s is a directory", u->path);
27 return -1;
28 }
29 snprintf(target, tsz, "%s", u->path);
30 return 0;
31 }
32
33 if (brix_dirlist(c, "/", 1, &ents, &n, st) != 0) {
34 return -1;
35 }
36 for (i = 0; i < n; i++) {
37 if (!ents[i].have_stat || (ents[i].st.flags & kXR_isDir)) {
38 continue;
39 }
40 if (ents[i].st.size > best) {
41 best = ents[i].st.size;
42 snprintf(target, tsz, "/%s", ents[i].name);
43 *sti = ents[i].st;
44 found = 1;
45 }
46 }
47 free(ents);
48 if (!found) {
49 brix_status_set(st, XRDC_EUSAGE, 0,
50 "no regular file under / to test (pass a file URL)");
51 return -1;
52 }
53 return 0;
54}

59/*
60 * topo_probe_locate — run the kXR_locate probe and record its outcome.
61 *
62 * WHAT: locates qpath on c; sets *locate_ok (1 when a non-empty result came
63 * back), copies the result blob into locate_result[rsz], and bumps *fails
64 * on any failure (locate error OR empty result). Prints a human probe line
65 * only when !a->json.
66 * WHY: the locate outcome feeds both the human report and the JSON node list, so
67 * it must accumulate into caller locals identically regardless of mode.
68 * HOW: (1) call brix_locate into a local buffer; (2) on success derive
69 * locate_ok/locate_result from the first byte and count an empty result as
70 * a failure; (3) on error zero locate_ok and count the failure; (4) emit
71 * the probe line only in human mode.
72 */
73static void
74topo_probe_locate(brix_conn *c, const char *qpath, const diag_args *a,
75 int *locate_ok, char *locate_result, size_t rsz, int *fails)
76{
77 char loc[4096];
78 brix_status st;
79
80 brix_status_clear(&st);
81 if (brix_locate(c, qpath, loc, sizeof(loc), &st) == 0) {
82 *locate_ok = (loc[0] != '\0');
83 snprintf(locate_result, rsz, "%s", loc[0] ? loc : "");
84 if (!*locate_ok) { (*fails)++; }
85 if (!a->json) {
86 probe("locate", loc[0] != '\0', "%s -> %s", qpath, loc[0] ? loc : "(empty)");
87 }
88 } else {
89 *locate_ok = 0;
90 (*fails)++;
91 if (!a->json) { probe("locate", 0, "%s", st.msg); }
92 }
93}

96/*
97 * topo_probe_redirect — verify a nonexistent path converges to NotFound.
98 *
99 * WHAT: stats a sentinel path that cannot exist; sets *redirect_ok when the stat
100 * failed with kXR_NotFound and bumps *fails otherwise. Prints a human probe
101 * line only when !a->json.
102 * WHY: a cluster redirect loop-guard regression would exhaust the redirect budget
103 * instead of returning NotFound; this probe is that regression test.
104 * HOW: (1) stat the sentinel path with a private status; (2) redirect_ok is true
105 * iff stat failed AND the kXR code is NotFound; (3) count a failure when not;
106 * (4) emit the probe line only in human mode.
107 */
108static void
109topo_probe_redirect(brix_conn *c, const diag_args *a, int *redirect_ok, int *fails)
110{
111 brix_statinfo s;
112 brix_status nst;
113 int rc;
114
115 brix_status_clear(&nst);
116 rc = brix_stat(c, "/nonexistent-xrddiag-probe-path", &s, &nst);
117 *redirect_ok = (rc != 0 && nst.kxr == kXR_NotFound);
118 if (!*redirect_ok) { (*fails)++; }
119 if (!a->json) {
120 probe("redirect-convergence", *redirect_ok,
121 "nonexistent path -> %s", rc != 0 ? brix_kxr_name(nst.kxr) : "SERVED?!");
122 }
123}

126/*
127 * topo_report_cluster — human-only /cluster HTTP dump.
128 *
129 * WHAT: when a->cluster_url is set, fetches it over HTTP and prints the body with
130 * its status code; on any failure (alloc, parse, or GET) prints a note.
131 * With no cluster_url, prints a usage hint. No return value.
132 * WHY: the cluster view is only meaningful in the human report, so it is skipped
133 * entirely in JSON mode by the caller and lives in its own helper.
134 * HOW: (1) with a cluster_url, allocate a 1 MiB body and require alloc, endpoint
135 * parse, and HTTP GET to all succeed before printing the body; otherwise
136 * note the failure; free the body; (2) with no cluster_url, note the hint.
137 */
138static void
139topo_report_cluster(const diag_args *a)
140{
141 if (a->cluster_url != NULL) {
142 brix_url cu;
143 brix_status cst;
144 char *body = malloc(1u << 20);
145 int http = 0;
146 brix_status_clear(&cst);
147 if (body != NULL && brix_endpoint_parse(a->cluster_url, &cu, &cst) == 0 &&
148 brix_http_get(cu.host, cu.port, cu.path[0] ? cu.path : "/", 5000,
149 &http, body, 1u << 20, NULL, &cst) == 0) {
150 printf(" /cluster (HTTP %d):\n%s\n", http, body);
151 } else {
152 note("cluster-json", "unavailable: %s", cst.msg);
153 }
154 free(body);
155 } else {
156 note("cluster-json", "pass --cluster-url http://host:port/brix/api/v1/cluster");
157 }
158}

161/*
162 * topo_emit_json — emit an array of nodes extracted from the locate-result blob.
163 *
164 * WHAT: prints [{"node":"<token>"}, ...] to stdout, one element per whitespace
165 * token of locate_result (empty locate yields []), and returns 1 when fails
166 * is nonzero else 0 (the process exit code for JSON mode).
167 * WHY: the human branch prints a single raw blob (not per-node lines), so
168 * splitting on whitespace is the minimal faithful translation. The top level
169 * must be an array so JSON consumers can iterate nodes without knowing the
170 * blob grammar.
171 * HOW: (1) open the array; (2) when locate succeeded and the blob is non-empty,
172 * strtok a stack copy and print one object per token, comma-separated;
173 * (3) close the array; (4) map the accumulated failure count to an exit code.
174 */
175static int
176topo_emit_json(int locate_ok, const char *locate_result, int fails)
177{
178 char lbuf[4096];
179 char *tok;
180 const char *sep = "";
181
182 printf("[");
183 if (locate_ok && locate_result[0] != '\0') {
184 snprintf(lbuf, sizeof(lbuf), "%s", locate_result);
185 for (tok = strtok(lbuf, " \t\r\n"); tok != NULL;
186 tok = strtok(NULL, " \t\r\n")) {
187 printf("%s{", sep);
188 brix_json_kv_str(stdout, "node", tok, 0);
189 printf("}");
190 sep = ",";
191 }
192 }
193 printf("]\n");
194 return fails ? 1 : 0;
195}

198/*
199 * do_topology — probe cluster topology at a live endpoint.
200 *
201 * WHAT: connects, runs kXR_locate + a redirect-convergence probe, then outputs
202 * a human-readable report or (--json) a single JSON object.
203 * WHY: JSON mode gives monitoring tools a structured, parseable topology snapshot.
204 * HOW: probe outcomes accumulate in locals; human path is guarded by !a->json;
205 * JSON emitted once at end — never partial, never on early connect error.
206 */
207int
208do_topology(const diag_args *a)
209{
210 brix_url u;
211 brix_conn c;
212 brix_status st;
213 const char *qpath;
214 /* JSON-mode accumulators */
215 int js_locate_ok = 0, js_redirect_ok = 0, js_fails = 0;
216 char js_locate_result[4096] = "";
217
218 brix_status_clear(&st);
219 if (brix_endpoint_parse(a->url, &u, &st) != 0) {
220 fprintf(stderr, "xrddiag: %s\n", st.msg);
221 return 50;
222 }
223 if (brix_connect(&c, &u, &a->conn, &st) != 0) {
224 fprintf(stderr, "xrddiag: connect %s:%d: %s\n", u.host, u.port, st.msg);
225 return brix_shellcode(&st);
226 }
227
228 qpath = (u.path[0] != '\0') ? u.path : "/";
229 if (!a->json) {
230 printf("Topology of %s:%d\n", u.host, u.port);
231 }
232
233 topo_probe_locate(&c, qpath, a, &js_locate_ok, js_locate_result,
234 sizeof(js_locate_result), &js_fails);
235 topo_probe_redirect(&c, a, &js_redirect_ok, &js_fails);
236 if (!a->json) {
237 topo_report_cluster(a);
238 }
239
240 brix_close(&c);
241
242 if (a->json) {
243 return topo_emit_json(js_locate_ok, js_locate_result, js_fails);
244 }
245 return g_fails ? 1 : 0;
246}

249/* Parse a "[http://]host[:port][/...]" WebDAV endpoint into host + port (8080). */
250void
251parse_http_hostport(const char *s, char *host, size_t hsz, int *port)
252{
253 const char *p = s, *e;
254 *port = 8080;
255 if (strncmp(p, "http://", 7) == 0) {
256 p += 7;
257 }
258 e = p;
259 while (*e != '\0' && *e != ':' && *e != '/') {
260 e++;
261 }
262 {
263 size_t n = (size_t) (e - p);
264 if (n >= hsz) { n = hsz - 1; }
265 memcpy(host, p, n);
266 host[n] = '\0';
267 }
268 if (*e == ':') {
269 *port = atoi(e + 1);
270 }
271}

276/*
277 * Resolve host ONCE to a numeric IP and classify whether it is loopback. The
278 * probe then connects to this SAME numeric IP (getaddrinfo on a literal address
279 * is deterministic), so a DNS-rebind / localhost.attacker.com cannot slip a
280 * non-loopback target past the gate between the check and the connect. 0 / -1.
281 */
282int
283resolve_once(const char *host, int port, char *ip, size_t ipsz, int *is_loop,
284 brix_status *st)
285{
286 struct addrinfo hints, *res = NULL;
287 char portstr[16];
288 int gai;
289
290 memset(&hints, 0, sizeof(hints));
291 /* Honor a session-wide IPv6→IPv4 demotion (netpref.c) for consistency with
292 * every other connect path: AF_UNSPEC normally, AF_INET once this process
293 * has fallen back to IPv4-only. */
294 hints.ai_family = brix_netpref_family();
295 hints.ai_socktype = SOCK_STREAM;
296 snprintf(portstr, sizeof(portstr), "%d", port);
297
298 gai = getaddrinfo(host, portstr, &hints, &res);
299 if (gai != 0 || res == NULL) {
300 brix_status_set(st, XRDC_ESOCK, 0, "resolve %s: %s", host,
301 gai_strerror(gai));
302 return -1;
303 }
304 *is_loop = 0;
305 if (res->ai_family == AF_INET) {
306 struct sockaddr_in *s4 = (struct sockaddr_in *) res->ai_addr;
307 *is_loop = ((ntohl(s4->sin_addr.s_addr) >> 24) == 127); /* 127.0.0.0/8 */
308 } else if (res->ai_family == AF_INET6) {
309 struct sockaddr_in6 *s6 = (struct sockaddr_in6 *) res->ai_addr;
310 *is_loop = IN6_IS_ADDR_LOOPBACK(&s6->sin6_addr);
311 }
312 getnameinfo(res->ai_addr, res->ai_addrlen, ip, (socklen_t) ipsz, NULL, 0,
313 NI_NUMERICHOST);
314 freeaddrinfo(res);
315 return 0;
316}

337/*
338 * Write a (possibly malformed) 24-byte header + body bytes straight onto the wire
339 * — bypassing brix_send so we can lie about dlen / use a bad opcode — then read
340 * one response. Returns 1 if the server REJECTED the abuse (kXR_error, an
341 * unexpected status, or a closed/timed-out connection), 0 if it SERVED it
342 * (kXR_ok/oksofar — a bug), -1 on our own write failure.
343 */
344int
345raw_send_expect_reject(brix_conn *c, const uint8_t hdr24[24],
346 const uint8_t *body, uint32_t bodylen,
347 int lie_dlen, uint32_t fake_dlen)
348{
349 uint8_t hdr[24];
350 brix_status st;
351 uint16_t status = 0;
352 uint8_t *rb = NULL;
353 uint32_t rl = 0, wire_dlen;
354 int rc;
355
356 memcpy(hdr, hdr24, 24);
357 hdr[0] = 0x7e; hdr[1] = 0x01; /* arbitrary streamid (recv accepts any) */
358 wire_dlen = lie_dlen ? fake_dlen : bodylen;
359 hdr[20] = (uint8_t) (wire_dlen >> 24);
360 hdr[21] = (uint8_t) (wire_dlen >> 16);
361 hdr[22] = (uint8_t) (wire_dlen >> 8);
362 hdr[23] = (uint8_t) wire_dlen;
363
364 brix_status_clear(&st);
365 if (brix_write_full(&c->io, hdr, 24, &st) != 0) {
366 return -1;
367 }
368 if (bodylen > 0 && body != NULL) {
369 if (brix_write_full(&c->io, body, bodylen, &st) != 0) {
370 return -1;
371 }
372 }
373 {
374 brix_resp_out out = { &status, &rb, &rl };
375 rc = brix_recv(c, 0xffff, &out, &st);
376 }
377 free(rb);
378 if (rc != 0) {
379 return 1; /* kXR_error / closed / timeout → rejected cleanly */
380 }
381 return (status == kXR_ok || status == kXR_oksofar) ? 0 : 1;
382}
