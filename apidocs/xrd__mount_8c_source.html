39/* `xrd ping [-c COUNT] <endpoint>` — connect once, then time COUNT (default 4) stat
40 * round-trips to "/" and report min/avg/max RTT. A simple, dependency-free liveness +
41 * latency probe (xrddiag has deeper net diagnostics). Exit nonzero on connect failure
42 * or if every probe failed. */
43int
44xrd_ping(int argc, char **argv)
45{
46 const char *endpoint = NULL;
47 int count = 4, i, ok = 0;
48 double tmin = 1e30, tmax = 0.0, tsum = 0.0;
49 brix_url u;
50 brix_opts o;
51 brix_conn c;
52 brix_status st;
53
54 for (i = 2; i < argc; i++) {
55 if (strcmp(argv[i], "-c") == 0 && i + 1 < argc) { count = atoi(argv[++i]); }
56 else if (endpoint == NULL && argv[i][0] != '-') { endpoint = argv[i]; }
57 }
58 if (endpoint == NULL || count <= 0) {
59 fprintf(stderr, "usage: xrd ping [-c COUNT] <endpoint>\n");
60 return 50;
61 }
62 memset(&o, 0, sizeof(o));
63 o.verify_host = 1;
64 brix_crypto_init();
65 brix_status_clear(&st);
66 if (brix_endpoint_parse(endpoint, &u, &st) != 0) {
67 fprintf(stderr, "xrd ping: %s\n", st.msg);
68 return 50;
69 }
70 if (brix_connect(&c, &u, &o, &st) != 0) {
71 fprintf(stderr, "xrd ping: connect %s:%d: %s\n", u.host, u.port, st.msg);
72 return brix_shellcode(&st);
73 }
74 printf("PING %s:%d (%d stat round-trips to /)\n", u.host, u.port, count);
75 for (i = 0; i < count; i++) {
76 struct timespec t0, t1;
77 brix_statinfo si;
78 brix_status pst;
79 double ms;
80 brix_status_clear(&pst);
81 clock_gettime(CLOCK_MONOTONIC, &t0);
82 if (brix_stat(&c, "/", &si, &pst) != 0) {
83 printf(" seq %d: FAILED (%s)\n", i + 1, pst.msg);
84 continue;
85 }
86 clock_gettime(CLOCK_MONOTONIC, &t1);
87 ms = (double) (t1.tv_sec - t0.tv_sec) * 1000.0
88 + (double) (t1.tv_nsec - t0.tv_nsec) / 1e6;
89 printf(" seq %d: %.3f ms\n", i + 1, ms);
90 ok++;
91 tsum += ms;
92 if (ms < tmin) { tmin = ms; }
93 if (ms > tmax) { tmax = ms; }
94 }
95 brix_close(&c);
96 if (ok == 0) {
97 printf("%d probes, 0 successful\n", count);
98 return 1;
99 }
100 printf("%d/%d ok min/avg/max = %.3f/%.3f/%.3f ms\n",
101 ok, count, tmin, tsum / ok, tmax);
102 return 0;
103}

204/* Fork + exec `cmd_argv` (PATH-searched) and wait. Returns the child's exit code,
205 * or 126 if it could not be exec'd (so callers can try a fallback tool), or -1 on
206 * fork failure. */
207int
208run_cmd(char *const cmd_argv[])
209{
210 pid_t pid = fork();
211 int status;
212
213 if (pid < 0) {
214 return -1;
215 }
216 if (pid == 0) {
217 execvp(cmd_argv[0], cmd_argv);
218 _exit(126); /* distinct from any normal tool exit → "couldn't exec" */
219 }
220 while (waitpid(pid, &status, 0) < 0 && errno == EINTR) {
221 /* retry */
222 }
223 return WIFEXITED(status) ? WEXITSTATUS(status) : -1;
224}

227/* Decode mountinfo octal escapes (\040 space, \011 tab, \012 nl, \134 backslash)
228 * from `in` into out[outsz]. */
229void
230mountinfo_unescape(const char *in, char *out, size_t outsz)
231{
232 size_t o = 0;
233 while (*in != '\0' && o + 1 < outsz) {
234 if (in[0] == '\\' && in[1] >= '0' && in[1] <= '7'
235 && in[2] >= '0' && in[2] <= '7' && in[3] >= '0' && in[3] <= '7') {
236 out[o++] = (char) ((in[1] - '0') * 64 + (in[2] - '0') * 8 + (in[3] - '0'));
237 in += 4;
238 } else {
239 out[o++] = *in++;
240 }
241 }
242 out[o] = '\0';
243}

246/* `xrd mount` (no args) / `xrd mounts` / `xrd mount -l` — list active XRootD FUSE
247 * mounts by parsing /proc/self/mountinfo (override with XRD_MOUNTINFO_PATH for tests).
248 * Matches fuse.xrootdfs* filesystem types, plus any fuse mount whose source looks like
249 * a root:// endpoint. Prints "ENDPOINT MOUNTPOINT DRIVER"; honest empty output (exit
250 * 0) when nothing is mounted. Pure procfs parse — no network, no credentials. */
251int
252xrd_list_mounts(void)
253{
254#ifndef __linux__
255 fprintf(stderr, "xrd mount: mount listing is only supported on Linux\n");
256 return 0;
257#else
258 const char *path = getenv("XRD_MOUNTINFO_PATH");
259 FILE *fp = fopen(path != NULL ? path : "/proc/self/mountinfo", "r");
260 char *line = NULL;
261 size_t cap = 0;
262 ssize_t r;
263 int header = 0;
264
265 if (fp == NULL) {
266 fprintf(stderr, "xrd mount: cannot read mountinfo: %s\n", strerror(errno));
267 return 1;
268 }
269 while ((r = getline(&line, &cap, fp)) >= 0) {
270 char *fields[48];
271 int nf = 0, sep = -1, i;
272 char *tok, *save;
273 const char *mp, *fstype, *src, *driver;
274 char mpbuf[PATH_MAX], srcbuf[PATH_MAX];
275
276 (void) r;
277 for (tok = strtok_r(line, " \n", &save); tok != NULL && nf < 48;
278 tok = strtok_r(NULL, " \n", &save)) {
279 fields[nf++] = tok;
280 }
281 for (i = 0; i < nf; i++) {
282 if (strcmp(fields[i], "-") == 0) { sep = i; break; }
283 }
284 if (sep < 5 || sep + 2 >= nf) { continue; } /* malformed / too few fields */
285 mp = fields[4];
286 fstype = fields[sep + 1];
287 src = fields[sep + 2];
288 if (strncmp(fstype, "fuse.xrootdfs", 13) != 0
289 && !(strncmp(fstype, "fuse", 4) == 0 && strstr(src, "root") != NULL)) {
290 continue;
291 }
292 driver = (strstr(fstype, "xrootdfs_legacy") != NULL) ? "legacy"
293 : (strncmp(fstype, "fuse.xrootdfs", 13) == 0) ? "aio"
294 : "fuse";
295 mountinfo_unescape(mp, mpbuf, sizeof(mpbuf));
296 mountinfo_unescape(src, srcbuf, sizeof(srcbuf));
297 if (!header) {
298 printf("%-36s %-28s %s\n", "ENDPOINT", "MOUNTPOINT", "DRIVER");
299 header = 1;
300 }
301 printf("%-36s %-28s %s\n", srcbuf, mpbuf, driver);
302 }
303 free(line);
304 fclose(fp);
305 return 0;
306#endif
307}

310/* `xrd mount [--legacy|--driver aio|legacy] [driver-opts] <endpoint> <mountpoint>
311 * [fuse-opts]` — mount an XRootD export via the single FUSE3 driver `xrootdfs`.
312 * Defaults to its resilient mode; --legacy is forwarded to xrootdfs to select its
313 * synchronous mode. Everything after the
314 * driver selector is forwarded verbatim in the driver's native arg order
315 * (`[opts] endpoint mountpoint [fuse-opts]`). The driver backgrounds itself unless
316 * a fuse -f/-d is passed. exec's the driver (does not return on success). */
317int
318xrd_mount(int argc, char **argv)
319{
320 /* One unified `xrootdfs` binary carries both drivers; `--legacy` selects the
321 * synchronous one at run time (passed through to the driver). */
322 const char *driver = "xrootdfs";
323 int i = 2, k = 0, list = 0, legacy = 0;
324 char **nv;
325 char endpoint[XRDC_PATH_MAX];
326
327 /* Optional driver selector / --list, only as the leading token(s). */
328 while (i < argc) {
329 if (strcmp(argv[i], "--legacy") == 0) { legacy = 1; i++; }
330 else if (strcmp(argv[i], "--aio") == 0) { legacy = 0; i++; }
331 else if (strcmp(argv[i], "-l") == 0 || strcmp(argv[i], "--list") == 0) {
332 list = 1; i++;
333 }
334 else if (strcmp(argv[i], "--driver") == 0 && i + 1 < argc) {
335 const char *d = argv[i + 1];
336 if (strcmp(d, "legacy") == 0) { legacy = 1; }
337 else if (strcmp(d, "aio") == 0 || strcmp(d, "resilient") == 0) { legacy = 0; }
338 else {
339 fprintf(stderr, "xrd mount: unknown driver '%s' (aio|legacy)\n", d);
340 return 50;
341 }
342 i += 2;
343 } else {
344 break;
345 }
346 }
347 /* No positional args (or an explicit --list) → list current XRootD mounts,
348 * mirroring mount(8)'s no-arg behavior. */
349 if (list || argc - i == 0) {
350 return xrd_list_mounts();
351 }
352 if (argc - i < 2) {
353 fprintf(stderr, "usage: xrd mount [--legacy] [driver-opts] <endpoint> "
354 "<mountpoint> [fuse-opts]\n"
355 " e.g. xrd mount root://store//data /mnt/xrd -o ro\n");
356 return 50;
357 }
358 /* +3: driver name, an optional "--legacy", and the NULL terminator. */
359 nv = (char **) malloc((size_t) (argc - i + 3) * sizeof(char *));
360 if (nv == NULL) {
361 fprintf(stderr, "xrd: out of memory\n");
362 return 51;
363 }
364 nv[k++] = (char *) driver;
365 if (legacy) {
366 nv[k++] = (char *) "--legacy";
367 }
368 /* Resolve a ~/.xrdrc alias for the endpoint (the first forwarded non-option
369 * token) so `xrd mount lab:/data /mnt` works like the rest of xrd; the driver
370 * itself doesn't expand aliases. A bare URL passes through verbatim. */
371 if (i < argc && argv[i][0] != '-') {
372 brix_alias_resolve(argv[i], endpoint, sizeof(endpoint));
373 nv[k++] = endpoint;
374 i++;
375 }
376 for (; i < argc; i++) {
377 nv[k++] = argv[i];
378 }
379 nv[k] = NULL;
380 exec_tool(driver, nv); /* does not return on success */
381 return 127; /* unreachable (exec_tool _exit's on failure) */
382}

385/* `xrd unmount [-z|--lazy] <mountpoint>` (alias: umount) — unmount a FUSE export,
386 * preferring fusermount3 (fuse3), then fusermount, then umount. -z/--lazy maps to
387 * the lazy-detach flag of whichever tool is used. */
388int
389xrd_unmount(int argc, char **argv)
390{
391 const char *mp = NULL;
392 int lazy = 0, i, rc;
393
394 for (i = 2; i < argc; i++) {
395 if (strcmp(argv[i], "-z") == 0 || strcmp(argv[i], "--lazy") == 0) {
396 lazy = 1;
397 } else if (mp == NULL && argv[i][0] != '-') {
398 mp = argv[i];
399 } else {
400 fprintf(stderr, "xrd unmount: unexpected argument '%s'\n", argv[i]);
401 return 50;
402 }
403 }
404 if (mp == NULL) {
405 fprintf(stderr, "usage: xrd unmount [-z] <mountpoint>\n");
406 return 50;
407 }
408 {
409 char *c[5]; int k = 0;
410 c[k++] = (char *) "fusermount3"; c[k++] = (char *) "-u";
411 if (lazy) { c[k++] = (char *) "-z"; }
412 c[k++] = (char *) mp; c[k] = NULL;
413 rc = run_cmd(c);
414 }
415 if (rc == 126) { /* fusermount3 not present → fusermount */
416 char *c[5]; int k = 0;
417 c[k++] = (char *) "fusermount"; c[k++] = (char *) "-u";
418 if (lazy) { c[k++] = (char *) "-z"; }
419 c[k++] = (char *) mp; c[k] = NULL;
420 rc = run_cmd(c);
421 }
422 if (rc == 126) { /* neither fusermount → umount (-l = lazy) */
423 char *c[4]; int k = 0;
424 c[k++] = (char *) "umount";
425 if (lazy) { c[k++] = (char *) "-l"; }
426 c[k++] = (char *) mp; c[k] = NULL;
427 rc = run_cmd(c);
428 }
429 if (rc == 126) {
430 fprintf(stderr, "xrd unmount: no fusermount3/fusermount/umount found\n");
431 return 127;
432 }
433 return (rc < 0) ? 1 : rc;
434}
