1/*
2 * xrdmapc.c — query a manager/redirector for the live cluster map of a path.
3 *
4 * WHAT: `xrdmapc [opts] root://host[:port][//path] [--verify]` — asks the
5 * endpoint (a cmsd/manager redirector, or a plain server) which data
6 * servers hold a path, prints them with read/write flag and free space,
7 * and (with --verify) probes each advertised holder to flag GHOST
8 * replicas (advertised but not actually serving) and UNREACHABLE nodes.
9 * WHY: Operational parity with this project's manager/cluster features and the
10 * "ghost replica" failure mode a plain transfer never surfaces (§15.4).
11 * A thin, libXrdCl-free front-end over the public libbrix — zero new wire.
12 * HOW: brix_locate(path) returns space-separated "S<r|w><host>:<port>" holder
13 * tokens (src/net/manager/registry.c / src/protocols/root/read/locate.c; IPv6 host bracketed).
14 * Parse them, brix_query(kXR_Qspace) for free/total, and for --verify open
15 * a fresh session to each holder and brix_stat(path): PASS=serves,
16 * GHOST=connects but NotFound, UNREACHABLE=connect fails.
17 *
18 * Clean-room: composes the public libbrix API only; the locate token grammar is
19 * this project's own (registry.c). No XrdCl / xrdmapc source consulted.
20 */

32/* Parse one "S<flag><host>:<port>" locate token (host may be "[v6]"). 0 / -1. */
33static int
34parse_holder(const char *tok, char *host, size_t hsz, int *port, char *rw)
35{
36 const char *rest;
37
38 if (tok[0] != 'S' || tok[1] == '\0') {
39 return -1;
40 }
41 *rw = tok[1]; /* 'r' read-capable, 'w' write-capable */
42 rest = tok + 2;
43
44 /* shared bracketed-IPv6-aware host:port split (libxrdproto). */
45 return brix_split_host_port(rest, host, hsz, port, 1094);
46}

48/* Build a connectable URL for a parsed holder (numeric/host literal). */
49static void
50holder_url(const brix_opts *o, const char *host, int port, char *out, size_t outsz)
51{
52 int v6 = (strchr(host, ':') != NULL && host[0] != '[');
53 snprintf(out, outsz, "%s://%s%s%s:%d/", o->want_tls ? "roots" : "root",
54 v6 ? "[" : "", host, v6 ? "]" : "", port);
55}

57/* ---- Print best-effort endpoint free/total space line ----
58 *
59 * WHAT: Queries the connected endpoint for kXR_Qspace on `path` and, on a
60 * non-empty answer, prints the "Space:" line. Silent on any failure or
61 * empty result (best-effort, redirect-aware) — returns nothing.
62 *
63 * WHY: Space reporting is advisory: a manager that cannot answer must not abort
64 * the cluster-map listing, so this is factored out with its own throwaway
65 * status and never propagates an error.
66 *
67 * HOW: 1. Clear a local status. 2. brix_query(kXR_Qspace). 3. If it succeeds
68 * and the answer buffer is non-empty, print "Space: <answer>".
69 */
70static void
71print_space(brix_conn *c, const char *path)
72{
73 char space[1024];
74 brix_status sst;
75
76 brix_status_clear(&sst);
77 if (brix_query(c, kXR_Qspace, path, space, sizeof(space), &sst) == 0
78 && space[0] != '\0') {
79 printf("Space: %s\n", space);
80 }
81}

83/* ---- Probe one advertised holder and classify it (--verify) ----
84 *
85 * WHAT: Opens a fresh session to <host:port> and stats `path`, printing one of
86 * PASS (serves, with byte size), GHOST (connects but does not serve), or
87 * UNREACHABLE (connect fails). Increments g_ghosts for GHOST/UNREACHABLE.
88 * Returns nothing.
89 *
90 * WHY: The "ghost replica" failure mode (advertised holder that does not serve)
91 * is invisible to a plain transfer; --verify surfaces it by contacting
92 * each holder directly. Isolating the probe keeps the classification and
93 * its g_ghosts side effect in one small, reviewable place.
94 *
95 * HOW: 1. Build a connectable URL via holder_url(). 2. Parse+connect; on
96 * failure print UNREACHABLE and bump g_ghosts. 3. Otherwise brix_stat():
97 * success → PASS with size; failure → GHOST with the kXR name + bump
98 * g_ghosts. 4. Close the probe session.
99 */
100static void
101probe_holder(const brix_opts *co, const char *path,
102 const char *host, int port, char rw)
103{
104 char hu[320];
105 brix_url hurl;
106 brix_conn hc;
107 brix_status hst;
108
109 holder_url(co, host, port, hu, sizeof(hu));
110 brix_status_clear(&hst);
111 if (brix_endpoint_parse(hu, &hurl, &hst) != 0
112 || brix_connect(&hc, &hurl, co, &hst) != 0) {
113 printf(" %s:%d [%s] UNREACHABLE (%s)\n", host, port,
114 rw == 'w' ? "rw" : "ro", hst.msg);
115 g_ghosts++;
116 return;
117 }
118
119 {
120 brix_statinfo si;
121 brix_status pst;
122 brix_status_clear(&pst);
123 if (brix_stat(&hc, path, &si, &pst) == 0) {
124 printf(" %s:%d [%s] PASS (%lld bytes)\n", host, port,
125 rw == 'w' ? "rw" : "ro", (long long) si.size);
126 } else {
127 printf(" %s:%d [%s] GHOST (advertised, %s)\n", host,
128 port, rw == 'w' ? "rw" : "ro",
129 brix_kxr_name(pst.kxr));
130 g_ghosts++;
131 }
132 brix_close(&hc);
133 }
134}

136/* ---- Handle one locate token: parse, then list or probe ----
137 *
138 * WHAT: Parses a single "S<flag><host>:<port>" locate token and, if valid,
139 * prints the holder line (plain ro/rw) or, when `verify` is set, probes
140 * it via probe_holder(). Returns 1 if the token was a valid holder (to be
141 * counted), 0 if it was malformed and skipped.
142 *
143 * WHY: Keeps the per-token decision (skip / list / verify) in one place so the
144 * holder loop stays a flat iteration and the counted-holder semantics
145 * (only well-formed tokens count) are expressed once.
146 *
147 * HOW: 1. parse_holder(); on failure return 0. 2. If not verifying, print the
148 * plain holder line and return 1. 3. Otherwise probe_holder() and return 1.
149 */
150static int
151map_holder(const brix_opts *co, const char *path, const char *tok, int verify)
152{
153 char host[256];
154 int port = 0;
155 char rw = '?';
156
157 if (parse_holder(tok, host, sizeof(host), &port, &rw) != 0) {
158 return 0;
159 }
160 if (!verify) {
161 printf(" %s:%d [%s]\n", host, port, rw == 'w' ? "rw" : "ro");
162 return 1;
163 }
164 probe_holder(co, path, host, port, rw);
165 return 1;
166}

168/* ---- Print the "Holders:" section from a locate answer ----
169 *
170 * WHAT: Tokenizes the locate answer, dispatches each token to map_holder(),
171 * prints "(none)" when no valid holder was found, and prints the "Total:"
172 * footer (with ", all serving" only when verifying and no ghosts). Returns
173 * nothing; ghost/unreachable counts accumulate in g_ghosts via probes.
174 *
175 * WHY: Separates the listing/aggregation loop from connection setup so do_map
176 * reads as a short sequence and the token-counting stays adjacent to the
177 * footer that reports it.
178 *
179 * HOW: 1. Copy the answer (strtok_r mutates). 2. For each whitespace-delimited
180 * token, add map_holder()'s return to the holder count. 3. Print "(none)"
181 * if zero. 4. Print the "Total:" line, appending ", all serving" only when
182 * verifying with no ghosts.
183 */
184static void
185map_holders(const brix_opts *co, const char *loc, const char *path, int verify)
186{
187 char copy[8192];
188 char *tok, *save;
189 int n = 0;
190
191 snprintf(copy, sizeof(copy), "%s", loc);
192 for (tok = strtok_r(copy, " \t\r\n", &save); tok != NULL;
193 tok = strtok_r(NULL, " \t\r\n", &save)) {
194 n += map_holder(co, path, tok, verify);
195 }
196 if (n == 0) {
197 printf(" (none)\n");
198 }
199 printf("Total: %d holder(s)%s\n", n,
200 verify ? (g_ghosts ? "" : ", all serving") : "");
201}

203/* ---- Query and print the cluster map for one endpoint/path ----
204 *
205 * WHAT: Connects to `url`, locates `path`, and prints the cluster map (header,
206 * best-effort space, holder list). Returns 50 on URL parse error, the
207 * brix_shellcode() of a connect/locate failure, 1 if any ghost/unreachable
208 * holder was seen (--verify), else 0.
209 *
210 * WHY: Top-level orchestrator: a flat early-return sequence over the connect →
211 * locate → report steps, with the per-section detail delegated to helpers.
212 *
213 * HOW: 1. Parse the endpoint URL (error → 50). 2. Connect resiliently (error →
214 * shellcode). 3. Default an empty path to "/". 4. brix_locate() (error →
215 * close + shellcode). 5. Print header, print_space(), map_holders().
216 * 6. Close and return 1 if g_ghosts else 0.
217 */
218static int
219do_map(const char *url, const brix_opts *co, int verify)
220{
221 brix_url u;
222 brix_conn c;
223 brix_status st;
224 char loc[8192];
225 const char *path;
226
227 brix_status_clear(&st);
228 if (brix_endpoint_parse(url, &u, &st) != 0) {
229 fprintf(stderr, "xrdmapc: %s\n", st.msg);
230 return 50;
231 }
232 if (brix_connect_resilient(&c, &u, co, &st) != 0) {
233 fprintf(stderr, "xrdmapc: connect %s:%d: %s\n", u.host, u.port, st.msg);
234 return brix_shellcode(&st);
235 }
236 path = (u.path[0] != '\0') ? u.path : "/";
237
238 if (brix_locate(&c, path, loc, sizeof(loc), &st) != 0) {
239 fprintf(stderr, "xrdmapc: locate %s: %s\n", path, st.msg);
240 brix_close(&c);
241 return brix_shellcode(&st);
242 }
243 printf("Cluster map for %s:%d path %s\n", u.host, u.port, path);
244
245 print_space(&c, path);
246 printf("Holders:\n");
247 map_holders(co, loc, path, verify);
248
249 brix_close(&c);
250 return g_ghosts ? 1 : 0;
251}

271int
272main(int argc, char **argv)
273{
274 brix_opts co;
275 const char *url = NULL;
276 int verify = 0, i;
277
278 brix_opts_init(&co);
279 brix_crypto_init();
280
281 for (i = 1; i < argc; i++) {
282 const char *a = argv[i];
283 if (a[0] == '-' && a[1] != '\0' && strcmp(a, "-") != 0) {
284 if (strcmp(a, "--verify") == 0) {
285 verify = 1;
286 } else {
287 int pr = brix_opts_parse_arg(&co, argc, argv, &i);
288 if (pr == 2) { usage_fp(stdout); return 0; } /* --help */
289 if (pr) { /* common conn flag, already handled */ }
290 else if (strcmp(a, "-h") == 0) { usage(); return 0; } /* C1 */
291 else { fprintf(stderr, "xrdmapc: unknown option '%s'\n", a); usage(); return 50; }
292 }
293 } else if (url == NULL) {
294 url = a;
295 } else {
296 fprintf(stderr, "xrdmapc: too many arguments\n");
297 return 50;
298 }
299 }
300 if (url == NULL) {
301 usage();
302 return 50;
303 }
304 return do_map(url, &co, verify);
305}
