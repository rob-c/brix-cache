1/*
2 * backend_async_queue.h — durable write-behind queue for backend namespace
3 * mutations (brix_backend_async).
4 *
5 * WHAT: A per-worker coalescing queue for backend MUTATIONS (unlink / rmdir /
6 * rename / mkdir). When `brix_backend_async on` is set for a server, a mutation
7 * is (1) fsync'd to a durable journal record, (2) appended to the worker's
8 * pending batch with a protocol-supplied completion callback, and (3) the
9 * client is parked. The batch is drained in bulk — driven against the backend
10 * with the pool-less VFS path primitives — when it reaches `batch` ops OR the
11 * oldest op has waited `wait_ms` (whichever comes first). Each op's real result
12 * is then delivered to its parked client via the callback, and the durable
13 * record is removed. On worker start, brix_baq_reconcile() re-applies any op
14 * left un-drained by a crash (idempotently), so a mutation that the client was
15 * told to wait for is never silently lost across a reboot.
16 *
17 * WHY: Operators front high-latency / rate-limited origins (tape, object stores,
18 * remote XRootD) where a burst of individual namespace mutations is far more
19 * expensive than one coalesced flush. The queue trades per-op latency (the
20 * client blocks, and may time out — an accepted contract) for bulk backend
21 * efficiency and, crucially, durability: the reply is only sent once the op is
22 * truly on the backend, and a crash mid-batch replays rather than loses it.
23 * Reads / stats / lists are never queued — only mutations, so the namespace a
24 * subsequent request sees is always the post-flush truth for THIS client.
25 *
26 * HOW: The durable substrate mirrors the stage engine's journal (a fsync'd fixed-
27 * size record per op, scanned + replayed on boot) but keeps its OWN record type
28 * and OWN journal subdirectory (<stage_journal_dir>/backend/) so the two
29 * concerns stay decoupled — a namespace mutation is not a byte transfer and must
30 * not ride the stage mover / deny-cred flush machinery. The pending list and the
31 * park state are per-worker (the worker that parked a client is the only one
32 * that may wake it), so no SHM is needed — unlike the cross-worker stage_waiter.
33 * The core is protocol-agnostic: the root:// and S3/WebDAV adapters each supply
34 * a brix_baq_done_pt that wakes their own client with the delivered errno.
35 */

43/* The four coalescable namespace mutations. Values are stable on-disk identities
44 * (a journal record survives a restart), so they are append-only — never renumber. */
45typedef enum {
46 BRIX_BAQ_UNLINK = 0, /* remove a regular file */
47 BRIX_BAQ_RMDIR = 1, /* remove an empty directory */
48 BRIX_BAQ_RENAME = 2, /* rename src_key -> dst_key */
49 BRIX_BAQ_MKDIR = 3 /* create a directory (mode) */
50} brix_baq_op_t;
51
52/* The durable on-disk record — one per queued mutation, fixed size (fsync'd whole,
53 * decoded size-exactly on replay). root_canon anchors the backend the op is
54 * re-driven against on reconcile; src_key/dst_key are export-relative logical
55 * paths (dst_key used only by RENAME). APPEND-ONLY layout: never reorder or shrink
56 * a field, or old journals become unreadable. */
57typedef struct {
58 char reqid[40]; /* minted request id (NUL-terminated) */
59 uint32_t op; /* brix_baq_op_t */
60 char root_canon[1024]; /* export root (canonical) to resolve backend */
61 char src_key[1024]; /* target path; RENAME source */
62 char dst_key[1024]; /* RENAME destination; "" for single-target ops */
63 uint32_t mode; /* MKDIR mode; 0 otherwise */
64 int64_t enqueued_at; /* wall-clock enqueue time (seconds) */
65 uint32_t attempts; /* re-drive count (durable across restart) */
66} brix_baq_rec_t;
67
68/*
69 * Completion callback: the core calls this exactly once per queued op when the op
70 * has been driven against the backend, on the SAME worker that enqueued it.
71 * `client` is the opaque token the adapter passed to brix_baq_enqueue (an
72 * ngx_connection_t* for root://, an ngx_http_request_t* for S3/WebDAV);
73 * `op_errno` is 0 on success or the POSIX errno the backend mutation failed with.
74 * The adapter delivers the corresponding wire/HTTP status and resumes the client.
75 */
76typedef void (*brix_baq_done_pt)(void *client, int op_errno);

78/*
79 * Configure the queue's durable journal subdirectory for this worker.
80 *
81 * WHAT: Derives <stage_journal_dir>/backend/ from the stage engine's journal dir
82 * (the single source of truth, set by brix_stage_engine_init) and creates it
83 * (0700). A NULL/empty stage journal dir leaves the queue in-memory only (no
84 * crash recovery) — the coalescing still works, only durability is off.
85 *
86 * WHY: Keeps the async-queue records physically separate from the stage engine's
87 * *.req records so each subsystem's boot reconcile scans only its own.
88 *
89 * HOW: Called from the worker stage-engine init, AFTER brix_stage_engine_init.
90 */
91void brix_baq_init(void);

93/*
94 * Enqueue one mutation and park the caller's client.
95 *
96 * WHAT: Persists a durable record (when a journal dir is configured), appends the
97 * op to the worker's pending batch with its completion callback, and — if
98 * the batch now meets the size trigger — drains immediately (which invokes
99 * `done` for every op including this one, inline). Otherwise the op stays
100 * parked until a later size/time-triggered drain.
101 *
102 * WHY: The single entry point every protocol adapter calls after it has resolved
103 * + authorized the mutation and decided (from brix_backend_async) to defer.
104 *
105 * HOW: root_canon / src_key / dst_key / mode populate the record; batch and
106 * wait_ms are this server's coalescing thresholds (stored per-op so a worker
107 * shared by servers with different tunables still flushes each correctly).
108 * Returns NGX_OK when the op is queued (or drained inline), NGX_ERROR on a
109 * bad argument or a full queue that could not be drained (the caller then
110 * runs the op inline / errors, never silently drops it).
111 */
112ngx_int_t brix_baq_enqueue(brix_baq_op_t op, const char *root_canon,
113 const char *src_key, const char *dst_key, uint32_t mode,
114 ngx_uint_t batch, ngx_msec_t wait_ms,
115 brix_baq_done_pt done, void *client);

117/*
118 * Per-worker timer tick: drain the batch if the time trigger has fired.
119 *
120 * WHAT: If any parked op has waited >= its wait_ms, drains the whole batch now
121 * (delivering each op's result to its client). A no-op when the batch is
122 * empty or nothing has aged out yet.
123 *
124 * WHY: The size trigger fires inline in enqueue; the time trigger needs a clock
125 * source. Called from the same per-worker timer that drives the stage
126 * scheduler tick.
127 *
128 * HOW: Compares ngx_current_msec against each pending op's enqueue stamp.
129 */
130void brix_baq_tick(void);
131
132/*
133 * Drop any parked op belonging to `client` without delivering (best-effort).
134 *
135 * WHAT: Removes pending ops whose opaque client token equals `client` (their
136 * durable records are LEFT on disk, so a still-un-drained mutation is
137 * replayed at next boot rather than lost). Used when a parked client
138 * disconnects before its batch drains.
139 *
140 * WHY: A woken callback must never touch a freed connection/request. The adapter
141 * calls this from its disconnect/cleanup handler so the drain skips the dead
142 * client — the mutation itself still lands (inline drain of the rest, or
143 * reconcile).
144 */
145void brix_baq_drop_client(void *client);
146
147/*
148 * Restart replay: re-apply every persisted mutation left un-drained by a crash.
149 *
150 * WHAT: Scans <stage_journal_dir>/backend/, decodes each record, re-runs the op
151 * against its backend IDEMPOTENTLY (unlink/rmdir of a missing target = done,
152 * mkdir of an existing dir = done, rename of an already-moved source = done),
153 * and removes the record on success. Called on worker 0 at startup.
154 *
155 * WHY: The client was told to wait for the op; durability means a crash between
156 * "journalled" and "drained" must not lose it.
157 *
158 * HOW: Mirrors brix_stage_reconcile's snapshot-then-drive loop; no client to wake
159 * (the parked connections died with the crash), so it only re-drives the
160 * backend effect.
161 */
162void brix_baq_reconcile(void);
