1/*
2 * write.c — kXR_write opcode handler.
3 *
4 * This file implements the standard XRootD write operation: client sends data,
5 * server writes it to a local file at the specified offset. The flow is:
6 * 1. Validate that the requested file handle is open and writable
7 * 2. If thread pool available → post async pwrite task (non-blocking)
8 * 3. Otherwise → synchronously write from recv buffer to disk
9 * 4. Track bytes written for access-log throughput calculation
10 * 5. Return kXR_ok response to client
11 */
12
13/* Write-through dirty state tracking section
14 * This is the core of the Policy File Cache (PFC) write-back feature.
15 * When wt_enabled = 1, every write increments two counters:
16 * - wt_bytes_written: cumulative total for session metrics reporting
17 * - wt_dirty_offset: highest offset that hasn't been flushed to origin yet
18 *
19 * These values are retained for a future close-time write-back implementation.
20 * At close time, if wt_dirty_offset > -1 (dirty data exists), the flush handler
21 * propagates those bytes back to an origin XRootD server before closing the handle.
22 */
23
24/* AIO (Async I/O) section
25 * When nginx has a thread pool configured, writes are posted to background threads
26 * so the main event loop doesn't block on disk I/O. The payload buffer is detached
27 * from ctx->recv.payload_buf and freed in the completion callback. This allows the main
28 * thread to continue reading the next request header while write happens elsewhere. */
29
30/* Synchronous fallback section
31 * When no thread pool is configured OR queue is full, writes happen synchronously
32 * on the main event loop thread using pwrite() directly from the recv buffer.
33 * This ensures writes always succeed even under degraded conditions. */
34
35/*
36 *
37 * WHAT: Protocol-level kXR_write handler — validates write handle (must be open and writable), handles zero-length writes as valid no-ops, dispatches to AIO thread pool when configured (pwrite in worker thread with detached payload buffer), falls back to synchronous inline pwrite on main event loop. Tracks bytes written for access-log throughput calculation and PFC write-through dirty state tracking (wt_bytes_written + wt_dirty_offset for future close-time origin flush). Access-log detail format: "<offset>+<requested-bytes>".
38 *
39 * WHY: AIO dispatch detaches the payload buffer from ctx->recv.payload_buf so the main thread can safely begin reading the next request header while write happens in a worker thread. PFC write-through dirty state tracking accumulates wt_bytes_written and wt_dirty_offset for future close-time origin propagation — mirroring XrdPfcFile::m_bytesWritten, m_dirtyOffset semantics. Short-write detection catches disk-full conditions before silently truncating client data. */

59/*
60 * brix_write_route_special — dispatch a validated write handle to its
61 * non-standard early-return path (SSI accumulate, zero-length no-op, whole-object
62 * staged commit, inline write-decompression, or recovery-journal replay skip)
63 * before the default AIO/sync fast path runs.
64 *
65 * WHAT: Checks, in wire order, every condition that must short-circuit
66 * brix_handle_write. On a match it performs the matching action and writes the
67 * handler return code to *out_rc, returning 1 (handled). Returns 0 when the
68 * handle takes the normal file-write path (out_rc untouched).
69 * WHY: Factoring these mutually-exclusive early returns out of brix_handle_write
70 * keeps the hot AIO/sync body — and its CCN — small while preserving the exact
71 * ordering and behavior of every special case.
72 * HOW: Ordered guards on ctx->files[idx] state (ssi / wlen==0 / staged /
73 * write_codec / wrts replay); each mirrors the original inline block verbatim.
74 */
75static ngx_flag_t
76brix_write_route_special(brix_ctx_t *ctx, ngx_connection_t *c,
77 const brix_write_desc_t *w, ngx_int_t *out_rc)
78{
79 int idx = w->idx;
80
81 /* §7 XrdSsi: an SSI handle accumulates the request body (no fd write). The
82 * write offset carries an XrdSsiRRInfo (raw big-endian bytes). Clean
83 * early-return — the normal write path below is unchanged for file handles. */
84 if (ctx->files[idx].ssi != NULL) {
85 BRIX_OP_OK(ctx, BRIX_OP_WRITE);
86 /* SSI reinterprets the offset field as a raw big-endian XrdSsiRRInfo;
87 * pass the wire bytes (offset field = body byte 4), not the decoded int. */
88 *out_rc = brix_ssi_write(ctx, c, idx, w->hdrbody + 4);
89 return 1;
90 }
91
92 if (w->wlen == 0) {
93 /* Zero-length writes are valid no-ops that still count as successful requests. */
94 BRIX_OP_OK(ctx, BRIX_OP_WRITE);
95 *out_rc = brix_send_ok(ctx, c, NULL, 0);
96 return 1;
97 }
98
99 /* Whole-object staged-commit adapter (phase-70): this handle writes to a
100 * backend with no random write, so the block is APPENDED to a VFS staged
101 * handle at the running offset and the object is PUT whole on sync/close.
102 * Isolated early-return keeps the AIO/journal/wt fast path below unchanged. */
103 if (ctx->files[idx].writer != NULL) {
104 *out_rc = brix_write_staged(ctx, c, idx, w->offset, w->wlen);
105 return 1;
106 }
107
108 /*
109 * Phase-42 W5: inline write decompression (opt-in, off by default). Routed
110 * to its own isolated synchronous handler so EVERYTHING below — the AIO write
111 * fast path and the write-recovery journal — stays byte-identical for the
112 * default (write_codec == 0) case. pgwrite/writev have their own handlers and
113 * never reach here, so their plaintext + per-page CRC32c invariant is kept.
114 */
115 if (ctx->files[idx].write_codec != 0) {
116 *out_rc = brix_write_compressed(ctx, c, idx, w->offset, w->wlen);
117 return 1;
118 }
119
120 /* kXR_recoverWrts replay detection
121 * If the journal already covers this (offset, wlen) range, this is a
122 * replayed write from a recovering XrdCl client. The bytes are already
123 * on disk — skip pwrite() and return kXR_ok so the client can advance.
124 */
125 if (ctx->files[idx].wrts_enabled &&
126 brix_wrts_is_replay(&ctx->files[idx], w->offset, (uint32_t) w->wlen))
127 {
128 ngx_log_debug(NGX_LOG_DEBUG_STREAM, c->log, 0,
129 "brix: write recovery replay skip offset=%L len=%uz",
130 w->offset, w->wlen);
131 BRIX_OP_OK(ctx, BRIX_OP_WRITE);
132 *out_rc = brix_send_ok(ctx, c, NULL, 0);
133 return 1;
134 }
135
136 return 0;
137}

139/*
140 * brix_write_finalize_sync — commit the accounting and journaling side effects
141 * of a completed synchronous pwrite and return the client response.
142 *
143 * WHAT: Given the byte count returned by the inline pwrite, emits the kXR error
144 * response on failure (negative / short write) or, on success, updates the
145 * file+session byte counters, bandwidth charge, dashboard slot, write-through
146 * dirty state and recovery journal, then returns the kXR_ok response.
147 * WHY: Splitting the post-write bookkeeping out of brix_handle_write keeps the
148 * handler's cyclomatic complexity and length within budget while leaving every
149 * counter update, error path and return code byte-identical.
150 * HOW: Mirrors the original inline tail verbatim; BRIX_RETURN_ERR / BRIX_RETURN_OK
151 * expand to a return, so this helper's return value is the handler's return.
152 */
153static ngx_int_t
154brix_write_finalize_sync(brix_ctx_t *ctx, ngx_connection_t *c,
155 const brix_write_desc_t *w, ssize_t nwritten,
156 const char *write_detail)
157{
158 int idx = w->idx;
159 int64_t offset = w->offset;
160
161 if (nwritten < 0) {
162 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
163 ctx->files[idx].path, write_detail,
164 kXR_IOError, strerror(errno));
165 }
166
167 if ((size_t) nwritten < w->wlen) {
168 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
169 ctx->files[idx].path, write_detail,
170 kXR_IOError, "short write (disk full?)");
171 }
172
173 ctx->files[idx].bytes_written += (size_t) nwritten;
174 ctx->totals.bytes_written += (size_t) nwritten;
175 brix_rl_charge_ctx(ctx, (size_t) nwritten); /* Phase 25 bandwidth */
176
177 if (ctx->files[idx].dashboard_slot >= 0 &&
178 ngx_brix_dashboard_shm_zone != NULL)
179 {
180 brix_transfer_slot_update(ngx_brix_dashboard_shm_zone->data,
181 ctx->files[idx].dashboard_slot,
182 (ngx_atomic_int_t) nwritten,
183 (int64_t) ngx_current_msec);
184 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
185 ctx->files[idx].dashboard_slot,
186 "write");
187 }
188
189 /* write-through dirty state tracking (mirrors XrdPfcFile::m_bytesWritten, m_dirtyOffset)
190 * When wt_enabled = 1 this handle has a cached DECISION to propagate writes
191 * back to the origin at close time. We track:
192 * - wt_bytes_written: cumulative bytes since last sync point (for metrics)
193 * - wt_dirty_offset: offset of most recent write that hasn't been flushed yet
194 *
195 * These fields are retained for a future close-time write-back implementation.
196 */
197 if (ctx->files[idx].wt_enabled) {
198 brix_wt_mark_dirty(ctx, idx,
199 offset + (int64_t) nwritten - 1,
200 (size_t) nwritten);
201 }
202
203 /* Record the committed write in the recovery journal. */
204 if (ctx->files[idx].wrts_enabled) {
205 brix_wrts_record(&ctx->files[idx], offset, (uint32_t) nwritten);
206 }
207
208 BRIX_RETURN_OK(ctx, c, BRIX_OP_WRITE, "WRITE",
209 ctx->files[idx].path, write_detail, (size_t) nwritten);
210}

212/*
213 * brix_write_require_pgwrite — hostile-network wire-integrity gate for the
214 * cleartext native write ops (kXR_write, kXR_writev).
215 *
216 * WHAT: When brix_require_pgwrite is on, refuses a cleartext data write on a
217 * writable file handle with kXR_Unsupported, so the deployment forces clients
218 * onto the per-page-CRC32c kXR_pgwrite path. Returns 1 (handled — the error
219 * reply is sent and *rc holds its return) or 0 (write proceeds; *rc untouched).
220 * WHY: Plain kXR_write/kXR_writev carry NO checksum (INVARIANT 1 only covers
221 * pgwrite), so a byte a hostile middlebox flips past the TCP checksum is
222 * committed undetected; brix_verify_write is a storage read-back, not a wire
223 * check. This is the native-write analogue of brix_webdav_require_digest /
224 * brix_gridftp_require_allo_size — a fail-closed integrity posture for
225 * deployments whose clients already speak pgwrite.
226 * HOW: Exempt the two payload-free cases so the knob never breaks control
227 * traffic: an SSI handle (accumulates a request body, not a file write) and a
228 * zero-length write (a valid no-op with nothing to corrupt). Everything else on
229 * a writable handle carries data and is refused.
230 */
231ngx_int_t
232brix_write_require_pgwrite(brix_ctx_t *ctx, ngx_connection_t *c, int idx,
233 size_t wlen, const char *op, ngx_int_t *rc)
234{
235 ngx_stream_brix_srv_conf_t *conf;
236
237 conf = ngx_stream_get_module_srv_conf(
238 (ngx_stream_session_t *) c->data, ngx_stream_brix_module);
239
240 if (!conf->common.require_pgwrite || wlen == 0
241 || ctx->files[idx].ssi != NULL)
242 {
243 return 0;
244 }
245
246 brix_log_access(ctx, c, op, ctx->files[idx].path, "-", 0,
247 kXR_Unsupported,
248 "cleartext write refused; use pgwrite "
249 "(brix_require_pgwrite)", 0);
250 BRIX_OP_ERR(ctx, BRIX_OP_WRITE);
251 *rc = brix_send_error(ctx, c, kXR_Unsupported,
252 "cleartext write refused; use pgwrite (brix_require_pgwrite)");
253 return 1;
254}

256/*
257 * brix_handle_write — handle kXR_write: write the request payload to an
258 * open file at the specified offset.
259 *
260 * Wire format (from ClientWriteRequest):
261 * fhandle[4]: open file handle (first byte is the slot index)
262 * offset: big-endian int64 file position
263 * dlen: payload byte count (ctx->recv.cur_dlen, already received)
264 *
265 * When a thread pool is configured, the pwrite is posted asynchronously and
266 * the payload buffer is detached from ctx->recv.payload_buf so the main thread can
267 * safely begin reading the next request header. The done callback frees the
268 * detached buffer and sends the kXR_ok response.
269 *
270 * Falls back to synchronous pwrite if no thread pool is configured or the
271 * thread pool queue is full.
272 *
273 * Access-log detail: "<offset>+<requested-bytes>"
274 */
275ngx_int_t
276brix_handle_write(brix_ctx_t *ctx, ngx_connection_t *c)
277{
278 const u_char *hdrbody = ((ClientRequestHdr *) ctx->recv.hdr_buf)->body;
279 xrdw_write_req_t req;
280 int idx;
281 int64_t offset;
282 size_t wlen = ctx->recv.cur_dlen;
283
284 xrdw_write_req_unpack(hdrbody, &req);
285 idx = (int)(unsigned char) req.fhandle[0];
286 offset = req.offset;
287 ngx_int_t rc = NGX_OK;
288 ssize_t nwritten = 0;
289 char write_detail[64];
290 brix_write_desc_t w = { idx, offset, wlen, hdrbody };
291
292 if (!brix_validate_write_handle(ctx, c, idx, "WRITE",
293 BRIX_OP_WRITE, &rc)) {
294 return rc;
295 }
296
297 /* INVARIANT 1 / hostile-network integrity: a cleartext kXR_write carries NO
298 * per-page CRC32c (unlike kXR_pgwrite), so a wire bit-flip past the TCP
299 * checksum is committed undetected. brix_require_pgwrite lets a deployment
300 * refuse cleartext data writes on a file handle, forcing clients onto the
301 * checksummed pgwrite path. SSI (request accumulation, not a file write) and
302 * zero-length no-ops carry no data to protect and stay exempt. */
303 if (brix_write_require_pgwrite(ctx, c, idx, wlen, "WRITE", &rc)) {
304 return rc;
305 }
306
307 if (brix_write_route_special(ctx, c, &w, &rc)) {
308 return rc;
309 }
310
311 {
312 ngx_flag_t posted = 0;
313
314 rc = brix_try_post_write_aio(ctx, c, idx, (off_t) offset,
315 ctx->recv.payload ? ctx->recv.payload : (u_char *) "",
316 wlen, offset, 0, ctx->recv.payload, NULL, 0,
317 "brix: thread_task_post failed, falling back to sync write",
318 &posted);
319 if (rc != NGX_OK) {
320 return rc;
321 }
322 if (posted) {
323 ctx->recv.payload = NULL;
324 ctx->recv.payload_buf = NULL;
325 ctx->recv.payload_buf_size = 0;
326 /*
327 * Write pipelining: account this pwrite as in-flight. The recv loop
328 * (which sees state == XRD_ST_AIO on return) keeps receiving the next
329 * write while this one runs, bounded by out_count + wr_inflight <
330 * ctx->out.pipeline_depth. brix_write_aio_done decrements wr_inflight and
331 * queues the ack asynchronously (no recv suspend).
332 */
333 ctx->out.wr_inflight++;
334 return NGX_OK;
335 }
336 } /* end NGX_THREADS block */
337
338
339 /* Synchronous fallback writes the request payload directly from the recv buffer. */
340 {
341 brix_vfs_job_t job;
342
343 brix_vfs_job_write_init(&job, ctx->files[idx].fd, (off_t) offset,
344 ctx->recv.payload ? ctx->recv.payload : (u_char *) "",
345 wlen);
346 job.csi = ctx->files[idx].csi; /* phase-59 W2: update page tags */
347 brix_vfs_job_set_obj(&job, &ctx->files[idx].sd_obj);
348 brix_vfs_io_execute(&job);
349 nwritten = job.nio;
350 if (job.io_errno != 0) {
351 errno = job.io_errno;
352 }
353 }
354
355 /* Access log detail format for writes is "<offset>+<requested-bytes>". */
356 snprintf(write_detail, sizeof(write_detail), "%lld+%zu",
357 (long long) offset, wlen);
358
359 return brix_write_finalize_sync(ctx, c, &w, nwritten, write_detail);
360}
