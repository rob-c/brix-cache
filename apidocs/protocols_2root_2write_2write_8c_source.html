1/*
2 * write.c — kXR_write opcode handler.
3 *
4 * This file implements the standard XRootD write operation: client sends data,
5 * server writes it to a local file at the specified offset. The flow is:
6 * 1. Validate that the requested file handle is open and writable
7 * 2. If thread pool available → post async pwrite task (non-blocking)
8 * 3. Otherwise → synchronously write from recv buffer to disk
9 * 4. Track bytes written for access-log throughput calculation
10 * 5. Return kXR_ok response to client
11 */
12
13/* Write-through dirty state tracking section
14 * This is the core of the Policy File Cache (PFC) write-back feature.
15 * When wt_enabled = 1, every write increments two counters:
16 * - wt_bytes_written: cumulative total for session metrics reporting
17 * - wt_dirty_offset: highest offset that hasn't been flushed to origin yet
18 *
19 * These values are retained for a future close-time write-back implementation.
20 * At close time, if wt_dirty_offset > -1 (dirty data exists), the flush handler
21 * propagates those bytes back to an origin XRootD server before closing the handle.
22 */
23
24/* AIO (Async I/O) section
25 * When nginx has a thread pool configured, writes are posted to background threads
26 * so the main event loop doesn't block on disk I/O. The payload buffer is detached
27 * from ctx->recv.payload_buf and freed in the completion callback. This allows the main
28 * thread to continue reading the next request header while write happens elsewhere. */
29
30/* Synchronous fallback section
31 * When no thread pool is configured OR queue is full, writes happen synchronously
32 * on the main event loop thread using pwrite() directly from the recv buffer.
33 * This ensures writes always succeed even under degraded conditions. */
34
35/*
36 *
37 * WHAT: Protocol-level kXR_write handler — validates write handle (must be open and writable), handles zero-length writes as valid no-ops, dispatches to AIO thread pool when configured (pwrite in worker thread with detached payload buffer), falls back to synchronous inline pwrite on main event loop. Tracks bytes written for access-log throughput calculation and PFC write-through dirty state tracking (wt_bytes_written + wt_dirty_offset for future close-time origin flush). Access-log detail format: "<offset>+<requested-bytes>".
38 *
39 * WHY: AIO dispatch detaches the payload buffer from ctx->recv.payload_buf so the main thread can safely begin reading the next request header while write happens in a worker thread. PFC write-through dirty state tracking accumulates wt_bytes_written and wt_dirty_offset for future close-time origin propagation — mirroring XrdPfcFile::m_bytesWritten, m_dirtyOffset semantics. Short-write detection catches disk-full conditions before silently truncating client data. */

46/*
47 * brix_handle_write — handle kXR_write: write the request payload to an
48 * open file at the specified offset.
49 *
50 * Wire format (from ClientWriteRequest):
51 * fhandle[4]: open file handle (first byte is the slot index)
52 * offset: big-endian int64 file position
53 * dlen: payload byte count (ctx->recv.cur_dlen, already received)
54 *
55 * When a thread pool is configured, the pwrite is posted asynchronously and
56 * the payload buffer is detached from ctx->recv.payload_buf so the main thread can
57 * safely begin reading the next request header. The done callback frees the
58 * detached buffer and sends the kXR_ok response.
59 *
60 * Falls back to synchronous pwrite if no thread pool is configured or the
61 * thread pool queue is full.
62 *
63 * Access-log detail: "<offset>+<requested-bytes>"
64 */
65ngx_int_t
66brix_handle_write(brix_ctx_t *ctx, ngx_connection_t *c)
67{
68 const u_char *hdrbody = ((ClientRequestHdr *) ctx->recv.hdr_buf)->body;
69 xrdw_write_req_t req;
70 int idx;
71 int64_t offset;
72 size_t wlen = ctx->recv.cur_dlen;
73
74 xrdw_write_req_unpack(hdrbody, &req);
75 idx = (int)(unsigned char) req.fhandle[0];
76 offset = req.offset;
77 ngx_int_t rc;
78 ssize_t nwritten;
79 char write_detail[64];
80
81 if (!brix_validate_write_handle(ctx, c, idx, "WRITE",
82 BRIX_OP_WRITE, &rc)) {
83 return rc;
84 }
85
86 /* §7 XrdSsi: an SSI handle accumulates the request body (no fd write). The
87 * write offset carries an XrdSsiRRInfo (raw big-endian bytes). Clean
88 * early-return — the normal write path below is unchanged for file handles. */
89 if (ctx->files[idx].ssi != NULL) {
90 BRIX_OP_OK(ctx, BRIX_OP_WRITE);
91 /* SSI reinterprets the offset field as a raw big-endian XrdSsiRRInfo;
92 * pass the wire bytes (offset field = body byte 4), not the decoded int. */
93 return brix_ssi_write(ctx, c, idx, hdrbody + 4);
94 }
95
96 if (wlen == 0) {
97 /* Zero-length writes are valid no-ops that still count as successful requests. */
98 BRIX_OP_OK(ctx, BRIX_OP_WRITE);
99 return brix_send_ok(ctx, c, NULL, 0);
100 }
101
102 /*
103 * Phase-42 W5: inline write decompression (opt-in, off by default). Routed
104 * to its own isolated synchronous handler so EVERYTHING below — the AIO write
105 * fast path and the write-recovery journal — stays byte-identical for the
106 * default (write_codec == 0) case. pgwrite/writev have their own handlers and
107 * never reach here, so their plaintext + per-page CRC32c invariant is kept.
108 */
109 if (ctx->files[idx].write_codec != 0) {
110 return brix_write_compressed(ctx, c, idx, offset, wlen);
111 }
112
113 /* kXR_recoverWrts replay detection
114 * If the journal already covers this (offset, wlen) range, this is a
115 * replayed write from a recovering XrdCl client. The bytes are already
116 * on disk — skip pwrite() and return kXR_ok so the client can advance.
117 */
118 if (ctx->files[idx].wrts_enabled &&
119 brix_wrts_is_replay(&ctx->files[idx], offset, (uint32_t) wlen))
120 {
121 ngx_log_debug(NGX_LOG_DEBUG_STREAM, c->log, 0,
122 "brix: write recovery replay skip offset=%L len=%uz",
123 offset, wlen);
124 BRIX_OP_OK(ctx, BRIX_OP_WRITE);
125 return brix_send_ok(ctx, c, NULL, 0);
126 }
127
128 {
129 ngx_flag_t posted;
130
131 rc = brix_try_post_write_aio(ctx, c, idx, (off_t) offset,
132 ctx->recv.payload ? ctx->recv.payload : (u_char *) "",
133 wlen, offset, 0, ctx->recv.payload, NULL, 0,
134 "brix: thread_task_post failed, falling back to sync write",
135 &posted);
136 if (rc != NGX_OK) {
137 return rc;
138 }
139 if (posted) {
140 ctx->recv.payload = NULL;
141 ctx->recv.payload_buf = NULL;
142 ctx->recv.payload_buf_size = 0;
143 /*
144 * Write pipelining: account this pwrite as in-flight. The recv loop
145 * (which sees state == XRD_ST_AIO on return) keeps receiving the next
146 * write while this one runs, bounded by out_count + wr_inflight <
147 * ctx->out.pipeline_depth. brix_write_aio_done decrements wr_inflight and
148 * queues the ack asynchronously (no recv suspend).
149 */
150 ctx->out.wr_inflight++;
151 return NGX_OK;
152 }
153 } /* end NGX_THREADS block */
154
155
156 /* Synchronous fallback writes the request payload directly from the recv buffer. */
157 {
158 brix_vfs_job_t job;
159
160 brix_vfs_job_write_init(&job, ctx->files[idx].fd, (off_t) offset,
161 ctx->recv.payload ? ctx->recv.payload : (u_char *) "",
162 wlen);
163 job.csi = ctx->files[idx].csi; /* phase-59 W2: update page tags */
164 brix_vfs_job_set_obj(&job, &ctx->files[idx].sd_obj);
165 brix_vfs_io_execute(&job);
166 nwritten = job.nio;
167 if (job.io_errno != 0) {
168 errno = job.io_errno;
169 }
170 }
171
172 /* Access log detail format for writes is "<offset>+<requested-bytes>". */
173 snprintf(write_detail, sizeof(write_detail), "%lld+%zu",
174 (long long) offset, wlen);
175
176 if (nwritten < 0) {
177 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
178 ctx->files[idx].path, write_detail,
179 kXR_IOError, strerror(errno));
180 }
181
182 if ((size_t) nwritten < wlen) {
183 BRIX_RETURN_ERR(ctx, c, BRIX_OP_WRITE, "WRITE",
184 ctx->files[idx].path, write_detail,
185 kXR_IOError, "short write (disk full?)");
186 }
187
188 ctx->files[idx].bytes_written += (size_t) nwritten;
189 ctx->totals.bytes_written += (size_t) nwritten;
190 brix_rl_charge_ctx(ctx, (size_t) nwritten); /* Phase 25 bandwidth */
191
192 if (ctx->files[idx].dashboard_slot >= 0 &&
193 ngx_brix_dashboard_shm_zone != NULL)
194 {
195 brix_transfer_slot_update(ngx_brix_dashboard_shm_zone->data,
196 ctx->files[idx].dashboard_slot,
197 (ngx_atomic_int_t) nwritten,
198 (int64_t) ngx_current_msec);
199 brix_transfer_slot_count_op(ngx_brix_dashboard_shm_zone->data,
200 ctx->files[idx].dashboard_slot,
201 "write");
202 }
203
204 /* write-through dirty state tracking (mirrors XrdPfcFile::m_bytesWritten, m_dirtyOffset)
205 * When wt_enabled = 1 this handle has a cached DECISION to propagate writes
206 * back to the origin at close time. We track:
207 * - wt_bytes_written: cumulative bytes since last sync point (for metrics)
208 * - wt_dirty_offset: offset of most recent write that hasn't been flushed yet
209 *
210 * These fields are retained for a future close-time write-back implementation.
211 */
212 if (ctx->files[idx].wt_enabled) {
213 brix_wt_mark_dirty(ctx, idx,
214 offset + (int64_t) nwritten - 1,
215 (size_t) nwritten);
216 }
217
218 /* Record the committed write in the recovery journal. */
219 if (ctx->files[idx].wrts_enabled) {
220 brix_wrts_record(&ctx->files[idx], offset, (uint32_t) nwritten);
221 }
222
223 BRIX_RETURN_OK(ctx, c, BRIX_OP_WRITE, "WRITE",
224 ctx->files[idx].path, write_detail, (size_t) nwritten);
225}

227/* HOW: Extracts idx from req->fhandle[0], offset from be64toh(req->offset), wlen from ctx->recv.cur_dlen. Validates write handle via brix_validate_write_handle() — returns early on failure. Zero-length writes return kXR_ok immediately as valid no-ops. NGX_THREADS block: calls brix_try_post_write_aio() with detached payload; if posted=1 sets ctx->recv.payload=NULL and returns NGX_OK (completion callback sends response); if posted=0 falls through to sync write. Synchronous fallback: pwrite(fd, payload, wlen, offset) inline. Logs access detail "<offset>+<wlen>". If wt_enabled updates wt_bytes_written and wt_dirty_offset for PFC write-through tracking. Returns BRIX_RETURN_OK. */
