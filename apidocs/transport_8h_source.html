4/*
5 * transport.h — origin-transport seam for the read-through cache.
6 *
7 * WHAT: A small vtable that abstracts "where the cached bytes come from" so the
8 * fill engine (fetch.c / slice_fill.c) is independent of the origin's wire
9 * protocol. Today the only driver is xroot:// (the historical
10 * origin_protocol.c functions); Phase 2 adds an HTTP(S) driver (libcurl)
11 * and Phase 3 a Pelican-federation driver layered on top of it.
12 *
13 * WHY: Without a seam, adding an HTTP or Pelican origin meant forking the fill
14 * loop and the commit/verify path per protocol. The seam keeps a single
15 * fill engine and a single checksum-on-fill integration (verify.c) that
16 * works identically for every transport: each driver only has to (a)
17 * stream a byte range to a local fd and (b) report the origin's advertised
18 * content digest. This mirrors the storage-driver seam in
19 * src/fs/backend/sd.h.
20 *
21 * HOW: brix_cache_origin_url_parse() classifies the configured origin /
22 * request URL into a scheme and selects the matching driver. The fill
23 * worker calls connect → open (learn size) → read_range* → checksum →
24 * close. All driver calls run in an nginx thread-pool worker (blocking
25 * I/O); none touch the event loop. A driver advertises what it can do via
26 * `caps` so the engine can skip unsupported steps (e.g. an origin that
27 * cannot report a checksum yields BRIX_CACHE_DIGEST_NONE and the
28 * best-effort verify policy commits the fill unverified).
29 */

71/*
72 * brix_cache_transport_t — the driver vtable.
73 *
74 * Every function runs in a fill thread-pool worker and reports failures through
75 * the brix_cache_fill_t error fields (brix_cache_set_error/_syserror), so the
76 * fill engine only checks the int return. `state` is an opaque per-fill driver
77 * handle the driver allocates in connect() and frees in close().
78 */
79typedef struct {
80 const char *name; /* "xroot" / "http" / "pelican" (for logs) */
81 uint32_t caps; /* BRIX_CACHE_CAP_* bitmask */

83 /* Establish a session to the origin for t (resolved url in *u). Returns 0 /
84 * -1 (t error set). On success *state is the driver handle. */
85 int (*connect)(brix_cache_fill_t *t, const brix_cache_origin_url_t *u,
86 void **state);
87 /* Open the source object for reading and learn its size into *out_size.
88 * Returns 0 / -1 (t error set). */
89 int (*open)(brix_cache_fill_t *t, void *state, uint64_t *out_size);
90 /* Stream [off, off+len) from the origin straight to dst_fd, writing *got
91 * bytes (may be < len at EOF). Returns 0 / -1 (t error set). */
92 int (*read_range)(brix_cache_fill_t *t, void *state, uint64_t off,

94 /* Report the origin's advertised content digest into *out. Sets out->alg[0]

97 int (*checksum)(brix_cache_fill_t *t, void *state,

100 void (*close)(brix_cache_fill_t *t, void *state);
101} brix_cache_transport_t;
